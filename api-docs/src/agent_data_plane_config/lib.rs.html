1//! ADP-native configuration model: the typed target of configuration translation.
2//!
3//! This crate owns the domain-shaped model types that translation produces and that ADP runtime
4//! code consumes: `SalukiConfiguration { control, shared, domains }`, `ControlConfiguration`,
5//! `SharedConfiguration`, `DomainConfiguration` and its per-domain structs.
6//!
7//! It does not embed component config structs (those stay in `saluki-components`, built from this
8//! model). It depends on neither the raw configuration map nor the Datadog source model, so a
9//! consumer can depend on it without inheriting either.
10//!
11//! Every field is plain, source-agnostic data. There are no source key names in identifiers and no
12//! source serde (these structs are serialized for the `/config/internal` view but never
13//! deserialized from a source language; that is the source adapter's job).
14
15use serde::Serialize;
16
17pub mod control;
18pub mod domains;
19pub mod live;
20pub mod shared;
21
22pub use control::{ControlConfiguration, ListenAddress, Logging};
23pub use domains::DomainConfiguration;
24pub use live::Live;
25pub use shared::SharedConfiguration;

27/// The complete ADP-native runtime configuration after translation.
28///
29/// Two writers fill it: the Datadog witness `drive` (schema fields) and `seed` (Saluki-only
30/// fields). They write disjoint fields. It is read by the orchestration layer (`control`) and by
31/// components at topology assembly (`shared` and `domains`).
32#[derive(Clone, Debug, Default, PartialEq, Serialize)]
33pub struct SalukiConfiguration {
34    /// Read first: decides which pipelines/topology to build. Orchestration layer only.
35    pub control: ControlConfiguration,
36    /// Cross-cutting values consumed by more than one domain, each with a single home.
37    pub shared: SharedConfiguration,
38    /// Per-domain resolved config, grouped by ownership domain.
39    pub domains: DomainConfiguration,
40}
