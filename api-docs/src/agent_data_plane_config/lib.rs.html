1//! ADP-native configuration model: the typed target of configuration translation.
2//!
3//! This crate owns the domain-shaped model types that translation produces and that ADP runtime
4//! code consumes: `SalukiConfiguration { control, shared, domains }`, `ControlConfiguration`,
5//! `SharedConfiguration`, `DomainConfiguration` and its per-domain structs.
6//!
7//! It does not embed component config structs (those stay in `saluki-components`, built from this
8//! model). It depends on neither the raw configuration map nor the Datadog source model, so a
9//! consumer can depend on it without inheriting either.
10//!
11//! Every field is plain, source-agnostic data. There are no source key names in identifiers and no
12//! source serde (these structs are serialized for the `/config/internal` view but never
13//! deserialized from a source language; that is the source adapter's job).
14

29/// The complete ADP-native runtime configuration after translation.
30///
31/// Two writers fill it: the Datadog witness `drive` (schema fields) and `seed` (Saluki-only
32/// fields). They write disjoint fields. It is read by the orchestration layer (`control`) and by
33/// components at topology assembly (`shared` and `domains`).
34#[derive(Clone, Debug, Default, PartialEq, Serialize)]
35pub struct SalukiConfiguration {
36    /// Read first: decides which pipelines/topology to build. Orchestration layer only.
37    pub control: ControlConfiguration,
38    /// Cross-cutting values consumed by more than one domain, each with a single home.
39    pub shared: SharedConfiguration,
40    /// Per-domain resolved config, grouped by ownership domain.
41    pub domains: DomainConfiguration,
42}
43
44/// An error produced while translating a `DatadogConfiguration` or `SalukiOnly` value into a
45/// `SalukiConfiguration` value.
46#[derive(Debug)]
47pub struct Error {
48    context: String,
49    error: Option<Box<dyn std::error::Error + Send + Sync + 'static>>,
50}
51
52/// A boxed source error that can cross thread boundaries.
53pub type BoxError = Box<dyn std::error::Error + Send + Sync + 'static>;
54
55impl Error {
56    /// Create an error that wraps an underlying error.
57    ///
58    /// Displays `{context}: {error}`.
59    pub fn new<S, E>(context: S, error: E) -> Self
60    where
61        S: Into<String>,
62        E: Into<BoxError>,
63    {
64        Self {
65            context: context.into(),
66            error: Some(error.into()),
67        }
68    }
69
70    /// Create an error without an underlying source error.
71    pub fn new_without_source<S>(context: S) -> Self
72    where
73        S: Into<String>,
74    {
75        Self {
76            context: context.into(),
77            error: None,
78        }
79    }
80}
81
82impl fmt::Display for Error {
83    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
84        write!(f, "{}", self.context)?;
85        if let Some(error) = &self.error {
86            write!(f, ": {error}")?;
87        }
88        Ok(())
89    }
90}
91
92impl std::error::Error for Error {
93    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
94        self.error.as_deref().map(|s| s as &(dyn std::error::Error + 'static))
95    }
96}
