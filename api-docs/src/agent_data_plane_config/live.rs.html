1//! `Live<T>`: a live, typed view of one projection of the current configuration.
2
3use std::fmt;
4use std::ops::Deref;
5use std::sync::Arc;
6
7use arc_swap::ArcSwap;
8use tokio::sync::watch;
9
10use crate::SalukiConfiguration;
11
12// Borrow-in, borrow-out so navigating deeper never clones the whole configuration; the caller clones
13// only the final `T` when it needs an owned value.
14type Projection<T> = Arc<dyn for<'a> Fn(&'a SalukiConfiguration) -> &'a T + Send + Sync>;
15
16/// A typed view of one projection of the current configuration.
17///
18/// A never-dynamic consumer can hold a plain `T`; a dynamic one holds `Live<T>` and never learns
19/// whether it is fixed or tracking the live config. Read with `Deref`/`current`; react with
20/// `changed`. The projection is over `SalukiConfiguration`, so this is a config view, not a
21/// general-purpose watcher.
22pub struct Live<T> {
23    inner: Inner<T>,
24}
25
26enum Inner<T> {
27    Fixed(T),
28    Dynamic {
29        cell: Arc<ArcSwap<SalukiConfiguration>>,
30        tick: watch::Receiver<()>,
31        project: Projection<T>,
32        // Last observed projected value; backs `Deref` and the change comparison.
33        snapshot: T,
34    },
35}
36
37impl<T: Clone + PartialEq + 'static> Live<T> {
38    /// A view that never changes (local mode, tests).
39    pub fn fixed(value: T) -> Self {
40        Self {
41            inner: Inner::Fixed(value),
42        }
43    }
44
45    /// A view projected from the live configuration. Called by the config system.
46    pub fn dynamic(
47        cell: Arc<ArcSwap<SalukiConfiguration>>, tick: watch::Receiver<()>,
48        project: impl for<'a> Fn(&'a SalukiConfiguration) -> &'a T + Send + Sync + 'static,
49    ) -> Self {
50        let snapshot = project(&cell.load()).clone();
51        Self {
52            inner: Inner::Dynamic {
53                cell,
54                tick,
55                project: Arc::new(project),
56                snapshot,
57            },
58        }
59    }
60
61    /// A fresh clone of the current projected value.
62    pub fn current(&self) -> T {
63        match &self.inner {
64            Inner::Fixed(value) => value.clone(),
65            Inner::Dynamic { cell, project, .. } => project(&cell.load()).clone(),
66        }
67    }
68
69    /// Resolves when the projected value changes. Parks forever when `Fixed` or the channel closed,
70    /// so a caller can `select!` on it unconditionally; a bare `.await` on a fixed view hangs. On
71    /// return, `Deref` reflects the new value.
72    pub async fn changed(&mut self) {
73        match &mut self.inner {
74            Inner::Fixed(_) => std::future::pending().await,
75            Inner::Dynamic {
76                cell,
77                tick,
78                project,
79                snapshot,
80            } => loop {
81                if tick.changed().await.is_err() {
82                    std::future::pending::<()>().await;
83                }
84                let guard = cell.load();
85                let latest = project(&guard);
86                if *latest != *snapshot {
87                    *snapshot = latest.clone();
88                    return;
89                }
90            },
91        }
92    }
93
94    /// Narrows this view to a child node or a single field. The child shares the same source and
95    /// notification and wakes only when the child value changes.
96    pub fn project<U>(&self, f: impl for<'a> Fn(&'a T) -> &'a U + Send + Sync + 'static) -> Live<U>
97    where
98        U: Clone + PartialEq + 'static,
99    {
100        match &self.inner {
101            Inner::Fixed(value) => Live::fixed(f(value).clone()),
102            Inner::Dynamic {
103                cell, tick, project, ..
104            } => {
105                let parent = Arc::clone(project);
106                Live::dynamic(Arc::clone(cell), tick.clone(), move |c| f(parent(c)))
107            }
108        }
109    }
110}
111
112impl<T> Deref for Live<T> {
113    type Target = T;
114    fn deref(&self) -> &T {
115        match &self.inner {
116            Inner::Fixed(value) => value,
117            Inner::Dynamic { snapshot, .. } => snapshot,
118        }
119    }
120}
121
122impl<T: Clone> Clone for Live<T> {
123    fn clone(&self) -> Self {
124        let inner = match &self.inner {
125            Inner::Fixed(value) => Inner::Fixed(value.clone()),
126            Inner::Dynamic {
127                cell,
128                tick,
129                project,
130                snapshot,
131            } => Inner::Dynamic {
132                cell: Arc::clone(cell),
133                tick: tick.clone(),
134                project: Arc::clone(project),
135                snapshot: snapshot.clone(),
136            },
137        };
138        Self { inner }
139    }
140}
141
142impl<T: fmt::Debug> fmt::Debug for Live<T> {
143    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
144        match &self.inner {
145            Inner::Fixed(value) => f.debug_tuple("Live::Fixed").field(value).finish(),
146            Inner::Dynamic { snapshot, .. } => f
147                .debug_struct("Live::Dynamic")
148                .field("snapshot", snapshot)
149                .finish_non_exhaustive(),
150        }
151    }
152}
