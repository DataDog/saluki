43/// How explicit OTLP histogram buckets are reported.
44#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Serialize)]
45pub enum HistogramMode {
46    /// Omit bucket metrics.
47    NoBuckets,
48
49    /// Report each bucket as a counter.
50    Counters,
51
52    /// Report buckets as distributions.
53    #[default]
54    Distributions,
55}
56
57impl FromStr for HistogramMode {
58    type Err = Error;
59
60    fn from_str(value: &str) -> Result<Self, Self::Err> {
61        match value {
62            "nobuckets" => Ok(Self::NoBuckets),
63            "counters" => Ok(Self::Counters),
64            "distributions" => Ok(Self::Distributions),
65            other => Err(Error::new_without_source(format!(
66                "unknown histogram mode `{other}`; expected `nobuckets`, `counters`, or `distributions`"
67            ))),
68        }
69    }
70}

72/// How cumulative monotonic sums are reported.
73#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Serialize)]
74pub enum CumulativeMonotonicMode {
75    /// Converts cumulative values to deltas and reports them as counts.
76    #[default]
77    ToDelta,
78
79    /// Reports cumulative values as gauges without converting them to deltas.
80    RawValue,
81}
82
83impl FromStr for CumulativeMonotonicMode {
84    type Err = Error;
85
86    fn from_str(value: &str) -> Result<Self, Self::Err> {
87        match value {
88            "to_delta" => Ok(Self::ToDelta),
89            "raw_value" => Ok(Self::RawValue),
90            other => Err(Error::new_without_source(format!(
91                "unknown cumulative monotonic sum mode `{other}`; expected `to_delta` or `raw_value`"
92            ))),
93        }
94    }
95}

97/// Controls how the first value of a cumulative monotonic sum is reported.
98#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Serialize)]
99pub enum InitialCumulativeMonotonicValue {
100    /// Reports the first value when its series started after the translator process.
101    #[default]
102    Auto,
103
104    /// Always drops the first value.
105    Drop,
106
107    /// Always reports the first value.
108    Keep,
109}
110
111impl FromStr for InitialCumulativeMonotonicValue {
112    type Err = Error;
113
114    fn from_str(value: &str) -> Result<Self, Self::Err> {
115        match value {
116            "auto" => Ok(Self::Auto),
117            "drop" => Ok(Self::Drop),
118            "keep" => Ok(Self::Keep),
119            other => Err(Error::new_without_source(format!(
120                "unknown initial cumulative monotonic value `{other}`; expected `auto`, `drop`, or `keep`"
121            ))),
122        }
123    }
124}

126/// OTLP sum translation settings.
127#[derive(Clone, Debug, Default, PartialEq, Serialize)]
128pub struct Sums {
129    /// Cumulative monotonic sum reporting mode.
130    ///
131    /// Defaults to `to_delta`, which converts cumulative values to delta counts. Set to `raw_value` to emit
132    /// cumulative values as gauges.
133    pub cumulative_monotonic_mode: CumulativeMonotonicMode,
134
135    /// Initial cumulative monotonic sum reporting behavior.
136    ///
137    /// Defaults to `auto`, which reports the value only when its series started after the translator process.
138    /// Set this to `drop` to always discard the first value or `keep` to always report it.
139    pub initial_cumulative_monotonic_value: InitialCumulativeMonotonicValue,
140}
