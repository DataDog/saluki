10pub fn deserialize_proto_enum<'de, E: Enum, D: Deserializer<'de>>(d: D) -> Result<EnumOrUnknown<E>, D::Error> {
11    struct DeserializeEnumVisitor<E: Enum>(PhantomData<E>);
12
13    impl<'de, E: Enum> serde::de::Visitor<'de> for DeserializeEnumVisitor<E> {
14        type Value = EnumOrUnknown<E>;
15
16        fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
17            write!(formatter, "an integer")
18        }
19
20        fn visit_i8<R>(self, v: i8) -> Result<Self::Value, R>
21        where
22            R: serde::de::Error,
23        {
24            Ok(EnumOrUnknown::from_i32(v as i32))
25        }
26
27        fn visit_i16<R>(self, v: i16) -> Result<Self::Value, R>
28        where
29            R: serde::de::Error,
30        {
31            Ok(EnumOrUnknown::from_i32(v as i32))
32        }
33
34        fn visit_i32<R>(self, v: i32) -> Result<Self::Value, R>
35        where
36            R: serde::de::Error,
37        {
38            Ok(EnumOrUnknown::from_i32(v))
39        }
40
41        fn visit_u8<R>(self, v: u8) -> Result<Self::Value, R>
42        where
43            R: serde::de::Error,
44        {
45            Ok(EnumOrUnknown::from_i32(v as i32))
46        }
47
48        fn visit_u16<R>(self, v: u16) -> Result<Self::Value, R>
49        where
50            R: serde::de::Error,
51        {
52            Ok(EnumOrUnknown::from_i32(v as i32))
53        }
54    }
55
56    d.deserialize_any(DeserializeEnumVisitor(PhantomData))
57}

59pub fn serialize_proto_message<T: Serialize, S: Serializer>(
60    message: &MessageField<T>, s: S,
61) -> Result<S::Ok, S::Error> {
62    match message.as_ref() {
63        Some(inner) => s.serialize_some(inner),
64        None => s.serialize_none(),
65    }
66}
