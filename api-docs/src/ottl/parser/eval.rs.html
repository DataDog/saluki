14/// Apply indexes to a value. Used only for converter/editor return value indexing (e.g. Split(...)[0]).
15#[inline]
16pub(crate) fn apply_indexes(value: Value, indexes: &[IndexExpr]) -> Result<Value> {
17    let mut current = value;
18    for index in indexes {
19        current = apply_index(&current, index)?;
20    }
21    Ok(current)
22}
23
24/// Apply an index to a value
25fn apply_index(value: &Value, index: &IndexExpr) -> Result<Value> {
26    match (value, index) {
27        (Value::List(list), IndexExpr::Int(i)) => list
28            .get(*i)
29            .cloned()
30            .ok_or_else(|| format!("Index {} out of bounds", i).into()),
31        (Value::Map(map), IndexExpr::String(key)) => map
32            .get(key)
33            .cloned()
34            .ok_or_else(|| format!("Key '{}' not found", key).into()),
35        (Value::String(s), IndexExpr::Int(i)) => s
36            .chars()
37            .nth(*i)
38            .map(|c| Value::string(c.to_string()))
39            .ok_or_else(|| format!("Index {} out of bounds", i).into()),
40        _ => Err(format!("Cannot index {:?} with {:?}", value, index).into()),
41    }
42}

48/// Zero-allocation argument evaluator for function calls.
49/// Implements lazy evaluation - arguments are only evaluated when requested.
50struct ArenaArgs<'a> {
51    arena: &'a AstArena,
52    args: &'a [ArenaArgExpr],
53    ctx: &'a mut EvalContext,
54}
55
56impl<'a> ArenaArgs<'a> {
57    /// Builds an [`ArenaArgs`] for a call: borrows arena, argument list, and context.
58    #[inline]
59    fn new(arena: &'a AstArena, args: &'a [ArenaArgExpr], ctx: &'a mut EvalContext) -> Self {
60        Self { arena, args, ctx }
61    }
62}
63
64/// Implementation of [`Args`] for arena-based function/editor calls.
65///
66/// Arguments are stored as [`ArenaArgExpr`] refs; evaluation is lazy and uses the arena
67/// and context to compute values on demand without allocating an argument vector.
68impl<'a> Args for ArenaArgs<'a> {
69    /// Returns exclusive access to the evaluation context for the duration of the call.
70    #[inline]
71    fn ctx(&mut self) -> &mut EvalContext {
72        self.ctx
73    }
74
75    /// Returns the number of arguments (positional and named).
76    #[inline]
77    fn len(&self) -> usize {
78        self.args.len()
79    }
80
81    /// Evaluates the argument at `index` via the arena and returns its [`Value`].
82    /// Supports both positional and named args; named args are indexed by position.
83    #[inline]
84    fn get(&mut self, index: usize) -> Result<Value> {
85        if index >= self.args.len() {
86            return Err(format!("Argument index {} out of bounds", index).into());
87        }
88
89        match &self.args[index] {
90            ArenaArgExpr::Positional(value_ref) => arena_evaluate_value_expr(*value_ref, self.arena, self.ctx),
91            ArenaArgExpr::Named { value, .. } => arena_evaluate_value_expr(*value, self.arena, self.ctx),
92        }
93    }
94
95    /// Returns the name of the argument at `index` if it is a named argument; `None` for positional.
96    #[inline]
97    fn name(&self, index: usize) -> Option<&str> {
98        if index >= self.args.len() {
99            return None;
100        }
101        match &self.args[index] {
102            ArenaArgExpr::Positional(_) => None,
103            ArenaArgExpr::Named { name, .. } => Some(name),
104        }
105    }
106
107    /// Sets the value at the path referenced by the argument at `index`.
108    /// Only supported when that argument is a path expression; otherwise returns an error.
109    #[inline]
110    fn set(&mut self, index: usize, value: &Value) -> Result<()> {
111        if index >= self.args.len() {
112            return Err(format!("Argument index {} out of bounds", index).into());
113        }
114
115        let value_ref = match &self.args[index] {
116            ArenaArgExpr::Positional(r) => *r,
117            ArenaArgExpr::Named { value, .. } => *value,
118        };
119
120        match self.arena.get_value(value_ref) {
121            ArenaValueExpr::Path(resolved_path) => {
122                resolved_path
123                    .accessor
124                    .set(self.ctx, &resolved_path.full_path, &resolved_path.indexes, value)
125            }
126            _ => Err("set: argument must be a path expression".into()),
127        }
128    }
129}

131/// Evaluate the arena-based root expression
132#[inline]
133pub fn arena_evaluate_root(root: &ArenaRootExpr, arena: &AstArena, ctx: &mut EvalContext) -> Result<Value> {
134    match root {
135        ArenaRootExpr::EditorStatement(stmt) => {
136            let should_execute = if let Some(cond_ref) = stmt.condition {
137                arena_evaluate_bool_expr(cond_ref, arena, ctx)?
138            } else {
139                true
140            };
141
142            if should_execute {
143                arena_evaluate_function_call(stmt.editor, arena, ctx)?;
144            }
145
146            Ok(Value::Nil)
147        }
148        ArenaRootExpr::BooleanExpression(expr_ref) => {
149            let result = arena_evaluate_bool_expr(*expr_ref, arena, ctx)?;
150            Ok(Value::Bool(result))
151        }
152        ArenaRootExpr::MathExpression(expr_ref) => arena_evaluate_math_expr(*expr_ref, arena, ctx),
153    }
154}

156/// Evaluate an arena-based boolean expression
157#[inline]
158fn arena_evaluate_bool_expr(expr_ref: BoolExprRef, arena: &AstArena, ctx: &mut EvalContext) -> Result<bool> {
159    match arena.get_bool(expr_ref) {
160        ArenaBoolExpr::Literal(b) => Ok(*b),
161        ArenaBoolExpr::Comparison { left, op, right } => {
162            let left_val = arena_evaluate_value_expr(*left, arena, ctx)?;
163            let right_val = arena_evaluate_value_expr(*right, arena, ctx)?;
164            compare(&left_val, op, &right_val)
165        }
166        ArenaBoolExpr::Converter(fc_ref) => {
167            let result = arena_evaluate_function_call(*fc_ref, arena, ctx)?;
168            match result {
169                Value::Bool(b) => Ok(b),
170                _ => Err("Converter did not return a boolean".into()),
171            }
172        }
173        ArenaBoolExpr::Path(resolved_path) => {
174            let value = resolved_path
175                .accessor
176                .get(ctx, &resolved_path.full_path, &resolved_path.indexes)?;
177            match value {
178                Value::Bool(b) => Ok(b),
179                _ => Err("Path did not return a boolean".into()),
180            }
181        }
182        ArenaBoolExpr::Not(inner_ref) => {
183            let result = arena_evaluate_bool_expr(*inner_ref, arena, ctx)?;
184            Ok(!result)
185        }
186        ArenaBoolExpr::And(left_ref, right_ref) => {
187            let left_result = arena_evaluate_bool_expr(*left_ref, arena, ctx)?;
188            if !left_result {
189                return Ok(false);
190            }
191            arena_evaluate_bool_expr(*right_ref, arena, ctx)
192        }
193        ArenaBoolExpr::Or(left_ref, right_ref) => {
194            let left_result = arena_evaluate_bool_expr(*left_ref, arena, ctx)?;
195            if left_result {
196                return Ok(true);
197            }
198            arena_evaluate_bool_expr(*right_ref, arena, ctx)
199        }
200    }
201}

203/// Evaluate an arena-based value expression
204#[inline]
205fn arena_evaluate_value_expr(expr_ref: ValueExprRef, arena: &AstArena, ctx: &mut EvalContext) -> Result<Value> {
206    match arena.get_value(expr_ref) {
207        ArenaValueExpr::Literal(v) => Ok(v.clone()),
208        ArenaValueExpr::Path(resolved_path) => {
209            resolved_path
210                .accessor
211                .get(ctx, &resolved_path.full_path, &resolved_path.indexes)
212        }
213        ArenaValueExpr::List(items) => {
214            let values: Result<Vec<Value>> = items
215                .iter()
216                .map(|item_ref| arena_evaluate_value_expr(*item_ref, arena, ctx))
217                .collect();
218            Ok(Value::List(values?))
219        }
220        ArenaValueExpr::Map(entries) => {
221            let mut map = std::collections::HashMap::new();
222            for (key, value_ref) in entries {
223                let value = arena_evaluate_value_expr(*value_ref, arena, ctx)?;
224                map.insert(key.clone(), value);
225            }
226            Ok(Value::Map(map))
227        }
228        ArenaValueExpr::FunctionCall(fc_ref) => arena_evaluate_function_call(*fc_ref, arena, ctx),
229        ArenaValueExpr::Math(math_ref) => arena_evaluate_math_expr(*math_ref, arena, ctx),
230    }
231}

233/// Evaluate an arena-based function call (ZERO ALLOCATION)
234#[inline]
235fn arena_evaluate_function_call(fc_ref: FunctionCallRef, arena: &AstArena, ctx: &mut EvalContext) -> Result<Value> {
236    let fc = arena.get_func(fc_ref);
237
238    let callback = fc
239        .callback
240        .as_ref()
241        .ok_or_else(|| -> BoxError { format!("Unknown function: {}", fc.name).into() })?;
242
243    let mut args = ArenaArgs::new(arena, &fc.args, ctx);
244    let result = callback(&mut args)?;
245    apply_indexes(result, &fc.indexes)
246}

248/// Evaluate an arena-based math expression
249#[inline]
250fn arena_evaluate_math_expr(expr_ref: MathExprRef, arena: &AstArena, ctx: &mut EvalContext) -> Result<Value> {
251    match arena.get_math(expr_ref) {
252        ArenaMathExpr::Primary(value_ref) => arena_evaluate_value_expr(*value_ref, arena, ctx),
253        ArenaMathExpr::Negate(inner_ref) => {
254            let value = arena_evaluate_math_expr(*inner_ref, arena, ctx)?;
255            match value {
256                Value::Int(i) => Ok(Value::Int(-i)),
257                Value::Float(f) => Ok(Value::Float(-f)),
258                _ => Err("Cannot negate non-numeric value".into()),
259            }
260        }
261        ArenaMathExpr::Binary {
262            left: left_ref,
263            op,
264            right: right_ref,
265        } => {
266            let left_val = arena_evaluate_math_expr(*left_ref, arena, ctx)?;
267            let right_val = arena_evaluate_math_expr(*right_ref, arena, ctx)?;
268            math_op(&left_val, op, &right_val)
269        }
270    }
