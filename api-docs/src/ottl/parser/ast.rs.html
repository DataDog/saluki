34/// Index for path or converter result

43// =====================================================================================================================
44// Arena-based AST Types (used during evaluation - cache-friendly)
45// we are using u32 as index on purpose: usize is considered as too fat
46// and cache locality will be better for 32 bits.
47// 16 bits are considered as too small for generated AST (test purposes)
48// =====================================================================================================================
49
50/// Index into the arena for BoolExpr nodes
51#[derive(Debug, Clone, Copy, PartialEq, Eq)]
52pub struct BoolExprRef(pub(crate) u32);
53
54/// Index into the arena for MathExpr nodes
55#[derive(Debug, Clone, Copy, PartialEq, Eq)]
56pub struct MathExprRef(pub(crate) u32);
57
58/// Index into the arena for ValueExpr nodes
59#[derive(Debug, Clone, Copy, PartialEq, Eq)]
60pub struct ValueExprRef(pub(crate) u32);
61
62/// Index into the arena for FunctionCall nodes
63#[derive(Debug, Clone, Copy, PartialEq, Eq)]
64pub struct FunctionCallRef(pub(crate) u32);

66/// Resolved path with pre-computed path string and accessor (resolved at parse time).
67/// Generic over context type `C` so the accessor operates on that context.
68#[derive(Clone)]
69pub struct ResolvedPath<C> {
70    /// Pre-computed full path string (e.g., "my.int.value")
71    pub full_path: String,
72    /// Pre-resolved accessor (resolved once at parse time, not at each execution)
73    pub accessor: Arc<dyn PathAccessor<C> + Send + Sync>,
74    /// Optional indexes for indexing into the result
75    pub indexes: Vec<IndexExpr>,
76}
77
78impl<C> std::fmt::Debug for ResolvedPath<C> {
79    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
80        f.debug_struct("ResolvedPath")
81            .field("full_path", &self.full_path)
82            .field("indexes", &self.indexes)
83            .finish()
84    }
85}

87/// Arena-based BoolExpr using indices instead of Box. Generic over context type `C`.
88#[derive(Debug, Clone)]
89pub enum ArenaBoolExpr<C> {
90    Literal(bool),
91    Comparison {
92        left: ValueExprRef,
93        op: CompOp,
94        right: ValueExprRef,
95    },
96    Converter(FunctionCallRef),
97    /// Path with pre-resolved accessor
98    Path(ResolvedPath<C>),
99    Not(BoolExprRef),
100    And(BoolExprRef, BoolExprRef),
101    Or(BoolExprRef, BoolExprRef),
102}

104/// Arena-based MathExpr using indices instead of Box
105#[derive(Debug, Clone)]
106pub enum ArenaMathExpr {
107    Primary(ValueExprRef),
108    Negate(MathExprRef),
109    Binary {
110        left: MathExprRef,
111        op: MathOp,
112        right: MathExprRef,
113    },
114}

116/// Arena-based ValueExpr using indices instead of Box. Generic over context type `C`.
117#[derive(Debug, Clone)]
118pub enum ArenaValueExpr<C> {
119    Literal(Value),
120    /// Path with pre-resolved accessor (no runtime lookup!)
121    Path(ResolvedPath<C>),
122    List(Vec<ValueExprRef>),
123    Map(Vec<(String, ValueExprRef)>),
124    FunctionCall(FunctionCallRef),
125    Math(MathExprRef),

128/// Arena-based FunctionCall using indices for args. Generic over context type `C`.
129#[derive(Clone)]
130pub struct ArenaFunctionCall<C> {
131    pub name: String,
132    pub is_editor: bool,
133    pub args: Vec<ArenaArgExpr>,
134    pub indexes: Vec<IndexExpr>,
135    pub callback: Option<CallbackFn<C>>,
136}

157/// Arena-based EditorStatement. Holds [`PhantomData<C>`] so [`ArenaRootExpr<C>`] is well-formed.

177/// Path expression (e.g., resource.attributes["key"])
178#[derive(Debug, Clone)]
179pub struct PathExpr {
180    /// Segments of the path (e.g., ["resource", "attributes"])
181    pub segments: Vec<String>,
182    /// Optional indexes
183    pub indexes: Vec<IndexExpr>,
184}
185
186/// Function invocation (Editor or Converter). Generic over context type `C`.
187#[derive(Clone)]
188pub struct FunctionCall<C> {
189    /// Function name
190    pub name: String,
191    /// Whether this is an editor (lowercase) or converter (uppercase)
192    pub is_editor: bool,
193    /// Arguments
194    pub args: Vec<ArgExpr<C>>,
195    /// Optional indexes (for converters)
196    pub indexes: Vec<IndexExpr>,
197    /// Callback reference (resolved at parse time)
198    pub callback: Option<CallbackFn<C>>,
199}
200
201impl<C: std::fmt::Debug> std::fmt::Debug for FunctionCall<C> {
202    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
203        f.debug_struct("FunctionCall")
204            .field("name", &self.name)
205            .field("is_editor", &self.is_editor)
206            .field("args", &self.args)
207            .field("indexes", &self.indexes)
208            .field("callback", &self.callback.is_some())
209            .finish()
210    }
211}

213/// Argument expression. Generic over context type `C`.
214#[derive(Debug, Clone)]
215pub enum ArgExpr<C> {
216    /// Positional argument
217    Positional(ValueExpr<C>),
218    /// Named argument

222/// Value expression - any value in OTTL. Generic over context type `C`.
223#[derive(Debug, Clone)]
224pub enum ValueExpr<C> {
225    /// Literal value
226    Literal(Value),
227    /// Path expression
228    Path(PathExpr),
229    /// List literal
230    List(Vec<ValueExpr<C>>),
231    /// Map literal
232    Map(Vec<(String, ValueExpr<C>)>),
233    /// Function call (converter or editor)
234    FunctionCall(Box<FunctionCall<C>>),
235    /// Math expression
236    Math(Box<MathExpr<C>>),
237}
238
239/// Math expression with operator precedence. Generic over context type `C`.
240#[derive(Debug, Clone)]
241pub enum MathExpr<C> {
242    /// Primary value (literal, path, converter, or grouped expression)
243    Primary(ValueExpr<C>),
244    /// Unary negation
245    Negate(Box<MathExpr<C>>),
246    /// Binary operation: term (+/-) or factor (*/)
247    Binary {
248        left: Box<MathExpr<C>>,
249        op: MathOp,
250        right: Box<MathExpr<C>>,
251    },
252}
253
254/// Boolean expression with operator precedence. Generic over context type `C`.
255#[derive(Debug, Clone)]
256pub enum BoolExpr<C> {
257    /// Literal boolean
258    Literal(bool),
259    /// Comparison expression
260    Comparison {
261        left: ValueExpr<C>,
262        op: CompOp,
263        right: ValueExpr<C>,
264    },
265    /// Converter call returning boolean
266    Converter(Box<FunctionCall<C>>),
267    /// Path that evaluates to boolean
268    Path(PathExpr),
269    /// Logical NOT
270    Not(Box<BoolExpr<C>>),
271    /// Logical AND
272    And(Box<BoolExpr<C>>, Box<BoolExpr<C>>),
273    /// Logical OR
274    Or(Box<BoolExpr<C>>, Box<BoolExpr<C>>),
275}

277/// Editor invocation statement. Generic over context type `C`.
278#[derive(Debug, Clone)]
279pub struct EditorStatement<C> {
280    /// The editor function call
281    pub editor: FunctionCall<C>,
282    /// Optional WHERE clause condition
283    pub condition: Option<BoolExpr<C>>,
284}

286/// Root AST node - either an editor statement, a boolean expression, or a math expression.
287/// Generic over context type `C`.
288#[derive(Debug, Clone)]
289pub enum RootExpr<C> {
290    /// Editor invocation with optional WHERE clause
291    EditorStatement(EditorStatement<C>),
292    /// Standalone boolean expression
293    BooleanExpression(BoolExpr<C>),
294    /// Standalone math expression
295    MathExpression(MathExpr<C>),
