1//! Arena-based AST storage and conversion from boxed AST
2//!
3//! Provides cache-friendly AST storage and optimizations like:
4//! - Path resolution at parse time
5//! - Constant folding for literals
6//! - linear memory with less inderections

15/// Arena storage for all AST nodes - enables cache-friendly traversal
16#[derive(Debug, Clone, Default)]
17pub struct AstArena {
18    pub bool_exprs: Vec<ArenaBoolExpr>,
19    pub math_exprs: Vec<ArenaMathExpr>,
20    pub value_exprs: Vec<ArenaValueExpr>,
21    pub function_calls: Vec<ArenaFunctionCall>,
22}
23
24impl AstArena {
25    pub fn new() -> Self {
26        Self::default()
27    }
28
29    #[inline]
30    pub fn alloc_bool(&mut self, expr: ArenaBoolExpr) -> BoolExprRef {
31        let idx = self.bool_exprs.len() as u32;
32        self.bool_exprs.push(expr);
33        BoolExprRef(idx)
34    }
35
36    #[inline]
37    pub fn alloc_math(&mut self, expr: ArenaMathExpr) -> MathExprRef {
38        let idx = self.math_exprs.len() as u32;
39        self.math_exprs.push(expr);
40        MathExprRef(idx)
41    }
42
43    #[inline]
44    pub fn alloc_value(&mut self, expr: ArenaValueExpr) -> ValueExprRef {
45        let idx = self.value_exprs.len() as u32;
46        self.value_exprs.push(expr);
47        ValueExprRef(idx)
48    }
49
50    #[inline]
51    pub fn alloc_func(&mut self, call: ArenaFunctionCall) -> FunctionCallRef {
52        let idx = self.function_calls.len() as u32;
53        self.function_calls.push(call);
54        FunctionCallRef(idx)
55    }
56
57    #[inline]
58    pub fn get_bool(&self, r: BoolExprRef) -> &ArenaBoolExpr {
59        &self.bool_exprs[r.0 as usize]
60    }
61
62    #[inline]
63    pub fn get_math(&self, r: MathExprRef) -> &ArenaMathExpr {
64        &self.math_exprs[r.0 as usize]
65    }
66
67    #[inline]
68    pub fn get_value(&self, r: ValueExprRef) -> &ArenaValueExpr {
69        &self.value_exprs[r.0 as usize]
70    }
71
72    #[inline]
73    pub fn get_func(&self, r: FunctionCallRef) -> &ArenaFunctionCall {
74        &self.function_calls[r.0 as usize]
75    }
76}

82/// Helper to resolve a PathExpr to ResolvedPath at parse time.
83/// Looks up the path in path_resolvers; returns error if no resolver is provided for this path.
84fn resolve_path(path: &PathExpr, path_resolvers: &PathResolverMap) -> Result<ResolvedPath> {
85    let full_path = path.segments.join(".");
86    let resolver = path_resolvers
87        .get(&full_path)
88        .ok_or_else(|| format!("No PathResolver provided for path \"{}\"", full_path))?;
89    let accessor = resolver()?;
90    Ok(ResolvedPath {
91        full_path,
92        accessor,
93        indexes: path.indexes.clone(),
94    })
95}

97/// Convert boxed AST to arena-based AST for cache-friendly execution
98/// Path resolution happens HERE (once at parse time), not at each execution!
99pub fn convert_to_arena(
100    root: &RootExpr, arena: &mut AstArena, path_resolvers: &PathResolverMap,
101) -> Result<ArenaRootExpr> {
102    match root {
103        RootExpr::EditorStatement(stmt) => {
104            let editor_ref = convert_function_call(&stmt.editor, arena, path_resolvers)?;
105            let condition_ref = match &stmt.condition {
106                Some(c) => Some(convert_bool_expr(c, arena, path_resolvers)?),
107                None => None,
108            };
109            Ok(ArenaRootExpr::EditorStatement(ArenaEditorStatement {
110                editor: editor_ref,
111                condition: condition_ref,
112            }))
113        }
114        RootExpr::BooleanExpression(expr) => {
115            let expr_ref = convert_bool_expr(expr, arena, path_resolvers)?;
116            Ok(ArenaRootExpr::BooleanExpression(expr_ref))
117        }
118        RootExpr::MathExpression(expr) => {
119            let expr_ref = convert_math_expr(expr, arena, path_resolvers)?;
120            Ok(ArenaRootExpr::MathExpression(expr_ref))
121        }
122    }
123}

129/// Check if a ValueExpr is a literal and get its value
130fn try_get_literal(expr: &ValueExpr) -> Option<&Value> {
131    match expr {
132        ValueExpr::Literal(v) => Some(v),
133        _ => None,
134    }
135}
136
137/// Check if arena bool expr is a literal
138fn arena_bool_is_literal(arena: &AstArena, r: BoolExprRef) -> Option<bool> {
139    match arena.get_bool(r) {
140        ArenaBoolExpr::Literal(b) => Some(*b),
141        _ => None,
142    }
143}
144
145/// Try to get literal value from a MathExpr
146fn try_get_math_literal(expr: &MathExpr) -> Option<&Value> {
147    match expr {
148        MathExpr::Primary(ValueExpr::Literal(v)) => Some(v),
149        _ => None,
150    }
151}

160/// Converts a boxed [`BoolExpr`] into arena-based form, allocating nodes in `arena` and returning a [`BoolExprRef`].
161///
162/// - Resolves paths via [`resolve_path`] (path → [`ResolvedPath`] with [`crate::PathAccessor`]).
163/// - Recursively converts nested bool/value expressions and function calls.
164/// - Applies **constant folding** where possible: literal comparisons, `not(literal)`, short-circuit
165///   `and`/`or` with literal operands, so the arena may contain fewer/simpler nodes.
166fn convert_bool_expr(expr: &BoolExpr, arena: &mut AstArena, path_resolvers: &PathResolverMap) -> Result<BoolExprRef> {
167    let arena_expr = match expr {
168        BoolExpr::Literal(b) => ArenaBoolExpr::Literal(*b),
169        BoolExpr::Comparison { left, op, right } => {
170            // CONSTANT FOLDING: if both sides are literals, compute at parse time!
171            if let (Some(left_val), Some(right_val)) = (try_get_literal(left), try_get_literal(right)) {
172                if let Ok(result) = compare(left_val, op, right_val) {
173                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(result)));
174                }
175            }
176            let left_ref = convert_value_expr(left, arena, path_resolvers)?;
177            let right_ref = convert_value_expr(right, arena, path_resolvers)?;
178            ArenaBoolExpr::Comparison {
179                left: left_ref,
180                op: *op,
181                right: right_ref,
182            }
183        }
184        BoolExpr::Converter(fc) => {
185            let fc_ref = convert_function_call(fc, arena, path_resolvers)?;
186            ArenaBoolExpr::Converter(fc_ref)
187        }
188        BoolExpr::Path(path) => {
189            let resolved = resolve_path(path, path_resolvers)?;
190            ArenaBoolExpr::Path(resolved)
191        }
192        BoolExpr::Not(inner) => {
193            let inner_ref = convert_bool_expr(inner, arena, path_resolvers)?;
194            // CONSTANT FOLDING: not(literal) => !literal
195            if let Some(b) = arena_bool_is_literal(arena, inner_ref) {
196                return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(!b)));
197            }
198            ArenaBoolExpr::Not(inner_ref)
199        }
200        BoolExpr::And(left, right) => {
201            let left_ref = convert_bool_expr(left, arena, path_resolvers)?;
202            // CONSTANT FOLDING: false && x => false, true && x => x
203            if let Some(left_val) = arena_bool_is_literal(arena, left_ref) {
204                if !left_val {
205                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(false)));
206                }
207                return convert_bool_expr(right, arena, path_resolvers);
208            }
209            let right_ref = convert_bool_expr(right, arena, path_resolvers)?;
210            // CONSTANT FOLDING: x && false => false, x && true => x
211            if let Some(right_val) = arena_bool_is_literal(arena, right_ref) {
212                if !right_val {
213                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(false)));
214                }
215                return Ok(left_ref);
216            }
217            ArenaBoolExpr::And(left_ref, right_ref)
218        }
219        BoolExpr::Or(left, right) => {
220            let left_ref = convert_bool_expr(left, arena, path_resolvers)?;
221            // CONSTANT FOLDING: true || x => true, false || x => x
222            if let Some(left_val) = arena_bool_is_literal(arena, left_ref) {
223                if left_val {
224                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(true)));
225                }
226                return convert_bool_expr(right, arena, path_resolvers);
227            }
228            let right_ref = convert_bool_expr(right, arena, path_resolvers)?;
229            // CONSTANT FOLDING: x || true => true, x || false => x
230            if let Some(right_val) = arena_bool_is_literal(arena, right_ref) {
231                if right_val {
232                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(true)));
233                }
234                return Ok(left_ref);
235            }
236            ArenaBoolExpr::Or(left_ref, right_ref)
237        }
238    };
239    Ok(arena.alloc_bool(arena_expr))
240}

242/// Converts a boxed [`MathExpr`] into arena-based form, allocating nodes in `arena` and returning a [`MathExprRef`].
243///
244/// - Recursively converts primary value expressions and nested math expressions.
245/// - Applies **constant folding**: negate of literal, binary `literal op literal` → computed literal,
246///   so the arena may contain a single [`ArenaMathExpr::Primary`] where the source had a tree.
247fn convert_math_expr(expr: &MathExpr, arena: &mut AstArena, path_resolvers: &PathResolverMap) -> Result<MathExprRef> {
248    let arena_expr = match expr {
249        MathExpr::Primary(v) => {
250            let v_ref = convert_value_expr(v, arena, path_resolvers)?;
251            ArenaMathExpr::Primary(v_ref)
252        }
253        MathExpr::Negate(inner) => {
254            // CONSTANT FOLDING: -literal => literal negated
255            if let Some(val) = try_get_math_literal(inner) {
256                let negated = match val {
257                    Value::Int(i) => Some(Value::Int(-i)),
258                    Value::Float(f) => Some(Value::Float(-f)),
259                    _ => None,
260                };
261                if let Some(v) = negated {
262                    let v_ref = arena.alloc_value(ArenaValueExpr::Literal(v));
263                    return Ok(arena.alloc_math(ArenaMathExpr::Primary(v_ref)));
264                }
265            }
266            let inner_ref = convert_math_expr(inner, arena, path_resolvers)?;
267            ArenaMathExpr::Negate(inner_ref)
268        }
269        MathExpr::Binary { left, op, right } => {
270            // CONSTANT FOLDING: literal op literal => computed literal
271            if let (Some(left_val), Some(right_val)) = (try_get_math_literal(left), try_get_math_literal(right)) {
272                if let Some(result) = try_math_op(left_val, op, right_val) {
273                    let v_ref = arena.alloc_value(ArenaValueExpr::Literal(result));
274                    return Ok(arena.alloc_math(ArenaMathExpr::Primary(v_ref)));
275                }
276            }
277            let left_ref = convert_math_expr(left, arena, path_resolvers)?;
278            let right_ref = convert_math_expr(right, arena, path_resolvers)?;
279            ArenaMathExpr::Binary {
280                left: left_ref,
281                op: *op,
282                right: right_ref,
283            }
284        }
285    };
286    Ok(arena.alloc_math(arena_expr))
287}

289/// Converts a boxed [`ValueExpr`] into arena-based form, allocating nodes in `arena` and returning a [`ValueExprRef`].
290///
291/// - Resolves paths via [`resolve_path`] (path → [`ResolvedPath`] with [`crate::PathAccessor`]).
292/// - Recursively converts list items, map entries, nested function calls, and math expressions.
293/// - Literals are stored as-is without further conversion.
294fn convert_value_expr(
295    expr: &ValueExpr, arena: &mut AstArena, path_resolvers: &PathResolverMap,
296) -> Result<ValueExprRef> {
297    let arena_expr = match expr {
298        ValueExpr::Literal(v) => ArenaValueExpr::Literal(v.clone()),
299        ValueExpr::Path(path) => {
300            let resolved = resolve_path(path, path_resolvers)?;
301            ArenaValueExpr::Path(resolved)
302        }
303        ValueExpr::List(items) => {
304            let refs: Result<Vec<_>> = items
305                .iter()
306                .map(|i| convert_value_expr(i, arena, path_resolvers))
307                .collect();
308            ArenaValueExpr::List(refs?)
309        }
310        ValueExpr::Map(entries) => {
311            let refs: Result<Vec<_>> = entries
312                .iter()
313                .map(|(k, v)| Ok((k.clone(), convert_value_expr(v, arena, path_resolvers)?)))
314                .collect();
315            ArenaValueExpr::Map(refs?)
316        }
317        ValueExpr::FunctionCall(fc) => {
318            let fc_ref = convert_function_call(fc, arena, path_resolvers)?;
319            ArenaValueExpr::FunctionCall(fc_ref)
320        }
321        ValueExpr::Math(m) => {
322            let m_ref = convert_math_expr(m, arena, path_resolvers)?;
323            ArenaValueExpr::Math(m_ref)
324        }
325    };
326    Ok(arena.alloc_value(arena_expr))
327}

329/// Converts a boxed [`FunctionCall`] into arena-based form, allocating in `arena` and returning a [`FunctionCallRef`].
330///
331/// - Converts all arguments (positional and named) to arena value refs via [`convert_value_expr`].
332/// - Preserves `name`, `is_editor`, `indexes`, and the pre-resolved `callback` from the source AST.
333/// - Does not resolve callbacks here; the parser is expected to have attached [`CallbackFn`] when applicable.
334fn convert_function_call(
335    fc: &FunctionCall, arena: &mut AstArena, path_resolvers: &PathResolverMap,
336) -> Result<FunctionCallRef> {
337    let args: Result<Vec<_>> = fc
338        .args
339        .iter()
340        .map(|arg| match arg {
341            ArgExpr::Positional(v) => Ok(ArenaArgExpr::Positional(convert_value_expr(v, arena, path_resolvers)?)),
342            ArgExpr::Named { name, value } => Ok(ArenaArgExpr::Named {
343                name: name.clone(),
344                value: convert_value_expr(value, arena, path_resolvers)?,
345            }),
346        })
347        .collect();
348
349    let arena_fc = ArenaFunctionCall {
350        name: fc.name.clone(),
351        is_editor: fc.is_editor,
352        args: args?,
353        indexes: fc.indexes.clone(),
354        callback: fc.callback.clone(),
355    };
356    Ok(arena.alloc_func(arena_fc))
