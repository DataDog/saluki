1//! Arena-based AST storage and conversion from boxed AST
2//!
3//! Provides cache-friendly AST storage and optimizations like:
4//! - Path resolution at parse time
5//! - Constant folding for literals
6//! - linear memory with less inderections

17/// Arena storage for all AST nodes - enables cache-friendly traversal.
18///
19/// The type parameter `F` is the [`EvalContextFamily`] used by resolved path accessors.
20#[derive(Debug)]
21pub struct AstArena<F: EvalContextFamily> {
22    pub bool_exprs: Vec<ArenaBoolExpr<F>>,
23    pub math_exprs: Vec<ArenaMathExpr>,
24    pub value_exprs: Vec<ArenaValueExpr<F>>,
25    pub function_calls: Vec<ArenaFunctionCall>,
26    _marker: PhantomData<F>,
27}
28
29impl<F: EvalContextFamily> Clone for AstArena<F> {
30    fn clone(&self) -> Self {
31        Self {
32            bool_exprs: self.bool_exprs.clone(),
33            math_exprs: self.math_exprs.clone(),
34            value_exprs: self.value_exprs.clone(),
35            function_calls: self.function_calls.clone(),
36            _marker: PhantomData,
37        }
38    }
39}
40
41impl<F: EvalContextFamily> AstArena<F> {
42    pub fn new() -> Self {
43        Self {
44            bool_exprs: Vec::new(),
45            math_exprs: Vec::new(),
46            value_exprs: Vec::new(),
47            function_calls: Vec::new(),
48            _marker: PhantomData,
49        }
50    }
51
52    #[inline]
53    pub fn alloc_bool(&mut self, expr: ArenaBoolExpr<F>) -> BoolExprRef {
54        let idx = self.bool_exprs.len() as u32;
55        self.bool_exprs.push(expr);
56        BoolExprRef(idx)
57    }
58
59    #[inline]
60    pub fn alloc_math(&mut self, expr: ArenaMathExpr) -> MathExprRef {
61        let idx = self.math_exprs.len() as u32;
62        self.math_exprs.push(expr);
63        MathExprRef(idx)
64    }
65
66    #[inline]
67    pub fn alloc_value(&mut self, expr: ArenaValueExpr<F>) -> ValueExprRef {
68        let idx = self.value_exprs.len() as u32;
69        self.value_exprs.push(expr);
70        ValueExprRef(idx)
71    }
72
73    #[inline]
74    pub fn alloc_func(&mut self, call: ArenaFunctionCall) -> FunctionCallRef {
75        let idx = self.function_calls.len() as u32;
76        self.function_calls.push(call);
77        FunctionCallRef(idx)
78    }
79
80    #[inline]
81    pub fn get_bool(&self, r: BoolExprRef) -> &ArenaBoolExpr<F> {
82        &self.bool_exprs[r.0 as usize]
83    }
84
85    #[inline]
86    pub fn get_math(&self, r: MathExprRef) -> &ArenaMathExpr {
87        &self.math_exprs[r.0 as usize]
88    }
89
90    #[inline]
91    pub fn get_value(&self, r: ValueExprRef) -> &ArenaValueExpr<F> {
92        &self.value_exprs[r.0 as usize]
93    }
94
95    #[inline]
96    pub fn get_func(&self, r: FunctionCallRef) -> &ArenaFunctionCall {
97        &self.function_calls[r.0 as usize]
98    }
99}

105/// Helper to resolve a PathExpr to ResolvedPath at parse time.
106/// Looks up the path in path_resolvers; returns error if no resolver is provided for this path.
107fn resolve_path<F: EvalContextFamily>(path: &PathExpr, path_resolvers: &PathResolverMap<F>) -> Result<ResolvedPath<F>> {
108    let full_path = path.segments.join(".");
109    let resolver = path_resolvers
110        .get(&full_path)
111        .ok_or_else(|| format!("No PathResolver provided for path \"{}\"", full_path))?;
112    let accessor = resolver()?;
113    Ok(ResolvedPath {
114        full_path,
115        accessor,
116        indexes: path.indexes.clone(),
117    })
118}

120/// Convert boxed AST to arena-based AST for cache-friendly execution.
121/// Path resolution happens HERE (once at parse time), not at each execution!
122pub fn convert_to_arena<F: EvalContextFamily>(
123    root: &RootExpr, arena: &mut AstArena<F>, path_resolvers: &PathResolverMap<F>,
124) -> Result<ArenaRootExpr> {
125    match root {
126        RootExpr::EditorStatement(stmt) => {
127            let editor_ref = convert_function_call(&stmt.editor, arena, path_resolvers)?;
128            let condition_ref = match &stmt.condition {
129                Some(c) => Some(convert_bool_expr(c, arena, path_resolvers)?),
130                None => None,
131            };
132            Ok(ArenaRootExpr::EditorStatement(ArenaEditorStatement {
133                editor: editor_ref,
134                condition: condition_ref,
135            }))
136        }
137        RootExpr::BooleanExpression(expr) => {
138            let expr_ref = convert_bool_expr(expr, arena, path_resolvers)?;
139            Ok(ArenaRootExpr::BooleanExpression(expr_ref))
140        }
141        RootExpr::MathExpression(expr) => {
142            let expr_ref = convert_math_expr(expr, arena, path_resolvers)?;
143            Ok(ArenaRootExpr::MathExpression(expr_ref))
144        }
145    }
146}

152/// Check if a ValueExpr is a literal and get its value
153fn try_get_literal(expr: &ValueExpr) -> Option<&Value> {
154    match expr {
155        ValueExpr::Literal(v) => Some(v),
156        _ => None,
157    }
158}
159
160/// Check if arena bool expr is a literal
161fn arena_bool_is_literal<F: EvalContextFamily>(arena: &AstArena<F>, r: BoolExprRef) -> Option<bool> {
162    match arena.get_bool(r) {
163        ArenaBoolExpr::Literal(b) => Some(*b),
164        _ => None,
165    }
166}
167
168/// Try to get literal value from a MathExpr
169fn try_get_math_literal(expr: &MathExpr) -> Option<&Value> {
170    match expr {
171        MathExpr::Primary(ValueExpr::Literal(v)) => Some(v),
172        _ => None,
173    }
174}

183/// Converts a boxed [`BoolExpr`] into arena-based form, allocating nodes in `arena` and returning a [`BoolExprRef`].
184///
185/// - Resolves paths via [`resolve_path`] (path → [`ResolvedPath`] with [`crate::PathAccessor`]).
186/// - Recursively converts nested bool/value expressions and function calls.
187/// - Applies **constant folding** where possible: literal comparisons, `not(literal)`, short-circuit
188///   `and`/`or` with literal operands, so the arena may contain fewer/simpler nodes.
189fn convert_bool_expr<F: EvalContextFamily>(
190    expr: &BoolExpr, arena: &mut AstArena<F>, path_resolvers: &PathResolverMap<F>,
191) -> Result<BoolExprRef> {
192    let arena_expr = match expr {
193        BoolExpr::Literal(b) => ArenaBoolExpr::Literal(*b),
194        BoolExpr::Comparison { left, op, right } => {
195            // CONSTANT FOLDING: if both sides are literals, compute at parse time!
196            if let (Some(left_val), Some(right_val)) = (try_get_literal(left), try_get_literal(right)) {
197                if let Ok(result) = compare(left_val, op, right_val) {
198                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(result)));
199                }
200            }
201            let left_ref = convert_value_expr(left, arena, path_resolvers)?;
202            let right_ref = convert_value_expr(right, arena, path_resolvers)?;
203            ArenaBoolExpr::Comparison {
204                left: left_ref,
205                op: *op,
206                right: right_ref,
207            }
208        }
209        BoolExpr::Converter(fc) => {
210            let fc_ref = convert_function_call(fc, arena, path_resolvers)?;
211            ArenaBoolExpr::Converter(fc_ref)
212        }
213        BoolExpr::Path(path) => {
214            let resolved = resolve_path(path, path_resolvers)?;
215            ArenaBoolExpr::Path(resolved)
216        }
217        BoolExpr::Not(inner) => {
218            let inner_ref = convert_bool_expr(inner, arena, path_resolvers)?;
219            // CONSTANT FOLDING: not(literal) => !literal
220            if let Some(b) = arena_bool_is_literal(arena, inner_ref) {
221                return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(!b)));
222            }
223            ArenaBoolExpr::Not(inner_ref)
224        }
225        BoolExpr::And(left, right) => {
226            let left_ref = convert_bool_expr(left, arena, path_resolvers)?;
227            // CONSTANT FOLDING: false && x => false, true && x => x
228            if let Some(left_val) = arena_bool_is_literal(arena, left_ref) {
229                if !left_val {
230                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(false)));
231                }
232                return convert_bool_expr(right, arena, path_resolvers);
233            }
234            let right_ref = convert_bool_expr(right, arena, path_resolvers)?;
235            // CONSTANT FOLDING: x && false => false, x && true => x
236            if let Some(right_val) = arena_bool_is_literal(arena, right_ref) {
237                if !right_val {
238                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(false)));
239                }
240                return Ok(left_ref);
241            }
242            ArenaBoolExpr::And(left_ref, right_ref)
243        }
244        BoolExpr::Or(left, right) => {
245            let left_ref = convert_bool_expr(left, arena, path_resolvers)?;
246            // CONSTANT FOLDING: true || x => true, false || x => x
247            if let Some(left_val) = arena_bool_is_literal(arena, left_ref) {
248                if left_val {
249                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(true)));
250                }
251                return convert_bool_expr(right, arena, path_resolvers);
252            }
253            let right_ref = convert_bool_expr(right, arena, path_resolvers)?;
254            // CONSTANT FOLDING: x || true => true, x || false => x
255            if let Some(right_val) = arena_bool_is_literal(arena, right_ref) {
256                if right_val {
257                    return Ok(arena.alloc_bool(ArenaBoolExpr::Literal(true)));
258                }
259                return Ok(left_ref);
260            }
261            ArenaBoolExpr::Or(left_ref, right_ref)
262        }
263    };
264    Ok(arena.alloc_bool(arena_expr))
265}

267/// Converts a boxed [`MathExpr`] into arena-based form, allocating nodes in `arena` and returning a [`MathExprRef`].
268///
269/// - Recursively converts primary value expressions and nested math expressions.
270/// - Applies **constant folding**: negate of literal, binary `literal op literal` → computed literal,
271///   so the arena may contain a single [`ArenaMathExpr::Primary`] where the source had a tree.
272fn convert_math_expr<F: EvalContextFamily>(
273    expr: &MathExpr, arena: &mut AstArena<F>, path_resolvers: &PathResolverMap<F>,
274) -> Result<MathExprRef> {
275    let arena_expr = match expr {
276        MathExpr::Primary(v) => {
277            let v_ref = convert_value_expr(v, arena, path_resolvers)?;
278            ArenaMathExpr::Primary(v_ref)
279        }
280        MathExpr::Negate(inner) => {
281            // CONSTANT FOLDING: -literal => literal negated
282            if let Some(val) = try_get_math_literal(inner) {
283                let negated = match val {
284                    Value::Int(i) => Some(Value::Int(-i)),
285                    Value::Float(f) => Some(Value::Float(-f)),
286                    _ => None,
287                };
288                if let Some(v) = negated {
289                    let v_ref = arena.alloc_value(ArenaValueExpr::Literal(v));
290                    return Ok(arena.alloc_math(ArenaMathExpr::Primary(v_ref)));
291                }
292            }
293            let inner_ref = convert_math_expr(inner, arena, path_resolvers)?;
294            ArenaMathExpr::Negate(inner_ref)
295        }
296        MathExpr::Binary { left, op, right } => {
297            // CONSTANT FOLDING: literal op literal => computed literal
298            if let (Some(left_val), Some(right_val)) = (try_get_math_literal(left), try_get_math_literal(right)) {
299                if let Some(result) = try_math_op(left_val, op, right_val) {
300                    let v_ref = arena.alloc_value(ArenaValueExpr::Literal(result));
301                    return Ok(arena.alloc_math(ArenaMathExpr::Primary(v_ref)));
302                }
303            }
304            let left_ref = convert_math_expr(left, arena, path_resolvers)?;
305            let right_ref = convert_math_expr(right, arena, path_resolvers)?;
306            ArenaMathExpr::Binary {
307                left: left_ref,
308                op: *op,
309                right: right_ref,
310            }
311        }
312    };
313    Ok(arena.alloc_math(arena_expr))
314}

316/// Converts a boxed [`ValueExpr`] into arena-based form, allocating nodes in `arena` and returning a [`ValueExprRef`].
317///
318/// - Resolves paths via [`resolve_path`] (path → [`ResolvedPath`] with [`crate::PathAccessor`]).
319/// - Recursively converts list items, map entries, nested function calls, and math expressions.
320/// - Literals are stored as-is without further conversion.
321fn convert_value_expr<F: EvalContextFamily>(
322    expr: &ValueExpr, arena: &mut AstArena<F>, path_resolvers: &PathResolverMap<F>,
323) -> Result<ValueExprRef> {
324    let arena_expr = match expr {
325        ValueExpr::Literal(v) => ArenaValueExpr::Literal(v.clone()),
326        ValueExpr::Path(path) => {
327            let resolved = resolve_path(path, path_resolvers)?;
328            ArenaValueExpr::Path(resolved)
329        }
330        ValueExpr::List(items) => {
331            let refs: Result<Vec<_>> = items
332                .iter()
333                .map(|i| convert_value_expr(i, arena, path_resolvers))
334                .collect();
335            ArenaValueExpr::List(refs?)
336        }
337        ValueExpr::Map(entries) => {
338            let refs: Result<Vec<_>> = entries
339                .iter()
340                .map(|(k, v)| Ok((k.clone(), convert_value_expr(v, arena, path_resolvers)?)))
341                .collect();
342            ArenaValueExpr::Map(refs?)
343        }
344        ValueExpr::FunctionCall(fc) => {
345            let fc_ref = convert_function_call(fc, arena, path_resolvers)?;
346            ArenaValueExpr::FunctionCall(fc_ref)
347        }
348        ValueExpr::Math(m) => {
349            let m_ref = convert_math_expr(m, arena, path_resolvers)?;
350            ArenaValueExpr::Math(m_ref)
351        }
352    };
353    Ok(arena.alloc_value(arena_expr))
354}

356/// Converts a boxed [`FunctionCall`] into arena-based form, allocating in `arena` and returning a [`FunctionCallRef`].
357///
358/// - Converts all arguments (positional and named) to arena value refs via [`convert_value_expr`].
359/// - Preserves `name`, `is_editor`, `indexes`, and the pre-resolved `callback` from the source AST.
360/// - Does not resolve callbacks here; the parser is expected to have attached [`CallbackFn`] when applicable.
361fn convert_function_call<F: EvalContextFamily>(
362    fc: &FunctionCall, arena: &mut AstArena<F>, path_resolvers: &PathResolverMap<F>,
363) -> Result<FunctionCallRef> {
364    let args: Result<Vec<_>> = fc
365        .args
366        .iter()
367        .map(|arg| match arg {
368            ArgExpr::Positional(v) => Ok(ArenaArgExpr::Positional(convert_value_expr(v, arena, path_resolvers)?)),
369            ArgExpr::Named { name, value } => Ok(ArenaArgExpr::Named {
370                name: name.clone(),
371                value: convert_value_expr(value, arena, path_resolvers)?,
372            }),
373        })
374        .collect();
375
376    let arena_fc = ArenaFunctionCall {
377        name: fc.name.clone(),
378        is_editor: fc.is_editor,
379        args: args?,
380        indexes: fc.indexes.clone(),
381        callback: fc.callback.clone(),
382    };
383    Ok(arena.alloc_func(arena_fc))
