15/// Unescape a quoted string literal (removes quotes and handles escapes)
16#[inline]
17fn unescape(s: &str) -> String {
18    s[1..s.len() - 1].replace("\\\"", "\"").replace("\\\\", "\\")
19}

25/// Parser for literal values
26fn literal_parser<'a>() -> impl chumsky::Parser<'a, TokenInput<'a>, ValueExpr, ParserExtra<'a>> + Clone {
27    let string_literal = select_ref! {
28        Token::StringLiteral(s) => Value::string(unescape(s))
29    };
30
31    let int_literal = just(&Token::Minus)
32        .or_not()
33        .then(select_ref! {
34            Token::IntLiteral(s) => s.parse::<i64>().unwrap_or(0)
35        })
36        .map(|(neg, val)| {
37            let val = if neg.is_some() { -val } else { val };
38            Value::Int(val)
39        });
40
41    let float_literal = just(&Token::Minus)
42        .or_not()
43        .then(select_ref! {
44            Token::FloatLiteral(s) => s.parse::<f64>().unwrap_or(0.0)
45        })
46        .map(|(neg, val)| {
47            let val = if neg.is_some() { -val } else { val };
48            Value::Float(val)
49        });
50
51    let bytes_literal = select_ref! {
52        Token::BytesLiteral(s) => {
53            let hex = &s[2..];
54            let bytes: Vec<u8> = (0..hex.len())
55                .step_by(2)
56                .map(|i| {
57                    let end = (i + 2).min(hex.len());
58                    u8::from_str_radix(&hex[i..end], 16).unwrap_or(0)
59                })
60                .collect();
61            Value::bytes(bytes)
62        }
63    };
64
65    let bool_literal = select_ref! {
66        Token::True => Value::Bool(true),
67        Token::False => Value::Bool(false),
68    };
69
70    let nil_literal = just(&Token::Nil).to(Value::Nil);
71
72    choice((
73        float_literal,
74        int_literal,
75        string_literal,
76        bytes_literal,
77        bool_literal,
78        nil_literal,
79    ))
80    .map(ValueExpr::Literal)
81}

83/// Parser for index expressions: "[" (string | int) "]"
84fn index_parser<'a>() -> impl chumsky::Parser<'a, TokenInput<'a>, IndexExpr, ParserExtra<'a>> + Clone {
85    choice((
86        select_ref! { Token::StringLiteral(s) => IndexExpr::String(unescape(s)) },
87        select_ref! { Token::IntLiteral(s) => IndexExpr::Int(s.parse::<usize>().unwrap_or(0)) },
88    ))
89    .delimited_by(just(&Token::LBracket), just(&Token::RBracket))
90}

92/// Parser for identifiers (lower or upper case)
93fn ident_parser<'a>(upper: bool) -> impl chumsky::Parser<'a, TokenInput<'a>, String, ParserExtra<'a>> + Clone {
94    if upper {
95        select_ref! { Token::UpperIdent(s) => s.to_string() }.boxed()
96    } else {
97        select_ref! { Token::LowerIdent(s) => s.to_string() }.boxed()
98    }
99}

101/// Parser for path expressions: lower_ident ("." ident_segment)* index*
102fn path_parser<'a>() -> impl chumsky::Parser<'a, TokenInput<'a>, PathExpr, ParserExtra<'a>> + Clone {
103    let lower_ident = ident_parser(false);
104    let ident_segment = ident_parser(false).or(ident_parser(true));
105    let index = index_parser();
106
107    lower_ident
108        .then(
109            just(&Token::Dot)
110                .ignore_then(ident_segment)
111                .repeated()
112                .collect::<Vec<_>>(),
113        )
114        .then(index.repeated().collect::<Vec<_>>())
115        .map(|((first, rest), indexes)| {
116            let mut segments = vec![first];
117            segments.extend(rest);
118            PathExpr { segments, indexes }
119        })
120}

122/// Parser for comparison operators
123fn comp_op_parser<'a>() -> impl chumsky::Parser<'a, TokenInput<'a>, CompOp, ParserExtra<'a>> + Clone {
124    choice((
125        just(&Token::Eq).to(CompOp::Eq),
126        just(&Token::NotEq).to(CompOp::NotEq),
127        just(&Token::LessEq).to(CompOp::LessEq),
128        just(&Token::GreaterEq).to(CompOp::GreaterEq),
129        just(&Token::Less).to(CompOp::Less),
130        just(&Token::Greater).to(CompOp::Greater),
131    ))
132}

134/// Parser for argument list
135fn arg_list_parser<'a>(
136    arg_value: impl chumsky::Parser<'a, TokenInput<'a>, ValueExpr, ParserExtra<'a>> + Clone + 'a,
137) -> impl chumsky::Parser<'a, TokenInput<'a>, Vec<ArgExpr>, ParserExtra<'a>> + Clone + 'a {
138    let named_arg = ident_parser(false)
139        .then_ignore(just(&Token::Assign))
140        .then(arg_value.clone())
141        .map(|(name, value)| ArgExpr::Named { name, value });
142
143    let positional_arg = arg_value.map(ArgExpr::Positional);
144
145    named_arg
146        .or(positional_arg)
147        .separated_by(just(&Token::Comma))
148        .allow_trailing()
149        .collect::<Vec<_>>()
150}

152/// Wraps a MathExpr into ValueExpr, unwrapping simple Primary values
153fn math_to_value_expr(math: MathExpr) -> ValueExpr {
154    match math {
155        MathExpr::Primary(v) => v,
156        other => ValueExpr::Math(Box::new(other)),
157    }
158}

160/// Creates a math expression parser
161fn make_math_expr<'a>(
162    value_expr: impl chumsky::Parser<'a, TokenInput<'a>, ValueExpr, ParserExtra<'a>> + Clone + 'a,
163) -> impl chumsky::Parser<'a, TokenInput<'a>, MathExpr, ParserExtra<'a>> + Clone + 'a {
164    recursive(move |math_expr| {
165        let math_literal = choice((
166            select_ref! {
167                Token::FloatLiteral(s) => {
168                    let val: f64 = s.parse().unwrap_or(0.0);
169                    MathExpr::Primary(ValueExpr::Literal(Value::Float(val)))
170                }
171            },
172            select_ref! {
173                Token::IntLiteral(s) => {
174                    let val: i64 = s.parse().unwrap_or(0);
175                    MathExpr::Primary(ValueExpr::Literal(Value::Int(val)))
176                }
177            },
178        ));
179
180        let paren_math = math_expr
181            .clone()
182            .delimited_by(just(&Token::LParen), just(&Token::RParen));
183
184        let math_value = value_expr.clone().try_map(|v, span| match &v {
185            ValueExpr::FunctionCall(_) => Ok(MathExpr::Primary(v)),
186            ValueExpr::Path(_) => Ok(MathExpr::Primary(v)),
187            ValueExpr::Literal(_) => Ok(MathExpr::Primary(v)),
188            ValueExpr::List(_) => Ok(MathExpr::Primary(v)),
189            ValueExpr::Map(_) => Ok(MathExpr::Primary(v)),
190            _ => Err(Rich::custom(span, "Expected converter, path, literal, list, or map")),
191        });
192
193        let primary = choice((paren_math, math_literal, math_value));
194
195        let unary_op = choice((just(&Token::Plus).to(true), just(&Token::Minus).to(false)));
196
197        let factor = unary_op.or_not().then(primary).map(|(op, expr)| match op {
198            Some(false) => MathExpr::Negate(Box::new(expr)),
199            _ => expr,
200        });
201
202        let mul_op = choice((
203            just(&Token::Multiply).to(MathOp::Mul),
204            just(&Token::Divide).to(MathOp::Div),
205        ));
206
207        let term = factor
208            .clone()
209            .foldl(mul_op.then(factor).repeated(), |left, (op, right)| MathExpr::Binary {
210                left: Box::new(left),
211                op,
212                right: Box::new(right),
213            });
214
215        let add_op = choice((just(&Token::Plus).to(MathOp::Add), just(&Token::Minus).to(MathOp::Sub)));
216
217        term.clone()
218            .foldl(add_op.then(term).repeated(), |left, (op, right)| MathExpr::Binary {
219                left: Box::new(left),
220                op,
221                right: Box::new(right),
222            })
223    })
224}

230/// Build the chumsky parser for OTTL
231pub fn build_parser<'a>(
232    editors_map: &'a CallbackMap, converters_map: &'a CallbackMap, enums_map: &'a EnumMap,
233) -> impl chumsky::Parser<'a, TokenInput<'a>, RootExpr, ParserExtra<'a>> + 'a {
234    let literal = literal_parser();
235    let index = index_parser();
236    let path = path_parser();
237    let comp_op = comp_op_parser();
238
239    let enum_parser = {
240        ident_parser(true).try_map(move |name, span| {
241            if let Some(&val) = enums_map.get(&name) {
242                Ok(ValueExpr::Literal(Value::Int(val)))
243            } else {
244                Err(Rich::custom(span, format!("Unknown enum: {}", name)))
245            }
246        })
247    };
248
249    let path_for_bool = path.clone();
250
251    let value_expr = recursive(|value_expr| {
252        let list = value_expr
253            .clone()
254            .separated_by(just(&Token::Comma))
255            .allow_trailing()
256            .collect::<Vec<_>>()
257            .delimited_by(just(&Token::LBracket), just(&Token::RBracket))
258            .map(ValueExpr::List);
259
260        let map_entry = select_ref! { Token::StringLiteral(s) => unescape(s) }
261            .then_ignore(just(&Token::Colon))
262            .then(value_expr.clone());
263
264        let map = map_entry
265            .separated_by(just(&Token::Comma))
266            .allow_trailing()
267            .collect::<Vec<_>>()
268            .delimited_by(just(&Token::LBrace), just(&Token::RBrace))
269            .map(ValueExpr::Map);
270
271        let arg_value = make_math_expr(value_expr.clone()).map(math_to_value_expr);
272        let arg_list = arg_list_parser(arg_value);
273
274        let converter_call = ident_parser(true)
275            .then(arg_list.delimited_by(just(&Token::LParen), just(&Token::RParen)))
276            .then(index.clone().repeated().collect::<Vec<_>>())
277            .map(move |((name, args), indexes)| {
278                let callback = converters_map.get(&name).cloned();
279                ValueExpr::FunctionCall(Box::new(FunctionCall {
280                    name,
281                    is_editor: false,
282                    args,
283                    indexes,
284                    callback,
285                }))
286            });
287
288        choice((
289            converter_call,
290            list,
291            map,
292            enum_parser.clone(),
293            path.clone().map(ValueExpr::Path),
294            literal.clone(),
295        ))
296    });
297
298    let editor_call = {
299        let arg_value = make_math_expr(value_expr.clone()).map(math_to_value_expr);
300        let arg_list = arg_list_parser(arg_value);
301
302        ident_parser(false)
303            .then(arg_list.delimited_by(just(&Token::LParen), just(&Token::RParen)))
304            .map(move |(name, args)| FunctionCall {
305                callback: editors_map.get(&name).cloned(),
306                name,
307                is_editor: true,
308                args,
309                indexes: Vec::new(),
310            })
311    };
312
313    let math_expr_for_comparison = make_math_expr(value_expr.clone());
314    let comparison_value = math_expr_for_comparison.clone().map(|m| ValueExpr::Math(Box::new(m)));
315
316    let bool_expr = recursive({
317        let value_expr = value_expr.clone();
318        let comparison_value = comparison_value.clone();
319        let path = path_for_bool.clone();
320        move |bool_expr| {
321            let bool_literal_expr = select_ref! {
322                Token::True => BoolExpr::Literal(true),
323                Token::False => BoolExpr::Literal(false),
324            };
325
326            let comparison = comparison_value
327                .clone()
328                .then(comp_op.clone())
329                .then(comparison_value.clone())
330                .map(|((left, op), right)| BoolExpr::Comparison { left, op, right });
331
332            let bool_converter = value_expr.clone().try_map(|v, span| {
333                if let ValueExpr::FunctionCall(fc) = v {
334                    Ok(BoolExpr::Converter(fc))
335                } else {
336                    Err(Rich::custom(span, "Expected converter call"))
337                }
338            });
339
340            let bool_path = path.clone().map(BoolExpr::Path);
341
342            let bool_primary = choice((
343                bool_expr
344                    .clone()
345                    .delimited_by(just(&Token::LParen), just(&Token::RParen)),
346                comparison,
347                bool_literal_expr,
348                bool_converter,
349                bool_path,
350            ));
351
352            let bool_factor = just(&Token::Not).or_not().then(bool_primary).map(|(not, expr)| {
353                if not.is_some() {
354                    BoolExpr::Not(Box::new(expr))
355                } else {
356                    expr
357                }
358            });
359
360            let bool_term = bool_factor
361                .clone()
362                .foldl(just(&Token::And).ignore_then(bool_factor).repeated(), |left, right| {
363                    BoolExpr::And(Box::new(left), Box::new(right))
364                });
365
366            bool_term
367                .clone()
368                .foldl(just(&Token::Or).ignore_then(bool_term).repeated(), |left, right| {
369                    BoolExpr::Or(Box::new(left), Box::new(right))
370                })
371        }
372    });
373
374    let math_expr = make_math_expr(value_expr.clone());
375
376    let math_expr_with_ops = math_expr.clone().try_map(|m, span| {
377        if let MathExpr::Binary { .. } = &m {
378            Ok(m)
379        } else if let MathExpr::Negate(_) = &m {
380            Ok(m)
381        } else {
382            Err(Rich::custom(span, "Expected math expression with operators"))
383        }
384    });
385
386    let where_clause = just(&Token::Where).ignore_then(bool_expr.clone());
387
388    let editor_statement = editor_call
389        .then(where_clause.or_not())
390        .map(|(editor, condition)| RootExpr::EditorStatement(EditorStatement { editor, condition }));
391
392    choice((
393        editor_statement.then_ignore(end()),
394        math_expr_with_ops.map(RootExpr::MathExpression).then_ignore(end()),
395        bool_expr.map(RootExpr::BooleanExpression).then_ignore(end()),
396        math_expr.map(RootExpr::MathExpression).then_ignore(end()),
397    ))
