1//! Helper utilities for OTTL integrators.
2//!
3//! This module provides functions that implement common behavior expected when
4//! integrating OTTL (e.g. implementing [`crate::PathAccessor`]). Index resolution
5//! is the integrator's responsibility; this helper is provided for convenience
6//! in tests and reference implementations.
7
8use crate::{IndexExpr, Value};
9
10/// Applies a sequence of index expressions to a value and returns the result.
11///
12/// Supports indexing into [`Value::List`] and [`Value::Map`] with [`IndexExpr::Int`]
13/// and [`IndexExpr::String`] respectively, and into [`Value::String`] with
14/// [`IndexExpr::Int`] (character index). Any other combination returns an error.
15///
16/// # Errors
17///
18/// Returns an error if an index is out of bounds, a map key is missing, or
19/// the value type does not support the given index type.
20///
21/// # Example
22///
23/// ```ignore
24/// use ottl::helpers::apply_indexes;
25/// use ottl::{Value, IndexExpr};
26///
27/// let list = Value::List(vec![Value::Int(1), Value::Int(2)]);
28/// let indexes = [IndexExpr::Int(0)];
29/// let v = apply_indexes(list, &indexes)?;
30/// assert!(matches!(v, Value::Int(1)));
31/// ```
32pub fn apply_indexes(value: Value, indexes: &[IndexExpr]) -> crate::Result<Value> {
33    let mut current = value;
34    for index in indexes {
35        current = match (&current, index) {
36            (Value::List(list), IndexExpr::Int(i)) => list
37                .get(*i)
38                .cloned()
39                .ok_or_else(|| -> crate::BoxError { format!("Index {} out of bounds", i).into() })?,
40            (Value::Map(map), IndexExpr::String(key)) => map
41                .get(key)
42                .cloned()
43                .ok_or_else(|| -> crate::BoxError { format!("Key '{}' not found", key).into() })?,
44            (Value::String(s), IndexExpr::Int(i)) => s
45                .chars()
46                .nth(*i)
47                .map(|c| Value::string(c.to_string()))
48                .ok_or_else(|| -> crate::BoxError { format!("Index {} out of bounds", i).into() })?,
49            _ => return Err(format!("Cannot index {:?} with {:?}", current, index).into()),
50        };
51    }
52    Ok(current)
53}
