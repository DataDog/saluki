3use antithesis_sdk::random::random_choice;
4use rand::distr::Distribution;
5use rand::Rng;
6
7use super::common::{self, Vibe};
8use crate::rand::{Boundary, Wide};
9
10const METRIC_TYPES: &[&[u8]] = &[b"c", b"g", b"ms", b"h", b"s", b"d"];
11
12/// Sample-rate payloads (the `@` field).
13const COMPLIANT_RATE: &[&[u8]] = &[b"1", b"0.5", b"0.25", b"0.1", b"0.001"];
14
15/// Container-id payloads (the `c:` field).
16const COMPLIANT_CONTAINER: &[&[u8]] = &[b"ci-0a1b2c3d4e5f", b"cid-deadbeef", b"in-4026531840"];
17
18/// External-data items (the `e:` field), joined by ',' at runtime.
19const COMPLIANT_EXT: &[&[u8]] = &[
20    b"it-true",
21    b"it-false",
22    b"cn-redis",
23    b"cn-web",
24    b"pu-810fe89d",
25    b"pu-abc",
26];
27
28/// Cardinality payloads (the `card:` field).
29const COMPLIANT_CARD: &[&[u8]] = &[b"none", b"low", b"orchestrator", b"high"];
30
31/// The `e:` external-data item separator.
32const EXT_SEPARATORS: &[u8] = b",";
33
34/// How to build a value.
35#[derive(Clone, Copy)]
36enum ValueKind {
37    Aberrant,
38    Int,
39    Wide,
40}
41
42/// A metric extension field.
43#[derive(Clone, Copy)]
44enum Ext {
45    Rate,
46    Container,
47    Timestamp,
48    External,
49    Cardinality,
50}
51
52/// Append one metric line `<NAME>:<VALUE>|<TYPE>[|ext...]` to `buf`.
53pub(crate) fn write<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe) {
54    common::write_words(rng, buf, vibe);
55    buf.push(b':');
56    write_value(rng, buf, vibe);
57    buf.push(b'|');
58    if let Some(&t) = random_choice(METRIC_TYPES) {
59        buf.extend_from_slice(t);
60    }
61    common::write_tags(rng, buf, vibe);
62    write_extensions(rng, buf, vibe);
63    buf.push(b'\n');
64}
65
66/// Clean: a wide log-uniform value. Feral: an aberrant literal, a wide integer,
67/// or a wide float in a compact or cursed-but-equivalent expanded encoding.
68fn write_value<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe) {
69    let mut ryu = ryu::Buffer::new();
70    match vibe {
71        Vibe::Clean => {
72            let v: f64 = Wide.sample(rng);
73            buf.extend_from_slice(ryu.format(v).as_bytes());
74        }
75        Vibe::Feral => match random_choice(&[ValueKind::Aberrant, ValueKind::Int, ValueKind::Wide]) {
76            Some(ValueKind::Aberrant) => {
77                if let Some(&v) = random_choice(common::ABERRANT_VALUES) {
78                    buf.extend_from_slice(v);
79                }
80            }
81            Some(ValueKind::Wide) => {
82                let v: f64 = Wide.sample(rng);
83                common::write_number(rng, buf, ryu.format(v).as_bytes());
84            }
85            _ => {
86                let mut itoa = itoa::Buffer::new();
87                let v: i64 = Wide.sample(rng);
88                common::write_number(rng, buf, itoa.format(v).as_bytes());
89            }
90        },
91    }
92}

94/// A boundary-sampled count of extension fields, each a random kind. Repeats and
95/// zero are allowed.
96fn write_extensions<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe) {
97    let count = Boundary::<u8>::new().sample(rng);
98    for _ in 0..count {
99        match random_choice(&[
100            Ext::Rate,
101            Ext::Container,
102            Ext::Timestamp,
103            Ext::External,
104            Ext::Cardinality,
105        ]) {
106            Some(Ext::Rate) => common::write_field(buf, vibe, b"@", COMPLIANT_RATE, common::ABERRANT_VALUES),
107            Some(Ext::Container) => common::write_field(buf, vibe, b"c:", COMPLIANT_CONTAINER, common::ABERRANT_WORD),
108            Some(Ext::Timestamp) => common::write_field(buf, vibe, b"T", common::COMPLIANT_TS, common::ABERRANT_VALUES),
109            Some(Ext::External) => {
110                buf.extend_from_slice(b"|e:");
111                common::write_segments(rng, buf, vibe, COMPLIANT_EXT, common::ABERRANT_WORD, EXT_SEPARATORS);
112            }
113            _ => common::write_field(buf, vibe, b"card:", COMPLIANT_CARD, common::ABERRANT_WORD),
114        }
115    }
