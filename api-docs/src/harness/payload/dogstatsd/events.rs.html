1//! Feral `DogStatsD` event generation.
2
3use antithesis_sdk::random::random_choice;
4use rand::distr::Distribution;
5use rand::Rng;
6
7use super::common::{self, Vibe};
8use crate::rand::Boundary;
9
10/// Priority payloads (the `p:` field).
11const COMPLIANT_PRIO: &[&[u8]] = &[b"normal", b"low"];
12
13/// Alert-type payloads (the `t:` field).
14const COMPLIANT_ALERT: &[&[u8]] = &[b"error", b"warning", b"info", b"success"];
15
16/// An event optional field.
17#[derive(Clone, Copy)]
18enum Opt {
19    Timestamp,
20    Hostname,
21    AggKey,
22    Priority,
23    Source,
24    Alert,
25}
26
27/// Append one event `_e{<TLEN>,<XLEN>}:<TITLE>|<TEXT>[|opt...]` to `buf`.
28pub(crate) fn write<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe) {
29    let mut title = Vec::new();
30    common::write_words(rng, &mut title, vibe);
31    let mut text = Vec::new();
32    common::write_words(rng, &mut text, vibe);
33
34    buf.extend_from_slice(b"_e{");
35    write_len(rng, buf, vibe, title.len());
36    buf.push(b',');
37    write_len(rng, buf, vibe, text.len());
38    buf.extend_from_slice(b"}:");
39    buf.extend_from_slice(&title);
40    buf.push(b'|');
41    buf.extend_from_slice(&text);
42
43    let count = Boundary::<u8>::new().sample(rng);
44    for _ in 0..count {
45        match random_choice(&[
46            Opt::Timestamp,
47            Opt::Hostname,
48            Opt::AggKey,
49            Opt::Priority,
50            Opt::Source,
51            Opt::Alert,
52        ]) {
53            Some(Opt::Timestamp) => {
54                common::write_field(buf, vibe, b"d:", common::COMPLIANT_TS, common::ABERRANT_VALUES);
55            }
56            Some(Opt::Hostname) => {
57                buf.extend_from_slice(b"|h:");
58                common::write_words(rng, buf, vibe);
59            }
60            Some(Opt::AggKey) => {
61                buf.extend_from_slice(b"|k:");
62                common::write_words(rng, buf, vibe);
63            }
64            Some(Opt::Priority) => common::write_field(buf, vibe, b"p:", COMPLIANT_PRIO, common::ABERRANT_WORD),
65            Some(Opt::Source) => {
66                buf.extend_from_slice(b"|s:");
67                common::write_words(rng, buf, vibe);
68            }
69            _ => common::write_field(buf, vibe, b"t:", COMPLIANT_ALERT, common::ABERRANT_WORD),
70        }
71    }
72
73    common::write_tags(rng, buf, vibe);
74    buf.push(b'\n');
75}
76
77/// The event header length. Clean writes the true byte length; feral writes a
78/// boundary-sampled lie — the malformed-event surface.
79fn write_len<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe, actual: usize) {
80    let mut itoa = itoa::Buffer::new();
81    match vibe {
82        Vibe::Clean => buf.extend_from_slice(itoa.format(actual).as_bytes()),
83        Vibe::Feral => buf.extend_from_slice(itoa.format(Boundary::<u64>::new().sample(rng)).as_bytes()),
84    }
85}
