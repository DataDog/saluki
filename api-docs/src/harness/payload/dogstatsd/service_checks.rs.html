1//! Feral `DogStatsD` service-check generation.
2
3use rand::distr::Distribution;
4use rand::seq::IndexedRandom;
5use rand::Rng;
6
7use super::common::{self, Vibe};
8use crate::rand::Boundary;
9
10/// Status payloads: OK, warning, critical, unknown.
11const COMPLIANT_STATUS: &[&[u8]] = &[b"0", b"1", b"2", b"3"];
12
13/// A service-check optional field.
14#[derive(Clone, Copy)]
15enum Opt {
16    Hostname,
17    Message,
18}
19
20/// Append one service check `_sc|<NAME>|<STATUS>[|opt...]` to `buf`.
21pub(crate) fn write<R: Rng + ?Sized>(rng: &mut R, buf: &mut Vec<u8>, vibe: Vibe) {
22    buf.extend_from_slice(b"_sc|");
23    common::write_words(rng, buf, vibe);
24    buf.push(b'|');
25    // Status grammar is exactly [0-3] with zero slack — any aberrant status rejects
26    // the whole check, so feral strangeness lives in the name and options, not here.
27    common::extend_choice(rng, buf, vibe, COMPLIANT_STATUS, COMPLIANT_STATUS);
28
29    let count = Boundary::<u8>::new().sample(rng);
30    for _ in 0..count {
31        if let Some(Opt::Hostname) = [Opt::Hostname, Opt::Message].choose(rng) {
32            buf.extend_from_slice(b"|h:");
33            common::write_words(rng, buf, vibe);
34        } else {
35            buf.extend_from_slice(b"|m:");
36            common::write_words(rng, buf, vibe);
37        }
38    }
39
40    common::write_tags(rng, buf, vibe);
41    buf.push(b'\n');
42}
