1//! Annotations for DogStatsD prefix filter transform configuration keys.
2use crate::config_registry::{generated::schema, structs, SalukiAnnotation, SchemaEntry, SupportLevel, ValueType};
3
4// The Agent schema uses `statsd_metric_namespace_blacklist` (generated/schema.rs:
5// STATSD_METRIC_NAMESPACE_BLACKLIST, yaml_path "statsd_metric_namespace_blacklist").
6//
7// ADP renamed the field to `statsd_metric_namespace_blocklist` for inclusive language, but the
8// struct has NO serde alias for the old spelling. This means Agent config files that set
9// `statsd_metric_namespace_blacklist` are silently ignored by ADP — the customized list is
10// dropped and ADP falls back to the hardcoded default. The default lists happen to be identical
11// today, so this only affects users who have customized the key in their Agent config.
12//
13// Fix: add `#[serde(alias = "statsd_metric_namespace_blacklist")]` to the struct field in
14// `transforms/dogstatsd_prefix_filter/mod.rs` and switch this static to reference the generated
15// STATSD_METRIC_NAMESPACE_BLACKLIST schema entry with the old key as additional_yaml_paths.
16// TODO: https://github.com/DataDog/saluki/issues/1480
17static STATSD_METRIC_NAMESPACE_BLOCKLIST_SCHEMA: SchemaEntry = SchemaEntry {
18    yaml_path: "statsd_metric_namespace_blocklist",
19    env_vars: &[],
20    value_type: ValueType::StringList,
21    default: None,
22};
