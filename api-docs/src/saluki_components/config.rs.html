1//! Datadog-specific configuration providers and remappers.
2use figment::{
3    providers::Serialized,
4    value::{Dict, Map},
5    Error, Metadata, Profile, Provider,
6};
7
8/// Key aliases to pass to [`ConfigurationLoader::with_key_aliases`][saluki_config::ConfigurationLoader::with_key_aliases].
9///
10/// Each entry maps a nested dot-separated path to a flat key name. When the nested path is found in a loaded
11/// config file, its value is also emitted under the flat key — but only if the flat key is not already
12/// explicitly set. This ensures both YAML nested format and flat env var format produce the same Figment key,
13/// so source precedence (env vars > file) works correctly.
14pub const KEY_ALIASES: &[(&str, &str)] = &[
15    // The Datadog Agent config file uses `proxy: http:` and `proxy: https:` (nested), while env
16    // vars produce `proxy_http` and `proxy_https` (flat). Figment treats these as different keys,
17    // so without this alias env var precedence over YAML is silently broken for proxy config.
18    ("proxy.http", "proxy_http"),
19    ("proxy.https", "proxy_https"),
20    ("proxy.no_proxy", "proxy_no_proxy"),
21];
22
23/// Remappings from environment variable names to canonical config keys.
24///
25/// Matching is case-insensitive.
26const ENV_REMAPPINGS: &[(&str, &str)] = &[("http_proxy", "proxy_http"), ("https_proxy", "proxy_https")];
27
28/// A Figment provider that remaps canonical environment variable names to our desired config keys.
29///
30/// Reads environment variables case-insensitively and maps them to config keys (e.g. `HTTP_PROXY` →
31/// `proxy_http`). Values are snapshotted at construction time.
32///
33/// Add this provider to a [`ConfigurationLoader`][saluki_config::ConfigurationLoader] *after* file-based
34/// providers and *before* vendor-prefixed env providers (e.g. `DD_`) to achieve the correct precedence:
35/// file < remapped env vars < `DD_`-prefixed.
36///
37/// For YAML key aliasing (e.g. `proxy.http` → `proxy_http`), pass [`KEY_ALIASES`] to
38/// [`ConfigurationLoader::with_key_aliases`][saluki_config::ConfigurationLoader::with_key_aliases] instead —
39/// that is handled at file-load time.
40pub struct DatadogRemapper {
41    values: serde_json::Map<String, serde_json::Value>,
42}
43
44impl DatadogRemapper {
45    /// Constructs a `DatadogRemapper` by eagerly snapshotting env var remappings.
46    pub fn new() -> Self {
47        let mut values = serde_json::Map::new();
48
49        for (env_key, env_value) in std::env::vars() {
50            let lower = env_key.to_lowercase();
51            for &(from, to) in ENV_REMAPPINGS {
52                if lower == from && !values.contains_key(to) {
53                    values.insert(to.to_string(), serde_json::Value::String(env_value.clone()));
54                }
55            }
56        }
57
58        Self { values }
59    }
60}
61
62impl Default for DatadogRemapper {
63    fn default() -> Self {
64        Self::new()
65    }
66}
67
68impl Provider for DatadogRemapper {
69    fn metadata(&self) -> Metadata {
70        Metadata::named("Datadog config remapper")
71    }
72
73    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
74        if self.values.is_empty() {
75            return Ok(Map::new());
76        }
77        Serialized::defaults(serde_json::Value::Object(self.values.clone())).data()
78    }
79}

81#[cfg(test)]
82mod tests {
83    use super::*;
84
85    static ENV_MUTEX: std::sync::Mutex<()> = std::sync::Mutex::new(());
86
87    #[test]
88    fn env_var_remapped_case_insensitively() {
89        let _guard = ENV_MUTEX.lock().unwrap();
90
91        std::env::set_var("HTTP_PROXY", "http://proxy.example.com");
92        let remapper = DatadogRemapper::new();
93        std::env::remove_var("HTTP_PROXY");
94
95        assert_eq!(
96            remapper.values.get("proxy_http").and_then(|v| v.as_str()),
97            Some("http://proxy.example.com"),
98        );
99    }
100
101    #[test]
102    fn env_var_not_remapped_when_absent() {
103        let _guard = ENV_MUTEX.lock().unwrap();
104
105        std::env::remove_var("HTTP_PROXY");
106        std::env::remove_var("http_proxy");
107        std::env::remove_var("HTTPS_PROXY");
108        std::env::remove_var("https_proxy");
109
110        let remapper = DatadogRemapper::new();
111
112        assert!(remapper.values.get("proxy_http").is_none());
113        assert!(remapper.values.get("proxy_https").is_none());
114    }
115}
