17const DEPRECATED_RATE_KEY: &str = "_sampling_priority_rate_v1";
18
19/// Priority sampler for traces with sampling priority set by the tracer.
20pub struct PrioritySampler {
21    agent_env: MetaString,
22    sampler: Sampler,
23    catalog: ServiceKeyCatalog,
24}
25// the logic for this class is taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/sampler/prioritysampler.go#L39
26// note that any logic involving tracers were removed because ADP does not currently support tracers.
27impl PrioritySampler {
28    /// Creates a new priority sampler with the given configuration.
29    pub(super) fn new(agent_env: MetaString, extra_sample_rate: f64, target_tps: f64) -> Self {
30        PrioritySampler {
31            agent_env,
32            sampler: Sampler::new(extra_sample_rate, target_tps),
33            catalog: ServiceKeyCatalog::new(),
34        }
35    }
36
37    /// Updates the target traces per second.
38    pub(super) fn update_target_tps(&mut self, target_tps: f64) {
39        self.sampler.update_target_tps(target_tps);
40    }
41
42    /// Returns the current target traces per second.
43    pub(super) fn get_target_tps(&self) -> f64 {
44        self.sampler.target_tps()
45    }
46
47    /// Sample a trace that already has a sampling priority set.
48    ///
49    /// The decision is based on the priority value; the sampler only updates
50    /// feedback rates for auto-priority traces.
51    pub(super) fn sample(
52        &mut self, now: SystemTime, trace: &mut Trace, root_idx: usize, priority: i32, client_dropped_p0s_weight: f64,
53    ) -> bool {
54        if trace.spans().is_empty() || root_idx >= trace.spans().len() {
55            return false;
56        }
57
58        let sampled = priority == PRIORITY_AUTO_KEEP || priority == PRIORITY_USER_KEEP;
59
60        // Only auto-priority traces (0 or 1) participate in the feedback loop.
61        if !(PRIORITY_AUTO_DROP..=PRIORITY_AUTO_KEEP).contains(&priority) {
62            return sampled;
63        }
64
65        let (service_name, tracer_env, weight) = {
66            let root = &trace.spans()[root_idx];
67            let tracer_env = get_trace_env(trace, root_idx).map(|env| env.as_ref()).unwrap_or("");
68            let weight = weight_root(root) + client_dropped_p0s_weight as f32;
69            (root.service(), tracer_env, weight)
70        };
71
72        let sampler_env = to_sampler_env(tracer_env, &self.agent_env);
73        let svc_sig = ServiceSignature::new(service_name, sampler_env);
74        let signature = self.catalog.register(svc_sig);
75
76        let _ = self.sampler.count_weighted_sig(now, &signature, weight);
77
78        if sampled {
79            self.apply_rate(trace, root_idx, &signature);
80        }
81
82        sampled
83    }
84
85    fn apply_rate(&self, trace: &mut Trace, root_idx: usize, signature: &Signature) -> f64 {
86        let root = &mut trace.spans_mut()[root_idx];
87        if root.parent_id() != 0 {
88            return 1.0;
89        }
90
91        // ignore the tracer specific logic
92
93        let rate = self.sampler.get_signature_sample_rate(signature);
94        root.metrics_mut()
95            .insert(MetaString::from_static(DEPRECATED_RATE_KEY), rate);
96        rate
97    }
98}
99
100fn to_sampler_env(tracer_env: &str, agent_env: &MetaString) -> MetaString {
101    if tracer_env.is_empty() {
102        agent_env.clone()
103    } else {
104        MetaString::from(tracer_env)
105    }
106}

159    fn test_priority_sample() {
160        let test_cases = [
161            (PRIORITY_USER_DROP, false), // user drop
162            (PRIORITY_AUTO_DROP, false), // auto drop
163            (PRIORITY_AUTO_KEEP, true),  // auto keep
164            (PRIORITY_USER_KEEP, true),  // user keep
165        ];
166
167        for (idx, (priority, expected_sampled)) in test_cases.iter().copied().enumerate() {
168            let mut sampler = get_test_priority_sampler(0.0);
169            let (mut trace, root_idx) = get_test_trace_with_service("service-a", idx as u64 + 1);
170            let sampled = sampler.sample(SystemTime::now(), &mut trace, root_idx, priority, 0.0);
171            assert_eq!(
172                sampled, expected_sampled,
173                "priority {} should sample={}",
174                priority, expected_sampled
175            );
176        }
177    }

179    #[test]
180    fn test_priority_sampler_tps_feedback_loop() {
181        struct TestCase {
182            target_tps: f64,
183            generated_tps: f64,
184            service: &'static str,
185            expected_tps: f64,
186            relative_error: f64,
187        }
188
189        let test_cases = [
190            TestCase {
191                target_tps: 5.0,
192                generated_tps: 50.0,
193                expected_tps: 5.0,
194                relative_error: 0.25,
195                service: "bim",
196            },
197            TestCase {
198                target_tps: 3.0,
199                generated_tps: 200.0,
200                expected_tps: 3.0,
201                relative_error: 0.25,
202                service: "2",
203            },
204            TestCase {
205                target_tps: 10.0,
206                generated_tps: 10.0,
207                expected_tps: 10.0,
208                relative_error: 0.03,
209                service: "4",
210            },
211            TestCase {
212                target_tps: 10.0,
213                generated_tps: 3.0,
214                expected_tps: 3.0,
215                relative_error: 0.03,
216                service: "10",
217            },
218            TestCase {
219                target_tps: 0.5,
220                generated_tps: 100.0,
221                expected_tps: 0.5,
222                relative_error: 0.6,
223                service: "0.5",
224            },
225        ];
226
227        for tc in test_cases {
228            let mut sampler = get_test_priority_sampler(tc.target_tps);
229            let signature = ServiceSignature::new(tc.service, "agent-env").hash();
230            let expected_rate = tc.expected_tps / tc.generated_tps;
231
232            let warm_up_duration = 5;
233            let test_duration = 20;
234            let mut test_time = SystemTime::now();
235
236            let mut sampled_count = 0;
237            let mut handled_count = 0;
238
239            for time_elapsed in 0..(warm_up_duration + test_duration) {
240                let traces_per_period = (tc.generated_tps * BUCKET_DURATION.as_secs_f64()) as usize;
241                test_time += BUCKET_DURATION;
242
243                for i in 0..traces_per_period {
244                    let trace_id = (time_elapsed as u64) << 32 | i as u64;
245                    let (mut trace, root_idx) = get_test_trace_with_service(tc.service, trace_id);
246                    let sampled = sampler.sample(test_time, &mut trace, root_idx, PRIORITY_AUTO_KEEP, 0.0);
247
248                    if time_elapsed < warm_up_duration {
249                        continue;
250                    }
251
252                    let rate = sampler.sampler.get_signature_sample_rate(&signature);
253                    assert!(
254                        (rate - expected_rate).abs() <= expected_rate * tc.relative_error,
255                        "rate mismatch for service {}: got {}, want {}",
256                        tc.service,
257                        rate,
258                        expected_rate
259                    );
260
261                    handled_count += 1;
262                    if sampled {
263                        sampled_count += 1;
264                    }
265                }
266            }
267
268            assert_eq!(
269                sampled_count, handled_count,
270                "auto-keep priority should sample every handled trace"
271            );
272        }
273    }
