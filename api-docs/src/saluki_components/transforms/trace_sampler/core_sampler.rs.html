8const NUM_BUCKETS: usize = 6;
9const BUCKET_DURATION: Duration = Duration::from_secs(5);
10const MAX_RATE_INCREASE: f64 = 1.2;

12#[derive(Default)]
13pub struct Sampler {
14    /// maps each Signature to a circular buffer of per-bucket (bucket_id) counts covering the last NUM_BUCKETS * BUCKET_DURATION window.
15    seen: FastHashMap<Signature, [f32; NUM_BUCKETS]>,
16
17    /// all_sigs_seen counts all signatures in a circular buffer of NUM_BUCKETS of BUCKET_DURATION
18    all_sigs_seen: [f32; NUM_BUCKETS],
19
20    ///  last_bucket_id is the index of the last bucket on which traces were counted
21    last_bucket_id: u64,
22
23    /// rates maps sampling rate in %
24    rates: FastHashMap<Signature, f64>,
25
26    /// lowest_rate is the lowest rate of all signatures
27    lowest_rate: f64,
28
29    /// Maximum limit to the total number of traces per second to sample
30    target_tps: f64,
31
32    /// extra_rate is an extra raw sampling rate to apply on top of the sampler rate
33    extra_rate: f64,
34}

36// zeroAndGetMax zeroes expired buckets and returns the max count
37// logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/sampler/coresampler.go#L185
38fn zero_and_get_max(buckets: &mut [f32; NUM_BUCKETS], previous_bucket: u64, new_bucket: u64) -> f32 {
39    // A bucket is a BUCKET_DURATION slice (5s) that stores the count of traces that fell in the interval.
40    // An intuitive understanding of the function is that we start just after previous_buckets and iterate for a full window of buckets (NUM_BUCKETS)
41    // and zero out any buckets older then new_buckets (expired), then we compute the max_count among the buckets that are in the current window
42    let mut max_bucket = 0.0;
43    for i in (previous_bucket + 1)..=previous_bucket + NUM_BUCKETS as u64 {
44        let index = i as usize % NUM_BUCKETS;
45        // if a complete rotation (time between previous_bucket and new_bucket is more then NUM_BUCKETS * BUCKET_DURATION) happened between previous_bucket and new_bucket
46        // all buckets will be zeroed
47        if i < new_bucket {
48            buckets[index] = 0.0;
49            continue;
50        }
51        let value = buckets[index];
52        if value > max_bucket {
53            max_bucket = value;
54        }
55        // zeroing after taking in account the previous value of the bucket
56        // overridden by this rotation. This allows to take in account all buckets
57        if i == new_bucket {
58            buckets[index] = 0.0;
59        }
60    }
61    max_bucket
62}

64// compute_tps_per_sig distributes TPS looking at the seen_tps of all signatures.
65// By default it spreads uniformly the TPS on all signatures. If a signature
66// is low volume and does not use all of its TPS, the remaining is spread uniformly
67// on all other signatures. The returned sig_target is the final per_signature TPS target
68// logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/sampler/coresampler.go#L167
69fn compute_tps_per_sig(target_tps: f64, seen_tps: &[f64]) -> f64 {
70    // Example: target_tps = 30, seen_tps = [5, 10, 100] → sorted stays [5, 10, 100], Initial sig_target = 30 / 3 = 10
71    // Loop:
72    // 1) c = 5 (< 10), so subtract: target_tps = 30 - 5 = 25
73    // Recompute sig_target = 25 / 2 = 12.5
74    // 2) c = 10 (< 12.5), subtract: target_tps = 25 - 10 = 15
75    // Recompute sig_target = 15 / 1 = 15
76    // 3) Next is last element, break.
77    // Return sig_target = 15.
78    // Interpretation: the low‑volume signatures "use up" 5 and 10 TPS, and the remaining budget (15) is the per‑signature target for the higher‑volume signature(s).
79
80    if seen_tps.is_empty() {
81        return 0.0;
82    }
83    let mut sorted: Vec<f64> = seen_tps.to_vec();
84    sorted.sort_by(|a, b| a.total_cmp(b));
85    // compute the initial per_signature TPS budget by splitting target_tps across all signatures.
86    let mut remaining_tps = target_tps;
87    let mut sig_target = remaining_tps / sorted.len() as f64;
88
89    for (i, c) in sorted.iter().enumerate() {
90        if *c >= sig_target || i == sorted.len() - 1 {
91            break;
92        }
93        remaining_tps -= c;
94        sig_target = remaining_tps / (sorted.len() - i - 1) as f64;
95    }
96    sig_target
97}

99impl Sampler {

101        Self {
102            extra_rate,
103            target_tps,
104            ..Default::default()
105        }
106    }

108    pub(super) fn count_weighted_sig(&mut self, now: SystemTime, signature: &Signature, n: f32) -> bool {
109        // All traces within the same `BUCKET_DURATION` interval share the same bucket_id
110        let bucket_id = now.duration_since(UNIX_EPOCH).unwrap_or_default().as_secs() / BUCKET_DURATION.as_secs();
111        let prev_bucket_id = self.last_bucket_id;
112        self.last_bucket_id = bucket_id;
113        // If the bucket_id changed then the sliding window advanced and we need to recompute rates
114        let update_rate = prev_bucket_id != bucket_id;
115        if update_rate {
116            self.update_rates(prev_bucket_id, bucket_id);
117        }
118
119        let buckets = self.seen.entry(*signature).or_insert([0 as f32; NUM_BUCKETS]);
120        self.all_sigs_seen[(bucket_id % (NUM_BUCKETS as u64)) as usize] += n;
121        buckets[(bucket_id % (NUM_BUCKETS as u64)) as usize] += n;
122        update_rate
123    }
124
125    // update_rates distributes TPS on each signature and apply it to the moving
126    // max of seen buckets.
127    // Rates increase are bounded by 20% increases, it requires 13 evaluations (1.2**13 = 10.6)
128    // to increase a sampling rate by 10 fold in about 1min.
129    fn update_rates(&mut self, previous_bucket: u64, new_bucket: u64) {
130        let seen_len = self.seen.len();
131        if seen_len == 0 {
132            return;
133        }
134        let mut rates: FastHashMap<Signature, f64> =
135            FastHashMap::with_capacity_and_hasher(seen_len, FastBuildHasher::default());
136        // seen_tps is a vector of per-signature peak rates, we get the maximum bucket value (which represents the number/weight of traces in a BUCKET_DURATION interval)
137        // in the sliding window and convert that to traces per second. Each element is one TPS per signature.
138        let mut seen_tps_vec = Vec::with_capacity(seen_len);
139        let mut sigs = Vec::with_capacity(seen_len);
140        let mut sigs_to_remove = Vec::new();
141
142        for (sig, buckets) in self.seen.iter_mut() {
143            let max_bucket = zero_and_get_max(buckets, previous_bucket, new_bucket);
144            let seen_tps = max_bucket as f64 / BUCKET_DURATION.as_secs() as f64;
145            seen_tps_vec.push(seen_tps);
146            sigs.push(*sig);
147        }
148        zero_and_get_max(&mut self.all_sigs_seen, previous_bucket, new_bucket);
149        let tps_per_sig = compute_tps_per_sig(self.target_tps, &seen_tps_vec);
150        self.lowest_rate = 1.0;
151
152        for (i, sig) in sigs.iter().enumerate() {
153            let seen_tps = seen_tps_vec[i];
154            let mut rate = 1.0;
155            if tps_per_sig < seen_tps && seen_tps > 0.0 {
156                rate = tps_per_sig / seen_tps;
157            }
158
159            // Cap increase rate to 20%
160            if let Some(prev_rate) = self.rates.get(sig) {
161                if *prev_rate != 0.0 && rate / prev_rate > MAX_RATE_INCREASE {
162                    rate = prev_rate * MAX_RATE_INCREASE;
163                }
164            }
165
166            // Ensure rate doesn't exceed 1.0
167            if rate > 1.0 {
168                rate = 1.0;
169            }
170
171            // No traffic on this signature, mark it for cleanup
172            if rate == 1.0 && seen_tps == 0.0 {
173                sigs_to_remove.push(*sig);
174                continue;
175            }
176
177            // Update lowest rate
178            if rate < self.lowest_rate {
179                self.lowest_rate = rate;
180            }
181
182            rates.insert(*sig, rate);
183        }
184
185        // Clean up signatures with no traffic
186        for sig in sigs_to_remove {
187            self.seen.remove(&sig);
188        }
189
190        self.rates = rates;
191    }

193    /// Gets the sampling rate for a specific signature.
194    /// Returns the rate multiplied by the extra rate factor.
195    pub fn get_signature_sample_rate(&self, sig: &Signature) -> f64 {
196        self.rates
197            .get(sig)
198            .map(|rate| rate * self.extra_rate)
199            .unwrap_or_else(|| self.default_rate())
200    }
201
202    /// Gets all signature sample rates.
203    /// Returns a tuple of (rates map, default rate).
204    pub fn get_all_signature_sample_rates(&self) -> (FastHashMap<Signature, f64>, f64) {
205        let mut rates = FastHashMap::with_capacity_and_hasher(self.rates.len(), FastBuildHasher::default());
206        for (sig, rate) in self.rates.iter() {
207            rates.insert(*sig, rate * self.extra_rate);
208        }
209        (rates, self.default_rate())
210    }
211
212    pub fn update_target_tps(&mut self, target_tps: f64) {
213        let prev_target = self.target_tps;
214        self.target_tps = target_tps;
215
216        if prev_target == 0.0 {
217            return;
218        }
219        let ratio = target_tps / prev_target;
220        for rate in self.rates.values_mut() {
221            let new_rate = (*rate * ratio).min(1.0);
222            *rate = new_rate;
223        }
224    }
225
226    pub fn target_tps(&self) -> f64 {
227        self.target_tps
228    }
229
230    /// Computes the default rate for unknown signatures.
231    /// Based on the moving max of all signatures seen and the lowest stored rate.
232    fn default_rate(&self) -> f64 {
233        if self.target_tps == 0.0 {
234            return 0.0;
235        }
236
237        let mut max_seen = 0.0_f32;
238        for &count in self.all_sigs_seen.iter() {
239            if count > max_seen {
240                max_seen = count;
241            }
242        }
243
244        let seen_tps = max_seen as f64 / BUCKET_DURATION.as_secs() as f64;
245        let mut rate = 1.0;
246
247        if self.target_tps < seen_tps && seen_tps > 0.0 {
248            rate = self.target_tps / seen_tps;
249        }
250
251        if self.lowest_rate < rate && self.lowest_rate != 0.0 {
252            return self.lowest_rate;
253        }
254
255        rate
256    }
257
258    /// Returns the number of signatures being tracked.
259    pub fn size(&self) -> i64 {
260        self.seen.len() as i64
261    }
