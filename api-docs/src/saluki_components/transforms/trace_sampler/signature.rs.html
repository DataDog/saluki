1//! Trace signature computation utilities.
2//!
3//! This module currently provides:
4//! - a small FNV-1a 32-bit helper (used by probabilistic sampling)
5//! - a signature newtype + compute helper (for score/TPS samplers)

7use saluki_core::data_model::event::trace::{Span, Trace};

12const OFFSET_32: u32 = 2166136261;
13const PRIME_32: u32 = 16777619;
14const KEY_HTTP_STATUS_CODE: &str = "http.status_code";
15const KEY_ERROR_TYPE: &str = "error.type";
16
17fn write_hash(mut hash: u32, bytes: &[u8]) -> u32 {
18    for &b in bytes {
19        hash ^= b as u32;
20        hash = hash.wrapping_mul(PRIME_32);
21    }
22    hash
23}
24
25pub(super) fn fnv1a_32(seed: &[u8], bytes: &[u8]) -> u32 {
26    let hash = write_hash(OFFSET_32, seed);
27    write_hash(hash, bytes)
28}

33/// Service identifier for sampling rate lookups.
34///
35/// Represents a unique (service name, environment) pair used as a key
36/// for storing and retrieving sampling rates in distributed sampling.
37#[derive(Clone, Debug, Default, PartialEq, Eq, Hash)]
38pub(super) struct ServiceSignature {
39    name: MetaString,
40    env: MetaString,
41}
42
43impl ServiceSignature {
44    /// Creates a new ServiceSignature from name and environment.
45    pub(super) fn new(name: impl Into<MetaString>, env: impl Into<MetaString>) -> Self {
46        Self {
47            name: name.into(),
48            env: env.into(),
49        }
50    }
51
52    /// Computes FNV-1a hash matching Go's ServiceSignature.Hash().
53    ///
54    /// The hash is computed over: `name + "," + env`
55    pub(super) fn hash(&self) -> Signature {
56        let mut h = OFFSET_32;
57        h = write_hash(h, self.name.as_ref().as_bytes());
58        h = write_hash(h, b",");
59        h = write_hash(h, self.env.as_ref().as_bytes());
60        Signature(h as u64)
61    }
62}
63
64impl std::fmt::Display for ServiceSignature {
65    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
66        write!(f, "service:{},env:{}", self.name.as_ref(), self.env.as_ref())
67    }
68}

70pub(super) fn compute_signature_with_root_and_env(trace: &Trace, root_idx: usize) -> Signature {
71    // Mirrors datadog-agent/pkg/trace/sampler/signature.go:computeSignatureWithRootAndEnv.
72    //
73    // Signature based on the hash of (env, service, name, resource, is_error) for the root, plus the set of
74    // (env, service, name, is_error) of each span.
75    let spans = trace.spans();
76    let Some(root) = spans.get(root_idx) else {
77        return Signature(0);
78    };
79
80    let env = get_trace_env(trace, root_idx).map(|v| v.as_ref()).unwrap_or("");
81    let root_hash = compute_span_hash(root, env, true);
82    let mut span_hashes: Vec<u32> = spans.iter().map(|span| compute_span_hash(span, env, false)).collect();
83
84    if span_hashes.is_empty() {
85        return Signature(root_hash as u64);
86    }
87
88    // Sort, dedupe then merge all the hashes to build the signature.
89    span_hashes.sort_unstable();
90    span_hashes.dedup();
91
92    let mut trace_hash = span_hashes[0] ^ root_hash;
93    for &h in span_hashes.iter().skip(1) {
94        trace_hash ^= h;
95    }
96
97    Signature(trace_hash as u64)
98}

100fn compute_span_hash(span: &Span, env: &str, with_resource: bool) -> u32 {
101    let mut h = OFFSET_32;
102    h = write_hash(h, env.as_bytes());
103    h = write_hash(h, span.service().as_bytes());
104    h = write_hash(h, span.name().as_bytes());
105    h = write_hash(h, &[span.error() as u8]);
106    if with_resource {
107        h = write_hash(h, span.resource().as_bytes());
108    }
109    if let Some(code) = span.meta().get(KEY_HTTP_STATUS_CODE) {
110        h = write_hash(h, code.as_ref().as_bytes());
111    }
112    if let Some(typ) = span.meta().get(KEY_ERROR_TYPE) {
113        h = write_hash(h, typ.as_ref().as_bytes());
114    }
115    h
116}
