22/// ScoreSampler for traces
23///
24/// ScoreSampler samples pieces of traces by computing a signature based on spans (service, name, rsc, http.status, error.type)
25/// scoring it and applying a rate.
26/// The rates are applied on the TraceID to maximize the number of chunks with errors caught for the same traceID.
27/// For a set traceID: P(chunk1 kept and chunk2 kept) = min(P(chunk1 kept), P(chunk2 kept))
28///
29/// # Missing
30///
31/// TODO: Add SampleV1 method for legacy trace format support
32pub struct ScoreSampler {
33    sampler: Sampler,
34    sampling_rate_key: &'static str,
35    disabled: bool,
36    // When shrinking, the shrink allowlist represents the currently active signatures while new ones get collapsed
37    shrink_allow_list: Option<FastHashMap<Signature, f64>>,
38}

61impl ScoreSampler {
62    /// Create a new ScoreSampler with the given sampling rate key and target TPS.
63    pub fn new(sampling_rate_key: &'static str, disabled: bool, target_tps: f64, extra_sample_rate: f64) -> Self {
64        Self {
65            sampler: Sampler::new(extra_sample_rate, target_tps),
66            sampling_rate_key,
67            disabled,
68            shrink_allow_list: None,
69        }
70    }
71
72    /// Sample counts an incoming trace and tells if it is a sample which has to be kept
73    pub fn sample(&mut self, now: SystemTime, trace: &mut Trace, root_idx: usize) -> bool {
74        // logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/sampler/scoresampler.go#L71
75        if self.disabled {
76            return false;
77        }
78
79        let spans_len = trace.spans().len();
80        if spans_len == 0 || root_idx >= spans_len {
81            return false;
82        }
83
84        // Compute signature before mutably borrowing the root span.
85        let signature = compute_signature_with_root_and_env(trace, root_idx);
86        let signature = self.shrink(signature);
87
88        // Update sampler state by counting this trace
89        let weight = {
90            let spans = trace.spans();
91            let root = &spans[root_idx];
92            weight_root(root)
93        };
94        self.sampler.count_weighted_sig(now, &signature, weight);
95
96        // Get the sampling rate for this signature
97        let rate = self.sampler.get_signature_sample_rate(&signature);
98
99        // Apply the sampling decision
100        let root = &mut trace.spans_mut()[root_idx];
101        self.apply_sample_rate(root, rate)
102    }
103
104    /// Apply the sampling rate to determine if the trace should be kept.
105    fn apply_sample_rate(&self, root: &mut Span, rate: f64) -> bool {
106        let initial_rate = get_global_rate(root);
107        let new_rate = initial_rate * rate;
108        let trace_id = root.trace_id();
109        let sampled = sample_by_rate(trace_id, new_rate);
110
111        if sampled {
112            self.set_sampling_rate_metric(root, rate);
113        }
114
115        sampled
116    }
117
118    /// Shrink limits the number of signatures stored in the sampler.
119    /// After a cardinality above shrinkCardinality/2 is reached
120    /// signatures are spread uniformly on a fixed set of values.
121    /// This ensures that ScoreSamplers are memory capped.
122    /// When the shrink is triggered, previously active signatures
123    /// stay unaffected. New signatures may share the same TPS computation.
124    fn shrink(&mut self, sig: Signature) -> Signature {
125        // logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/sampler/scoresampler.go#L151
126        if self.sampler.size() < (SHRINK_CARDINALITY / 2) as i64 {
127            self.shrink_allow_list = None;
128            return sig;
129        }
130
131        if self.shrink_allow_list.is_none() {
132            let (rates, _) = self.sampler.get_all_signature_sample_rates();
133            self.shrink_allow_list = Some(rates);
134        }
135
136        if let Some(ref map) = self.shrink_allow_list {
137            if map.contains_key(&sig) {
138                return sig;
139            }
140        }
141
142        // Map to a limited set of signatures to bound cardinality and force
143        // new signatures to share the same bucket and TPS computation.
144        Signature(sig.0 % (SHRINK_CARDINALITY as u64 / 2))
145    }
146
147    /// Set the sampling rate metric on a span.
148    pub fn set_sampling_rate_metric(&self, span: &mut Span, rate: f64) {
149        span.metrics_mut()
150            .insert(MetaString::from(self.sampling_rate_key), rate);
151    }
152}

169/// Calculate the weight from the span's global rate and presampler rate.
170fn weight_root(span: &Span) -> f32 {
171    let client_rate = span
172        .metrics()
173        .get(KEY_SAMPLING_RATE_GLOBAL)
174        .copied()
175        .filter(|&r| r > 0.0 && r <= 1.0)
176        .unwrap_or(1.0);
177
178    let pre_sampler_rate = span
179        .metrics()
180        .get(KEY_SAMPLING_RATE_PRE_SAMPLER)
181        .copied()
182        .filter(|&r| r > 0.0 && r <= 1.0)
183        .unwrap_or(1.0);
184
185    (1.0 / (pre_sampler_rate * client_rate)) as f32
186}

188/// Get the cumulative sample rate of the trace to which this span belongs.
189fn get_global_rate(span: &Span) -> f64 {
190    span.metrics().get(KEY_SAMPLING_RATE_GLOBAL).copied().unwrap_or(1.0)
