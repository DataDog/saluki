40/// Credit card obfuscator with configuration.
41pub struct CreditCardObfuscator {
42    luhn: bool,
43    keep_values: FastHashSet<MetaString>,
44}

46impl CreditCardObfuscator {
47    /// Creates a new credit card obfuscator from configuration.
48    pub fn new(config: &CreditCardObfuscationConfig) -> Self {
49        // Only store user-provided keep_values.
50        // Static allowlist is checked separately via `ALLOWLISTED_TAGS.contains()`.
51        let keep_values: FastHashSet<MetaString> = config
52            .keep_values()
53            .iter()
54            .map(|s| MetaString::from(s.as_str()))
55            .collect();
56
57        Self {
58            luhn: config.luhn(),
59            keep_values,
60        }
61    }
62
63    /// Obfuscates credit card numbers in a value for the given key.
64    /// Returns `Some(replacement)` if a credit card number is detected, `None` if unchanged.
65    pub fn obfuscate_credit_card_number(&self, key: &str, val: &str) -> Option<MetaString> {
66        if key.starts_with('_') {
67            return None;
68        }
69
70        // Check static allowlist first.
71        if ALLOWLISTED_TAGS.contains(&key) {
72            return None;
73        }
74
75        // Check user-provided `keep_values`.
76        if self.keep_values.contains(key) {
77            return None;
78        }
79
80        if self.is_card_number(val) {
81            return Some("?".into());
82        }
83
84        None
85    }
86
87    fn is_card_number(&self, b: &str) -> bool {
88        if b.is_empty() || b.len() < 12 {
89            return false;
90        }
91
92        let first_char = b.chars().next().unwrap();
93        if first_char != ' ' && first_char != '-' && !first_char.is_ascii_digit() {
94            return false;
95        }
96
97        let mut prefix = 0; // Holds up to first 6 digits as numeric value
98        let mut count = 0; // Counts digits encountered
99        let mut found_prefix = false;
100        let mut digits = Vec::new(); // For Luhn validation
101
102        for ch in b.chars() {
103            match ch {
104                ' ' | '-' => continue,
105                '0'..='9' => {
106                    count += 1;
107                    let digit = ch as u8 - b'0';
108
109                    if self.luhn {
110                        digits.push(digit);
111                    }
112
113                    if !found_prefix {
114                        prefix = prefix * 10 + digit as i32;
115                        let (maybe, yes) = valid_card_prefix(prefix);
116                        if yes {
117                            found_prefix = true;
118                        } else if !maybe {
119                            return false;
120                        }
121                    }
122                }
123                _ => return false,
124            }
125        }
126
127        if !found_prefix {
128            return false;
129        }
130
131        if !(12..=19).contains(&count) {
132            return false;
133        }
134
135        if self.luhn {
136            return luhn_valid(&digits);
137        }
138
139        true
140    }
141}
142
143fn luhn_valid(digits: &[u8]) -> bool {
144    let mut sum = 0;
145    let mut alt = false;
146    let n = digits.len();
147
148    for i in (0..n).rev() {
149        let mut digit = digits[i] as i32;
150        if alt {
151            digit *= 2;
152            if digit > 9 {
153                digit = (digit % 10) + 1;
154            }
155        }
156        alt = !alt;
157        sum += digit;
158    }
159
160    sum % 10 == 0
161}

163fn valid_card_prefix(n: i32) -> (bool, bool) {
164    if n > 699999 {
165        return (false, false);
166    }
167
168    if n < 10 {
169        return match n {
170            1 | 4 => (false, true),
171            2 | 3 | 5 | 6 => (true, false),
172            _ => (false, false),
173        };
174    }
175
176    if n < 100 {
177        if (34..=39).contains(&n) || (51..=55).contains(&n) || n == 62 || n == 65 {
178            return (false, true);
179        }
180        if n == 30
181            || n == 63
182            || n == 64
183            || n == 50
184            || n == 60
185            || (22..=27).contains(&n)
186            || (56..=58).contains(&n)
187            || (60..=69).contains(&n)
188        {
189            return (true, false);
190        }
191        return (false, false);
192    }
193
194    if n < 1000 {
195        if (300..=305).contains(&n) || (644..=649).contains(&n) || n == 309 || n == 636 {
196            return (false, true);
197        }
198        if (352..=358).contains(&n)
199            || n == 501
200            || n == 601
201            || (222..=272).contains(&n)
202            || (500..=509).contains(&n)
203            || (560..=589).contains(&n)
204            || (600..=699).contains(&n)
205        {
206            return (true, false);
207        }
208        return (false, false);
209    }
210
211    if n < 10000 {
212        if (3528..=3589).contains(&n) || n == 5019 || n == 6011 {
213            return (false, true);
214        }
215        if (2221..=2720).contains(&n)
216            || (5000..=5099).contains(&n)
217            || (5600..=5899).contains(&n)
218            || (6000..=6999).contains(&n)
219        {
220            return (true, false);
221        }
222        return (false, false);
223    }
224
225    if n < 100000 {
226        if (22210..=27209).contains(&n)
227            || (50000..=50999).contains(&n)
228            || (56000..=58999).contains(&n)
229            || (60000..=69999).contains(&n)
230        {
231            return (true, false);
232        }
233        return (false, false);
234    }
235
236    if n < 1000000
237        && ((222100..=272099).contains(&n)
238            || (500000..=509999).contains(&n)
239            || (560000..=589999).contains(&n)
240            || (600000..=699999).contains(&n))
241    {
242        return (false, true);
243    }
244
245    (false, false)
246}

548        let config = CreditCardObfuscationConfig {
549            enabled: true,
550            luhn: false,
551            keep_values: vec!["skip_me".to_string()],
552        };
553        let obfuscator = CreditCardObfuscator::new(&config);

555        let possible_card = "378282246310005";
556
557        // skip_me is in keep_values, so no obfuscation (returns None)
558        assert_eq!(obfuscator.obfuscate_credit_card_number("skip_me", possible_card), None);
559
560        // obfuscate_me is not in keep_values, so obfuscation happens (returns Some("?"))
561        assert_eq!(
562            obfuscator.obfuscate_credit_card_number("obfuscate_me", possible_card),
563            Some("?".into())
564        );
