86#[async_trait]
87impl TransformBuilder for DogstatsDPrefixFilterConfiguration {
88    fn input_event_type(&self) -> EventType {
89        EventType::Metric
90    }
91
92    fn outputs(&self) -> &[OutputDefinition] {
93        static OUTPUTS: &[OutputDefinition] = &[OutputDefinition::default_output(EventType::Metric)];
94        OUTPUTS
95    }
96
97    async fn build(&self, _: ComponentContext) -> Result<Box<dyn Transform + Send>, GenericError> {
98        // Ensure our metric prefix has a trailing period so that we don't have to check for, and possibly add it, when we're
99        // actually processing metrics.
100        let mut metric_prefix = self.metric_prefix.clone();
101        if !metric_prefix.is_empty() && !metric_prefix.ends_with(".") {
102            metric_prefix.push('.');
103        }
104
105        // Prioritize `metric_filterlist` over `statsd_metric_blocklist`.
106        let (blocklist_metrics, blocklist_match_prefix) = if !self.metric_filterlist.is_empty() {
107            (&self.metric_filterlist, self.metric_filterlist_match_prefix)
108        } else {
109            (&self.metric_blocklist, self.metric_blocklist_match_prefix)
110        };
111
112        Ok(Box::new(DogstatsDPrefixFilter {
113            metric_prefix,
114            metric_prefix_blocklist: self.metric_prefix_blocklist.clone(),
115            blocklist: Blocklist::new(blocklist_metrics, blocklist_match_prefix),
116            configuration: self.configuration.clone(),
117        }))
118    }
119}

130#[derive(Debug, Default)]
131struct Blocklist {
132    data: Vec<String>,
133    match_prefix: bool,
134}
135
136impl Blocklist {
137    fn new(data: &[String], match_prefix: bool) -> Self {
138        let mut data = data.to_owned();
139        data.sort();
140
141        // Only keep values with unique prefixes.
142        if match_prefix && !data.is_empty() {
143            let mut i = 0;
144            for j in 1..data.len() {
145                if !data[j].starts_with(&data[i]) {
146                    i += 1;
147                    data[i] = data[j].clone(); // Move item to next position
148                }
149            }
150            data.truncate(i + 1);
151        }
152
153        Blocklist { data, match_prefix }
154    }
155
156    fn contains(&self, name: &str) -> bool {
157        if self.data.is_empty() {
158            return false;
159        }
160
161        let i = self.data.binary_search_by(|k| k.as_str().cmp(name));
162
163        // Check for prefix match when match_prefix is true
164        if self.match_prefix {
165            let index = i.unwrap_or_else(|idx| idx);
166            if index > 0 && name.starts_with(&self.data[index - 1]) {
167                return true;
168            }
169        }
170
171        if let Ok(index) = i {
172            return name == self.data[index];
173        }
174
175        false
176    }
177}

179#[derive(Debug)]
180pub struct DogstatsDPrefixFilter {
181    metric_prefix: String,
182    metric_prefix_blocklist: Vec<String>,
183    blocklist: Blocklist,
184    configuration: Option<GenericConfiguration>,
185}
186
187impl DogstatsDPrefixFilter {
188    fn process_metric(&self, metric: &mut Metric) -> bool {
189        let metric_name = metric.context().name().deref();
190
191        if self.metric_prefix.is_empty() {
192            for s in &self.blocklist.data {
193                if s == metric_name || self.blocklist.match_prefix && metric_name.starts_with(s) {
194                    debug!("Metric {} excluded due to blocklist.", metric_name);
195                    return false;
196                }
197            }
198        } else {
199            // We don't want to prefix the metric if it has a prefix that is on our _prefix_ blocklist,
200            // which ensures we don't prefix metrics that are already prefixed.
201            let new_metric_name = if self.has_excluded_prefix(metric_name) {
202                metric.context().name().clone()
203            } else {
204                let mut prefixed_metric_name = self.metric_prefix.clone();
205                prefixed_metric_name.push_str(metric_name);
206                prefixed_metric_name.into()
207            };
208
209            if self.blocklist.contains(&new_metric_name) {
210                debug!("Metric {} excluded due to blocklist.", new_metric_name);
211                return false;
212            }
213
214            // Update metric with new name.
215            let new_context = metric.context().with_name(new_metric_name);
216            let existing_context = metric.context_mut();
217            *existing_context = new_context;
218        }
219
220        true
221    }
222
223    fn has_excluded_prefix(&self, metric_name: &str) -> bool {
224        !self.metric_prefix.is_empty()
225            && self
226                .metric_prefix_blocklist
227                .iter()
228                .any(|prefix| metric_name.starts_with(prefix))
229    }
230}

232#[async_trait]
233impl Transform for DogstatsDPrefixFilter {
234    async fn run(mut self: Box<Self>, mut context: TransformContext) -> Result<(), GenericError> {
235        let mut health = context.take_health_handle();
236        health.mark_ready();
237
238        let config = self.configuration.as_ref().unwrap();
239        let mut filterlist_watcher = config.watch_for_updates("metric_filterlist");
240        let mut blocklist_watcher = config.watch_for_updates("statsd_metric_blocklist");
241
242        debug!("DogStatsD Prefix Filter transform started.");
243
244        loop {
245            select! {
246                _ = health.live() => continue,
247                maybe_events = context.events().next() => match maybe_events {
248                    Some(mut events) => {
249                        events.remove_if(|event| match event.try_as_metric_mut() {
250                            // `process_metric` returns `true` if the metric should be kept, so we have to invert that
251                            // here to match the predicate structure, which will _remove_ the event if `true` is returned.
252                            Some(metric) => !self.process_metric(metric),
253                            None => true,
254                        });
255
256                        if let Err(e) = context.dispatcher().dispatch(events).await {
257                            error!(error = %e, "Failed to dispatch events.");
258                        }
259                    },
260                    None => break,
261                },
262                (_, maybe_new_metric_filterlist) = filterlist_watcher.changed::<Vec<String>>() => {
263                    if let Some(new_filterlist) = maybe_new_metric_filterlist {
264                        self.blocklist = Blocklist::new(&new_filterlist, self.blocklist.match_prefix);
265                        debug!(?new_filterlist, "Updated metric filterlist.");
266                    }
267                },
268                (_, maybe_new_blocklist) = blocklist_watcher.changed::<Vec<String>>() => {
269                    if let Some(new_blocklist) = maybe_new_blocklist {
270                        self.blocklist = Blocklist::new(&new_blocklist, self.blocklist.match_prefix);
271                        debug!(?new_blocklist, "Updated metric blocklist.");
272                    }
273                },
274            }
275        }
276
277        debug!("DogStatsD Prefix Filter transform stopped.");
278
279        Ok(())
280    }
