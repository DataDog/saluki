8#[derive(Clone)]
9struct MetricTypedGauge {
10    for_counter: Gauge,
11    for_gauge: Gauge,
12    for_rate: Gauge,
13    for_set: Gauge,
14    for_histogram: Gauge,
15    for_distribution: Gauge,
16}
17
18impl MetricTypedGauge {
19    pub fn new(builder: &MetricsBuilder, name: &'static str) -> Self {
20        Self {
21            for_counter: builder.register_debug_gauge_with_tags(name, ["metric_type:counter"]),
22            for_gauge: builder.register_debug_gauge_with_tags(name, ["metric_type:gauge"]),
23            for_rate: builder.register_debug_gauge_with_tags(name, ["metric_type:rate"]),
24            for_set: builder.register_debug_gauge_with_tags(name, ["metric_type:set"]),
25            for_histogram: builder.register_debug_gauge_with_tags(name, ["metric_type:histogram"]),
26            for_distribution: builder.register_debug_gauge_with_tags(name, ["metric_type:distribution"]),
27        }
28    }
29
30    #[cfg(test)]
31    pub fn noop() -> Self {
32        Self {
33            for_counter: Gauge::noop(),
34            for_gauge: Gauge::noop(),
35            for_rate: Gauge::noop(),
36            for_set: Gauge::noop(),
37            for_histogram: Gauge::noop(),
38            for_distribution: Gauge::noop(),
39        }
40    }
41
42    pub fn for_values(&self, values: &MetricValues) -> &Gauge {
43        match values {
44            MetricValues::Counter(_) => &self.for_counter,
45            MetricValues::Gauge(_) => &self.for_gauge,
46            MetricValues::Rate(_, _) => &self.for_rate,
47            MetricValues::Set(_) => &self.for_set,
48            MetricValues::Histogram(_) => &self.for_histogram,
49            MetricValues::Distribution(_) => &self.for_distribution,
50        }
51    }
52}

54#[derive(Clone)]
55pub struct Telemetry {
56    active_contexts: Gauge,
57    active_contexts_by_type: MetricTypedGauge,
58    active_contexts_bytes_by_type: MetricTypedGauge,
59    events_dropped: Counter,
60    flushes: Counter,
61    series_flushed: Counter,
62    sketches_flushed: Counter,
63    passthrough_metrics: Counter,
64    passthrough_flushes: Counter,
65    passthrough_batch_duration: Histogram,
66}
67
68impl Telemetry {
69    pub fn new(builder: &MetricsBuilder) -> Self {
70        Self {
71            active_contexts: builder.register_debug_gauge("aggregate_active_contexts"),
72            active_contexts_by_type: MetricTypedGauge::new(builder, "aggregate_active_contexts_by_type"),
73            active_contexts_bytes_by_type: MetricTypedGauge::new(builder, "aggregate_active_contexts_bytes_by_type"),
74            events_dropped: builder
75                .register_debug_counter_with_tags("component_events_dropped_total", ["intentional:true"]),
76            flushes: builder.register_debug_counter("aggregate_flushes_total"),
77            series_flushed: builder.register_debug_counter_with_tags("aggregate_flushed_total", ["data_type:series"]),
78            sketches_flushed: builder
79                .register_debug_counter_with_tags("aggregate_flushed_total", ["data_type:sketches"]),
80            passthrough_metrics: builder.register_debug_counter("aggregate_passthrough_metrics_total"),
81            passthrough_flushes: builder.register_debug_counter("aggregate_passthrough_flushes_total"),
82            passthrough_batch_duration: builder.register_debug_histogram("aggregate_passthrough_batch_duration_secs"),
83        }
84    }
85
86    #[cfg(test)]
87    pub fn noop() -> Self {
88        Self {
89            active_contexts: Gauge::noop(),
90            active_contexts_by_type: MetricTypedGauge::noop(),
91            active_contexts_bytes_by_type: MetricTypedGauge::noop(),
92            events_dropped: Counter::noop(),
93            flushes: Counter::noop(),
94            series_flushed: Counter::noop(),
95            sketches_flushed: Counter::noop(),
96            passthrough_metrics: Counter::noop(),
97            passthrough_flushes: Counter::noop(),
98            passthrough_batch_duration: Histogram::noop(),
99        }
100    }
101
102    pub fn increment_contexts(&self, context: &Context, values: &MetricValues) {
103        self.active_contexts.increment(1);
104        self.active_contexts_by_type.for_values(values).increment(1);
105        self.active_contexts_bytes_by_type
106            .for_values(values)
107            .increment(context.size_of() as f64);
108    }
109
110    pub fn decrement_contexts(&self, context: &Context, values: &MetricValues) {
111        self.active_contexts.decrement(1);
112        self.active_contexts_by_type.for_values(values).decrement(1);
113        self.active_contexts_bytes_by_type
114            .for_values(values)
115            .decrement(context.size_of() as f64);
116    }
117
118    pub fn increment_events_dropped(&self) {
119        self.events_dropped.increment(1);
120    }
121
122    pub fn increment_flushes(&self) {
123        self.flushes.increment(1);
124    }
125
126    pub fn increment_flushed(&self, values: &MetricValues) {
127        if values.is_serie() {
128            self.series_flushed.increment(1);
129        } else if values.is_sketch() {
130            self.sketches_flushed.increment(1);
131        }
132    }
133
134    pub fn increment_passthrough_metrics(&self) {
135        self.passthrough_metrics.increment(1);
136    }
137
138    pub fn increment_passthrough_flushes(&self) {
139        self.passthrough_flushes.increment(1);
140    }
141
142    pub fn record_passthrough_batch_duration(&self, duration: Duration) {
143        self.passthrough_batch_duration.record(duration.as_secs_f64());
144    }
