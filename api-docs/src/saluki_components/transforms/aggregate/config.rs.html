5/// A histogram statistic to calculate.
6#[derive(Clone)]
7pub enum HistogramStatistic {
8    /// Total count of values in the histogram.
9    Count,
10
11    /// Sum of all values in the histogram.
12    Sum,
13
14    /// Minimum value in the histogram.
15    Minimum,
16
17    /// Maximum value in the histogram.
18    Maximum,
19
20    /// Average value in the histogram.
21    Average,
22
23    /// Median value in the histogram.
24    Median,
25
26    /// Calculate the given percentile from the histogram.
27    Percentile {
28        /// Quantile value to calculate.
29        q: f64,
30
31        /// Suffix to append to the metric name, representing the percentile.
32        ///
33        /// For example, a percentile of 95% (quantile value of 0.95) would have a suffix of `95percentile`.
34        suffix: MetaString,
35    },
36}
37
38impl HistogramStatistic {
39    /// Returns the suffix to be used for metrics representing this statistic.
40    pub fn suffix(&self) -> &str {
41        match self {
42            HistogramStatistic::Count => "count",
43            HistogramStatistic::Sum => "sum",
44            HistogramStatistic::Minimum => "min",
45            HistogramStatistic::Maximum => "max",
46            HistogramStatistic::Average => "avg",
47            HistogramStatistic::Median => "median",
48            HistogramStatistic::Percentile { suffix, .. } => suffix,
49        }
50    }
51
52    /// Returns `true` if this statistics should be represented as a rate.
53    pub fn is_rate_statistic(&self) -> bool {
54        matches!(self, HistogramStatistic::Count)
55    }
56
57    /// Returns the value of this statistic from the given histogram summary.
58    pub fn value_from_histogram(&self, summary: &HistogramSummary<'_>) -> f64 {
59        match self {
60            HistogramStatistic::Count => summary.count() as f64,
61            HistogramStatistic::Sum => summary.sum(),
62            HistogramStatistic::Minimum => summary.min().unwrap_or(0.0),
63            HistogramStatistic::Maximum => summary.max().unwrap_or(0.0),
64            HistogramStatistic::Average => summary.avg(),
65            HistogramStatistic::Median => summary.median().unwrap_or(0.0),
66            HistogramStatistic::Percentile { q, .. } => summary.quantile(*q).unwrap_or(0.0),
67        }
68    }
69}

71#[derive(Deserialize)]
72#[serde(default)]
73struct RawHistogramConfiguration {
74    /// Aggregates to calculate over histograms.
75    ///
76    /// Available aggregates: `count`, `sum`, `min`, `max`, `average`, and `median`
77    ///
78    /// The metric name generated for each aggregate will be in the form of `<original metric name>.<aggregate>`.
79    histogram_aggregates: Vec<String>,
80
81    /// Percentiles to calculate over histograms.
82    ///
83    /// Percentiles are expressed in quantile form: 95% becomes 0.95, and so on. Any floating-point number between
84    /// 0.0 and 1.0 (inclusive) is allowed, but values that extend beyond two decimal places (i.e. 0.999) will be
85    /// truncated.
86    ///
87    /// The metric name generated for each percentile will be in the form of `<original metric
88    /// name>.<stripped_q>percentile`, where `stripped_q` represents the quantile multiplied by 100. For example, a
89    /// percentile of 0.95 would be represented as `95percentile`, while a percentile of 0.05 would be represented as
90    /// `5percentile`.
91    histogram_percentiles: Vec<String>,
92
93    /// Whether to copy histograms to distributions.
94    ///
95    /// Emits a copy of each histogram as a distribution, potentially with
96    /// a prefixed version of the histogram's name, based on the value of
97    /// `copy_histogram_to_distribution_prefix`.
98    ///
99    /// Defaults to `false`.
100    histogram_copy_to_distribution: bool,
101
102    /// Prefix to append to the name of distributions copied from histograms.
103    ///
104    /// Defaults to an empty string (no prefixing).
105    histogram_copy_to_distribution_prefix: String,
106}
107
108impl Default for RawHistogramConfiguration {
109    fn default() -> Self {
110        Self {
111            histogram_aggregates: vec!["max".into(), "median".into(), "avg".into(), "count".into()],
112            histogram_percentiles: vec!["0.95".into()],
113            histogram_copy_to_distribution: false,
114            histogram_copy_to_distribution_prefix: "".into(),
115        }
116    }
117}

174impl TryFrom<RawHistogramConfiguration> for HistogramConfiguration {
175    type Error = String;
176
177    fn try_from(raw: RawHistogramConfiguration) -> Result<Self, Self::Error> {
178        let mut statistics = Vec::new();
179
180        for aggregate in raw.histogram_aggregates {
181            match aggregate.as_str() {
182                "count" => statistics.push(HistogramStatistic::Count),
183                "sum" => statistics.push(HistogramStatistic::Sum),
184                "min" => statistics.push(HistogramStatistic::Minimum),
185                "max" => statistics.push(HistogramStatistic::Maximum),
186                "avg" => statistics.push(HistogramStatistic::Average),
187                "median" => statistics.push(HistogramStatistic::Median),
188                _ => return Err(format!("Unknown histogram aggregate: {}", aggregate)),
189            }
190        }
191
192        for faux_percentile in raw.histogram_percentiles {
193            let quantile = faux_percentile
194                .parse::<f64>()
195                .map_err(|_| format!("Invalid percentile: {}", faux_percentile))?;
196            if !(0.0..=1.0).contains(&quantile) {
197                return Err(format!("Percentile out of range: {}", faux_percentile));
198            }
199
200            // Truncate the quantile value and then generate the string representation for the metric name suffix.
201            //
202            // We do the weird multiplication and division when we generate the suffix because you can end up with
203            // rounding errors, such that you might get `28` when doing `(0.29 * 100) as u32`.
204            let quantile = (quantile * 100.0).trunc() / 100.0;
205            let suffix = format!("{}percentile", ((quantile * 1000.0) as u32 / 10)).into();
206            statistics.push(HistogramStatistic::Percentile { q: quantile, suffix });
207        }
208
209        Ok(Self {
210            statistics,
211            copy_to_distribution: raw.histogram_copy_to_distribution,
212            copy_to_distribution_prefix: raw.histogram_copy_to_distribution_prefix,
213        })
214    }
