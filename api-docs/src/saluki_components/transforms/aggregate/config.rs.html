5/// A histogram statistic to calculate.
6#[derive(Clone)]
7#[cfg_attr(test, derive(Debug, PartialEq, serde::Serialize))]
8pub enum HistogramStatistic {
9    /// Total count of values in the histogram.
10    Count,
11
12    /// Sum of all values in the histogram.
13    Sum,
14
15    /// Minimum value in the histogram.
16    Minimum,
17
18    /// Maximum value in the histogram.
19    Maximum,
20
21    /// Average value in the histogram.
22    Average,
23
24    /// Median value in the histogram.
25    Median,
26
27    /// Calculate the given percentile from the histogram.
28    Percentile {
29        /// Quantile value to calculate.
30        q: f64,
31
32        /// Suffix to append to the metric name, representing the percentile.
33        ///
34        /// For example, a percentile of 95% (quantile value of 0.95) would have a suffix of `95percentile`.
35        suffix: MetaString,
36    },
37}
38
39impl HistogramStatistic {
40    /// Returns the suffix to be used for metrics representing this statistic.
41    pub fn suffix(&self) -> &str {
42        match self {
43            HistogramStatistic::Count => "count",
44            HistogramStatistic::Sum => "sum",
45            HistogramStatistic::Minimum => "min",
46            HistogramStatistic::Maximum => "max",
47            HistogramStatistic::Average => "avg",
48            HistogramStatistic::Median => "median",
49            HistogramStatistic::Percentile { suffix, .. } => suffix,
50        }
51    }
52
53    /// Returns `true` if this statistics should be represented as a rate.
54    pub fn is_rate_statistic(&self) -> bool {
55        matches!(self, HistogramStatistic::Count)
56    }
57
58    /// Returns the value of this statistic from the given histogram summary.
59    pub fn value_from_histogram(&self, summary: &HistogramSummary<'_>) -> f64 {
60        match self {
61            HistogramStatistic::Count => summary.count() as f64,
62            HistogramStatistic::Sum => summary.sum(),
63            HistogramStatistic::Minimum => summary.min().unwrap_or(0.0),
64            HistogramStatistic::Maximum => summary.max().unwrap_or(0.0),
65            HistogramStatistic::Average => summary.avg(),
66            HistogramStatistic::Median => summary.median().unwrap_or(0.0),
67            HistogramStatistic::Percentile { q, .. } => summary.quantile(*q).unwrap_or(0.0),
68        }
69    }
70}

73#[serde(default)]
74struct RawHistogramConfiguration {
75    /// Aggregates to calculate over histograms.
76    ///
77    /// Available aggregates: `count`, `sum`, `min`, `max`, `average`, and `median`
78    ///
79    /// The metric name generated for each aggregate will be in the form of `<original metric name>.<aggregate>`.
80    histogram_aggregates: Vec<String>,
81
82    /// Percentiles to calculate over histograms.
83    ///
84    /// Percentiles are expressed in quantile form: 95% becomes 0.95, and so on. Any floating-point number between
85    /// 0.0 and 1.0 (inclusive) is allowed, but values that extend beyond two decimal places are rounded to the nearest
86    /// whole percentile to match the Datadog Agent.
87    ///
88    /// The metric name generated for each percentile will be in the form of `<original metric
89    /// name>.<stripped_q>percentile`, where `stripped_q` represents the quantile multiplied by 100. For example, a
90    /// percentile of 0.95 would be represented as `95percentile`, while a percentile of 0.05 would be represented as
91    /// `5percentile`.
92    histogram_percentiles: Vec<String>,
93
94    /// Whether to copy histograms to distributions.
95    ///
96    /// Emits a copy of each histogram as a distribution, potentially with
97    /// a prefixed version of the histogram's name, based on the value of
98    /// `copy_histogram_to_distribution_prefix`.
99    ///
100    /// Defaults to `false`.
101    histogram_copy_to_distribution: bool,
102
103    /// Prefix to append to the name of distributions copied from histograms.
104    ///
105    /// Defaults to an empty string (no prefixing).
106    histogram_copy_to_distribution_prefix: String,
107}
108
109impl Default for RawHistogramConfiguration {
110    fn default() -> Self {
111        Self {

114            histogram_copy_to_distribution: false,
115            histogram_copy_to_distribution_prefix: "".into(),
116        }
117    }

176impl TryFrom<RawHistogramConfiguration> for HistogramConfiguration {
177    type Error = String;
178
179    fn try_from(raw: RawHistogramConfiguration) -> Result<Self, Self::Error> {
180        let mut statistics = Vec::new();
181
182        for aggregate in raw.histogram_aggregates {
183            match aggregate.as_str() {
184                "count" => statistics.push(HistogramStatistic::Count),
185                "sum" => statistics.push(HistogramStatistic::Sum),
186                "min" => statistics.push(HistogramStatistic::Minimum),
187                "max" => statistics.push(HistogramStatistic::Maximum),
188                "avg" => statistics.push(HistogramStatistic::Average),
189                "median" => statistics.push(HistogramStatistic::Median),
190                _ => return Err(format!("Unknown histogram aggregate: {}", aggregate)),
191            }
192        }
193
194        for faux_percentile in raw.histogram_percentiles {
195            let quantile = faux_percentile
196                .parse::<f64>()
197                .map_err(|_| format!("Invalid percentile: {}", faux_percentile))?;
198            if !(0.0..=1.0).contains(&quantile) {
199                return Err(format!("Percentile out of range: {}", faux_percentile));
200            }
201
202            let percentile = (quantile * 100.0 + 0.5) as u32;
203            let quantile = f64::from(percentile) / 100.0;
204            let suffix = format!("{}percentile", percentile).into();
205            statistics.push(HistogramStatistic::Percentile { q: quantile, suffix });
206        }
207
208        Ok(Self {
209            statistics,
210            copy_to_distribution: raw.histogram_copy_to_distribution,
211            copy_to_distribution_prefix: raw.histogram_copy_to_distribution_prefix,
212        })
213    }
214}
