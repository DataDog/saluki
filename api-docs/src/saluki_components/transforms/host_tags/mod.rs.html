21/// Host Tags synchronous transform.
22///
23/// Temporarily adds host tags to metrics to compensate for backend delays when a new host comes online,
24/// preventing gaps in queryability until the backend starts adding these tags automatically.
25pub struct HostTagsConfiguration {
26    client: RemoteAgentClient,
27    host_tags_context_string_interner_bytes: ByteSize,
28    expected_tags_duration: u64,
29}
30
31const DEFAULT_EXPECTED_TAGS_DURATION: u64 = 0;
32const DEFAULT_HOST_TAGS_CONTEXT_STRING_INTERNER_BYTES: ByteSize = ByteSize::kib(64);
33
34impl HostTagsConfiguration {
35    /// Creates a new `HostTagsConfiguration` from the given configuration.
36    pub async fn from_configuration(config: &GenericConfiguration) -> Result<Self, GenericError> {
37        let client = RemoteAgentClient::from_configuration(config).await?;
38        let expected_tags_duration = config
39            .try_get_typed::<u64>("expected_tags_duration")?
40            .unwrap_or(DEFAULT_EXPECTED_TAGS_DURATION);
41        let host_tags_context_string_interner_bytes = config
42            .try_get_typed::<ByteSize>("host_tags_context_string_interner_bytes")?
43            .unwrap_or(DEFAULT_HOST_TAGS_CONTEXT_STRING_INTERNER_BYTES);
44
45        Ok(Self {
46            client,
47            host_tags_context_string_interner_bytes,
48            expected_tags_duration,
49        })
50    }
51}
52
53#[async_trait]
54impl SynchronousTransformBuilder for HostTagsConfiguration {
55    async fn build(&self, context: ComponentContext) -> Result<Box<dyn SynchronousTransform + Send>, GenericError> {
56        // Make an initial request of the host tags from the Datadog Agent.
57        //
58        // We only pay attention to the "system" tags, as the "google_cloud_platform" tags are not relevant here.
59        let host_tags_reply = self.client.get_host_tags().await?.into_inner();
60        let host_tags = host_tags_reply
61            .system
62            .into_iter()
63            .map(|s| Arc::from(s.as_str()))
64            .map(MetaString::from)
65            .map(Tag::from)
66            .collect::<SharedTagSet>();
67
68        let context_string_interner_size =
69            NonZeroUsize::new(self.host_tags_context_string_interner_bytes.as_u64() as usize)
70                .ok_or_else(|| generic_error!("host_tags_context_string_interner_bytes must be greater than 0"))
71                .unwrap();
72        let context_resolver =
73            ContextResolverBuilder::from_name(format!("{}/host_tags/primary", context.component_id()))
74                .expect("resolver name is not empty")
75                .with_interner_capacity_bytes(context_string_interner_size)
76                .with_idle_context_expiration(Duration::from_secs(30))
77                .build();
78        Ok(Box::new(HostTagsEnrichment {
79            start: Instant::now(),
80            context_resolver: Some(context_resolver),
81            expected_tags_duration: Duration::from_secs(self.expected_tags_duration),
82            host_tags: Some(host_tags),
83        }))
84    }
85}

87impl MemoryBounds for HostTagsConfiguration {
88    fn specify_bounds(&self, builder: &mut MemoryBoundsBuilder) {
89        builder
90            // Capture the size of the heap allocation when the component is built.
91            .minimum()
92            .with_single_value::<HostTagsEnrichment>("component struct")
93            // We also allocate the backing storage for the string interner up front, which is used by our context
94            // resolver.
95            .with_fixed_amount(
96                "string interner",
97                self.host_tags_context_string_interner_bytes.as_u64() as usize,
98            );
99    }

102pub struct HostTagsEnrichment {
103    start: Instant,
104    context_resolver: Option<ContextResolver>,
105    expected_tags_duration: Duration,
106    host_tags: Option<SharedTagSet>,
107}
108
109impl HostTagsEnrichment {
110    fn enrich_metric(&mut self, metric: &mut Metric) {
111        // Get our context resolver and host tags.
112        //
113        // If they're not available, then we skip adding host tags.
114        let (resolver, host_tags) = match (self.context_resolver.as_mut(), self.host_tags.as_ref()) {
115            (Some(resolver), Some(host_tags)) => (resolver, host_tags),
116            _ => return,
117        };
118
119        let tags = metric.context().tags().into_iter().chain(host_tags);
120        let origin_tags = metric.context().origin_tags().clone();
121
122        if let Some(context) = resolver.resolve_with_origin_tags(metric.context().name(), tags, origin_tags) {
123            *metric.context_mut() = context;
124        }
125    }
126}
127
128impl SynchronousTransform for HostTagsEnrichment {
129    fn transform_buffer(&mut self, event_buffer: &mut EventsBuffer) {
130        // Skip adding host tags if duration has elapsed.
131        if self.start.elapsed() >= self.expected_tags_duration {
132            self.context_resolver = None;
133            self.host_tags = None;
134            return;
135        }
136
137        for event in event_buffer {
138            if let Some(metric) = event.try_as_metric_mut() {
139                self.enrich_metric(metric);
140            }
141        }
142    }
143}

145#[cfg(test)]
146mod tests {
147    use std::time::{Duration, Instant};
148
149    use saluki_context::{Context, ContextResolverBuilder};
150    use saluki_core::data_model::event::metric::Metric;
151
152    use super::*;
153
154    #[test]
155    fn basic() {
156        let context_resolver = ContextResolverBuilder::for_tests().build();
157        let host_tags = SharedTagSet::from_iter(vec![Tag::from("hosttag1"), Tag::from("hosttag2")]);
158        let mut host_tags_enrichment = HostTagsEnrichment {
159            start: Instant::now(),
160            context_resolver: Some(context_resolver),
161            expected_tags_duration: Duration::from_secs(30),
162            host_tags: Some(host_tags.clone()),
163        };
164
165        // Enrich a metric, and ensure the host tags are added.
166        let mut metric1 = Metric::gauge(Context::from_static_parts("test", &[]), 1.0);
167        host_tags_enrichment.enrich_metric(&mut metric1);
168        assert_eq!(metric1.context().tags().len(), host_tags.len());
169        for tag in &host_tags {
170            assert!(metric1.context().tags().has_tag(tag));
171        }
172
173        // Simulate exceeding our configured enrichment duration by clearing the context resolver and host tags.
174        host_tags_enrichment.context_resolver = None;
175        host_tags_enrichment.host_tags = None;
176
177        // We should no longer enrich the metric with host tags.
178        let mut metric2 = Metric::gauge(Context::from_static_parts("test", &[]), 1.0);
179        host_tags_enrichment.enrich_metric(&mut metric2);
180        assert_eq!(metric2.context().tags().len(), 0);
181    }
