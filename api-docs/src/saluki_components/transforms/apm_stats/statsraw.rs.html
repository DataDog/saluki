13struct GroupedStats {
14    hits: f64,
15    top_level_hits: f64,
16    errors: f64,
17    duration: f64,
18    ok_distribution: DDSketch,
19    err_distribution: DDSketch,
20    peer_tags: Vec<MetaString>,
21}
22
23impl GroupedStats {
24    // Note: The code in the agent uses `ddsketch.LogCollapsingLowestDenseDDSketch(0.01, 2048)`
25    fn new() -> Self {
26        Self {
27            hits: 0.0,
28            top_level_hits: 0.0,
29            errors: 0.0,
30            duration: 0.0,
31            ok_distribution: DDSketch::with_relative_accuracy(0.01)
32                .expect("Relative accuracy should be valid (0 <= 0.01 <= 1)"),
33            err_distribution: DDSketch::with_relative_accuracy(0.01)
34                .expect("Relative accuracy should be valid (0 <= 0.01 <= 1)"),
35            peer_tags: Vec::new(),
36        }
37    }
38
39    fn export(self, bucket_agg_key: BucketsAggregationKey) -> ClientGroupedStats {
40        let ok_summary = convert_to_ddsketch_proto_bytes(&self.ok_distribution);
41        let error_summary = convert_to_ddsketch_proto_bytes(&self.err_distribution);
42
43        ClientGroupedStats::new(bucket_agg_key.service, bucket_agg_key.name, bucket_agg_key.resource)
44            .with_http_status_code(bucket_agg_key.status_code)
45            .with_span_type(bucket_agg_key.span_type)
46            .with_hits(round(self.hits))
47            .with_errors(round(self.errors))
48            .with_duration(round(self.duration))
49            .with_top_level_hits(round(self.top_level_hits))
50            .with_ok_summary(ok_summary)
51            .with_error_summary(error_summary)
52            .with_synthetics(bucket_agg_key.synthetics)
53            .with_span_kind(bucket_agg_key.span_kind)
54            .with_peer_tags(self.peer_tags)
55            .with_is_trace_root(bucket_agg_key.is_trace_root)
56            .with_grpc_status_code(bucket_agg_key.grpc_status_code)
57            .with_http_method(bucket_agg_key.http_method)
58            .with_http_endpoint(bucket_agg_key.http_endpoint)
59    }
60}

67pub struct RawBucket {
68    /// Timestamp of the bucket start.
69    start: u64,
70    /// Bucket duration in nanoseconds.
71    duration: u64,
72    data: FastHashMap<Aggregation, GroupedStats>,
73    /// Map of container ID to container tags.
74    pub(super) container_tags_by_id: FastHashMap<MetaString, SharedTagSet>,
75    /// Map of process tags hash to process tags.
76    pub(super) process_tags_by_hash: PrehashedHashMap<u64, MetaString>,
77}
78
79impl RawBucket {
80    pub fn new(start: u64, duration: u64) -> Self {
81        Self {
82            start,
83            duration,
84            data: FastHashMap::default(),
85            container_tags_by_id: FastHashMap::default(),
86            process_tags_by_hash: PrehashedHashMap::default(),
87        }
88    }
89
90    /// Store process tags for later export.
91    pub fn set_process_tags(&mut self, hash: u64, tags: MetaString) {
92        self.process_tags_by_hash.insert(hash, tags);
93    }
94
95    /// Store container tags for later export.
96    pub fn set_container_tags(&mut self, container_id: MetaString, tags: impl Into<SharedTagSet>) {
97        self.container_tags_by_id.insert(container_id, tags.into());
98    }
99
100    /// Get container tags by container ID.
101    #[allow(unused)]
102    pub fn get_container_tags(&self, container_id: &MetaString) -> Option<&SharedTagSet> {
103        self.container_tags_by_id.get(container_id)
104    }
105
106    /// Get process tags by hash.
107    #[allow(unused)]
108    pub fn get_process_tags(&self, hash: u64) -> Option<&MetaString> {
109        self.process_tags_by_hash.get(&hash)
110    }
111
112    pub fn handle_span(&mut self, span: &StatSpan, weight: f64, origin: &str, payload_key: PayloadAggregationKey) {
113        if payload_key.env.is_empty() {
114            error!("PayloadAggregationKey env should never be empty");
115            return;
116        }
117        let aggr = new_aggregation_from_span(span, origin, payload_key);
118        self.add(span, weight, aggr);
119    }
120
121    fn add(&mut self, span: &StatSpan, weight: f64, aggr: Aggregation) {
122        let gs = self.data.entry(aggr).or_insert_with(|| {
123            let mut gs = GroupedStats::new();
124            gs.peer_tags = span.matching_peer_tags.clone();
125            gs
126        });
127
128        if span.is_top_level {
129            gs.top_level_hits += weight;
130        }
131        gs.hits += weight;
132
133        if span.error != 0 {
134            gs.errors += weight;
135        }
136
137        gs.duration += (span.duration as f64) * weight;
138
139        let trunc_dur = ns_timestamp_to_float(span.duration);
140        if span.error != 0 {
141            gs.err_distribution.add(trunc_dur);
142        } else {
143            gs.ok_distribution.add(trunc_dur);
144        }
145    }
146
147    pub fn export(self) -> ExportedBucket {
148        let mut data = FastHashMap::default();
149
150        for (agg, gs) in self.data {
151            let (bucket_agg_key, payload_agg_key) = agg.into_parts();
152            let grouped_stats = gs.export(bucket_agg_key);
153
154            let bucket = data
155                .entry(payload_agg_key)
156                .or_insert_with(|| ClientStatsBucket::new(self.start, self.duration, Vec::new()));
157            bucket.stats_mut().push(grouped_stats);
158        }
159
160        ExportedBucket {
161            data,
162            container_tags_by_id: self.container_tags_by_id,
163            process_tags_by_hash: self.process_tags_by_hash,
164        }
165    }
166}

174pub(super) fn new_aggregation_from_span(
175    span: &StatSpan, origin: &str, payload_key: PayloadAggregationKey,
176) -> Aggregation {
177    let synthetics = origin.starts_with(TAG_SYNTHETICS);
178    let is_trace_root = if span.parent_id == 0 { Some(true) } else { Some(false) };
179
180    Aggregation {
181        payload_key,
182        bucket_key: BucketsAggregationKey {
183            service: span.service.clone(),
184            name: span.name.clone(),
185            resource: span.resource.clone(),
186            span_type: span.typ.clone(),
187            span_kind: span.span_kind.clone(),
188            status_code: span.status_code,
189            synthetics,
190            is_trace_root,
191            grpc_status_code: span.grpc_status_code.clone(),
192            peer_tags_hash: tags_fnv_hash(&span.matching_peer_tags),
193            http_method: span.http_method.clone(),
194            http_endpoint: span.http_endpoint.clone(),
195        },
196    }
197}
198// Convert a nanosec timestamp into a float nanosecond timestamp truncated to a fixed precision.
199fn ns_timestamp_to_float(ns: u64) -> f64 {
200    let f = ns as f64;
201    let bits = f.to_bits();
202    // IEEE-754
203    // the mask include 1 bit sign 11 bits exponent (0xfff)
204    // then we filter the mantissa to 10bits (0xff8) (9 bits as it has implicit value of 1)
205    // 10 bits precision (any value will be +/- 1/1024)
206    // https://en.wikipedia.org/wiki/Double-precision_floating-point_format
207    let truncated_bits = bits & 0xffff_f800_0000_0000;
208    f64::from_bits(truncated_bits)
209}

211fn round(f: f64) -> u64 {
212    let i = f as u64;
213    let frac = f - (i as f64);
214    if rand::rng().random::<f64>() < frac {
215        i + 1
216    } else {
217        i
218    }
219}

221fn convert_to_ddsketch_proto_bytes(sketch: &DDSketch) -> Vec<u8> {
222    let proto = sketch.to_proto();
223    match proto.write_to_bytes() {
224        Ok(bytes) => bytes,
225        Err(e) => {
226            error!(error = ?e, "Failed to serialize DDSketch to canonical Protocol Buffers representation.");
227            Vec::new()
228        }
229    }
230}

237    fn test_ns_timestamp_to_float() {
238        let ns: u64 = 1_000_000_000;
239        let truncated = ns_timestamp_to_float(ns);
240        let relative_error = ((truncated - ns as f64) / (ns as f64)).abs();
241        assert!(relative_error < 0.001, "relative error too large: {}", relative_error);
242
243        assert_eq!(ns_timestamp_to_float(0), 0.0);
244
245        let small: u64 = 1000;
246        let truncated_small = ns_timestamp_to_float(small);
247        assert!(truncated_small > 0.0);
248    }
