113pub struct SpanConcentrator {
114    /// Whether to compute stats based on span.kind
115    compute_stats_by_span_kind: bool,
116
117    /// Whether peer tags aggregation is enabled
118    peer_tags_aggregation: bool,
119
120    /// Configured peer tag keys for aggregation (base + custom)
121    peer_tag_keys: Vec<MetaString>,
122
123    /// Bucket duration in nanoseconds (10s)
124    bsize: u64,
125
126    /// Timestamp of oldest allowed bucket (prevents stats for already-flushed buckets)
127    oldest_ts: u64,
128
129    /// Number of buckets to buffer before flushing
130    buffer_len: u64,
131
132    /// Time-bucketed raw stats: bucket_timestamp -> RawBucket
133    buckets: FastHashMap<u64, RawBucket>,
134
135    /// Shared registry for aggregation keys.
136    /// Keys are stored here and referenced by hash in individual buckets.
137    key_registry: AggregationRegistry,
138}
139
140impl SpanConcentrator {
141    pub fn new(
142        compute_stats_by_span_kind: bool, peer_tags_aggregation: bool, custom_peer_tags: &[MetaString], now: u64,
143    ) -> Self {
144        let mut peer_tag_keys: Vec<MetaString> = BASE_PEER_TAGS.iter().map(|s| MetaString::from_static(s)).collect();
145        for tag in custom_peer_tags {
146            if !peer_tag_keys.iter().any(|t| t == tag) {
147                peer_tag_keys.push(tag.clone());
148            }
149        }
150
151        Self {
152            compute_stats_by_span_kind,
153            peer_tags_aggregation,
154            peer_tag_keys,
155            bsize: BUCKET_DURATION_NS,
156            oldest_ts: align_ts(now, BUCKET_DURATION_NS),
157            buffer_len: DEFAULT_BUFFER_LEN,
158            buckets: FastHashMap::default(),
159            key_registry: AggregationRegistry::new(),
160        }
161    }
162
163    pub fn new_stat_span_from_span(&self, span: &Span) -> Option<StatSpan> {
164        self.new_stat_span(span)
165    }
166
167    pub fn add_span(
168        &mut self, stat_span: &StatSpan, weight: f64, payload_key: &PayloadAggregationKey, infra_tags: &InfraTags,
169        origin: &str,
170    ) {
171        self.add_span_internal(stat_span, weight, payload_key, infra_tags, origin);
172    }
173
174    pub fn flush(&mut self, now: u64, force: bool) -> Vec<ClientStatsPayload> {
175        let mut m = FastHashMap::<PayloadAggregationKey, Vec<ClientStatsBucket>>::default();
176        let mut container_tags_by_id = FastHashMap::<MetaString, SharedTagSet>::default();
177        let mut process_tags_by_hash = FastHashMap::<u64, MetaString>::default();
178
179        let timestamps: Vec<u64> = self.buckets.keys().copied().collect();
180
181        for ts in timestamps {
182            if !force && ts + self.buffer_len * self.bsize > now {
183                continue;
184            }
185
186            if let Some(bucket) = self.buckets.remove(&ts) {
187                // Export bucket data, which also decrements reference counts for keys inline
188                let exported = bucket.export(&mut self.key_registry);
189
190                for (k, b) in exported.data {
191                    if let Some(ctags) = exported.container_tags_by_id.get(&k.container_id) {
192                        container_tags_by_id.insert(k.container_id.clone(), ctags.clone());
193                    }
194                    if let Some(ptags) = exported.process_tags_by_hash.get(&k.process_tags_hash) {
195                        process_tags_by_hash.insert(k.process_tags_hash, ptags.clone());
196                    }
197                    m.entry(k).or_default().push(b);
198                }
199            }
200        }
201
202        let new_oldest_ts = align_ts(now, self.bsize).saturating_sub((self.buffer_len - 1) * self.bsize);
203        if new_oldest_ts > self.oldest_ts {
204            self.oldest_ts = new_oldest_ts;
205        }
206
207        let mut sb: Vec<ClientStatsPayload> = Vec::with_capacity(m.len());
208        for (k, s) in m {
209            let p = ClientStatsPayload::new(k.hostname, k.env, k.version)
210                .with_stats(s)
211                .with_tags(container_tags_by_id.get(&k.container_id).cloned().unwrap_or_default())
212                .with_container_id(k.container_id)
213                .with_git_commit_sha(k.git_commit_sha)
214                .with_image_tag(k.image_tag)
215                .with_lang(k.lang)
216                .with_process_tags(
217                    process_tags_by_hash
218                        .get(&k.process_tags_hash)
219                        .cloned()
220                        .unwrap_or_default(),
221                )
222                .with_process_tags_hash(k.process_tags_hash);
223            sb.push(p);
224        }
225        sb
226    }

228    fn is_span_eligible(&self, span: &Span) -> bool {
229        if let Some(&val) = span.metrics().get(METRIC_TOP_LEVEL) {
230            if val == 1.0 {
231                return true;
232            }
233        }
234        if let Some(&val) = span.metrics().get(METRIC_MEASURED) {
235            if val == 1.0 {
236                return true;
237            }
238        }
239        if self.compute_stats_by_span_kind {
240            if let Some(kind) = span.meta().get(TAG_SPAN_KIND) {
241                return compute_stats_for_span_kind(kind);
242            }
243        }
244        false
245    }
246
247    fn new_stat_span(&self, span: &Span) -> Option<StatSpan> {
248        if !self.is_span_eligible(span) {
249            return None;
250        }
251
252        if is_partial_snapshot(span) {
253            return None;
254        }
255
256        let span_kind = span.meta().get(TAG_SPAN_KIND).cloned().unwrap_or_default();
257        let status_code = get_status_code(span.meta(), span.metrics());
258        let grpc_status_code = get_grpc_status_code(span.meta(), span.metrics()).to_metastring();
259        let is_top_level = span.metrics().get(METRIC_TOP_LEVEL).map(|&v| v == 1.0).unwrap_or(false);
260        let matching_peer_tags = self.matching_peer_tags(span, &span_kind);
261
262        Some(StatSpan {
263            service: MetaString::from(span.service()),
264            resource: MetaString::from(span.resource()),
265            name: MetaString::from(span.name()),
266            typ: MetaString::from(span.span_type()),
267            span_kind,
268            status_code,
269            error: span.error(),
270            parent_id: span.parent_id(),
271            start: span.start(),
272            duration: span.duration(),
273            is_top_level,
274            matching_peer_tags,
275            grpc_status_code,
276            http_method: MetaString::default(),
277            http_endpoint: MetaString::default(),
278        })
279    }
280
281    fn matching_peer_tags(&self, span: &Span, span_kind: &str) -> Vec<MetaString> {
282        let mut peer_tags = Vec::new();
283
284        let keys_to_check = self.peer_tag_keys_to_aggregate_for_span(span_kind, span.meta().get(TAG_BASE_SERVICE));
285        for key in keys_to_check {
286            if let Some(value) = span.meta().get(key) {
287                if !value.is_empty() {
288                    peer_tags.push(MetaString::from(format!("{}:{}", key, value)));
289                }
290            }
291        }
292
293        peer_tags
294    }
295
296    fn peer_tag_keys_to_aggregate_for_span(&self, span_kind: &str, base_service: Option<&MetaString>) -> &[MetaString] {
297        static EMPTY_PEER_TAGS: &[MetaString] = &[];
298        static BASE_SERVICE_PEER_TAGS: &[MetaString] = &[MetaString::from_static(TAG_BASE_SERVICE)];
299
300        if !self.peer_tags_aggregation || self.peer_tag_keys.is_empty() {
301            return EMPTY_PEER_TAGS;
302        }
303
304        if (span_kind.is_empty() || span_kind.eq_ignore_ascii_case("internal"))
305            && base_service.map(|s| !s.is_empty()).unwrap_or(false)
306        {
307            return BASE_SERVICE_PEER_TAGS;
308        }
309
310        if span_kind.eq_ignore_ascii_case("client")
311            || span_kind.eq_ignore_ascii_case("producer")
312            || span_kind.eq_ignore_ascii_case("consumer")
313        {
314            return &self.peer_tag_keys;
315        }
316
317        EMPTY_PEER_TAGS
318    }
319
320    fn add_span_internal(
321        &mut self, s: &StatSpan, weight: f64, agg_key: &PayloadAggregationKey, tags: &InfraTags, origin: &str,
322    ) {
323        let end = s.start + s.duration;
324        let btime = (end - end % self.bsize).max(self.oldest_ts);
325
326        let b = self
327            .buckets
328            .entry(btime)
329            .or_insert_with(|| RawBucket::new(btime, self.bsize));
330
331        if tags.process_tags_hash != 0 && !tags.process_tags.is_empty() {
332            b.set_process_tags(tags.process_tags_hash, tags.process_tags.clone());
333        }
334        if !tags.container_id.is_empty() && !tags.container_tags.is_empty() {
335            b.set_container_tags(tags.container_id.clone(), tags.container_tags.clone());
336        }
337
338        b.handle_span(s, weight, origin, agg_key.clone(), &mut self.key_registry);
339    }
340}

342/// Align timestamp to bucket boundary.
343#[inline]
344fn align_ts(ts: u64, bsize: u64) -> u64 {
345    ts - ts % bsize
346}
347
348pub const fn compute_stats_for_span_kind(kind: &str) -> bool {
349    kind.eq_ignore_ascii_case("server")
350        || kind.eq_ignore_ascii_case("client")
351        || kind.eq_ignore_ascii_case("producer")
352        || kind.eq_ignore_ascii_case("consumer")
353}
354
355fn is_partial_snapshot(span: &Span) -> bool {
356    match span.metrics().get(METRIC_PARTIAL_VERSION) {
357        Some(&v) => v >= 0.0,
358        None => false,
359    }
