100/// Configuration for the Datadog Traces encoder.
101///
102/// This encoder converts trace events into Datadog's TracerPayload protobuf format and sends them
103/// to the Datadog traces intake endpoint (`/api/v0.2/traces`). It handles batching, compression,
104/// and enrichment with metadata such as hostname, environment, and container tags.
105#[derive(Deserialize)]
106pub struct DatadogTraceConfiguration {
107    #[serde(
108        rename = "serializer_compressor_kind",  // renames the field in the user_configuration from "serializer_compressor_kind" to "compressor_kind".
109        default = "default_serializer_compressor_kind"
110    )]
111    compressor_kind: String,
112
113    #[serde(
114        rename = "serializer_zstd_compressor_level",
115        default = "default_zstd_compressor_level"
116    )]
117    zstd_compressor_level: i32,
118
119    /// Flush timeout for pending requests, in seconds.
120    ///
121    /// When the encoder has written traces to the in-flight request payload, but it has not yet reached the
122    /// payload size limits that would force the payload to be flushed, the encoder will wait for a period of time
123    /// before flushing the in-flight request payload.
124    ///
125    /// Defaults to 2 seconds.
126    #[serde(default = "default_flush_timeout_secs")]
127    flush_timeout_secs: u64,
128
129    #[serde(skip)]
130    default_hostname: Option<String>,
131
132    #[serde(skip)]
133    version: String,
134
135    #[serde(default = "default_env")]
136    env: String,
137
138    #[serde(default = "default_ignore_missing_datadog_fields")]
139    ignore_missing_datadog_fields: bool,
140}

178        let metrics_builder = MetricsBuilder::from_component_context(&context);
179        let telemetry = ComponentTelemetry::from_builder(&metrics_builder);
180        let compression_scheme = CompressionScheme::new(&self.compressor_kind, self.zstd_compressor_level);

200        let flush_timeout = match self.flush_timeout_secs {
201            // Give ourselves a minimum flush timeout of 10ms to allow for minimal batching
202            0 => std::time::Duration::from_millis(10),
203            secs => std::time::Duration::from_secs(secs),
204        };
205
206        Ok(Box::new(DatadogTrace {
207            trace_rb,
208            telemetry,
209            flush_timeout,

235// Encodes Trace events to TracerPayloads.
236#[async_trait]
237impl Encoder for DatadogTrace {
238    async fn run(mut self: Box<Self>, mut context: EncoderContext) -> Result<(), GenericError> {
239        let Self {
240            trace_rb,
241            telemetry,
242            flush_timeout,
243        } = *self;
244
245        let mut health = context.take_health_handle();
246
247        // The encoder runs two async loops, the main encoder loop and the request builder loop,
248        // this channel is used to send events from the main encoder loop to the request builder loop safely.
249        let (events_tx, events_rx) = mpsc::channel(8);
250        // adds a channel to send payloads to the dispatcher and a channel to receive them.
251        let (payloads_tx, mut payloads_rx) = mpsc::channel(8);
252        let request_builder_fut = run_request_builder(trace_rb, telemetry, events_rx, payloads_tx, flush_timeout);
253        // Spawn the request builder task on the global thread pool, this task is responsible for encoding traces and flushing requests.
254        let request_builder_handle = context
255            .topology_context()
256            .global_thread_pool() // Use the shared Tokio runtime thread pool.
257            .spawn_traced_named("dd-traces-request-builder", request_builder_fut);
258
259        health.mark_ready();
260        debug!("Datadog Trace encoder started.");
261
262        loop {
263            select! {
264                biased; // makes the branches of the select statement be evaluated in order.
265
266                _ = health.live() => continue,
267                maybe_payload = payloads_rx.recv() => match maybe_payload {
268                    Some(payload) => {
269                        // Dispatch an HTTP payload to the dispatcher.
270                        if let Err(e) = context.dispatcher().dispatch(payload).await {
271                            error!("Failed to dispatch payload: {}", e);
272                        }
273                    }
274                    None => break,
275                },
276                maybe_event_buffer = context.events().next() => match maybe_event_buffer {
277                    Some(event_buffer) => events_tx.send(event_buffer).await
278                        .error_context("Failed to send event buffer to request builder task.")?,
279                    None => break,
280                },
281            }
282        }
283
284        // Drop the events sender, which signals the request builder task to stop.
285        drop(events_tx);
286
287        // Continue draining the payloads receiver until it is closed.
288        while let Some(payload) = payloads_rx.recv().await {
289            if let Err(e) = context.dispatcher().dispatch(payload).await {
290                error!("Failed to dispatch payload: {}", e);
291            }
292        }
293
294        // Request build task should now be stopped.
295        match request_builder_handle.await {
296            Ok(Ok(())) => debug!("Request builder task stopped."),
297            Ok(Err(e)) => error!(error = %e, "Request builder task failed."),
298            Err(e) => error!(error = %e, "Request builder task panicked."),
299        }
300
301        debug!("Datadog Trace encoder stopped.");
302
303        Ok(())
304    }
305}

307async fn run_request_builder(
308    mut trace_request_builder: RequestBuilder<TraceEndpointEncoder>, telemetry: ComponentTelemetry,
309    mut events_rx: Receiver<EventsBuffer>, payloads_tx: Sender<PayloadsBuffer>, flush_timeout: std::time::Duration,
310) -> Result<(), GenericError> {
311    let mut pending_flush = false;
312    let pending_flush_timeout = sleep(flush_timeout);
313    tokio::pin!(pending_flush_timeout);
314
315    loop {
316        select! {
317            Some(event_buffer) = events_rx.recv() => {
318                for event in event_buffer {
319                    let trace = match event {
320                        Event::Trace(trace) => trace,
321                        _ => continue,
322                    };
323
324                    // Encode the trace. If we get it back, that means the current request is full, and we need to
325                    // flush it before we can try to encode the trace again.
326                    let trace_to_retry = match trace_request_builder.encode(trace).await {
327                        Ok(None) => continue,
328                        Ok(Some(trace)) => trace,
329                        Err(e) => {
330                            error!(error = %e, "Failed to encode trace.");
331                            telemetry.events_dropped_encoder().increment(1);
332                            continue;
333                        }
334                    };
335
336                    let maybe_requests = trace_request_builder.flush().await;
337                    if maybe_requests.is_empty() {
338                        panic!("builder told us to flush, but gave us nothing");
339                    }
340
341                    for maybe_request in maybe_requests {
342                        match maybe_request {
343                            Ok((events, request)) => {
344                                let payload_meta = PayloadMetadata::from_event_count(events);
345                                let http_payload = HttpPayload::new(payload_meta, request);
346                                let payload = Payload::Http(http_payload);
347
348                                payloads_tx.send(payload).await
349                                    .map_err(|_| generic_error!("Failed to send payload to encoder."))?;
350                            },
351                            Err(e) => if e.is_recoverable() {
352                                // If the error is recoverable, we'll hold on to the trace to retry it later.
353                                continue;
354                            } else {
355                                return Err(GenericError::from(e).context("Failed to flush request."));
356                            }
357                        }
358                    }
359
360                    // Now try to encode the trace again.
361                    if let Err(e) = trace_request_builder.encode(trace_to_retry).await {
362                        error!(error = %e, "Failed to encode trace.");
363                        telemetry.events_dropped_encoder().increment(1);
364                    }
365                }
366
367                debug!("Processed event buffer.");
368
369                // If we're not already pending a flush, we'll start the countdown.
370                if !pending_flush {
371                    pending_flush_timeout.as_mut().reset(tokio::time::Instant::now() + flush_timeout);
372                    pending_flush = true;
373                }
374            },
375            _ = &mut pending_flush_timeout, if pending_flush => {
376                debug!("Flushing pending request(s).");
377
378                pending_flush = false;
379
380                // Once we've encoded and written all traces, we flush the request builders to generate a request with
381                // anything left over. Again, we'll enqueue those requests to be sent immediately.
382                let maybe_trace_requests = trace_request_builder.flush().await;
383                for maybe_request in maybe_trace_requests {
384                    match maybe_request {
385                        Ok((events, request)) => {
386                            let payload_meta = PayloadMetadata::from_event_count(events);
387                            let http_payload = HttpPayload::new(payload_meta, request);
388                            let payload = Payload::Http(http_payload);
389
390                            payloads_tx.send(payload).await
391                                .map_err(|_| generic_error!("Failed to send payload to encoder."))?;
392                        },
393                        Err(e) => if e.is_recoverable() {
394                            continue;
395                        } else {
396                            return Err(GenericError::from(e).context("Failed to flush request."));
397                        }
398                    }
399                }
400
401                debug!("All flushed requests sent to I/O task. Waiting for next event buffer...");
402            },
403
404            // Event buffers channel has been closed, and we have no pending flushing, so we're all done.
405            else => break,
406        }
407    }
408
409    Ok(())
410}

412#[derive(Debug)]
413struct TraceEndpointEncoder {
414    tracer_payload_scratch: Vec<u8>,
415    chunk_scratch: Vec<u8>,
416    tags_scratch: Vec<u8>,
417    // TODO: do we need additional tags or tag deplicator?
418    default_hostname: MetaString,
419    ignore_missing_datadog_fields: bool,
420    agent_hostname: String,
421    version: String,
422    env: String,
423}
424
425impl TraceEndpointEncoder {
426    fn new(default_hostname: MetaString, ignore_missing_datadog_fields: bool, version: String, env: String) -> Self {
427        Self {
428            tracer_payload_scratch: Vec::new(),
429            chunk_scratch: Vec::new(),
430            tags_scratch: Vec::new(),
431            agent_hostname: default_hostname.as_ref().to_string(),
432            default_hostname,
433            ignore_missing_datadog_fields,
434            version,
435            env,
436        }
437    }
438
439    fn encode_tracer_payload(&mut self, trace: &Trace, output_buffer: &mut Vec<u8>) -> Result<(), protobuf::Error> {
440        let resource_tags = trace.resource_tags();
441        let first_span = trace.spans().first();
442        let source = tags_to_source(resource_tags);
443
444        // Build AgentPayload (outer message) writing to output_buffer
445        let mut agent_payload_stream = CodedOutputStream::vec(output_buffer);
446
447        // Write AgentPayload fields defined in agent_payload.proto
448        agent_payload_stream.write_string(AGENT_PAYLOAD_HOSTNAME_FIELD_NUMBER, &self.agent_hostname)?;
449        agent_payload_stream.write_string(AGENT_PAYLOAD_ENV_FIELD_NUMBER, &self.env)?;
450
451        agent_payload_stream.write_string(AGENT_PAYLOAD_AGENT_VERSION_FIELD_NUMBER, &self.version)?;
452        // TODO: Remove hardcoded values for targetTPS and errorTPS when we have sampling.
453        agent_payload_stream.write_double(AGENT_PAYLOAD_TARGET_TPS_FIELD_NUMBER, 10.0)?;
454        agent_payload_stream.write_double(AGENT_PAYLOAD_ERROR_TPS_FIELD_NUMBER, 10.0)?;
455
456        // Build TracerPayload (nested message) in scratch buffer
457        self.tracer_payload_scratch.clear();
458        let mut tracer_payload_stream = CodedOutputStream::vec(&mut self.tracer_payload_scratch);
459
460        // Write TracerPayload fields
461        if let Some(container_id) = resolve_container_id(resource_tags, first_span) {
462            tracer_payload_stream.write_string(TRACER_PAYLOAD_CONTAINER_ID_FIELD_NUMBER, container_id)?;
463        }
464
465        if let Some(lang) = get_resource_tag_value(resource_tags, "telemetry.sdk.language") {
466            tracer_payload_stream.write_string(TRACER_PAYLOAD_LANGUAGE_NAME_FIELD_NUMBER, lang)?;
467        }
468
469        if let Some(sdk_version) = get_resource_tag_value(resource_tags, "telemetry.sdk.version") {
470            let tracer_version = format!("otlp-{}", sdk_version);
471            tracer_payload_stream.write_string(TRACER_PAYLOAD_TRACER_VERSION_FIELD_NUMBER, &tracer_version)?;
472        }
473
474        self.chunk_scratch.clear();
475        write_message_field(
476            &mut tracer_payload_stream,
477            TRACER_PAYLOAD_CHUNKS_FIELD_NUMBER,
478            &build_trace_chunk(trace),
479            &mut self.chunk_scratch,
480        )?;
481
482        self.tags_scratch.clear();
483        if let Some(tags) = resolve_container_tags(resource_tags, source.as_ref(), self.ignore_missing_datadog_fields) {
484            write_map_entry_string_string(
485                &mut tracer_payload_stream,
486                TRACER_PAYLOAD_TAGS_FIELD_NUMBER,
487                CONTAINER_TAGS_META_KEY,
488                tags.as_ref(),
489                &mut self.tags_scratch,
490            )?;
491        }
492
493        if let Some(env) = resolve_env(resource_tags, self.ignore_missing_datadog_fields) {
494            tracer_payload_stream.write_string(TRACER_PAYLOAD_ENV_FIELD_NUMBER, env)?;
495        }
496
497        if let Some(hostname) = resolve_hostname(
498            resource_tags,
499            source.as_ref(),
500            Some(self.default_hostname.as_ref()),
501            self.ignore_missing_datadog_fields,
502        ) {
503            tracer_payload_stream.write_string(TRACER_PAYLOAD_HOSTNAME_FIELD_NUMBER, hostname)?;
504        }
505
506        if let Some(app_version) = resolve_app_version(resource_tags) {
507            tracer_payload_stream.write_string(TRACER_PAYLOAD_APP_VERSION_FIELD_NUMBER, app_version)?;
508        }
509
510        tracer_payload_stream.flush()?;
511        // Drop tracer_payload_stream to release the mutable borrow of tracer_payload_scratch
512        drop(tracer_payload_stream);
513
514        // Write TracerPayload as a nested message in AgentPayload (repeated field)
515        agent_payload_stream.write_bytes(AGENT_PAYLOAD_TRACER_PAYLOADS_FIELD_NUMBER, &self.tracer_payload_scratch)?;
516        agent_payload_stream.flush()?;
517
518        Ok(())
519    }
520}

554fn build_trace_chunk(trace: &Trace) -> TraceChunk {
555    let spans: Vec<ProtoSpan> = trace.spans().iter().map(convert_span).collect();
556    let mut chunk = TraceChunk::new();
557    chunk.set_spans(spans);
558
559    // TODO: Remove this once we have sampling. We have to hardcode the priority to 1 for now so that intake does not drop the trace.

561    chunk.set_priority(PRIORITY_AUTO_KEEP);

566fn convert_span(span: &DdSpan) -> ProtoSpan {
567    let mut proto = ProtoSpan::new();
568    proto.set_service(span.service().to_string().into());
569    proto.set_name(span.name().to_string().into());
570    proto.set_resource(span.resource().to_string().into());
571    proto.set_traceID(span.trace_id());
572    proto.set_spanID(span.span_id());
573    proto.set_parentID(span.parent_id());
574    proto.set_start(span.start());
575    proto.set_duration(span.duration());
576    proto.set_error(span.error());
577    proto.set_type(span.span_type().to_string().into());
578
579    proto.set_meta(
580        span.meta()
581            .iter()
582            .map(|(k, v)| (k.to_string().into(), v.to_string().into()))
583            .collect(),
584    );
585    proto.set_metrics(span.metrics().iter().map(|(k, v)| (k.to_string().into(), *v)).collect());
586    proto.set_meta_struct(
587        span.meta_struct()
588            .iter()
589            .map(|(k, v)| (k.to_string().into(), Bytes::from(v.clone())))
590            .collect(),
591    );
592    proto.set_spanLinks(span.span_links().iter().map(convert_span_link).collect());
593    proto.set_spanEvents(span.span_events().iter().map(convert_span_event).collect());
594    proto
595}

597fn convert_span_link(link: &DdSpanLink) -> ProtoSpanLink {
598    let mut proto = ProtoSpanLink::new();
599    proto.set_traceID(link.trace_id());
600    proto.set_traceID_high(link.trace_id_high());
601    proto.set_spanID(link.span_id());
602    proto.set_attributes(
603        link.attributes()
604            .iter()
605            .map(|(k, v)| (k.to_string().into(), v.to_string().into()))
606            .collect(),
607    );
608    proto.set_tracestate(link.tracestate().to_string().into());
609    proto.set_flags(link.flags());
610    proto

634        AttributeValue::Bool(v) => {

679fn write_message_field<M: Message>(
680    output_stream: &mut CodedOutputStream<'_>, field_number: u32, message: &M, scratch_buf: &mut Vec<u8>,
681) -> Result<(), protobuf::Error> {
682    scratch_buf.clear();
683    {
684        // In protobuf, length-delimited is one of the wire types, it encodes data as [tag][length][value].
685        // We use a nested output stream to write the message to because output_stream requires the size of the message to be known before writing
686        // and the Message type size is not known as it depends on the actual data values and lengths and we
687        // don't know this until runtime.
688        let mut nested = CodedOutputStream::vec(scratch_buf);
689        message.write_to(&mut nested)?;
690        nested.flush()?;
691    }
692    output_stream.write_tag(field_number, WireType::LengthDelimited)?;
693    output_stream.write_raw_varint32(scratch_buf.len() as u32)?;
694    output_stream.write_raw_bytes(scratch_buf)?;
695    Ok(())
696}
697
698fn write_map_entry_string_string(
699    output_stream: &mut CodedOutputStream<'_>, field_number: u32, key: &str, value: &str, scratch_buf: &mut Vec<u8>,
700) -> Result<(), protobuf::Error> {
701    scratch_buf.clear();
702    {
703        let mut nested = CodedOutputStream::vec(scratch_buf);
704        // the field number 1 and 2 correspond to key and value
705        nested.write_string(1, key)?;
706        nested.write_string(2, value)?;
707        nested.flush()?;
708    }
709    output_stream.write_tag(field_number, WireType::LengthDelimited)?;
710    output_stream.write_raw_varint32(scratch_buf.len() as u32)?;
711    output_stream.write_raw_bytes(scratch_buf)?;
712    Ok(())
713}

719fn resolve_hostname<'a>(
720    resource_tags: &'a TagSet, source: Option<&'a OtlpSource>, default_hostname: Option<&'a str>,
721    ignore_missing_fields: bool,
722) -> Option<&'a str> {
723    let mut hostname = match source {
724        Some(src) => match src.kind {
725            OtlpSourceKind::HostnameKind => Some(src.identifier.as_str()),
726            _ => Some(""),
727        },
728        None => default_hostname,
729    };
730
731    if ignore_missing_fields {
732        hostname = Some("");
733    }
734
735    if let Some(value) = get_resource_tag_value(resource_tags, KEY_DATADOG_HOST) {
736        hostname = Some(value);
737    }
738
739    hostname
740}

756    for key in [KEY_DATADOG_CONTAINER_ID, CONTAINER_ID, K8S_POD_UID] {
757        if let Some(value) = get_resource_tag_value(resource_tags, key) {
758            return Some(value);

762    // https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/api/otlp.go#L414

765            if k == KEY_DATADOG_CONTAINER_ID || k == K8S_POD_UID {
766                return Some(v.as_ref());

768        }
769    }
770    None
771}

780fn resolve_container_tags(
781    resource_tags: &TagSet, source: Option<&OtlpSource>, ignore_missing_fields: bool,
782) -> Option<MetaString> {
783    // TODO: some refactoring is probably needed to normalize this function, the tags should already be normalized
784    // since we do so when we transform OTLP spans to DD spans however to make this class extensible for non otlp traces, we would
785    // need to normalize the tags here.
786    if let Some(tags) = get_resource_tag_value(resource_tags, KEY_DATADOG_CONTAINER_TAGS) {
787        if !tags.is_empty() {
788            return Some(MetaString::from(tags));
789        }
790    }
791
792    if ignore_missing_fields {
793        return None;
794    }
795    let mut container_tags = TagSet::default();
796    extract_container_tags_from_resource_tagset(resource_tags, &mut container_tags);
797    let is_fargate_source = source.is_some_and(|src| src.kind == OtlpSourceKind::AwsEcsFargateKind);
798    if container_tags.is_empty() && !is_fargate_source {
799        return None;
800    }
801
802    let mut flattened = flatten_container_tag(container_tags);
803    if is_fargate_source {
804        if let Some(src) = source {
805            append_tags(&mut flattened, &src.tag());
806        }
807    }
808
809    if flattened.is_empty() {
810        None
811    } else {
812        Some(MetaString::from(flattened))
813    }
814}

816fn flatten_container_tag(tags: TagSet) -> String {
817    let mut flattened = String::new();
818    for tag in tags {
819        if !flattened.is_empty() {
820            flattened.push(',');
821        }
822        flattened.push_str(tag.as_str());
823    }
824    flattened
825}

827fn append_tags(target: &mut String, tags: &str) {
828    if tags.is_empty() {
829        return;
830    }
831    if !target.is_empty() {
832        target.push(',');
833    }
834    target.push_str(tags);
