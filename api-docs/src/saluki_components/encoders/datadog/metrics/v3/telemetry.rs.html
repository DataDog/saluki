1use std::sync::Arc;
2
3use datadog_agent_metrics_v3::{V3ValueEncodingStats, COLUMN_NAMES};
4use metrics::Counter;
5use saluki_metrics::MetricsBuilder;
6
7#[derive(Clone, Copy, Debug)]
8pub(crate) enum V3PayloadSplitReason {
9    MaxPoints,
10    PayloadFull,
11    ItemTooBig,
12}
13
14struct V3ColumnSizeTelemetry {
15    compressed: Counter,
16    uncompressed: Counter,
17}
18
19struct V3SerializerTelemetryInner {
20    values_zero: Counter,
21    values_sint64: Counter,
22    values_float32: Counter,
23    values_float64: Counter,
24    item_too_big: Counter,
25    split_max_points: Counter,
26    split_payload_full: Counter,
27    split_item_too_big: Counter,
28    column_sizes: Vec<Option<V3ColumnSizeTelemetry>>,
29}
30
31#[derive(Clone)]
32pub(crate) struct V3SerializerTelemetry {
33    inner: Arc<V3SerializerTelemetryInner>,
34}
35
36impl V3SerializerTelemetry {
37    pub(crate) fn from_builder(builder: &MetricsBuilder) -> Self {
38        let mut column_sizes = Vec::with_capacity(COLUMN_NAMES.len());
39        column_sizes.push(None);
40        for column_name in COLUMN_NAMES.iter().skip(1) {
41            column_sizes.push(Some(V3ColumnSizeTelemetry {
42                compressed: builder.register_counter_with_tags(
43                    "serializer.v3_column_size",
44                    [("column", *column_name), ("compressed", "compressed")],
45                ),
46                uncompressed: builder.register_counter_with_tags(
47                    "serializer.v3_column_size",
48                    [("column", *column_name), ("compressed", "uncompressed")],
49                ),
50            }));
51        }
52
53        Self {
54            inner: Arc::new(V3SerializerTelemetryInner {
55                values_zero: builder.register_counter_with_tags("serializer.v3_values_count", ["type:zero"]),
56                values_sint64: builder.register_counter_with_tags("serializer.v3_values_count", ["type:sint64"]),
57                values_float32: builder.register_counter_with_tags("serializer.v3_values_count", ["type:float32"]),
58                values_float64: builder.register_counter_with_tags("serializer.v3_values_count", ["type:float64"]),
59                item_too_big: builder.register_counter("serializer.v3_item_too_big"),
60                split_max_points: builder
61                    .register_counter_with_tags("serializer.v3_payload_split_reason", ["reason:max_points"]),
62                split_payload_full: builder
63                    .register_counter_with_tags("serializer.v3_payload_split_reason", ["reason:payload_full"]),
64                split_item_too_big: builder
65                    .register_counter_with_tags("serializer.v3_payload_split_reason", ["reason:item_too_big"]),
66                column_sizes,
67            }),
68        }
69    }
70
71    pub(crate) fn record_values_count(&self, stats: V3ValueEncodingStats) {
72        self.inner.values_zero.increment(stats.zero);
73        self.inner.values_sint64.increment(stats.sint64);
74        self.inner.values_float32.increment(stats.float32);
75        self.inner.values_float64.increment(stats.float64);
76    }
77
78    pub(crate) fn record_column_size(&self, field_number: u32, uncompressed_size: u64, compressed_size: u64) {
79        let Some(Some(column)) = self.inner.column_sizes.get(field_number as usize) else {
80            return;
81        };
82
83        column.uncompressed.increment(uncompressed_size);
84        column.compressed.increment(compressed_size);
85    }
86
87    pub(crate) fn record_item_too_big(&self) {
88        self.inner.item_too_big.increment(1);
89    }
90
91    pub(crate) fn record_split_reason(&self, reason: V3PayloadSplitReason) {
92        match reason {
93            V3PayloadSplitReason::MaxPoints => self.inner.split_max_points.increment(1),
94            V3PayloadSplitReason::PayloadFull => self.inner.split_payload_full.increment(1),
95            V3PayloadSplitReason::ItemTooBig => self.inner.split_item_too_big.increment(1),
96        }
97    }
98}
