7/// Limits used when building V3 metrics payloads.
8#[derive(Clone, Copy, Debug)]
9pub(crate) struct V3PayloadLimits {
10    pub(crate) max_compressed_size: usize,
11    pub(crate) max_uncompressed_size: usize,
12    max_metrics_per_payload: usize,
13    pub(crate) max_points_per_payload: usize,
14}
15
16impl V3PayloadLimits {
17    pub(crate) const fn new(
18        max_compressed_size: usize, max_uncompressed_size: usize, max_metrics_per_payload: usize,
19        max_points_per_payload: usize,
20    ) -> Self {
21        Self {
22            max_compressed_size,
23            max_uncompressed_size,
24            max_metrics_per_payload,
25            max_points_per_payload,
26        }
27    }
28
29    pub(crate) fn request_fits(self, request: &V3EncodedRequest) -> bool {
30        request.compressed_len <= self.max_compressed_size && request.uncompressed_len <= self.max_uncompressed_size
31    }
32
33    pub(crate) fn point_count_fits(self, count: usize) -> bool {
34        count <= self.max_points_per_payload
35    }
36
37    pub(crate) fn should_flush_metric_count_limit(self, metrics: &[Metric]) -> bool {
38        metrics.len() >= self.max_metrics_per_payload
39    }
40
41    pub(crate) fn should_flush_point_count_limit(self, data_point_count: usize) -> bool {
42        data_point_count >= self.max_points_per_payload
43    }
44
45    pub(crate) fn point_count_exceeds_limit(self, data_point_count: usize) -> bool {
46        data_point_count > self.max_points_per_payload
47    }
48}
