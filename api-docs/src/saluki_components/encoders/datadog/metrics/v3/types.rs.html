22/// V3 value type values.
23///
24/// These are encoded in bits 4-7 of the types column and indicate which
25/// value array contains the metric's points.
26#[derive(Debug, Clone, Copy, PartialEq, Eq)]
27#[repr(u8)]
28pub enum V3ValueType {
29    /// Value is zero, not stored explicitly.
30    Zero = 0x00,
31
32    /// Value is stored in vals_sint64.
33    Sint64 = 0x10,
34
35    /// Value is stored in vals_float32.
36    Float32 = 0x20,
37
38    /// Value is stored in vals_float64.
39    Float64 = 0x30,
40}
41
42impl V3ValueType {
43    /// Returns the numeric value for encoding in the types column.
44    pub fn as_u64(self) -> u64 {
45        self as u64
46    }
47}

49/// Intermediate point classification for value type compaction.
50///
51/// This provides finer-grained classification than [`V3ValueType`] to avoid
52/// precision loss when combining different value types. In particular, it
53/// distinguishes small integers (that fit losslessly in f32) from large integers
54/// (that don't), so that mixing a large integer with a Float32 value correctly
55/// escalates to Float64 rather than silently truncating the integer.
56#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
57#[repr(u8)]
58enum PointKind {
59    /// Value is zero.
60    Zero = 0,
61    /// Integer with |v| <= 2^24, fits losslessly in both sint64 and f32.
62    Int24 = 1,
63    /// Integer with |v| > 2^24, fits in sint64 varint but NOT losslessly in f32.
64    Int48 = 2,
65    /// Fractional value exactly representable as f32.
66    Float32 = 3,
67    /// Everything else - requires full f64 precision.
68    Float64 = 4,
69}
70
71/// Maximum integer magnitude that fits losslessly in f32 (2^24).
72const F32_INT_MAX: i64 = 1 << 24;
73
74impl PointKind {
75    /// Classifies a single f64 value.
76    fn for_value(v: f64) -> Self {
77        if v == 0.0 {
78            return Self::Zero;
79        }
80
81        // Varint range that fits in 7 bytes or less (49 bits).
82        const VARINT_WIDTH: i32 = 7 * 7 - 1;
83        const MAX_INT: i64 = 1 << VARINT_WIDTH;
84        const MIN_INT: i64 = -MAX_INT;
85
86        let i = v as i64;
87        if (MIN_INT..MAX_INT).contains(&i) && (i as f64) == v {
88            if (-F32_INT_MAX..=F32_INT_MAX).contains(&i) {
89                return Self::Int24;
90            }
91            return Self::Int48;
92        }
93
94        if (v as f32 as f64) == v {
95            return Self::Float32;
96        }
97
98        Self::Float64
99    }
100
101    /// Combines two point kinds into the smallest kind that can represent both.
102    ///
103    /// This is `max(self, other)` in all cases **except**:
104    /// - `Int48 + Float32 = Float64` (and vice versa), because large integers
105    ///   lose precision in f32, and fractional values can't be stored as sint64.
106    fn union(self, other: Self) -> Self {
107        match (self, other) {
108            (Self::Int48, Self::Float32) | (Self::Float32, Self::Int48) => Self::Float64,
109            _ => self.max(other),
110        }
111    }
112
113    /// Converts to the wire-format value type.
114    fn to_value_type(self) -> V3ValueType {
115        match self {
116            Self::Zero => V3ValueType::Zero,
117            Self::Int24 | Self::Int48 => V3ValueType::Sint64,
118            Self::Float32 => V3ValueType::Float32,
119            Self::Float64 => V3ValueType::Float64,
120        }
121    }
122}
123
124/// Determines the best [`V3ValueType`] for a set of f64 values.
125///
126/// Uses [`PointKind`] internally to avoid precision loss when mixing
127/// large integers with fractional float32 values.
128pub(super) fn value_type_for_values(values: impl Iterator<Item = f64>) -> V3ValueType {
129    let mut kind = PointKind::Zero;
130    for v in values {
131        kind = kind.union(PointKind::for_value(v));
132    }
133    kind.to_value_type()
134}

136#[cfg(test)]
137mod tests {
138    use super::*;
139
140    #[test]
141    fn test_point_kind_classification() {
142        // Zero
143        assert_eq!(PointKind::for_value(0.0), PointKind::Zero);
144
145        // Small integers (fit in f32)
146        assert_eq!(PointKind::for_value(100.0), PointKind::Int24);
147        assert_eq!(PointKind::for_value(-100.0), PointKind::Int24);
148        assert_eq!(PointKind::for_value((1 << 24) as f64), PointKind::Int24);
149        assert_eq!(PointKind::for_value(-((1 << 24) as f64)), PointKind::Int24);
150
151        // Large integers (don't fit losslessly in f32)
152        assert_eq!(PointKind::for_value(((1 << 24) + 1) as f64), PointKind::Int48);
153        assert_eq!(PointKind::for_value((1i64 << 30) as f64), PointKind::Int48);
154
155        // Float32
156        assert_eq!(PointKind::for_value(1.5), PointKind::Float32);
157        assert_eq!(PointKind::for_value(2.75), PointKind::Float32);
158
159        // Float64
160        assert_eq!(PointKind::for_value(std::f64::consts::PI), PointKind::Float64);
161        let large = ((1i64 << 50) + 1) as f64;
162        assert_eq!(PointKind::for_value(large), PointKind::Float64);
163    }
164
165    #[test]
166    fn test_point_kind_union() {
167        // Standard widening (max)
168        assert_eq!(PointKind::Zero.union(PointKind::Int24), PointKind::Int24);
169        assert_eq!(PointKind::Int24.union(PointKind::Int48), PointKind::Int48);
170        assert_eq!(PointKind::Int24.union(PointKind::Float32), PointKind::Float32);
171        assert_eq!(PointKind::Float32.union(PointKind::Float64), PointKind::Float64);
172        assert_eq!(PointKind::Float64.union(PointKind::Zero), PointKind::Float64);
173
174        // The critical case: large integer + float32 must escalate to float64
175        assert_eq!(PointKind::Int48.union(PointKind::Float32), PointKind::Float64);
176        assert_eq!(PointKind::Float32.union(PointKind::Int48), PointKind::Float64);
177    }
178
179    #[test]
180    fn test_value_type_for_values() {
181        // All zeros
182        assert_eq!(value_type_for_values([0.0, 0.0].into_iter()), V3ValueType::Zero);
183
184        // Small integers
185        assert_eq!(value_type_for_values([100.0, 200.0].into_iter()), V3ValueType::Sint64);
186
187        // Large integers
188        assert_eq!(
189            value_type_for_values([(1i64 << 30) as f64, 200.0].into_iter()),
190            V3ValueType::Sint64
191        );
192
193        // Small integer + float32 → Float32 (safe, small int fits in f32)
194        assert_eq!(value_type_for_values([100.0, 1.5].into_iter()), V3ValueType::Float32);
195
196        // Large integer + float32 → Float64 (the bug fix!)
197        assert_eq!(
198            value_type_for_values([(1i64 << 30) as f64, 1.5].into_iter()),
199            V3ValueType::Float64
200        );
201
202        // Float64 value forces Float64
203        assert_eq!(
204            value_type_for_values([100.0, std::f64::consts::PI].into_iter()),
205            V3ValueType::Float64
206        );
207
208        // Empty iterator
209        assert_eq!(value_type_for_values(std::iter::empty()), V3ValueType::Zero);
210    }
