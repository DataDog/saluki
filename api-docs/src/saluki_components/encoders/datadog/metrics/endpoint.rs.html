1use saluki_context::tags::SharedTagSet;
2use saluki_core::data_model::event::metric::{Metric, MetricValues};
3use saluki_io::compression::CompressionScheme;

5/// Metrics intake endpoint.
6#[derive(Clone, Copy, Debug, Eq, PartialEq)]
7pub enum MetricsEndpoint {
8    /// V1 series metrics, encoded as JSON and sent to `/api/v1/series`.
9    ///
10    /// Includes counters, gauges, rates, and sets. Selected when `use_v2_api_series` is `false`.
11    SeriesV1,
12
13    /// V2 series metrics, encoded as Protocol Buffers and sent to `/api/v2/series`.
14    ///
15    /// Includes counters, gauges, rates, and sets. The default series encoding.
16    SeriesV2,
17
18    /// Sketch metrics, encoded as Protocol Buffers and sent to `/api/beta/sketches`.
19    ///
20    /// Includes histograms and distributions. Always uses the V2 endpoint regardless of `use_v2_api_series`.
21    Sketches,
22}
23
24impl MetricsEndpoint {
25    /// Creates a new `MetricsEndpoint` from the given metric.
26    pub fn from_metric(metric: &Metric) -> Self {
27        match metric.values() {
28            MetricValues::Counter(..) | MetricValues::Rate(..) | MetricValues::Gauge(..) | MetricValues::Set(..) => {
29                Self::SeriesV2
30            }
31            MetricValues::Histogram(..) | MetricValues::Distribution(..) => Self::Sketches,
32        }
33    }
34}
35
36pub struct EndpointConfiguration {
37    compression_scheme: CompressionScheme,
38    // `max_metrics_per_payload` controls the number of series/sketches
39    // per-payload to the endpoint, above which multiple payloads are split.
40    max_metrics_per_payload: usize,
41    // `max_series_points_per_payload` controls the number of total points per
42    // payload, across all series. Sketches are not constrained.
43    max_series_points_per_payload: usize,
44    additional_tags: SharedTagSet,
45}
46
47impl EndpointConfiguration {
48    pub fn new(
49        compression_scheme: CompressionScheme, max_metrics_per_payload: usize, max_series_points_per_payload: usize,
50        additional_tags: Option<SharedTagSet>,
51    ) -> Self {
52        Self {
53            compression_scheme,
54            max_metrics_per_payload,
55            max_series_points_per_payload,
56            additional_tags: additional_tags.unwrap_or_default(),
57        }
58    }
59
60    pub fn compression_scheme(&self) -> CompressionScheme {
61        self.compression_scheme
62    }
63
64    pub fn max_metrics_per_payload(&self) -> usize {
65        self.max_metrics_per_payload
66    }
67
68    pub fn max_series_points_per_payload(&self) -> usize {
69        self.max_series_points_per_payload
70    }
71
72    pub fn additional_tags(&self) -> &SharedTagSet {
73        &self.additional_tags
74    }
