170/// Configuration for OTLP histogram processing.
171#[derive(Debug, Default, Deserialize)]
172#[cfg_attr(test, derive(PartialEq, serde::Serialize))]
173pub struct HistogramsConfig {
174    /// Controls how histogram buckets are reported.
175    ///
176    /// Use `distributions` for distribution metrics, `counters` for one metric per bucket, or
177    /// `nobuckets` to omit bucket metrics. The `nobuckets` mode requires histogram aggregation
178    /// metrics to be enabled separately.
179    ///
180    /// Defaults to `distributions`.
181    #[serde(default, deserialize_with = "deserialize_histogram_mode")]
182    pub mode: HistogramMode,
183
184    /// Whether to emit histogram count, sum, minimum, and maximum metrics when available.
185    ///
186    /// When disabled, only configured bucket metrics are emitted. The `nobuckets` mode therefore
187    /// requires this setting.
188    ///
189    /// Defaults to `false`.
190    #[serde(default)]
191    pub send_aggregation_metrics: bool,
192}

261/// Configuration for OTLP sums.
262#[derive(Deserialize, Debug, Default)]
263#[cfg_attr(test, derive(PartialEq, serde::Serialize))]
264pub struct SumsConfig {
265    /// Controls how cumulative monotonic sums are emitted.
266    ///
267    /// The default `to_delta` converts each cumulative value to a delta and emits it as a count. Set this to
268    /// `raw_value` to emit the cumulative value as a gauge instead. This affects only cumulative monotonic sums;
269    /// delta sums and non-monotonic sums retain their existing behavior. Use `raw_value` when the receiver of the
270    /// translated metrics needs the original cumulative value.
271    ///
272    /// Corresponds to `otlp_config.metrics.sums.cumulative_monotonic_mode`.
273    #[serde(default, deserialize_with = "deserialize_cumulative_monotonic_mode")]
274    pub cumulative_monotonic_mode: CumulativeMonotonicMode,
275
276    /// Controls how the first value of a cumulative monotonic sum is emitted.
277    ///
278    /// The default `auto` reports the first value only when its series started after the translator process. Set this
279    /// to `drop` to always discard the first value or `keep` to always report it. This affects only cumulative
280    /// monotonic sums in `to_delta` mode; `raw_value` emits every value as a gauge.
281    ///
282    /// Corresponds to `otlp_config.metrics.sums.initial_cumulative_monotonic_value`.
283    #[serde(default, deserialize_with = "deserialize_initial_cumulative_monotonic_value")]
284    pub initial_cumulative_monotonic_value: InitialCumulativeMonotonicValue,
285}

317//TODO: remove env handling with typed config, unblocked by #2094
318impl MetricsConfig {
319    /// Applies flat environment-variable overrides to nested settings.
320    ///
321    /// Figment exposes single-underscore names as flat keys, which nested deserialization cannot read.
322    pub(crate) fn apply_env_overrides(&mut self, config: &GenericConfiguration) -> Result<(), GenericError> {
323        if let Some(send_aggregation_metrics) =
324            config.try_get_typed::<bool>("otlp_config_metrics_histograms_send_aggregation_metrics")?
325        {
326            self.histograms.send_aggregation_metrics = send_aggregation_metrics;
327        }
328        if let Some(raw_mode) = config.try_get_typed::<String>("otlp_config_metrics_sums_cumulative_monotonic_mode")? {
329            self.sums.cumulative_monotonic_mode = raw_mode.parse().map_err(|error| {
330                generic_error!(
331                    "invalid `otlp_config.metrics.sums.cumulative_monotonic_mode` environment override: {error}"
332                )
333            })?;
334        }
335        if let Some(raw_value) =
336            config.try_get_typed::<String>("otlp_config_metrics_sums_initial_cumulative_monotonic_value")?
337        {
338            self.sums.initial_cumulative_monotonic_value = raw_value.parse().map_err(|error| {
339                generic_error!(
340                    "invalid `otlp_config.metrics.sums.initial_cumulative_monotonic_value` environment override: {error}"
341                )
342            })?;
343        }
344        if let Some(raw_mode) = config.try_get_typed::<String>("otlp_config_metrics_summaries_mode")? {
345            self.summaries.mode = raw_mode.parse().map_err(|error| {
346                generic_error!("invalid `otlp_config.metrics.summaries.mode` environment override: {error}")
347            })?;
348        }
349        Ok(())
350    }
351}

353/// Configuration for OTLP traces processing.
354///
355/// Mirrors the Agent's `otlp_config.traces` configuration.
356#[derive(Clone, Deserialize, Debug)]
357#[cfg_attr(test, derive(PartialEq, serde::Serialize))]
358pub struct TracesConfig {
359    /// Whether to enable OTLP traces support.
360    ///
361    /// Defaults to `true`.
362    #[serde(default = "default_traces_enabled")]
363    pub enabled: bool,
364
365    /// Whether to skip deriving Datadog fields from standard OTLP attributes.
366    ///
367    /// When true, only uses explicit `datadog.*` prefixed attributes and skips
368    /// fallback resolution from OTLP semantic conventions.
369    ///
370    /// Corresponds to `otlp_config.traces.ignore_missing_datadog_fields` in the Agent.
371    ///
372    /// Defaults to `false`.
373    #[serde(default)]
374    pub ignore_missing_datadog_fields: bool,
375
376    /// When true, `_top_level` and `_dd.measured` are derived using the OTLP span kind.
377    ///
378    /// Corresponds to the `enable_otlp_compute_top_level_by_span_kind` feature flag
379    /// in the Agent's `apm_config.features`.
380    ///
381    /// Defaults to `true`.
382    #[serde(default = "default_enable_otlp_compute_top_level_by_span_kind")]
383    pub enable_otlp_compute_top_level_by_span_kind: bool,
384
385    /// Probabilistic sampler configuration for OTLP traces.
386    ///
387    /// Corresponds to `otlp_config.traces.probabilistic_sampler` in the Agent.
388    #[serde(default)]
389    pub probabilistic_sampler: ProbabilisticSampler,
390
391    /// Total size of the string interner used for OTLP traces.
392    ///
393    /// Defaults to 512 KiB.
394    #[serde(rename = "string_interner_size", default = "default_traces_string_interner_size")]
395    pub string_interner_bytes: ByteSize,
396
397    /// The internal port on the Core Agent to forward traces to.
398    ///
399    /// Defaults to 5003.
400    #[serde(default = "default_internal_port")]
401    #[allow(unused)]
402    pub internal_port: u16,
403}

444impl TracesConfig {
445    /// Applies env var overrides for keys whose `DD_`-stripped flat form can't reach the nested
446    /// struct through normal serde deserialization.
447    ///
448    /// `DD_OTLP_CONFIG_TRACES_PROBABILISTIC_SAMPLER_SAMPLING_PERCENTAGE` strips to flat Figment key
449    /// `otlp_config_traces_probabilistic_sampler_sampling_percentage`. KEY_ALIASES ensures YAML and
450    /// env var land on the same key, but a nested struct can't see a flat key—so we read it
451    /// explicitly and override.
452    pub(crate) fn apply_env_overrides(&mut self, config: &GenericConfiguration) -> Result<(), GenericError> {
453        if let Some(pct) =
454            config.try_get_typed::<f64>("otlp_config_traces_probabilistic_sampler_sampling_percentage")?
455        {
456            self.probabilistic_sampler.sampling_percentage = pct;
457        }
458        Ok(())
459    }
460}
