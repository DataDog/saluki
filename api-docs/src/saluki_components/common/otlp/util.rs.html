23/// The kind of source that produced telemetry data.
24#[derive(Debug, Clone, PartialEq)]
25pub enum SourceKind {
26    /// Hostname-based source.
27    HostnameKind,
28    /// AWS ECS Fargate source.
29    AwsEcsFargateKind,
30}
31
32/// Represents the source of telemetry data.
33#[derive(Debug, Clone)]
34pub struct Source {
35    /// The kind of source.
36    pub kind: SourceKind,
37    /// The identifier for this source.
38    pub identifier: String,
39}
40
41impl Source {
42    /// Returns a tag representation of this source.
43    pub fn tag(&self) -> String {
44        format!("{}:{}", self.kind.as_str(), self.identifier)
45    }
46}
47
48impl SourceKind {
49    fn as_str(&self) -> &'static str {
50        match self {
51            SourceKind::HostnameKind => "host",
52            SourceKind::AwsEcsFargateKind => "task_arn",
53        }
54    }
55}

101/// Extracts a string attribute value from OTLP KeyValue attributes by key.
102pub fn get_string_attribute<'a>(attributes: &'a [otlp_common::KeyValue], key: &str) -> Option<&'a str> {
103    attributes.iter().find_map(|kv| {
104        if kv.key == key {
105            if let Some(Value::StringValue(s_val)) = kv.value.as_ref().and_then(|v| v.value.as_ref()) {
106                Some(s_val.as_str())
107            } else {
108                None
109            }
110        } else {
111            None
112        }
113    })
114}

116/// Extracts container tags from OTLP resource attributes and inserts them into the provided TagSet.
117/// This function is based on the agent implementation here
118/// https://github.com/DataDog/datadog-agent/blob/main/pkg/opentelemetry-mapping-go/otlp/attributes/attributes.go#L277
119pub fn extract_container_tags_from_resource_attributes(attributes: &[otlp_common::KeyValue], tags: &mut TagSet) {
120    let mut extracted_tags = FastHashSet::default();
121
122    for kv in attributes {
123        if let Some(Value::StringValue(s_val)) = kv.value.as_ref().and_then(|v| v.value.as_ref()) {
124            // Semantic Conventions
125            if let Some(datadog_key) = CONTAINER_MAPPINGS.get(kv.key.as_str()) {
126                tags.insert_tag(format!("{}:{}", datadog_key, s_val));
127                extracted_tags.insert(*datadog_key);
128            }
129
130            // Custom (datadog.container.tag namespace)
131            if kv.key.starts_with(CUSTOM_CONTAINER_TAG_PREFIX) {
132                if let Some(custom_key) = kv.key.get(CUSTOM_CONTAINER_TAG_PREFIX.len()..) {
133                    if !custom_key.is_empty() {
134                        // Do not replace if set via semantic conventions mappings.
135                        if !extracted_tags.insert(custom_key) {
136                            tags.insert_tag(format!("{}:{}", custom_key, s_val));
137                        }
138                    }
139                }
140            }
141        }
142    }
143}

145/// Extracts container tags from a resource tagset and inserts them into the provided TagSet.
146///
147/// This mirrors `extract_container_tags_from_resource_attributes`, but operates on a `TagSet` representation of the
148/// resource.
149pub fn extract_container_tags_from_resource_tagset(resource_tags: &TagSet, tags: &mut TagSet) {
150    let mut extracted_tags = FastHashSet::default();
151
152    for tag in resource_tags {
153        let Some(value) = tag.value() else {
154            continue;
155        };
156
157        // Semantic Conventions
158        if let Some(datadog_key) = CONTAINER_MAPPINGS.get(tag.name()) {
159            tags.insert_tag(format!("{}:{}", datadog_key, value));
160            extracted_tags.insert(*datadog_key);
161        }
162
163        // Custom (datadog.container.tag namespace)
164        if tag.name().starts_with(CUSTOM_CONTAINER_TAG_PREFIX) {
165            if let Some(custom_key) = tag.name().get(CUSTOM_CONTAINER_TAG_PREFIX.len()..) {
166                if !custom_key.is_empty() {
167                    // Do not replace if set via semantic conventions mappings.
168                    if !extracted_tags.insert(custom_key) {
169                        tags.insert_tag(format!("{}:{}", custom_key, value));
170                    }
171                }
172            }
173        }
174    }
175}

177/// Resolves the source metadata from OTLP resource attributes.
178///
179/// This determines whether the telemetry came from a hostname or serverless environment.
180pub fn resource_to_source(resource: &otlp_protos::opentelemetry::proto::resource::v1::Resource) -> Option<Source> {
181    let attributes = &resource.attributes;
182
183    // AWS ECS Fargate
184    if get_string_attribute(attributes, CLOUD_PROVIDER) == Some("aws")
185        && get_string_attribute(attributes, opentelemetry_semantic_conventions::resource::CLOUD_PLATFORM)
186            == Some("aws_ecs")
187        && get_string_attribute(
188            attributes,
189            opentelemetry_semantic_conventions::resource::AWS_ECS_LAUNCHTYPE,
190        ) == Some("fargate")
191    {
192        if let Some(task_arn) = get_string_attribute(attributes, AWS_ECS_TASK_ARN) {
193            return Some(Source {
194                kind: SourceKind::AwsEcsFargateKind,
195                identifier: task_arn.to_string(),
196            });
197        }
198    }
199
200    // Hostname from attributes
201    if let Some(host_name) = get_string_attribute(attributes, opentelemetry_semantic_conventions::resource::HOST_NAME) {
202        return Some(Source {
203            kind: SourceKind::HostnameKind,
204            identifier: host_name.to_string(),
205        });
206    }
207
208    None
209}

211/// Resolves the source metadata from a resource `TagSet`.
212///
213/// This is equivalent to `resource_to_source`, but avoids the OTLP protobuf resource type.
214pub fn tags_to_source(resource_tags: &TagSet) -> Option<Source> {
215    let get = |key: &str| -> Option<&str> { resource_tags.get_single_tag(key).and_then(|t| t.value()) };
216
217    // AWS ECS Fargate
218    if get(CLOUD_PROVIDER) == Some("aws")
219        && get(opentelemetry_semantic_conventions::resource::CLOUD_PLATFORM) == Some("aws_ecs")
220        && get(opentelemetry_semantic_conventions::resource::AWS_ECS_LAUNCHTYPE) == Some("fargate")
221    {
222        if let Some(task_arn) = get(AWS_ECS_TASK_ARN) {
223            return Some(Source {
224                kind: SourceKind::AwsEcsFargateKind,
225                identifier: task_arn.to_string(),
226            });
227        }
228    }
229
230    // Hostname from attributes
231    if let Some(host_name) = get(opentelemetry_semantic_conventions::resource::HOST_NAME) {
232        return Some(Source {
233            kind: SourceKind::HostnameKind,
234            identifier: host_name.to_string(),
235        });
236    }
237
238    None
