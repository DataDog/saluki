21pub fn convert_trace_id(trace_id: &[u8]) -> u64 {
22    if trace_id.len() < 8 {
23        return 0;
24    }
25    u64::from_be_bytes((&trace_id[(trace_id.len() - 8)..]).try_into().unwrap_or_default())
26}
27
28pub fn convert_span_id(span_id: &[u8]) -> u64 {
29    if span_id.len() != 8 {
30        return 0;
31    }
32    u64::from_be_bytes(span_id.try_into().unwrap_or_default())
33}

35fn resource_attributes_to_tagset(
36    attributes: &[otlp_common::KeyValue], string_builder: &mut StringBuilder<GenericMapInterner>,
37) -> TagSet {
38    let mut tags = TagSet::with_capacity(attributes.len());
39    for kv in attributes {
40        if let Some(key_value) = &kv.value {
41            if let Some(value) = &key_value.value {
42                if let Some(string_value) = otlp_value_to_string(value) {
43                    string_builder.clear();
44                    let _ = string_builder.push_str(kv.key.as_str());
45                    let _ = string_builder.push(':');
46                    let _ = string_builder.push_str(string_value.as_str());
47                    tags.insert_tag(string_builder.to_meta_string());
48                }
49            }
50        }
51    }
52    tags
53}

61pub struct OtlpTracesTranslator {
62    config: TracesConfig,
63    interner: GenericMapInterner,
64    string_builder: StringBuilder<GenericMapInterner>,
65}
66
67impl OtlpTracesTranslator {
68    pub fn new(config: TracesConfig, interner_size: NonZeroUsize) -> Self {
69        let interner = GenericMapInterner::new(interner_size);
70        let string_builder = StringBuilder::new().with_interner(interner.clone());
71        Self {
72            config,
73            interner,
74            string_builder,
75        }
76    }
77
78    pub fn translate_spans(&mut self, resource_spans: ResourceSpans, metrics: &Metrics) -> impl Iterator<Item = Event> {
79        let resource: OtlpResource = resource_spans.resource.unwrap_or_default();
80        let ignore_missing_fields = self.config.ignore_missing_datadog_fields;
81        let compute_top_level = self.config.enable_otlp_compute_top_level_by_span_kind;
82        let interner = &self.interner;
83        let string_builder = &mut self.string_builder;
84        let resource_tags = resource_attributes_to_tagset(&resource.attributes, string_builder).into_shared();
85        let mut traces_by_id: FastHashMap<u64, TraceEntry> = FastHashMap::default();
86
87        for scope_spans in resource_spans.scope_spans {
88            let scope = scope_spans.scope;
89            let scope_ref = scope.as_ref();
90            metrics.spans_received().increment(scope_spans.spans.len() as u64);
91            for span in scope_spans.spans {
92                let trace_id = convert_trace_id(&span.trace_id);
93                let entry = traces_by_id.entry(trace_id).or_insert_with(|| TraceEntry {
94                    spans: Vec::new(),
95                    priority: None,
96                    trace_id_hex: None,
97                });
98
99                if entry.trace_id_hex.is_none() {
100                    entry.trace_id_hex = trace_id_hex_meta(&span.trace_id);
101                }
102
103                let trace_id_hex = entry.trace_id_hex.clone();
104                let dd_span = otel_span_to_dd_span(
105                    &span,
106                    &resource,
107                    scope_ref,
108                    ignore_missing_fields,
109                    compute_top_level,
110                    interner,
111                    string_builder,
112                    trace_id_hex,
113                );
114
115                // Track last-seen priority for this trace (overwrites previous values)
116                if let Some(&priority) = dd_span.metrics().get(SAMPLING_PRIORITY_METRIC_KEY) {
117                    entry.priority = Some(priority as i32);
118                }
119
120                entry.spans.push(dd_span);
121            }
122        }
123
124        OtlpTraceEventsIter {
125            resource_tags,
126            entries: traces_by_id.into_iter(),
127        }
128    }
129}

131struct OtlpTraceEventsIter {
132    resource_tags: SharedTagSet,
133    entries: IntoIter<u64, TraceEntry>,
134}
135
136impl Iterator for OtlpTraceEventsIter {
137    type Item = Event;
138
139    fn next(&mut self) -> Option<Self::Item> {
140        for (_, entry) in self.entries.by_ref() {
141            if entry.spans.is_empty() {
142                continue;
143            }
144
145            let mut trace = Trace::new(entry.spans, self.resource_tags.clone());
146
147            // Set the trace-level sampling priority if one was found
148            if let Some(priority) = entry.priority {
149                trace.set_sampling(Some(TraceSampling::new(false, Some(priority), None, None)));
150            }
151
152            return Some(Event::Trace(trace));
153        }
154
155        None
156    }
157}

159fn trace_id_hex_meta(trace_id: &[u8]) -> Option<MetaString> {
160    if trace_id.is_empty() {
161        return None;
162    }
163
164    let hex = bytes_to_hex_lowercase(trace_id);
165    if hex.is_empty() {
166        return None;
167    }
168
169    Some(MetaString::from(Arc::<str>::from(hex)))
170}
