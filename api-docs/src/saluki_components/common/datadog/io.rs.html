53/// A handle to the transaction forwarder.
54pub struct Handle<B>
55where
56    B: Buf + Clone,
57{
58    transactions_tx: mpsc::Sender<Transaction<B>>,
59    io_shutdown_rx: oneshot::Receiver<()>,
60}
61
62impl<B> Handle<B>
63where
64    B: Buf + Clone,
65{
66    /// Sends a transaction to the forwarder.
67    ///
68    /// # Errors
69    ///
70    /// If the endpoint I/O task has unexpectedly stopped and can no longer accept transactions, an error will be returned.
71    pub async fn send_transaction(&self, transaction: Transaction<B>) -> Result<(), GenericError> {
72        match self.transactions_tx.send(transaction).await {
73            Ok(()) => Ok(()),
74            Err(_) => Err(generic_error!("Failed to send request to I/O task: receiver dropped.")),
75        }
76    }
77
78    /// Triggers the forwarder to shutdown and waits to shutdown to complete.
79    pub async fn shutdown(self) {
80        let Self {
81            transactions_tx,
82            io_shutdown_rx,
83        } = self;
84
85        // Drop the sender side of the transaction channel, which will propagate the actual closure to the main I/O task.
86        drop(transactions_tx);
87
88        // Wait for the main I/O task to signal that it has shutdown.
89        io_shutdown_rx.await.expect("I/O task has already shutdown.");
90    }
91}

104#[derive(Clone, Copy, Debug, Eq, PartialEq)]
105enum TlsCertificateValidation {
106    Enabled,
107    Disabled,
108}
109
110impl TlsCertificateValidation {
111    const fn from_forwarder_config(config: &ForwarderConfiguration) -> Self {
112        if config.skip_ssl_validation() {
113            Self::Disabled
114        } else {
115            Self::Enabled
116        }
117    }
118
119    fn apply_to(self, client_builder: HttpClientBuilder) -> Result<HttpClientBuilder, GenericError> {
120        self.ensure_supported()?;
121
122        Ok(match self {
123            Self::Enabled => client_builder,
124            Self::Disabled => {
125                warn!(
126                    config_key = "skip_ssl_validation",
127                    "TLS certificate validation is disabled for Datadog intake forwarding."
128                );
129                client_builder.with_tls_config(|builder| builder.danger_accept_invalid_certs())
130            }
131        })
132    }
133
134    fn ensure_supported(self) -> Result<(), GenericError> {
135        #[cfg(feature = "fips")]
136        if matches!(self, Self::Disabled) {
137            return Err(generic_error!(
138                "`skip_ssl_validation: true` is unsupported in FIPS mode because disabling TLS certificate validation is not FIPS-compliant."
139            ));
140        }
141
142        Ok(())
143    }
144}

152    /// Creates a new `TransactionForwarder` instance from the given configuration.
153    pub fn from_config<F>(
154        context: ComponentContext, config: ForwarderConfiguration, configuration: Option<GenericConfiguration>,
155        endpoint_name: F, telemetry: ComponentTelemetry, metrics_builder: MetricsBuilder,
156    ) -> Result<Self, GenericError>
157    where
158        F: Fn(&Uri) -> Option<MetaString> + Send + Sync + 'static,
159    {
160        let endpoints = config.endpoint().build_resolved_endpoints(configuration)?;
161        let mut client_builder = HttpClient::builder()
162            .with_request_timeout(config.request_timeout())
163            .with_bytes_sent_counter(telemetry.bytes_sent().clone())
164            .with_endpoint_telemetry(metrics_builder.clone(), Some(endpoint_name));
165        if let Some(proxy) = config.proxy() {
166            client_builder = client_builder.with_proxies(proxy.build()?);
167        }
168
169        if config.connection_reset_interval() > Duration::ZERO {
170            client_builder = client_builder.with_connection_age_limit(config.connection_reset_interval());
171        }
172
173        client_builder = TlsCertificateValidation::from_forwarder_config(&config).apply_to(client_builder)?;
174
175        let client = client_builder.build()?;
176
177        Ok(Self {
178            context,
179            config,
180            telemetry,
181            metrics_builder,
182            client,
183            endpoints,
184            _marker: std::marker::PhantomData,
185        })
186    }
187
188    /// Spawns the I/O task for the forwarder, and any associated endpoint I/O tasks.
189    ///
190    /// Returns a `Handle` that can be used to send transactions to the forwarder, as well as eventually shut it down in
191    /// an orderly fashion.
192    pub async fn spawn(self) -> Handle<B> {
193        let (transactions_tx, transactions_rx) = mpsc::channel(8);
194        let (io_shutdown_tx, io_shutdown_rx) = oneshot::channel();
195
196        let Self {
197            context,
198            config,
199            telemetry,
200            metrics_builder,
201            client,
202            endpoints,
203            _marker,
204        } = self;
205
206        spawn_traced_named(
207            "dd-txn-forwarder-io-loop",
208            run_io_loop(
209                transactions_rx,
210                io_shutdown_tx,
211                context,
212                config,
213                client,
214                telemetry,
215                metrics_builder,
216                endpoints,
217            ),
218        );
219
220        Handle {
221            transactions_tx,
222            io_shutdown_rx,
223        }
224    }
225}

227async fn run_io_loop<B>(
228    mut transactions_rx: mpsc::Receiver<Transaction<B>>, io_shutdown_tx: oneshot::Sender<()>,
229    context: ComponentContext, config: ForwarderConfiguration, service: HttpClient, telemetry: ComponentTelemetry,
230    metrics_builder: MetricsBuilder, resolved_endpoints: Vec<ResolvedEndpoint>,
231) where
232    B: Body + Buf + Clone + Send + Sync + 'static,
233    B::Data: Send,
234    B::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
235{
236    // Spawn an endpoint I/O task for each endpoint we're configured to send to, which we'll forward transactions to.
237    let mut endpoint_txs = Vec::new();
238    let task_barrier = Arc::new(Barrier::new(resolved_endpoints.len() + 1));
239    let shared_txnq_telemetry = SharedTransactionQueueTelemetry::from_builder(&metrics_builder);
240
241    for resolved_endpoint in resolved_endpoints {
242        let endpoint_url = resolved_endpoint.endpoint().to_string();
243
244        let txnq_telemetry =
245            TransactionQueueTelemetry::from_builder(&metrics_builder, &endpoint_url, shared_txnq_telemetry.clone());
246
247        let (endpoint_tx, endpoint_rx) = mpsc::channel(8);
248        let task_barrier = Arc::clone(&task_barrier);
249
250        let task_name = format!("dd-txn-forwarder-io-loop-{}", resolved_endpoint.endpoint().authority());
251        spawn_traced_named(
252            task_name,
253            run_endpoint_io_loop(
254                endpoint_rx,
255                task_barrier,
256                context.clone(),
257                config.clone(),
258                service.clone(),
259                telemetry.clone(),
260                txnq_telemetry,
261                resolved_endpoint,
262            ),
263        );
264
265        endpoint_txs.push((endpoint_url, endpoint_tx));
266    }
267
268    // Listen for transactions to forward, and send a copy of each one to each endpoint I/O task.
269    while let Some(transaction) = transactions_rx.recv().await {
270        for (endpoint_url, endpoint_tx) in &endpoint_txs {
271            if endpoint_tx.send(transaction.clone()).await.is_err() {
272                error!(
273                    endpoint = endpoint_url,
274                    "Failed to send request to endpoint I/O task: receiver dropped."
275                );
276            }
277        }
278    }
279
280    debug!("Requests channel for main I/O task complete. Stopping endpoint I/O tasks and synchronizing on shutdown.");
281
282    // Drop our endpoint I/O task channels, which will cause them to shut down once they've processed all outstanding
283    // requests in their respective channel. We wait for that to happen by synchronizing on the task barrier.
284    //
285    // Once all tasks have completed, we signal back to the main component task that the I/O loop has shutdown.
286    drop(endpoint_txs);
287    task_barrier.wait().await;
288
289    debug!("All endpoint I/O tasks have stopped. Main I/O task shutting down.");
290
291    let _ = io_shutdown_tx.send(());
292}

294async fn run_endpoint_io_loop<B>(
295    mut txns_rx: mpsc::Receiver<Transaction<B>>, task_barrier: Arc<Barrier>, context: ComponentContext,
296    config: ForwarderConfiguration, service: HttpClient, telemetry: ComponentTelemetry,
297    txnq_telemetry: TransactionQueueTelemetry, endpoint: ResolvedEndpoint,
298) where
299    B: Body + Buf + Clone + Send + Sync + 'static,
300    B::Data: Send,
301    B::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
302{
303    let queue_id = generate_retry_queue_id(context, &endpoint);
304    let endpoint_url = endpoint.endpoint().to_string();
305    debug!(
306        endpoint_url,
307        num_workers = config.endpoint_concurrency(),
308        "Starting endpoint I/O task."
309    );
310
311    // Build our endpoint service.
312    //
313    // This is where we'll modify the incoming transaction for our our specific endpoint, such as setting the host portion
314    // of the URI, adding the API key as a header, and so on.
315    //
316    // The body type conversion from `TransactionBody<B>` to `ClientBody` happens as the innermost layer,
317    // after the retry circuit breaker. This ensures that `RetryCircuitBreakerError::Open(req)` returns
318    // the original `Request<TransactionBody<B>>` so we can reassemble it into a `Transaction<B>` for re-enqueuing.
319    let mut service = ServiceBuilder::new()
320        // Set the request's URI to the endpoint's URI, and add the API key as a header.
321        .map_request(for_resolved_endpoint(endpoint))
322        // Set the User-Agent and DD-Agent-Version headers indicating the version of the data plane sending the request.
323        .map_request(with_version_info())
324        .concurrency_limit(config.endpoint_concurrency())
325        .layer(RetryCircuitBreakerLayer::new(
326            config.retry().to_default_http_retry_policy(),
327        ))
328        .map_request(|req: Request<TransactionBody<B>>| req.map(into_client_body))
329        .service(service);
330
331    let mut retry_queue = RetryQueue::new(queue_id.clone(), config.retry().queue_max_size_bytes());
332
333    // If the storage size is set, enable disk persistence for the retry queue.
334    if config.retry().storage_max_size_bytes() > 0 {
335        retry_queue = retry_queue
336            .with_disk_persistence(
337                PathBuf::from(config.retry().storage_path()),
338                config.retry().storage_max_size_bytes(),
339                config.retry().storage_max_disk_ratio(),
340                Arc::new(DiskUsageRetrieverImpl::new(PathBuf::from(
341                    config.retry().storage_path(),
342                ))),
343            )
344            .await
345            .unwrap_or_else(|e| {
346                error!(endpoint_url, error = %e, "Failed to initialize disk persistence for retry queue. Transactions will not be persisted.");
347                RetryQueue::new(queue_id, config.retry().queue_max_size_bytes())
348            });
349    }
350    let mut pending_txns = PendingTransactions::new(config.endpoint_buffer_size(), retry_queue, txnq_telemetry);
351
352    let mut in_flight = JoinSet::new();
353    let mut done = false;
354
355    loop {
356        select! {
357            // Try and drain the next transaction from our channel, and push it into the pending transactions queue.
358            maybe_txn = txns_rx.recv(), if !done => match maybe_txn {
359                Some(txn) => match pending_txns.push_high_priority(txn).await {
360                    Ok(push_result) => {
361                        telemetry.track_dropped_items(push_result.items_dropped);
362                        telemetry.track_dropped_events(push_result.events_dropped);
363                    }
364                    Err(e) => error!(endpoint_url, error = %e, "Failed to enqueue transaction. Events may be permanently lost."),
365                },
366                None => {
367                    // Our transactions channel has been closed, so mark ourselves as done which will stop any further
368                    // transactions from being sent, but will allow in-flight transactions to complete.
369                    done = true;
370                    debug!(endpoint_url, "Requests channel for endpoint I/O task complete. Completing any in-flight requests...");
371                }
372            },
373
374            // While we're not done and there are pending transactions, wait for the service to become ready and then
375            // next the next available pending transaction.
376            svc = service.ready(), if !done && !pending_txns.is_empty() => match svc {
377                Ok(svc) => if let Some(txn) = pending_txns.pop().await {
378                    let (metadata, request) = txn.into_parts();
379                    in_flight.spawn(svc.call(request).map(move |result| (metadata, result)));
380
381                    debug!(endpoint_url, "Request sent.");
382                },
383                Err(e) => match e {
384                    RetryCircuitBreakerError::Service(e) => {
385                        telemetry.track_sent_request_error();
386                        error!(endpoint_url, error = %e, error_source = ?e.source(), "Unexpected error when querying service for readiness.");
387                        break;
388                    },
389                    RetryCircuitBreakerError::Open(_) => unreachable!("should not get open error when querying service for readiness"),
390                }
391            },
392
393            // Drive any in-flight transactions to completion.
394            maybe_result = in_flight.join_next(), if !in_flight.is_empty() => {
395                let task_result = maybe_result.expect("in_flight marked as not being empty");
396                match task_result {
397                    Ok((metadata, result)) => match result {
398                        // We got a response -- maybe successful, maybe not -- so just process that.
399                        Ok(http_response) => process_http_response(http_response, metadata, &telemetry, &endpoint_url).await,
400
401                        // The service itself encountered an error while sending the request or receiving the response:
402                        // connection reset by peer, I/O error, etc.
403                        Err(RetryCircuitBreakerError::Service(e)) => {
404                            telemetry.track_failed_transaction(&metadata, None);
405                            error!(endpoint_url, error = %e, error_source = ?e.source(), "Failed to send request.");
406                        },
407
408                        // Our endpoint circuit breaker is open, which means this request either didn't go through at
409                        // all or needs to be retried... so we'll re-enqueue it to the low-priority queue to be retried
410                        // later.
411                        Err(RetryCircuitBreakerError::Open(req)) => {
412                            let reassembled_txn = Transaction::reassemble(metadata, req);
413                            match pending_txns.push_low_priority(reassembled_txn).await {
414                                Ok(push_result) => {
415                                    telemetry.track_dropped_items(push_result.items_dropped);
416                                    telemetry.track_dropped_events(push_result.events_dropped);
417                                }
418                                Err(e) => error!(endpoint_url, error = %e, "Failed to re-enqueue failed transaction. Events may be permanently lost."),
419                            }
420                        },
421                    },
422
423                    // Our transaction task itself failed, which means something weirdly bad happened: panic, etc.
424                    Err(e) => {
425                        error!(endpoint_url, error = %e, error_source = ?e.source(), "Request task failed to run to completion.");
426                    }
427                }
428            },
429
430            else => break,
431        }
432    }
433
434    // Flush any outstanding transactions in the pending transactions queue, which will potentially enqueue them to disk
435    // if we have disk persistent enabled for the retry queue.
436    match pending_txns.flush().await {
437        // If we successfully flushed the pending transactions, track the number of events that were dropped, if any.
438        Ok(flush_result) => {
439            debug!(
440                items_dropped = flush_result.items_dropped,
441                events_dropped = flush_result.events_dropped,
442                "Flushed pending transactions prior to shutdown."
443            );
444            telemetry.track_dropped_items(flush_result.items_dropped);
445            telemetry.track_dropped_events(flush_result.events_dropped);
446        }
447        Err(e) => {
448            error!(endpoint_url, error = %e, "Failed to flush pending transactions. Events may be permanently lost.")
449        }
450    }
451
452    debug!(
453        endpoint_url,
454        "Requests channel for endpoint I/O task complete. Synchronizing on shutdown."
455    );
456
457    // Signal to the main I/O task that we've finished.
458    task_barrier.wait().await;
459}

461fn generate_retry_queue_id(context: ComponentContext, endpoint: &ResolvedEndpoint) -> String {
462    // TODO: This logic does not take into account cases where the API key is updated dynamically. While a running
463    // process would just keep using the existing retry queue, based on the queue ID we generate here... the next time
464    // the process restarted, the retry queue ID would change, which could leave behind old transactions that wouldn't
465    // end up being retried.
466    //
467    // The Core Agent is also susceptible to this, I believe... but we should double check that and see what they're
468    // doing if they actually _do_ handle this case.
469
470    // We set our queue ID/name to be unique for the component/endpoint/API key combination, which ensures that two
471    // instances of the same destination cannot collide with each other if they're using the same endpoint/API key
472    // combination.
473    let hash = hash_single_stable((context.component_id(), endpoint.endpoint(), endpoint.cached_api_key()));
474
475    let endpoint_host = endpoint
476        .endpoint()
477        .host_str()
478        .expect("resolved endpoint must have a host");
479    format!("{}/{}/{:x}", context.component_id(), endpoint_host, hash)
480}

482async fn process_http_response(
483    response: Response<Incoming>, metadata: Metadata, telemetry: &ComponentTelemetry, endpoint_url: &str,
484) {
485    let status = response.status();
486    if status.is_success() {
487        debug!(endpoint_url, %status, "Request completed.");
488
489        telemetry.track_successful_transaction(&metadata);
490    } else {
491        telemetry.track_failed_transaction(&metadata, Some(status));
492
493        match response.into_body().collect().await {
494            Ok(body) => {
495                let body = body.to_bytes();
496                let body_str = String::from_utf8_lossy(&body[..]);
497                error!(endpoint_url, %status, "Received non-success response. Body: {}", body_str);
498            }
499            Err(e) => {
500                error!(endpoint_url, %status, error = %e, "Failed to read response body of non-success response.");
501            }
502        }
503    }
504}

506/// A queue of pending transactions waiting to be sent.
507///
508/// This queue is split into two parts: a high-priority queue and a low-priority queue. The high-priority queue is used
509/// for brand-new transactions that are waiting to be processed for the first time. The low-priority queue contains
510/// transactions that have either been requeued to be retried again at a later time, or that could not fit in the
511/// high-priority queue due to it being full.
512///
513/// Ultimately, we use this construction to provide a fast path for new transactions, while limiting the overall number
514/// of outstanding transactions that are waiting to be processed, with a bias towards preserving the most recent
515/// transactions so that fresh data can be sent as soon as any temporary networking issues are resolved.
516struct PendingTransactions<T> {
517    high_priority: VecDeque<T>,
518    low_priority: RetryQueue<T>,
519    telemetry: TransactionQueueTelemetry,
520    incoming_bytes_per_sec: IncomingBytesPerSec,
521}
522
523// Mirrors the Datadog Agent retry queue duration traffic-rate input:
524// https://github.com/DataDog/datadog-agent/blob/main/comp/forwarder/defaultforwarder/internal/retry/queue_duration_capacity.go
525// https://github.com/DataDog/datadog-agent/blob/main/comp/forwarder/defaultforwarder/internal/retry/time_interval_accumulator.go
526struct IncomingBytesPerSec {
527    bucket_sum: Vec<u64>,
528    current_index: usize,
529    current_index_time_secs: Option<u64>,
530    start_index_time_secs: Option<u64>,
531    sum: u64,
532    bucket_duration_secs: u64,
533}
534
535impl IncomingBytesPerSec {
536    fn new(history_duration_secs: u64, bucket_duration_secs: u64) -> Self {
537        assert!(bucket_duration_secs > 0, "bucket duration must be at least one second");
538        assert!(
539            history_duration_secs >= bucket_duration_secs,
540            "history duration must be greater than or equal to bucket duration"
541        );
542
543        Self {
544            bucket_sum: vec![0; (history_duration_secs / bucket_duration_secs) as usize],
545            current_index: 0,
546            current_index_time_secs: None,
547            start_index_time_secs: None,
548            sum: 0,
549            bucket_duration_secs,
550        }
551    }
552
553    fn record(&mut self, now_secs: u64, bytes: u64) -> f64 {
554        if self.start_index_time_secs.is_none() {
555            self.start_index_time_secs = Some(now_secs);
556            self.current_index_time_secs = Some(now_secs);
557        }
558
559        while now_secs
560            >= self.current_index_time_secs.expect("current index time should be set") + self.bucket_duration_secs
561        {
562            self.current_index = (self.current_index + 1) % self.bucket_sum.len();
563            self.sum -= self.bucket_sum[self.current_index];
564            self.bucket_sum[self.current_index] = 0;
565
566            let current_index_time_secs =
567                self.current_index_time_secs.expect("current index time should be set") + self.bucket_duration_secs;
568            self.current_index_time_secs = Some(current_index_time_secs);
569
570            let start_index_time_secs = self.start_index_time_secs.expect("start index time should be set");
571            if current_index_time_secs
572                >= start_index_time_secs + self.bucket_sum.len() as u64 * self.bucket_duration_secs
573            {
574                self.start_index_time_secs = Some(start_index_time_secs + self.bucket_duration_secs);
575            }
576        }
577
578        self.bucket_sum[self.current_index] += bytes;
579        self.sum += bytes;
580        self.bytes_per_sec(now_secs)
581    }
582
583    fn bytes_per_sec(&self, now_secs: u64) -> f64 {
584        let Some(start_index_time_secs) = self.start_index_time_secs else {
585            return 0.0;
586        };
587
588        let duration_secs = now_secs.saturating_sub(start_index_time_secs) + 1;
589        self.sum as f64 / duration_secs as f64
590    }
591}

593impl<T: Retryable> PendingTransactions<T> {
594    /// Creates a new `PendingTransactions` instance.
595    ///
596    /// The high-priority queue will have a maximum capacity of `max_enqueued`, and the retry queue will be used as the
597    /// low-priority queue.
598    pub fn new(max_enqueued: usize, retry_queue: RetryQueue<T>, telemetry: TransactionQueueTelemetry) -> Self {
599        Self {
600            high_priority: VecDeque::with_capacity(max_enqueued),
601            low_priority: retry_queue,
602            telemetry,
603            incoming_bytes_per_sec: IncomingBytesPerSec::new(
604                RETRY_QUEUE_CAPACITY_HISTORY_DURATION_SECS,
605                RETRY_QUEUE_CAPACITY_BUCKET_DURATION_SECS,

610    /// Returns `true` if there are no pending transactions.
611    ///
612    /// This includes both the high-priority and low-priority queues.
613    pub fn is_empty(&self) -> bool {
614        self.high_priority.is_empty() && self.low_priority.is_empty()
615    }

617    /// Pushes a high-priority transaction into the queue.
618    ///
619    /// If the high-priority queue is full, the transaction will be pushed into the low-priority queue.
620    pub async fn push_high_priority(&mut self, transaction: T) -> Result<PushResult, GenericError> {
621        self.record_incoming_transaction_size(transaction.size_bytes());
622
623        if self.high_priority.len() < self.high_priority.capacity() {
624            self.high_priority.push_back(transaction);
625            self.telemetry.high_prio_queue_insertions().increment(1);
626
627            debug!(
628                high_prio_queue_len = self.high_priority.len(),
629                "Enqueued pending transaction to high-priority queue."
630            );
631
632            Ok(PushResult::default())
633        } else {
634            let push_result = self.low_priority.push(transaction).await?;
635            self.telemetry.low_prio_queue_insertions().increment(1);
636            self.record_retry_queue_size();
637
638            debug!(
639                low_prio_queue_len = self.low_priority.len(),
640                "Enqueued pending transaction to low-priority queue."
641            );
642
643            Ok(push_result)
644        }
645    }
646
647    /// Pushes a low-priority transaction into the queue.
648    pub async fn push_low_priority(&mut self, transaction: T) -> Result<PushResult, GenericError> {
649        let push_result = self.low_priority.push(transaction).await?;
650        self.telemetry.low_prio_queue_insertions().increment(1);
651        self.record_retry_queue_size();
652
653        debug!(
654            low_prio_queue_len = self.low_priority.len(),
655            "Enqueued pending transaction to low-priority queue."
656        );
657
658        Ok(push_result)
659    }
660
661    /// Pops the next transaction from the queue.
662    ///
663    /// The high-priority queue is drained first before attempting to pop from the low-priority queue.
664    pub async fn pop(&mut self) -> Option<T> {
665        // We bias towards handling enqueued transactions first, since those are our "high priority" transactions, and we
666        // want to keep them flowing as fast as possible.
667        loop {
668            if let Some(transaction) = self.high_priority.pop_front() {
669                self.telemetry.high_prio_queue_removals().increment(1);
670
671                debug!(
672                    high_prio_queue_len = self.high_priority.len(),
673                    "Dequeued pending transaction from high-priority queue."
674                );
675                return Some(transaction);
676            }
677
678            let pop_result = self.low_priority.pop().await;
679
680            let entries_dropped = self.low_priority.take_persisted_entries_dropped();
681            if entries_dropped > 0 {
682                self.telemetry
683                    .low_prio_queue_entries_dropped()
684                    .increment(entries_dropped);
685            }
686
687            match pop_result {
688                Ok(Some(transaction)) => {
689                    self.telemetry.low_prio_queue_removals().increment(1);
690                    self.record_retry_queue_size();
691
692                    debug!(
693                        low_prio_queue_len = self.low_priority.len(),
694                        "Dequeued pending transaction from low-priority queue."
695                    );
696                    return Some(transaction);
697                }
698                Ok(None) => {
699                    self.record_retry_queue_size();
700                    return None;
701                }
702                Err(e) => {
703                    error!(error = %e, "Failed to pop transaction from low-priority queue.");
704                    continue;
705                }
706            }
707        }
708    }
709
710    /// Flushes all transactions and finalizes the queue.
711    ///
712    /// This will flush any pending high-priority transactions to the low-priority queue, and then flush the
713    /// low-priority queue, which will persist any transactions that are still in the queue to disk if the retry queue
714    /// has disk persistence enabled.
715    ///
716    /// If disk persistence is not enabled, all pending transactions will be dropped.
717    ///
718    /// # Errors
719    ///
720    /// If an error occurs flushing transactions to the low-priority queue, or occurs while flushing the low-priority
721    /// queue itself, an error will be returned.
722    pub async fn flush(mut self) -> Result<PushResult, GenericError> {
723        let mut push_result = PushResult::default();
724
725        // Push all high-priority transactions into the low-priority queue.
726        while let Some(transaction) = self.high_priority.pop_front() {
727            self.telemetry.high_prio_queue_removals().increment(1);
728
729            let subpush_result = self.low_priority.push(transaction).await?;
730            self.telemetry.low_prio_queue_insertions().increment(1);
731            self.record_retry_queue_size();
732
733            push_result.merge(subpush_result);
734        }
735
736        // Flush the low-priority queue.
737        let flush_result = self.low_priority.flush().await?;
738        push_result.merge(flush_result);
739
740        Ok(push_result)
741    }
742
743    fn record_retry_queue_size(&self) {
744        self.telemetry.record_retry_queue_size(self.low_priority.len());
745    }
746
747    fn record_incoming_transaction_size(&mut self, bytes: u64) {
748        self.record_incoming_transaction_size_at(bytes, get_unix_timestamp());
749    }

751    fn record_incoming_transaction_size_at(&mut self, bytes: u64, now_secs: u64) {
752        let bytes_per_sec = self.incoming_bytes_per_sec.record(now_secs, bytes);
753        self.telemetry.record_retry_queue_bytes_per_sec(bytes_per_sec);
754    }

802        init_tls_crypto_provider();
803
804        let CertifiedKey { cert, signing_key } = generate_simple_self_signed(["localhost".to_string()]).unwrap();
805        let cert_chain = vec![cert.der().clone()];
806        let key = PrivateKeyDer::Pkcs8(PrivatePkcs8KeyDer::from(signing_key.serialize_der()));
807        let server_config = ServerConfig::builder()
808            .with_no_client_auth()
809            .with_single_cert(cert_chain, key)

811        let acceptor = TlsAcceptor::from(Arc::new(server_config));
812

814        let port = listener.local_addr().unwrap().port();
815        let (request_tx, request_rx) = mpsc::channel(4);
816
817        tokio::spawn(async move {
818            while let Ok((stream, _)) = listener.accept().await {
819                let acceptor = acceptor.clone();
820                let request_tx = request_tx.clone();
821
822                tokio::spawn(async move {
823                    let mut stream = match acceptor.accept(stream).await {
824                        Ok(stream) => stream,
825                        Err(_) => return,
826                    };
827
828                    let mut request = Vec::new();
829                    let mut buf = [0u8; 1024];
830                    loop {
831                        match stream.read(&mut buf).await {
832                            Ok(0) => return,
833                            Ok(n) => {
834                                request.extend_from_slice(&buf[..n]);
835                                if request.windows(4).any(|window| window == b"\r\n\r\n") {

841                    }

844                    let _ = request_tx.send(request).await;
845                    let _ = stream
846                        .write_all(b"HTTP/1.1 200 OK\r\nContent-Length: 0\r\nConnection: close\r\n\r\n")
847                        .await;
848                    let _ = stream.shutdown().await;
