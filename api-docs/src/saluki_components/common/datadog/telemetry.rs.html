15/// Component-specific telemetry.
16///
17/// This type covers high-level component telemetry, such as events/bytes sent, tailored to the Datadog destinations and
18/// the commonalities between them.
19#[derive(Clone)]
20pub struct ComponentTelemetry {
21    builder: MetricsBuilder,
22    events_sent: Counter,
23    events_sent_batch_size: Histogram,
24    bytes_sent: Counter,
25    data_points_sent_by_domain: Arc<Mutex<HashMap<String, Gauge>>>,
26    data_points_dropped_by_domain: Arc<Mutex<HashMap<String, Gauge>>>,
27    events_dropped_http: Counter,
28    events_dropped_encoder: Counter,
29    events_dropped_queue: Counter,
30    items_dropped_total: Counter,
31    http_failed_send: Counter,
32    sent_request_errors: Counter,
33    http_errors_by_code: Arc<Mutex<HashMap<StatusCode, Counter>>>,
34}

36impl ComponentTelemetry {
37    /// Creates a new `ComponentTelemetry` instance with default tags derived from the given component context.
38    pub fn from_builder(builder: &MetricsBuilder) -> Self {
39        Self {
40            builder: builder.clone(),
41            events_sent: builder.register_counter("component_events_sent_total"),
42            events_sent_batch_size: builder.register_trace_histogram("component_events_sent_batch_size"),
43            bytes_sent: builder.register_counter("component_bytes_sent_total"),
44            data_points_sent_by_domain: Arc::new(Mutex::new(HashMap::new())),
45            data_points_dropped_by_domain: Arc::new(Mutex::new(HashMap::new())),
46            events_dropped_http: builder.register_counter_with_tags(
47                "component_events_dropped_total",
48                ["intentional:false", "drop_reason:http_failure"],
49            ),
50            events_dropped_encoder: builder.register_counter_with_tags(
51                "component_events_dropped_total",
52                ["intentional:false", "drop_reason:encoder_failure"],
53            ),
54            events_dropped_queue: builder.register_counter_with_tags(
55                "component_events_dropped_total",
56                ["intentional:true", "drop_reason:queue_limit"],
57            ),
58            items_dropped_total: builder.register_counter("component_items_dropped_total"),
59            http_failed_send: builder.register_counter_with_tags("component_errors_total", ["error_type:http_send"]),
60            sent_request_errors: builder.register_counter_with_tags(
61                NETWORK_HTTP_REQUESTS_ERRORS_TOTAL,
62                [
63                    ("error_type", ERROR_TYPE_SENT_REQUEST),
64                    ("error_scope", ERROR_SCOPE_TRANSACTION),
65                ],
66            ),
67            http_errors_by_code: Arc::new(Mutex::new(HashMap::new())),
68        }
69    }

71    /// Returns a reference to the "bytes sent" counter.
72    pub fn bytes_sent(&self) -> &Counter {
73        &self.bytes_sent
74    }
75
76    /// Returns a reference to the "events dropped (encoder)" counter.
77    pub fn events_dropped_encoder(&self) -> &Counter {
78        &self.events_dropped_encoder
79    }

81    /// Tracks a successful transaction.
82    pub fn track_successful_transaction(&self, metadata: &Metadata, domain: &str) {
83        self.events_sent.increment(metadata.event_count as u64);
84        self.events_sent_batch_size.record(metadata.event_count as f64);
85        self.track_data_points_sent(domain, metadata.data_point_count as u64);
86    }
87
88    /// Tracks sent metric data points.
89    pub fn track_data_points_sent(&self, domain: &str, data_point_count: u64) {
90        if data_point_count == 0 {
91            return;
92        }
93
94        let mut gauges = self.data_points_sent_by_domain.lock().unwrap();
95        let gauge = gauges.entry(domain.to_string()).or_insert_with(|| {
96            self.builder
97                .register_gauge_with_tags("component_data_points_sent_total", [("domain", domain.to_string())])
98        });
99        gauge.increment(data_point_count as f64);
100    }
101
102    /// Tracks dropped metric data points.
103    pub fn track_data_points_dropped(&self, domain: &str, data_point_count: u64) {
104        if data_point_count == 0 {
105            return;
106        }
107
108        let mut gauges = self.data_points_dropped_by_domain.lock().unwrap();
109        let gauge = gauges.entry(domain.to_string()).or_insert_with(|| {
110            self.builder
111                .register_gauge_with_tags("component_data_points_dropped_total", [("domain", domain.to_string())])
112        });
113        gauge.increment(data_point_count as f64);
114    }
115
116    /// Tracks a failed transaction.
117    ///
118    /// When `status` is `Some`, the metric is additionally emitted with an `error_code` tag for the specific status.
119    pub fn track_failed_transaction(&self, metadata: &Metadata, status: Option<StatusCode>) {
120        match status {
121            None => self.http_failed_send.increment(1),
122            Some(status) => {
123                let mut map = self.http_errors_by_code.lock().unwrap();
124                let counter = map.entry(status).or_insert_with(|| {
125                    self.builder.register_counter_with_tags(
126                        "component_errors_total",
127                        [
128                            ("error_type", "http_send".to_string()),
129                            ("error_code", status.as_str().to_string()),
130                        ],
131                    )
132                });
133                counter.increment(1);
134            }
135        }
136        self.events_dropped_http.increment(metadata.event_count as u64);
137    }
138
139    /// Tracks a permanently failed transaction.
140    pub fn track_permanently_failed_transaction(&self, metadata: &Metadata, status: Option<StatusCode>, domain: &str) {
141        self.track_failed_transaction(metadata, status);
142        self.track_data_points_dropped(domain, metadata.data_point_count as u64);
143    }
144
145    /// Tracks a failure before a transaction is submitted to the HTTP client.
146    pub fn track_sent_request_error(&self) {
147        self.sent_request_errors.increment(1);
148    }
149
150    /// Tracks dropped items from queue eviction.
151    pub fn track_dropped_items(&self, item_count: u64) {
152        self.items_dropped_total.increment(item_count);
153    }
154
155    /// Tracks dropped events from queue eviction.
156    pub fn track_dropped_events(&self, event_count: u64) {
157        self.events_dropped_queue.increment(event_count);
158    }
159}

176/// Shared transaction queue telemetry.
177#[derive(Clone)]
178pub struct SharedTransactionQueueTelemetry {
179    inner: Arc<Mutex<SharedTransactionQueueTelemetryInner>>,
180}
181
182struct SharedTransactionQueueTelemetryInner {
183    per_endpoint: FastHashMap<MetaString, RetryQueueStats>,
184    retry_queue_size: Gauge,
185    retry_queue_bytes_per_sec: Gauge,
186}
187
188#[derive(Default)]
189struct RetryQueueStats {
190    size: usize,
191    bytes_per_sec: f64,
192}
193
194impl SharedTransactionQueueTelemetry {
195    /// Creates a new `SharedTransactionQueueTelemetry` instance.
196    pub fn from_builder(builder: &MetricsBuilder) -> Self {
197        Self {
198            inner: Arc::new(Mutex::new(SharedTransactionQueueTelemetryInner {
199                per_endpoint: FastHashMap::default(),
200                retry_queue_size: builder.register_gauge("network_http_retry_queue_size"),
201                retry_queue_bytes_per_sec: builder.register_gauge("network_http_retry_queue_bytes_per_sec"),
202            })),
203        }
204    }
205
206    fn record_retry_queue_size(&self, endpoint_id: &MetaString, len: usize) {
207        let mut inner = self.inner.lock().unwrap();
208        inner.per_endpoint.entry(endpoint_id.clone()).or_default().size = len;
209
210        let total_size = inner.per_endpoint.values().map(|stats| stats.size).sum::<usize>();
211        inner.retry_queue_size.set(total_size as f64);
212    }
213
214    fn record_retry_queue_bytes_per_sec(&self, endpoint_id: &MetaString, bytes_per_sec: f64) {
215        let mut inner = self.inner.lock().unwrap();
216        inner.per_endpoint.entry(endpoint_id.clone()).or_default().bytes_per_sec = bytes_per_sec;
217
218        let total_bytes_per_sec = inner
219            .per_endpoint
220            .values()
221            .map(|stats| stats.bytes_per_sec)
222            .sum::<f64>();
223        inner.retry_queue_bytes_per_sec.set(total_bytes_per_sec);
224    }
225
226    #[cfg(test)]
227    pub(crate) fn aggregate_snapshot(&self) -> (usize, f64) {
228        let inner = self.inner.lock().unwrap();
229        let total_size = inner.per_endpoint.values().map(|stats| stats.size).sum::<usize>();
230        let total_bytes_per_sec = inner
231            .per_endpoint
232            .values()
233            .map(|stats| stats.bytes_per_sec)
234            .sum::<f64>();
235        (total_size, total_bytes_per_sec)
236    }
237}

287#[cfg(test)]
288mod tests {
289    use super::*;
290
291    #[test]
292    fn shared_transaction_queue_telemetry_aggregates_endpoint_snapshots() {
293        let builder = MetricsBuilder::default();
294        let shared = SharedTransactionQueueTelemetry::from_builder(&builder);
295        let first = TransactionQueueTelemetry::from_builder(&builder, "https://one.example", shared.clone());
296        let second = TransactionQueueTelemetry::from_builder(&builder, "https://two.example", shared.clone());
297
298        first.record_retry_queue_size(3);
299        first.record_retry_queue_bytes_per_sec(10.0);
300        second.record_retry_queue_size(5);
301        second.record_retry_queue_bytes_per_sec(2.5);
302
303        assert_eq!(shared.aggregate_snapshot(), (8, 12.5));
304
305        first.record_retry_queue_size(1);
306        first.record_retry_queue_bytes_per_sec(4.0);
307
308        assert_eq!(shared.aggregate_snapshot(), (6, 6.5));
309    }
