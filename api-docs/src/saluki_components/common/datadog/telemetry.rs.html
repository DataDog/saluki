37#[derive(Clone)]
38pub(super) struct TransactionRetryCounters {
39    retries: Counter,
40    requeued: Counter,
41}
42
43impl TransactionRetryCounters {
44    /// Tracks a completed retry dispatch.
45    pub(super) fn increment_retries(&self) {
46        self.retries.increment(1);
47    }
48
49    /// Tracks a retry that was requeued before dispatch.
50    pub(super) fn increment_requeued(&self) {
51        self.requeued.increment(1);
52    }
53}
54
55impl TransactionRetryTelemetry {
56    /// Creates retry telemetry for a fixed intake domain.
57    pub(super) fn from_builder(builder: &MetricsBuilder, domain: &str) -> Self {
58        Self {
59            builder: builder.clone(),
60            domain: MetaString::from(domain),
61            counters_by_endpoint: FastHashMap::default(),
62        }
63    }
64
65    /// Returns cached retry counter handles for the logical endpoint, registering them on first use.
66    pub(super) fn counters_for(&mut self, endpoint: &str) -> TransactionRetryCounters {
67        if let Some(counters) = self.counters_by_endpoint.get(endpoint) {
68            return counters.clone();
69        }
70
71        let endpoint = MetaString::from(endpoint);
72        let tags = [("domain", self.domain.to_string()), ("endpoint", endpoint.to_string())];
73        let counters = TransactionRetryCounters {
74            retries: self
75                .builder
76                .register_counter_with_tags(NETWORK_HTTP_REQUESTS_RETRIES_TOTAL, tags.clone()),
77            requeued: self
78                .builder
79                .register_counter_with_tags(NETWORK_HTTP_REQUESTS_REQUEUED_TOTAL, tags),
80        };
81        self.counters_by_endpoint.insert(endpoint, counters.clone());
82        counters
83    }
84}

167    /// Tracks a successful transaction.
168    pub fn track_successful_transaction(&self, metadata: &Metadata, domain: &str) {
169        self.events_sent.increment(metadata.event_count as u64);
170        self.events_sent_batch_size.record(metadata.event_count as f64);
171        self.track_data_points_sent(domain, metadata.data_point_count as u64);

174    /// Tracks sent metric data points.
175    pub fn track_data_points_sent(&self, domain: &str, data_point_count: u64) {
176        if data_point_count == 0 {
177            return;
178        }
179
180        let mut gauges = self.data_points_sent_by_domain.lock().unwrap();
181        let gauge = gauges.entry(domain.to_string()).or_insert_with(|| {
182            self.builder
183                .register_gauge_with_tags("component_data_points_sent_total", [("domain", domain.to_string())])
184        });
185        gauge.increment(data_point_count as f64);
186    }
187
188    /// Tracks dropped metric data points.
189    pub fn track_data_points_dropped(&self, domain: &str, data_point_count: u64) {
190        if data_point_count == 0 {
191            return;
192        }
193
194        let mut gauges = self.data_points_dropped_by_domain.lock().unwrap();
195        let gauge = gauges.entry(domain.to_string()).or_insert_with(|| {
196            self.builder
197                .register_gauge_with_tags("component_data_points_dropped_total", [("domain", domain.to_string())])
198        });
199        gauge.increment(data_point_count as f64);
200    }

202    /// Tracks a failed transaction.
203    ///
204    /// When `status` is `Some`, the metric is additionally emitted with an `error_code` tag for the specific status.
205    pub fn track_failed_transaction(&self, metadata: &Metadata, status: Option<StatusCode>) {
206        match status {
207            None => self.http_failed_send.increment(1),
208            Some(status) => {
209                let mut map = self.http_errors_by_code.lock().unwrap();
210                let counter = map.entry(status).or_insert_with(|| {
211                    self.builder.register_counter_with_tags(
212                        "component_errors_total",
213                        [
214                            ("error_type", "http_send".to_string()),
215                            ("error_code", status.as_str().to_string()),
216                        ],
217                    )
218                });
219                counter.increment(1);
220            }
221        }
222        self.events_dropped_http.increment(metadata.event_count as u64);
223    }
224
225    /// Tracks a permanently failed transaction.
226    pub fn track_permanently_failed_transaction(&self, metadata: &Metadata, status: Option<StatusCode>, domain: &str) {
227        self.track_failed_transaction(metadata, status);
228        self.track_data_points_dropped(domain, metadata.data_point_count as u64);
229    }
230
231    /// Tracks a failure before a transaction is submitted to the HTTP client.
232    pub fn track_sent_request_error(&self) {
233        self.sent_request_errors.increment(1);
234    }
235
236    /// Tracks dropped items from queue eviction.
237    pub fn track_dropped_items(&self, item_count: u64) {
238        self.items_dropped_total.increment(item_count);
239    }
240
241    /// Tracks dropped events from queue eviction.
242    pub fn track_dropped_events(&self, event_count: u64) {
243        self.events_dropped_queue.increment(event_count);
244    }

268struct SharedTransactionQueueTelemetryInner {
269    builder: MetricsBuilder,
270    per_endpoint: FastHashMap<MetaString, RetryQueueStats>,
271    capacity: RetryQueueCapacityAggregator,
272    retry_queue_size: Gauge,
273    retry_queue_bytes_per_sec: Gauge,
274    retry_queue_bytes_per_sec_by_domain: FastHashMap<String, Gauge>,
275    retry_queue_capacity_secs_by_domain: FastHashMap<String, Gauge>,
276    retry_queue_capacity_bytes_by_domain: FastHashMap<String, Gauge>,
277}
278
279#[derive(Default)]
280struct RetryQueueStats {
281    size: usize,
282    bytes_per_sec: f64,
283}
284
285impl SharedTransactionQueueTelemetryInner {
286    fn from_builder(builder: &MetricsBuilder) -> Self {
287        Self {
288            builder: builder.clone(),
289            per_endpoint: FastHashMap::default(),
290            capacity: RetryQueueCapacityAggregator::new(),
291            retry_queue_size: builder.register_gauge("network_http_retry_queue_size"),
292            retry_queue_bytes_per_sec: builder.register_gauge("network_http_retry_queue_bytes_per_sec"),
293            retry_queue_bytes_per_sec_by_domain: FastHashMap::default(),
294            retry_queue_capacity_secs_by_domain: FastHashMap::default(),
295            retry_queue_capacity_bytes_by_domain: FastHashMap::default(),
296        }
297    }
298
299    fn record_size(&mut self, endpoint_id: &MetaString, len: usize) {
300        self.per_endpoint.entry(endpoint_id.clone()).or_default().size = len;
301
302        let total_size = self.per_endpoint.values().map(|stats| stats.size).sum::<usize>();
303        self.retry_queue_size.set(total_size as f64);
304    }
305
306    fn record_bytes_per_sec(&mut self, endpoint_id: &MetaString, bytes_per_sec: f64) {
307        self.per_endpoint.entry(endpoint_id.clone()).or_default().bytes_per_sec = bytes_per_sec;
308
309        let total_bytes_per_sec = self.per_endpoint.values().map(|stats| stats.bytes_per_sec).sum::<f64>();
310        self.retry_queue_bytes_per_sec.set(total_bytes_per_sec);
311    }
312
313    fn record_capacity_stats(
314        &mut self, endpoint_id: &MetaString, domain: &MetaString, bytes_per_sec: f64, in_memory_capacity_bytes: u64,
315        disk_available_capacity_bytes: u64,
316    ) {
317        self.capacity.update_endpoint(
318            endpoint_id.clone(),
319            domain.clone(),
320            bytes_per_sec,
321            in_memory_capacity_bytes,
322            disk_available_capacity_bytes,
323        );
324        self.retry_queue_bytes_per_sec.set(self.capacity.total_bytes_per_sec());
325
326        let domain_stats = self
327            .capacity
328            .domain_stats()
329            .map(|(domain, stats)| (domain.to_string(), stats))
330            .collect::<Vec<_>>();
331
332        for (domain, stats) in domain_stats {
333            let bytes_per_sec = stats.bytes_per_sec;
334            let capacity_secs = stats.capacity_secs;
335            let capacity_bytes = stats.capacity_bytes;
336
337            let bytes_per_sec_gauge = self
338                .retry_queue_bytes_per_sec_by_domain
339                .entry(domain.clone())
340                .or_insert_with(|| {
341                    self.builder.register_gauge_with_tags(
342                        "network_http_retry_queue_bytes_per_sec",
343                        [("domain", domain.clone())],
344                    )
345                });
346            bytes_per_sec_gauge.set(bytes_per_sec);
347
348            let capacity_secs_gauge = self
349                .retry_queue_capacity_secs_by_domain
350                .entry(domain.clone())
351                .or_insert_with(|| {
352                    self.builder.register_gauge_with_tags(
353                        "network_http_retry_queue_capacity_secs",
354                        [("domain", domain.clone())],
355                    )
356                });
357            capacity_secs_gauge.set(capacity_secs);
358
359            let capacity_bytes_gauge = self
360                .retry_queue_capacity_bytes_by_domain
361                .entry(domain.clone())
362                .or_insert_with(|| {
363                    self.builder
364                        .register_gauge_with_tags("network_http_retry_queue_capacity_bytes", [("domain", domain)])
365                });
366            capacity_bytes_gauge.set(capacity_bytes);
367        }
368    }
369}

471    fn aggregate_snapshot(shared: &SharedTransactionQueueTelemetry) -> (usize, f64) {
472        let inner = shared.inner.lock().unwrap();
473        let total_size = inner.per_endpoint.values().map(|stats| stats.size).sum::<usize>();
474        let total_bytes_per_sec = inner
475            .per_endpoint
476            .values()
477            .map(|stats| stats.bytes_per_sec)
478            .sum::<f64>();
479
480        (total_size, total_bytes_per_sec)
481    }

483    fn domain_capacity_snapshot(shared: &SharedTransactionQueueTelemetry, domain: &str) -> Option<(f64, f64, f64)> {
484        let inner = shared.inner.lock().unwrap();
485        let snapshot = inner
486            .capacity
487            .domain_stats()
488            .find_map(|(stats_domain, stats)| (stats_domain == domain).then_some(stats))?;
489
490        Some((snapshot.bytes_per_sec, snapshot.capacity_secs, snapshot.capacity_bytes))
491    }
