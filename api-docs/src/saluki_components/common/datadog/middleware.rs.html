1use http::{
2    uri::{Authority, Scheme},
3    HeaderName, HeaderValue, Request, Uri,

8static DD_AGENT_VERSION_HEADER: HeaderName = HeaderName::from_static("dd-agent-version");
9static DD_API_KEY_HEADER: HeaderName = HeaderName::from_static("dd-api-key");
10
11/// Builds a middleware function that will update the request's URI to use the resolved endpoint's URI, and add the API
12/// key as a header.
13///
14/// The request's URI must already container a relative path component. Any existing scheme, host, and port components
15/// will be overridden by the resolved endpoint's URI.
16pub fn for_resolved_endpoint<B>(mut endpoint: ResolvedEndpoint) -> impl FnMut(Request<B>) -> Request<B> + Clone {
17    let new_uri_authority = Authority::try_from(endpoint.endpoint().authority())
18        .expect("should not fail to construct new endpoint authority");
19    let new_uri_scheme =
20        Scheme::try_from(endpoint.endpoint().scheme()).expect("should not fail to construct new endpoint scheme");
21    move |mut request| {
22        let path_and_query = request.uri().path_and_query().expect("request path must exist").clone();
23
24        let authority = if path_and_query.as_str().starts_with("/api/v2/logs") {
25            endpoint
26                .logs_authority()
27                .cloned()
28                .unwrap_or_else(|| new_uri_authority.clone())
29        } else {
30            new_uri_authority.clone()
31        };
32        // Build an updated URI by taking the endpoint URL and slapping the request's URI path on the end of it.
33        let new_uri = Uri::builder()
34            .scheme(new_uri_scheme.clone())
35            .authority(authority)
36            .path_and_query(path_and_query)
37            .build()
38            .expect("should not fail to construct new URI");
39        let api_key = endpoint.api_key();
40        let api_key_value = HeaderValue::from_str(api_key).expect("should not fail to construct API key header value");
41        *request.uri_mut() = new_uri;
42
43        // Add the API key as a header.
44        request
45            .headers_mut()
46            .insert(DD_API_KEY_HEADER.clone(), api_key_value.clone());
47
48        request
49    }
50}

52/// Builds a middleware function that adds request headers indicating the version of the data plane making the request.
53///
54/// Adds the `dd-agent-version` header with the version of the data plane (`x.y.z`), and the `User-Agent` header with
55/// the name of data plane and the version, such as `agent-data-plane/x.y.z`.`
56pub fn with_version_info<B>() -> impl Fn(Request<B>) -> Request<B> + Clone {
57    let app_details = saluki_metadata::get_app_details();
58    let formatted_full_name = app_details
59        .full_name()
60        .replace(" ", "-")
61        .replace("_", "-")
62        .to_lowercase();
63
64    let agent_version_header_value = HeaderValue::from_static(app_details.version().raw());
65    let raw_user_agent_header_value = format!("{}/{}", formatted_full_name, app_details.version().raw());
66    let user_agent_header_value = HeaderValue::from_maybe_shared(raw_user_agent_header_value)
67        .expect("should not fail to construct User-Agent header value");
68
69    move |mut request| {
70        request
71            .headers_mut()
72            .insert(DD_AGENT_VERSION_HEADER.clone(), agent_version_header_value.clone());
73        request
74            .headers_mut()
75            .insert(http::header::USER_AGENT, user_agent_header_value.clone());
76        request
77    }
78}
