12/// Builds a middleware function that will update the request's URI to use the resolved endpoint's URI, and add the API
13/// key as a header.
14///
15/// The request's URI must already container a relative path component. Any existing scheme, host, and port components
16/// will be overridden by the resolved endpoint's URI.
17pub fn for_resolved_endpoint<B>(mut endpoint: ResolvedEndpoint) -> impl FnMut(Request<B>) -> Request<B> + Clone {
18    let new_uri_authority = Authority::try_from(endpoint.endpoint().authority())
19        .expect("should not fail to construct new endpoint authority");
20    let new_uri_scheme =
21        Scheme::try_from(endpoint.endpoint().scheme()).expect("should not fail to construct new endpoint scheme");
22    move |mut request| {
23        let path_and_query = request.uri().path_and_query().expect("request path must exist").clone();
24
25        let authority = if path_and_query.as_str().starts_with("/api/v2/logs") {
26            endpoint
27                .logs_authority()
28                .cloned()
29                .unwrap_or_else(|| new_uri_authority.clone())
30        } else if path_and_query.as_str().starts_with("/api/v0.2/traces")
31            || path_and_query.as_str().starts_with("/api/v0.2/stats")
32        {
33            endpoint
34                .traces_authority()
35                .cloned()
36                .unwrap_or_else(|| new_uri_authority.clone())
37        } else {
38            new_uri_authority.clone()
39        };
40        // Build an updated URI by taking the endpoint URL and slapping the request's URI path on the end of it.
41        let new_uri = Uri::builder()
42            .scheme(new_uri_scheme.clone())
43            .authority(authority)
44            .path_and_query(path_and_query)
45            .build()
46            .expect("should not fail to construct new URI");
47        let api_key = endpoint.api_key();
48        let api_key_value = HeaderValue::from_str(api_key).expect("should not fail to construct API key header value");
49        *request.uri_mut() = new_uri;
50
51        // Add the API key as a header.
52        request
53            .headers_mut()
54            .insert(DD_API_KEY_HEADER.clone(), api_key_value.clone());
55
56        request
57    }
58}

60/// Builds a middleware function that optionally signals backend support for arbitrary tag values.
61///
62/// When enabled, adds `Allow-Arbitrary-Tag-Value: true` to every outbound request. This mirrors the Datadog Agent's
63/// forwarder behavior for `allow_arbitrary_tags`; it does not perform local tag validation.
64pub fn with_allow_arbitrary_tags<B>(enabled: bool) -> impl Fn(Request<B>) -> Request<B> + Clone {
65    move |mut request| {
66        if enabled {
67            request.headers_mut().insert(
68                ALLOW_ARBITRARY_TAG_VALUE_HEADER.clone(),
69                HeaderValue::from_static("true"),
70            );
71        }
72
73        request
74    }
75}

77/// Builds a middleware function that adds request headers indicating the version of the data plane making the request.
78///
79/// Adds the `dd-agent-version` header with the version of the data plane (`x.y.z`), and the `User-Agent` header with
80/// the name of data plane and the version, such as `agent-data-plane/x.y.z`.`
81pub fn with_version_info<B>() -> impl Fn(Request<B>) -> Request<B> + Clone {
82    let app_details = saluki_metadata::get_app_details();
83    let formatted_full_name = app_details
84        .full_name()
85        .replace(" ", "-")
86        .replace("_", "-")
87        .to_lowercase();
88
89    let agent_version_header_value = HeaderValue::from_static(app_details.version().raw());
90    let raw_user_agent_header_value = format!("{}/{}", formatted_full_name, app_details.version().raw());
91    let user_agent_header_value = HeaderValue::from_maybe_shared(raw_user_agent_header_value)
92        .expect("should not fail to construct User-Agent header value");
93
94    move |mut request| {
95        request
96            .headers_mut()
97            .insert(DD_AGENT_VERSION_HEADER.clone(), agent_version_header_value.clone());
98        request
99            .headers_mut()
100            .insert(http::header::USER_AGENT, user_agent_header_value.clone());
101        request
102    }
103}

105#[cfg(test)]
106mod tests {
107    use http::Request;
108
109    use super::with_allow_arbitrary_tags;
110
111    #[test]
112    fn allow_arbitrary_tags_header_is_added_when_enabled() {
113        let middleware = with_allow_arbitrary_tags(true);
114        let request = Request::builder()
115            .uri("/api/v1/series")
116            .body(())
117            .expect("request should build");
118
119        let request = middleware(request);
120
121        assert_eq!(
122            request.headers().get("allow-arbitrary-tag-value"),
123            Some(&http::HeaderValue::from_static("true"))
124        );
125    }
126
127    #[test]
128    fn allow_arbitrary_tags_header_is_not_added_when_disabled() {
129        let middleware = with_allow_arbitrary_tags(false);
130        let request = Request::builder()
131            .uri("/api/v1/series")
132            .body(())
133            .expect("request should build");
134
135        let request = middleware(request);
136
137        assert!(request.headers().get("allow-arbitrary-tag-value").is_none());
138    }
