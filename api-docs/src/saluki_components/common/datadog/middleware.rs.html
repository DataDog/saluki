11/// Builds a middleware function that will update the request's URI to use the resolved endpoint's URI, and add the API
12/// key as a header.
13///
14/// The request's URI must already container a relative path component. Any existing scheme, host, and port components
15/// will be overridden by the resolved endpoint's URI.
16pub fn for_resolved_endpoint<B>(mut endpoint: ResolvedEndpoint) -> impl FnMut(Request<B>) -> Request<B> + Clone {
17    let new_uri_authority = Authority::try_from(endpoint.endpoint().authority())
18        .expect("should not fail to construct new endpoint authority");
19    let new_uri_scheme =
20        Scheme::try_from(endpoint.endpoint().scheme()).expect("should not fail to construct new endpoint scheme");
21    move |mut request| {
22        let path_and_query = request.uri().path_and_query().expect("request path must exist").clone();
23
24        let authority = if path_and_query.as_str().starts_with("/api/v2/logs") {
25            endpoint
26                .logs_authority()
27                .cloned()
28                .unwrap_or_else(|| new_uri_authority.clone())
29        } else if path_and_query.as_str().starts_with("/api/v0.2/traces") {
30            endpoint
31                .traces_authority()
32                .cloned()
33                .unwrap_or_else(|| new_uri_authority.clone())
34        } else {
35            new_uri_authority.clone()
36        };
37        // Build an updated URI by taking the endpoint URL and slapping the request's URI path on the end of it.
38        let new_uri = Uri::builder()
39            .scheme(new_uri_scheme.clone())
40            .authority(authority)
41            .path_and_query(path_and_query)
42            .build()
43            .expect("should not fail to construct new URI");
44        let api_key = endpoint.api_key();
45        let api_key_value = HeaderValue::from_str(api_key).expect("should not fail to construct API key header value");
46        *request.uri_mut() = new_uri;
47
48        // Add the API key as a header.
49        request
50            .headers_mut()
51            .insert(DD_API_KEY_HEADER.clone(), api_key_value.clone());
52
53        request
54    }
55}

57/// Builds a middleware function that adds request headers indicating the version of the data plane making the request.
58///
59/// Adds the `dd-agent-version` header with the version of the data plane (`x.y.z`), and the `User-Agent` header with
60/// the name of data plane and the version, such as `agent-data-plane/x.y.z`.`
61pub fn with_version_info<B>() -> impl Fn(Request<B>) -> Request<B> + Clone {
62    let app_details = saluki_metadata::get_app_details();
63    let formatted_full_name = app_details
64        .full_name()
65        .replace(" ", "-")
66        .replace("_", "-")
67        .to_lowercase();
68
69    let agent_version_header_value = HeaderValue::from_static(app_details.version().raw());
70    let raw_user_agent_header_value = format!("{}/{}", formatted_full_name, app_details.version().raw());
71    let user_agent_header_value = HeaderValue::from_maybe_shared(raw_user_agent_header_value)
72        .expect("should not fail to construct User-Agent header value");
73
74    move |mut request| {
75        request
76            .headers_mut()
77            .insert(DD_AGENT_VERSION_HEADER.clone(), agent_version_header_value.clone());
78        request
79            .headers_mut()
80            .insert(http::header::USER_AGENT, user_agent_header_value.clone());
81        request
82    }
