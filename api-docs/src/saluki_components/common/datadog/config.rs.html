22/// Forwarder configuration based on the Datadog Agent's forwarder configuration.
23///
24/// This adapter provides a simple way to utilize the existing configuration values that are passed to the Datadog
25/// Agent, which are used to control the behavior of its forwarder, such as retries and concurrency, in conjunction with
26/// with existing primitives, as such retry policies in [`saluki_io::util::retry`].
27#[derive(Clone, Deserialize)]
28pub struct ForwarderConfiguration {
29    /// Maximum number of concurrent requests for an individual endpoint.
30    ///
31    /// Defaults to 1.
32    #[serde(default = "default_endpoint_concurrency", rename = "forwarder_num_workers")]
33    endpoint_concurrency: usize,
34
35    /// Request timeout, in seconds.
36    ///
37    /// Defaults to 20 seconds.
38    #[serde(default = "default_request_timeout_secs", rename = "forwarder_timeout")]
39    request_timeout_secs: u64,
40
41    /// Maximum number of pending requests for an individual endpoint.
42    ///
43    /// Defaults to 16.
44    #[serde(default = "default_endpoint_buffer_size", rename = "forwarder_high_prio_buffer_size")]
45    endpoint_buffer_size: usize,
46
47    /// Endpoint configuration.
48    #[serde(flatten)]
49    pub(crate) endpoint: EndpointConfiguration,
50
51    /// Retry configuration.
52    #[serde(flatten)]
53    retry: RetryConfiguration,
54
55    /// Proxy configuration.
56    #[serde(flatten)]
57    proxy: Option<ProxyConfiguration>,
58
59    /// Connection reset interval, in seconds.
60    ///
61    /// Defaults to 0.
62    #[serde(
63        default = "default_forwarder_connection_reset_interval",
64        rename = "forwarder_connection_reset_interval"
65    )]
66    connection_reset_interval_secs: u64,
67}
68
69impl ForwarderConfiguration {
70    /// Returns the maximum number of concurrent requests for an individual endpoint.
71    pub const fn endpoint_concurrency(&self) -> usize {
72        self.endpoint_concurrency
73    }
74
75    /// Returns the request timeout.
76    pub const fn request_timeout(&self) -> Duration {
77        Duration::from_secs(self.request_timeout_secs)
78    }
79
80    /// Returns the maximum number of pending requests for an individual endpoint.
81    pub const fn endpoint_buffer_size(&self) -> usize {
82        self.endpoint_buffer_size
83    }
84
85    /// Returns a reference to the endpoint configuration.
86    pub const fn endpoint(&self) -> &EndpointConfiguration {
87        &self.endpoint
88    }
89
90    /// Returns a mutable reference to the endpoint configuration.
91    pub fn endpoint_mut(&mut self) -> &mut EndpointConfiguration {
92        &mut self.endpoint
93    }
94
95    /// Returns a reference to the retry configuration.
96    pub const fn retry(&self) -> &RetryConfiguration {
97        &self.retry
98    }
99
100    /// Returns a reference to the proxy configuration.
101    pub const fn proxy(&self) -> &Option<ProxyConfiguration> {
102        &self.proxy
103    }
104
105    /// Returns the connection reset interval.
106    pub const fn connection_reset_interval(&self) -> Duration {
107        Duration::from_secs(self.connection_reset_interval_secs)
108    }
109}
