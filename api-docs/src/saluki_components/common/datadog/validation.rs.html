29/// Readiness decision produced by API key validation.
30#[derive(Clone, Copy, Debug, Eq, PartialEq)]
31pub(crate) enum ValidationReadiness {
32    /// At least one key is valid, or validation could not prove every key invalid.
33    Ready,
34    /// Validation proved that every known key is invalid.
35    NotReady,
36}

61    pub(crate) fn spawn(self) -> ApiKeyValidationHandle {

78/// Handle for API key validation readiness updates.
79pub(crate) struct ApiKeyValidationHandle {
80    task: JoinHandle<()>,
81    readiness_rx: Option<mpsc::Receiver<ValidationReadiness>>,
82}

84impl ApiKeyValidationHandle {
85    /// Waits until API key validation produces a readiness update.
86    pub(crate) async fn wait_for_change(&mut self) -> ValidationReadiness {
87        let Some(rx) = &mut self.readiness_rx else {
88            return future::pending().await;
89        };
90
91        match rx.recv().await {
92            Some(readiness) => readiness,
93            None => {
94                self.readiness_rx = None;
95                debug!("Datadog API key validation task stopped.");
96                future::pending().await
97            }
98        }
99    }
100
101    /// Stops the validation task.
102    pub(crate) fn abort(&self) {
103        self.task.abort();
104    }
105}

127fn spawn_validation_task(
128    endpoints: Vec<RoutableEndpoint>, client: HttpClient, live_config: Option<GenericConfiguration>,
129    interval: Duration, readiness_tx: mpsc::Sender<ValidationReadiness>,
130) -> JoinHandle<()> {
131    spawn_traced_named(
132        "dd-api-key-validation",
133        run_validation_loop(endpoints, client, live_config, interval, readiness_tx),
134    )
135}
136
137async fn run_validation_loop(
138    mut endpoints: Vec<RoutableEndpoint>, mut client: HttpClient, live_config: Option<GenericConfiguration>,
139    interval: Duration, readiness_tx: mpsc::Sender<ValidationReadiness>,
140) {
141    if !validate_and_send_readiness(&mut endpoints, &mut client, &readiness_tx).await {
142        return;
143    }
144
145    let mut interval = time::interval(interval);
146    interval.set_missed_tick_behavior(MissedTickBehavior::Delay);
147    // The startup validation above is the immediate tick.
148    interval.tick().await;
149
150    let mut config_updates_rx = live_config
151        .as_ref()
152        .and_then(GenericConfiguration::subscribe_for_updates);
153
154    loop {
155        select! {
156            _ = interval.tick() => {
157                if !validate_and_send_readiness(&mut endpoints, &mut client, &readiness_tx).await {
158                    return;
159                }
160            },
161            _ = wait_for_validation_config_change(&mut config_updates_rx) => {
162                if !validate_and_send_readiness(&mut endpoints, &mut client, &readiness_tx).await {
163                    return;
164                }
165            },
166        }
167    }
168}
169
170async fn wait_for_validation_config_change(
171    rx: &mut Option<broadcast::Receiver<saluki_config::dynamic::ConfigChangeEvent>>,
172) {
173    let Some(rx) = rx else {
174        std::future::pending::<()>().await;
175        return;
176    };
177
178    loop {
179        match rx.recv().await {
180            Ok(event) if is_validation_trigger_key(&event.key) => return,
181            Ok(_) => {}
182            Err(broadcast::error::RecvError::Lagged(_)) => return,
183            Err(broadcast::error::RecvError::Closed) => {
184                std::future::pending::<()>().await;
185                return;
186            }
187        }
188    }
189}

191fn is_validation_trigger_key(key: &str) -> bool {
192    key == "api_key"
193        || key == "multi_region_failover.api_key"
194        || key == "additional_endpoints"
195        || key.starts_with("additional_endpoints.")
196}

198async fn validate_and_send_readiness(
199    endpoints: &mut [RoutableEndpoint], client: &mut HttpClient, readiness_tx: &mpsc::Sender<ValidationReadiness>,
200) -> bool {
201    let targets = collect_validation_targets(endpoints);
202    let readiness = validate_targets(client, &targets).await;
203
204    if readiness_tx.send(readiness).await.is_err() {
205        debug!("API key validation readiness receiver dropped; stopping validation task.");
206        return false;
207    }
208
209    true
210}

212async fn validate_targets(client: &mut HttpClient, targets: &[ValidationTarget]) -> ValidationReadiness {
213    if targets.is_empty() {
214        warn!("No Datadog API keys are available for validation; marking forwarder ready.");
215        return ValidationReadiness::Ready;
216    }
217
218    let mut saw_error = false;
219
220    for target in targets {
221        match validate_target(client, target).await {
222            KeyValidationResult::Valid => return ValidationReadiness::Ready,
223            KeyValidationResult::Invalid => {}
224            KeyValidationResult::Error => saw_error = true,
225        }
226    }
227
228    if saw_error {
229        ValidationReadiness::Ready
230    } else {
231        ValidationReadiness::NotReady
232    }
233}
234
235fn collect_validation_targets(endpoints: &mut [RoutableEndpoint]) -> Vec<ValidationTarget> {
236    let mut seen = HashSet::new();
237    let mut targets = Vec::new();
238
239    for routable in endpoints {
240        // `api_key()` lazily refreshes the cached key from live configuration before validation.
241        let endpoint = routable.endpoint_mut();
242        let api_key = endpoint.api_key().trim().to_string();
243        if api_key.is_empty() {
244            continue;
245        }
246
247        let validation_base_url = validation_base_url(endpoint.endpoint());
248        let key = ValidationTargetKey {
249            validation_base_url: validation_base_url.to_string(),
250            api_key: api_key.clone(),
251        };
252
253        if seen.insert(key) {
254            targets.push(ValidationTarget {
255                endpoint: endpoint.endpoint().clone(),
256                validation_base_url,
257                api_key,
258            });
259        }
260    }
261
262    targets
263}

265async fn validate_target(client: &mut HttpClient, target: &ValidationTarget) -> KeyValidationResult {
266    if target.api_key == FAKE_API_KEY {
267        debug!(endpoint = %target.endpoint, "Treating fake Datadog API key as valid.");
268        return KeyValidationResult::Valid;
269    }
270
271    let request = match build_validation_request(target) {
272        Ok(request) => request,
273        Err(e) => {
274            debug!(endpoint = %target.endpoint, error = %e, "Could not build Datadog API key validation request.");
275            return KeyValidationResult::Error;
276        }
277    };
278
279    match client.send(request).await {
280        Ok(response) => match response.status() {
281            StatusCode::OK => {
282                debug!(endpoint = %target.endpoint, validation_endpoint = %target.validation_base_url, "Datadog API key is valid.");
283                KeyValidationResult::Valid
284            }
285            StatusCode::FORBIDDEN => {
286                warn!(endpoint = %target.endpoint, validation_endpoint = %target.validation_base_url, "Datadog API key is invalid.");
287                KeyValidationResult::Invalid
288            }
289            status => {
290                debug!(
291                    endpoint = %target.endpoint,
292                    validation_endpoint = %target.validation_base_url,
293                    %status,
294                    "Datadog API key validation returned an unexpected status."
295                );
296                KeyValidationResult::Error
297            }
298        },
299        Err(e) => {
300            debug!(
301                endpoint = %target.endpoint,
302                validation_endpoint = %target.validation_base_url,
303                error = %e,
304                "Datadog API key validation request failed."
305            );
306            KeyValidationResult::Error
307        }
308    }
309}

311fn build_validation_request(target: &ValidationTarget) -> Result<Request<Empty<Bytes>>, GenericError> {
312    let mut url = target.validation_base_url.clone();
313    url.set_path(VALIDATE_PATH);
314    url.set_query(None);
315    url.query_pairs_mut().append_pair("api_key", &target.api_key);
316
317    let uri = url
318        .as_str()
319        .parse::<Uri>()
320        .map_err(|e| generic_error!("Failed to parse validation URL as URI: {}", e))?;
321
322    Request::builder()
323        .method("GET")
324        .uri(uri)
325        .body(Empty::<Bytes>::new())
326        .map_err(|e| generic_error!("Failed to build validation request: {}", e))
327}
328
329fn validation_base_url(endpoint: &Url) -> Url {
330    let Some(host) = endpoint.host_str() else {
331        return endpoint.clone();
332    };
333
334    if let Some(matched) = DATADOG_API_DOMAIN_RE.find(host) {
335        let site = matched.as_str().trim_end_matches('.');
336        let mut api_endpoint = endpoint.clone();
337        let _ = api_endpoint.set_scheme("https");
338        let _ = api_endpoint.set_host(Some(&format!("api.{site}")));
339        api_endpoint.set_path("");
340        api_endpoint.set_query(None);
341        return api_endpoint;
342    }
343
344    endpoint.clone()
345}

596    async fn start_validation_server(status: StatusCode) -> String {
597        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
598        let port = listener.local_addr().unwrap().port();
599        let router = Router::new().route(VALIDATE_PATH, get(move || async move { status }));
600
601        tokio::spawn(async move {
602            axum::serve(listener, router).await.unwrap();
603        });
604
605        format!("http://127.0.0.1:{port}/")
606    }

608    async fn start_counting_validation_server(status: StatusCode, requests: Arc<AtomicUsize>) -> String {
609        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
610        let port = listener.local_addr().unwrap().port();
611        let router = Router::new().route(
612            VALIDATE_PATH,
613            get(move || {
