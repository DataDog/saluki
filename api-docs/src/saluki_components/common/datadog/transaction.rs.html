15/// Data type for the body of `TransactionBody<B>`.
16pub enum TransactionBodyData<B>
17where
18    B: Body,
19{
20    /// Original body data.
21    Original(B::Data),
22
23    /// Rehydrated body data.
24    Rehydrated(Bytes),
25}
26
27impl<B> Buf for TransactionBodyData<B>
28where
29    B: Body,
30{
31    fn remaining(&self) -> usize {
32        match self {
33            Self::Original(data) => data.remaining(),
34            Self::Rehydrated(data) => data.remaining(),
35        }
36    }
37
38    fn chunk(&self) -> &[u8] {
39        match self {
40            Self::Original(data) => data.chunk(),
41            Self::Rehydrated(data) => data.chunk(),
42        }
43    }
44
45    fn advance(&mut self, cnt: usize) {
46        match self {
47            Self::Original(data) => data.advance(cnt),
48            Self::Rehydrated(data) => data.advance(cnt),
49        }
50    }
51}

53/// Custom body type that can abstract over an "original" body type `B` and a rehydrated body type based on `Bytes`.
54///
55/// In order for `Transaction<B>` to support being serialized and deserialized, we need to be able to serialize and
56/// deserialize the body of the request. However, the body type `B` may not be `Serialize` or `Deserialize` itself. To
57/// compensate for this, `TransactionBody<B>` provides the necessary serialization and deserialization logic by bounding
58/// `B` in a way that ensures we can clone it and serialize it to disk, and then rehydrate it to a compatible body type
59/// after deserialization.
60#[derive(Clone, Deserialize)]
61#[serde(bound = "", from = "Vec<u8>")]
62#[pin_project(project = TransactionBodyProj)]
63pub enum TransactionBody<B> {
64    /// Original body.
65    Original(#[pin] B),
66
67    /// Rehydrated body.
68    Rehydrated(Option<Bytes>),
69}
70
71impl<B> Buf for TransactionBody<B>
72where
73    B: Buf,
74{
75    fn remaining(&self) -> usize {
76        match self {
77            Self::Original(body) => body.remaining(),
78            Self::Rehydrated(body) => body.as_ref().map_or(0, |body| body.remaining()),
79        }
80    }
81
82    fn chunk(&self) -> &[u8] {
83        match self {
84            Self::Original(body) => body.chunk(),
85            Self::Rehydrated(body) => body.as_ref().map_or(&[], |body| body.chunk()),
86        }
87    }
88
89    fn advance(&mut self, cnt: usize) {
90        match self {
91            Self::Original(body) => body.advance(cnt),
92            Self::Rehydrated(body) => match body {
93                Some(body) => body.advance(cnt),
94                None => panic!("attempted to advance a rehydrated body that was consumed"),
95            },
96        }
97    }
98}

100impl<B> Body for TransactionBody<B>
101where
102    B: Body,
103{
104    type Data = TransactionBodyData<B>;
105    type Error = B::Error;
106
107    fn poll_frame(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Result<Frame<Self::Data>, Self::Error>>> {
108        let this = self.project();
109        match this {
110            TransactionBodyProj::Original(body) => body.poll_frame(cx).map(|maybe_frame| {
111                maybe_frame.map(|res| res.map(|frame| frame.map_data(|data| TransactionBodyData::Original(data))))
112            }),
113            TransactionBodyProj::Rehydrated(body) => Poll::Ready(
114                body.take()
115                    .map(|body| Ok(Frame::data(TransactionBodyData::Rehydrated(body)))),
116            ),
117        }
118    }
119
120    fn size_hint(&self) -> http_body::SizeHint {
121        match self {
122            Self::Original(body) => body.size_hint(),
123            Self::Rehydrated(body) => match body.as_ref() {
124                Some(body) => http_body::SizeHint::with_exact(body.len() as u64),
125                None => http_body::SizeHint::with_exact(0),
126            },
127        }
128    }
129}

131impl<B> Serialize for TransactionBody<B>
132where
133    B: Buf + Clone,
134{
135    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
136    where
137        S: Serializer,
138    {
139        match self {
140            Self::Original(body) => {
141                // We have to clone the body and then run through all of the chunks to serialize them.
142                //
143                // TODO: It's not clear to me if this is actually as optimized as we can make it, in terms of
144                // serialization.
145                let mut new_body = body.clone();
146                let mut seq = serializer.serialize_seq(Some(new_body.remaining()))?;
147                while new_body.remaining() > 0 {
148                    let chunk = new_body.chunk();
149                    for b in chunk {
150                        seq.serialize_element(b)?;
151                    }
152
153                    new_body.advance(chunk.len());
154                }
155
156                seq.end()
157            }
158            Self::Rehydrated(body) => match body.as_ref() {
159                Some(body) => body.serialize(serializer),
160                None => Err(serde::ser::Error::custom(
161                    "attempted to serialize a rehydrated body that was consumed",
162                )),
163            },
164        }
165    }
166}

168impl<B> From<Vec<u8>> for TransactionBody<B> {
169    fn from(buf: Vec<u8>) -> Self {
170        Self::Rehydrated(Some(Bytes::from(buf)))
171    }
172}

174/// Transaction metadata.
175#[derive(Clone, Serialize, Deserialize)]
176pub struct Metadata {
177    /// Number of events represented by this transaction.
178    pub event_count: usize,
179
180    /// Number of metric data points represented by this transaction.
181    #[serde(default)]
182    pub data_point_count: usize,

202/// A generic HTTP transaction that can be serialized and deserialized.
203///
204/// In order to support using the retry queue, which may need to serialize and deserialize the transaction to disk, we
205/// need a generic container for HTTP requests that can carry the necessary metadata (event count, etc), and the request
206/// itself (headers, path, body, etc).
207///
208/// `Transaction<B>` supports this by allowing for wrapping an in-memory body type `B` (for example, `ReadIoBuffer`) or wrapping
209/// a body that has been rehydrated from a string (for example, `Bytes`). This means that `B` can be a complex type that can't
210/// actually be rehydrated from a single string input (such as `FrozenChunkedBytesBuffer`) and we maintain optimal
211/// memory usage, and performance, regardless of which body type was used to construct `Transaction<B>`.
212#[derive(Clone, Deserialize, Serialize)]
213#[serde(bound = "")]
214pub struct Transaction<B>
215where
216    B: Buf + Clone,
217{
218    metadata: Metadata,
219
220    #[serde(with = "http_serde_ext::request")]
221    request: Request<TransactionBody<B>>,
222}
223
224impl<B> Transaction<B>
225where
226    B: Buf + Clone,
227{
228    /// Create a new `Transaction` from an original request.
229    pub fn from_original(metadata: Metadata, request: Request<B>) -> Self {
230        Self {
231            metadata,
232            request: request.map(TransactionBody::Original),
233        }
234    }
235
236    /// Reassembles a `Transaction` from a decomposed `Transaction<B>`.
237    pub fn reassemble(metadata: Metadata, request: Request<TransactionBody<B>>) -> Self {
238        Self { metadata, request }
239    }
240
241    /// Returns a reference to the transaction metadata.
242    pub const fn metadata(&self) -> &Metadata {
243        &self.metadata
244    }
245
246    /// Returns the transaction request URI.
247    pub fn request_uri(&self) -> &http::Uri {
248        self.request.uri()
249    }
250
251    /// Consumes the `Transaction` and returns the transaction metadata and original request.
252    pub fn into_parts(self) -> (Metadata, http::Request<TransactionBody<B>>) {
253        (self.metadata, self.request)
254    }
255}

279#[cfg(test)]
280mod tests {
281    use std::collections::VecDeque;
282
283    use bytes::Buf as _;
284    use http::Request;
285    use saluki_io::net::util::retry::EventContainer as _;
286
287    use super::{Metadata, Transaction};
288
289    #[test]
290    fn basic_transaction_ser_deser_roundtrip() {
291        // Create a basic transaction with a simple body.
292        let metadata = Metadata::from_event_and_data_point_count(1, 7);
293        let body = VecDeque::from("hello, world!".as_bytes().to_vec());
294        let request = Request::builder().uri("http://example.com").body(body.clone()).unwrap();
295
296        // Create our `Transaction`, and then roundtrip it: serialize and then deserialize.
297        let transaction = Transaction::from_original(metadata.clone(), request);
298        let serialized = serde_json::to_string(&transaction).unwrap();
299        let deserialized: Transaction<VecDeque<u8>> = serde_json::from_str(&serialized).unwrap();
300
301        // Check some basic properties.
302        assert_eq!(deserialized.metadata.event_count, metadata.event_count);
303        assert_eq!(deserialized.metadata.data_point_count, metadata.data_point_count);
304        assert_eq!(deserialized.event_count(), metadata.event_count as u64);
305        assert_eq!(deserialized.data_point_count(), metadata.data_point_count as u64);
306        assert_eq!(deserialized.request.uri(), "http://example.com");
307
308        // Clone / drain the request body, and make sure it matches the original body.
309        let req_body_raw = deserialized.request.body().clone();
310        let req_body = VecDeque::from(req_body_raw.chunk().to_vec());
311        assert_eq!(req_body, body);
312    }
