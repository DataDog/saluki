42/// Datadog Agent-specific forwarder retry configuration.
43#[derive(Clone, Deserialize)]
44pub struct RetryConfiguration {
45    /// The minimum backoff factor to use when retrying requests.
46    ///
47    /// Controls the the interval range that a calculated backoff duration can fall within, such that with a minimum
48    /// backoff factor of 2.0, calculated backoff durations will fall between `d/2` and `d`, where `d` is the calculated
49    /// backoff duration using a purely exponential growth strategy.
50    ///
51    /// Defaults to 2.
52    #[serde(default = "default_request_backoff_factor", rename = "forwarder_backoff_factor")]
53    backoff_factor: f64,
54
55    /// The base growth rate of the backoff duration when retrying requests, in seconds.
56    ///
57    /// Defaults to 2 seconds.
58    #[serde(default = "default_request_backoff_base", rename = "forwarder_backoff_base")]
59    backoff_base: f64,
60
61    /// The upper bound of the backoff duration when retrying requests, in seconds.
62    ///
63    /// Defaults to 64 seconds.
64    #[serde(default = "default_request_backoff_max", rename = "forwarder_backoff_max")]
65    backoff_max: f64,
66
67    /// The amount to decrease the error count by when a request is successful.
68    ///
69    /// This essentially controls how quickly we forget about the number of previous errors when calculating the next
70    /// backoff duration for a request that must be retried.
71    ///
72    /// Increasing this value should be done with caution, as it can lead to more retries being attempted in the same
73    /// period of time when downstream services are flapping.
74    ///
75    /// Defaults to 2.
76    #[serde(
77        default = "default_request_recovery_error_decrease_factor",
78        rename = "forwarder_recovery_interval"
79    )]
80    recovery_error_decrease_factor: u32,
81
82    /// Whether or not a successful request should completely reset the error count.
83    ///
84    /// Defaults to `false`.
85    #[serde(default = "default_request_recovery_reset", rename = "forwarder_recovery_reset")]
86    recovery_reset: bool,
87
88    /// The maximum in-memory size of the retry queue, in bytes.
89    ///
90    /// Defaults to 15MiB.
91    #[serde(rename = "forwarder_retry_queue_payloads_max_size")]
92    retry_queue_payloads_max_size: Option<u64>,
93
94    /// The maximum in-memory size of the retry queue, in bytes. (deprecated)
95    ///
96    /// Defaults to 0.
97    #[serde(rename = "forwarder_retry_queue_max_size")]
98    retry_queue_max_size: Option<u64>,
99
100    /// The maximum size of the retry queue on disk, in bytes.
101    ///
102    /// Defaults to 0 (disabled).
103    #[serde(
104        rename = "forwarder_storage_max_size_in_bytes",
105        default = "default_storage_max_size_bytes"
106    )]
107    storage_max_size_bytes: u64,
108
109    /// The path to the directory where the retry queue will be stored on disk.
110    ///
111    /// Defaults to `/opt/datadog-agent/run/transactions_to_retry`.
112    #[serde(default, rename = "forwarder_storage_path")]
113    storage_path: PathBuf,
114
115    /// The maximum disk usage ratio for storing transactions on disk.
116    ///
117    /// Defaults to 0.80.
118    ///
119    /// `0.8` means the Agent can store transactions on disk until `forwarder_storage_max_size_in_bytes`
120    /// is reached or when the disk mount for `forwarder_storage_path` exceeds 80% of the disk capacity,
121    /// whichever is lower.
122    #[serde(
123        default = "default_storage_max_disk_ratio",
124        rename = "forwarder_storage_max_disk_ratio"
125    )]
126    storage_max_disk_ratio: f64,
127}
128
129impl RetryConfiguration {
130    pub(super) fn fix_empty_storage_path(&mut self, config: &GenericConfiguration) {
131        // If `forwarder_storage_path` is empty, try setting it to a default path based on `run_path`.
132        if self.storage_path.parent().is_none() {
133            let storage_path = match config.try_get_typed::<PathBuf>("run_path") {
134                Ok(Some(mut run_path)) => {
135                    run_path.push(RETRY_TXN_DIR);
136                    run_path
137                }
138                Ok(None) => {
139                    debug!("`forwarder_storage_path` and `run_path` were empty. Cannot calculate default storage path for forwarder.");
140                    return;
141                }
142                Err(e) => {
143                    debug!(error = %e, "Failed to read `run_path` from configuration. Cannot calculate default storage path for forwarder.");
144                    return;
145                }
146            };
147
148            self.storage_path = storage_path;
149        }
150    }
151
152    /// Returns the maximum size of the retry queue in bytes.
153    ///
154    /// Preferentially uses `forwarder_retry_queue_payloads_max_size` if set, otherwise uses `forwarder_retry_queue_max_size`. If neither
155    /// are set, defaults to 15MiB.
156    pub fn queue_max_size_bytes(&self) -> u64 {
157        self.retry_queue_payloads_max_size
158            .or(self.retry_queue_max_size)
159            .unwrap_or(FORWARDER_RETRY_QUEUE_PAYLOADS_MAX_SIZE_BYTES)
160    }
161
162    /// Returns the maximum size of the retry queue on disk, in bytes.
163    pub const fn storage_max_size_bytes(&self) -> u64 {
164        self.storage_max_size_bytes
165    }
166
167    /// Returns the path to the directory where the retry queue will be stored on disk.
168    pub fn storage_path(&self) -> &Path {
169        &self.storage_path
170    }
171
172    /// Returns the maximum disk usage ratio for storing transactions on disk.
173    pub const fn storage_max_disk_ratio(&self) -> f64 {
174        self.storage_max_disk_ratio
175    }
176
177    /// Creates a new [`DefaultHttpRetryPolicy`] based on the forwarder configuration.
178    pub fn to_default_http_retry_policy(&self) -> DefaultHttpRetryPolicy {
179        let retry_backoff = ExponentialBackoff::with_jitter(
180            Duration::from_secs_f64(self.backoff_base),
181            Duration::from_secs_f64(self.backoff_max),
182            self.backoff_factor,
183        );
184
185        let recovery_error_decrease_factor = (!self.recovery_reset).then_some(self.recovery_error_decrease_factor);
186        DefaultHttpRetryPolicy::with_backoff(retry_backoff)
187            .with_recovery_error_decrease_factor(recovery_error_decrease_factor)
188    }
189}

198    #[tokio::test]
199    async fn fix_empty_storage_path_sets_path_from_run_path() {
200        const RUN_PATH: &str = "/my/little/run_path";
201
202        // Create a base configuration with only `run_path` set.
203        let base_config_values = json!({ "run_path": RUN_PATH });
204        let (config, _) = ConfigurationLoader::for_tests(Some(base_config_values), None, false).await;
205
206        // Read our retry configuration, and make sure we start out with the expected empty `storage_path`.
207        let mut retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
208        assert_eq!(retry_config.storage_path(), PathBuf::new());
209
210        // Try to fix up the empty storage path, and make sure the updated storage path is based on the `run_path` we
211        // set on our base configuration.
212        retry_config.fix_empty_storage_path(&config);
213
214        let expected = PathBuf::from(RUN_PATH).join(RETRY_TXN_DIR);
215        assert_eq!(expected, retry_config.storage_path());
216    }

218    #[tokio::test]
219    async fn fix_empty_storage_path_does_nothing_when_path_already_set() {
220        const RUN_PATH: &str = "/my/little/run_path";
221        const FORWARDER_STORAGE_PATH: &str = "/custom/path/to/storage";
222
223        // Create a base configuration with both `run_path` and `forwarder_storage_path` set.
224        let base_config_values = json!({ "run_path": RUN_PATH, "forwarder_storage_path": FORWARDER_STORAGE_PATH });
225        let (config, _) = ConfigurationLoader::for_tests(Some(base_config_values), None, false).await;
226
227        // Read our retry configuration, and make sure we see the storage path that we initially set.
228        let mut retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
229
230        let initial_storage_path = retry_config.storage_path().to_path_buf();
231        assert_eq!(initial_storage_path, PathBuf::from(FORWARDER_STORAGE_PATH));
232
233        // Try to fix up the storage path, and make sure nothing changes since it's not actually empty.
234        retry_config.fix_empty_storage_path(&config);
235        assert_eq!(initial_storage_path, retry_config.storage_path());
236    }

254    #[tokio::test]
255    async fn queue_max_size_bytes_fallback_behavior() {
256        const OVERRIDE_FALLBACK_SIZE_BYTES: u64 = 1024;
257        const OVERRIDE_PRIMARY_SIZE_BYTES: u64 = 2048;
258
259        // When neither field is set, returns the default (15 MiB).
260        let (config, _) = ConfigurationLoader::for_tests(None, None, false).await;
261        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
262        assert_eq!(
263            retry_config.queue_max_size_bytes(),
264            FORWARDER_RETRY_QUEUE_PAYLOADS_MAX_SIZE_BYTES
265        );
266
267        // When only the deprecated field is set, uses it.
268        let values = json!({ "forwarder_retry_queue_max_size": OVERRIDE_FALLBACK_SIZE_BYTES });
269        let (config, _) = ConfigurationLoader::for_tests(Some(values), None, false).await;
270        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
271        assert_eq!(retry_config.queue_max_size_bytes(), OVERRIDE_FALLBACK_SIZE_BYTES);
272
273        // When both fields are set, the newer field takes priority.
274        let values = json!({
275            "forwarder_retry_queue_payloads_max_size": OVERRIDE_PRIMARY_SIZE_BYTES,
276            "forwarder_retry_queue_max_size": OVERRIDE_FALLBACK_SIZE_BYTES,
277        });
278        let (config, _) = ConfigurationLoader::for_tests(Some(values), None, false).await;
279        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
280        assert_eq!(retry_config.queue_max_size_bytes(), OVERRIDE_PRIMARY_SIZE_BYTES);
281    }
