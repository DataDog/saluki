47/// Datadog Agent-specific forwarder retry configuration.
48#[derive(Clone, Deserialize, Facet)]
49#[cfg_attr(test, derive(Debug, PartialEq, serde::Serialize))]
50pub struct RetryConfiguration {
51    /// The minimum backoff factor to use when retrying requests.
52    ///
53    /// Controls the interval range that a calculated backoff duration can fall within, such that with a minimum
54    /// backoff factor of 2.0, calculated backoff durations will fall between `d/2` and `d`, where `d` is the calculated
55    /// backoff duration using a purely exponential growth strategy.
56    ///
57    /// Defaults to 2.
58    #[serde(default = "default_request_backoff_factor", rename = "forwarder_backoff_factor")]
59    backoff_factor: f64,
60
61    /// The base growth rate of the backoff duration when retrying requests, in seconds.
62    ///
63    /// Defaults to 2 seconds.
64    #[serde(default = "default_request_backoff_base", rename = "forwarder_backoff_base")]
65    backoff_base: f64,
66
67    /// The upper bound of the backoff duration when retrying requests, in seconds.
68    ///
69    /// Defaults to 64 seconds.
70    #[serde(default = "default_request_backoff_max", rename = "forwarder_backoff_max")]
71    backoff_max: f64,
72
73    /// The amount to decrease the error count by when a request is successful.
74    ///
75    /// This essentially controls how quickly we forget about the number of previous errors when calculating the next
76    /// backoff duration for a request that must be retried.
77    ///
78    /// Increasing this value should be done with caution, as it can lead to more retries being attempted in the same
79    /// period of time when downstream services are flapping.
80    ///
81    /// Defaults to 2.
82    #[serde(
83        default = "default_request_recovery_error_decrease_factor",
84        rename = "forwarder_recovery_interval"
85    )]
86    recovery_error_decrease_factor: u32,
87
88    /// Whether or not a successful request should completely reset the error count.
89    ///
90    /// Defaults to `false`.
91    #[serde(default = "default_request_recovery_reset", rename = "forwarder_recovery_reset")]
92    recovery_reset: bool,
93
94    /// The maximum in-memory size of the retry queue, in bytes.
95    ///
96    /// Defaults to 15MiB.
97    #[serde(rename = "forwarder_retry_queue_payloads_max_size")]
98    retry_queue_payloads_max_size: Option<u64>,
99
100    /// The maximum in-memory size of the retry queue, in bytes. (deprecated)
101    ///
102    /// Defaults to 0.
103    #[serde(rename = "forwarder_retry_queue_max_size")]
104    retry_queue_max_size: Option<u64>,
105
106    /// The maximum size of the retry queue on disk, in bytes.
107    ///
108    /// Defaults to 0 (disabled).
109    #[serde(
110        rename = "forwarder_storage_max_size_in_bytes",
111        default = "default_storage_max_size_bytes"
112    )]
113    storage_max_size_bytes: u64,
114
115    /// The path to the directory where the retry queue will be stored on disk.
116    ///
117    /// Defaults to `/opt/datadog-agent/run/transactions_to_retry`.
118    #[serde(default, rename = "forwarder_storage_path")]
119    storage_path: PathBuf,
120
121    /// The maximum disk usage ratio for storing transactions on disk.
122    ///
123    /// Defaults to 0.80.
124    ///
125    /// `0.8` means the Agent can store transactions on disk until `forwarder_storage_max_size_in_bytes`
126    /// is reached or when the disk mount for `forwarder_storage_path` exceeds 80% of the disk capacity,
127    /// whichever is lower.
128    #[serde(
129        default = "default_storage_max_disk_ratio",
130        rename = "forwarder_storage_max_disk_ratio"
131    )]
132    storage_max_disk_ratio: f64,
133}

135impl RetryConfiguration {
136    pub(super) fn fix_empty_storage_path(&mut self, config: &GenericConfiguration) {
137        // If `forwarder_storage_path` is empty, try setting it to a default path based on `run_path`.
138        if self.storage_path.parent().is_none() {
139            let storage_path = match config.try_get_typed::<PathBuf>("run_path") {
140                Ok(Some(mut run_path)) => {
141                    run_path.push(RETRY_TXN_DIR);
142                    run_path
143                }
144                Ok(None) => {
145                    debug!("`forwarder_storage_path` and `run_path` were empty. Cannot calculate default storage path for forwarder.");
146                    return;
147                }
148                Err(e) => {
149                    debug!(error = %e, "Failed to read `run_path` from configuration. Cannot calculate default storage path for forwarder.");
150                    return;
151                }
152            };
153
154            self.storage_path = storage_path;
155        }
156    }
157
158    /// Returns the maximum size of the retry queue in bytes.
159    ///
160    /// Preferentially uses `forwarder_retry_queue_payloads_max_size` if set, otherwise uses `forwarder_retry_queue_max_size`. If neither
161    /// are set, defaults to 15MiB.
162    pub fn queue_max_size_bytes(&self) -> u64 {
163        self.retry_queue_payloads_max_size
164            .or(self.retry_queue_max_size)
165            .unwrap_or(FORWARDER_RETRY_QUEUE_PAYLOADS_MAX_SIZE_BYTES)
166    }
167
168    /// Returns the maximum size of the retry queue on disk, in bytes.
169    pub const fn storage_max_size_bytes(&self) -> u64 {
170        self.storage_max_size_bytes
171    }
172
173    /// Returns the path to the directory where the retry queue will be stored on disk.
174    pub fn storage_path(&self) -> &Path {
175        &self.storage_path
176    }
177
178    /// Returns the maximum disk usage ratio for storing transactions on disk.
179    pub const fn storage_max_disk_ratio(&self) -> f64 {
180        self.storage_max_disk_ratio
181    }

183    /// Creates a new [`DefaultHttpRetryPolicy`] based on the forwarder configuration.
184    ///
185    /// If a [`GenericConfiguration`] is supplied, the policy captures it and checks whether
186    /// secrets management is active on every 403 Forbidden response. This allows the retry gate to
187    /// pick up runtime changes pushed via the config stream without rebuilding the service. When no
188    /// configuration is supplied, 403 responses retain their default non-retriable behavior.
189    pub fn to_default_http_retry_policy<B: 'static>(
190        &self, live_config: Option<GenericConfiguration>,
191    ) -> DefaultHttpRetryPolicy<B> {
192        let retry_backoff = ExponentialBackoff::with_jitter(
193            Duration::from_secs_f64(self.backoff_base),
194            Duration::from_secs_f64(self.backoff_max),
195            self.backoff_factor,
196        );
197
198        let classifier = if let Some(config) = live_config {
199            let gate: HttpRetryPredicate<B> =
200                Arc::new(move |response| response.status() == StatusCode::FORBIDDEN && secrets_in_use(&config));
201            StandardHttpClassifier::new().with_predicate(gate)
202        } else {
203            StandardHttpClassifier::new()
204        };
205
206        let recovery_error_decrease_factor = (!self.recovery_reset).then_some(self.recovery_error_decrease_factor);
207        DefaultHttpRetryPolicy::with_backoff_and_classifier(retry_backoff, classifier)
208            .with_recovery_error_decrease_factor(recovery_error_decrease_factor)
209    }
210}
211
212fn secrets_in_use(config: &GenericConfiguration) -> bool {
213    matches!(config.try_get_typed::<u64>("secret_refresh_on_api_key_failure_interval"), Ok(Some(value)) if value > 0)
214        || matches!(config.try_get_typed::<String>("secret_backend_command"), Ok(Some(value)) if !value.trim().is_empty())
215}

315        const OVERRIDE_FALLBACK_SIZE_BYTES: u64 = 1024;
316        const OVERRIDE_PRIMARY_SIZE_BYTES: u64 = 2048;
317
318        // When neither field is set, returns the default (15 MiB).
319        let (config, _) = ConfigurationLoader::for_tests(None, None, false).await;
320        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
321        assert_eq!(
322            retry_config.queue_max_size_bytes(),
323            FORWARDER_RETRY_QUEUE_PAYLOADS_MAX_SIZE_BYTES
324        );
325
326        // When only the deprecated field is set, uses it.
327        let values = json!({ "forwarder_retry_queue_max_size": OVERRIDE_FALLBACK_SIZE_BYTES });
328        let (config, _) = ConfigurationLoader::for_tests(Some(values), None, false).await;
329        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
330        assert_eq!(retry_config.queue_max_size_bytes(), OVERRIDE_FALLBACK_SIZE_BYTES);
331
332        // When both fields are set, the newer field takes priority.
333        let values = json!({
334            "forwarder_retry_queue_payloads_max_size": OVERRIDE_PRIMARY_SIZE_BYTES,
335            "forwarder_retry_queue_max_size": OVERRIDE_FALLBACK_SIZE_BYTES,
336        });
337        let (config, _) = ConfigurationLoader::for_tests(Some(values), None, false).await;
338        let retry_config: RetryConfiguration = config.as_typed().expect("should deserialize");
339        assert_eq!(retry_config.queue_max_size_bytes(), OVERRIDE_PRIMARY_SIZE_BYTES);
340    }

384    #[tokio::test]
385    async fn policy_403_gate_reflects_dynamic_secrets_config_change() {
386        use std::time::Duration as StdDuration;
387
388        use saluki_config::dynamic::ConfigUpdate;
389
390        let (config, sender) = ConfigurationLoader::for_tests(Some(json!({})), None, true).await;
391        let sender = sender.expect("dynamic configuration sender should be present");
392
393        // Apply an empty initial snapshot and wait for readiness.
394        sender
395            .send(ConfigUpdate::Snapshot(json!({})))
396            .await
397            .expect("should send initial snapshot");
398        config.ready().await;
399
400        let retry_config = test_retry_config();
401        let mut policy = retry_config.to_default_http_retry_policy(Some(config.clone()));
402
403        // Before secrets are configured, 403 must not be retried.
404        assert!(!would_retry(&mut policy, ok_response(StatusCode::FORBIDDEN)));
405
406        // Push a config update that enables secrets management.
407        let mut watcher = config.watch_for_updates("secret_backend_command");
408        sender
409            .send(ConfigUpdate::Partial {
410                key: "secret_backend_command".to_string(),
411                value: json!("/bin/true"),
412            })
413            .await
414            .expect("should send partial update");
415
416        tokio::time::timeout(StdDuration::from_secs(2), watcher.changed::<String>())
417            .await
418            .expect("timed out waiting for secret_backend_command update");
419
420        // The same policy instance must now retry 403 because the predicate reads the live cached secrets flag.
421        assert!(would_retry(&mut policy, ok_response(StatusCode::FORBIDDEN)));
422    }
