25/// APM configuration.
26///
27/// This configuration mirrors the Agent's trace agent configuration..
28#[derive(Clone, Debug, Deserialize)]
29struct ApmConfiguration {
30    #[serde(default)]
31    apm_config: ApmConfig,
32}
33
34#[derive(Clone, Debug, Deserialize)]
35pub struct ApmConfig {
36    /// Target traces per second for priority sampling.
37    ///
38    /// Defaults to 10.0.
39    #[serde(default = "default_target_traces_per_second")]
40    target_traces_per_second: f64,
41
42    /// Target traces per second for error sampling.
43    ///
44    /// Defaults to 10.0.
45    #[serde(default = "default_errors_per_second")]
46    errors_per_second: f64,
47
48    /// Enables an additional stats computation check on spans to see if they have an eligible `span.kind` (server, consumer, client, producer).
49    /// If enabled, a span with an eligible `span.kind` will have stats computed. If disabled, only top-level and measured spans will have stats computed.
50    ///
51    /// Defaults to `true`.
52    #[serde(default = "default_compute_stats_by_span_kind")]
53    compute_stats_by_span_kind: bool,
54
55    /// Enables aggregation of peer related tags (e.g., `peer.service`, `db.instance`, etc.) in the Agent.
56    ///
57    /// Defaults to `true`.
58    #[serde(default = "default_peer_tags_aggregation")]
59    peer_tags_aggregation: bool,
60
61    /// Optional list of supplementary peer tags that go beyond the defaults. The Datadog backend validates all tags
62    /// and will drop ones that are unapproved.
63    ///
64    /// Defaults to an empty list.
65    #[serde(default)]
66    peer_tags: Vec<MetaString>,
67
68    /// Default environment to use when traces don't provide one.
69    ///
70    /// Defaults to "none".
71    #[serde(default = "default_env")]
72    default_env: MetaString,
73
74    /// Default hostname to use when traces don't provide one.
75    ///
76    /// Defaults to empty string (no fallback).
77    #[serde(skip)]
78    hostname: MetaString,
79}
80
81impl ApmConfig {
82    pub fn from_configuration(config: &GenericConfiguration) -> Result<Self, GenericError> {
83        let wrapper = config.as_typed::<ApmConfiguration>()?;
84        Ok(wrapper.apm_config)
85    }
86
87    /// Returns the target traces per second for priority sampling.
88    pub const fn target_traces_per_second(&self) -> f64 {
89        self.target_traces_per_second
90    }
91
92    /// Returns the target traces per second for error sampling.
93    pub const fn errors_per_second(&self) -> f64 {
94        self.errors_per_second
95    }
96
97    /// Returns if stats computation by span kind is enabled.
98    pub const fn compute_stats_by_span_kind(&self) -> bool {
99        self.compute_stats_by_span_kind
100    }
101
102    /// Returns if peer tags aggregation is enabled.
103    pub const fn peer_tags_aggregation(&self) -> bool {
104        self.peer_tags_aggregation
105    }
106
107    /// Returns the supplementary peer tags.
108    pub fn peer_tags(&self) -> &[MetaString] {
109        &self.peer_tags
110    }
111
112    /// Returns the default environment to use when traces don't provide one.
113    pub fn default_env(&self) -> &MetaString {
114        &self.default_env
115    }
116
117    /// Returns the default hostname to use when traces don't provide one.
118    pub fn hostname(&self) -> &MetaString {
119        &self.hostname
120    }
121
122    /// Sets the hostname if it is currently empty.
123    pub fn set_hostname_if_empty(&mut self, hostname: impl Into<MetaString>) {
124        if self.hostname.is_empty() {
125            self.hostname = hostname.into();
126        }
127    }
128}
129
130impl Default for ApmConfig {
131    fn default() -> Self {
132        Self {
133            target_traces_per_second: default_target_traces_per_second(),
134            errors_per_second: default_errors_per_second(),
135            compute_stats_by_span_kind: default_compute_stats_by_span_kind(),
136            peer_tags_aggregation: default_peer_tags_aggregation(),
137            peer_tags: Vec::new(),
138            default_env: default_env(),
139            hostname: MetaString::default(),
140        }
141    }
