79/// Returns whether to keep a trace, based on its ID and a sampling rate.
80///
81/// This assumes trace IDs are nearly uniformly distributed.
82pub fn sample_by_rate(trace_id: u64, rate: f64) -> bool {
83    if rate < 1.0 {
84        trace_id.wrapping_mul(SAMPLER_HASHER) < (rate * MAX_TRACE_ID_FLOAT) as u64
85    } else {
86        true
87    }
88}

90pub fn get_trace_env(trace: &Trace, root_span_idx: usize) -> Option<&MetaString> {
91    // logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/traceutil/trace.go#L19-L20
92    let env = trace
93        .spans()
94        .get(root_span_idx)
95        .and_then(|span| span.attributes.get("env").and_then(AttributeValue::as_string));
96    if let Some(env) = env {
97        return Some(env);
98    }
99    for span in trace.spans().iter() {
100        if let Some(env) = span.attributes.get("env").and_then(AttributeValue::as_string) {
101            return Some(env);
102        }
103    }
104    // Fall back to the payload-level env (set from tracer payload headers or OTLP resource attributes).
105    if !trace.payload.env.is_empty() {
106        return Some(&trace.payload.env);
107    }
108    None
109}
