37/// Hasher used for deterministic sampling.
38pub const SAMPLER_HASHER: u64 = 1111111111111111111;

40/// Returns whether to keep a trace, based on its ID and a sampling rate.
41///
42/// This assumes trace IDs are nearly uniformly distributed.
43pub fn sample_by_rate(trace_id: u64, rate: f64) -> bool {
44    if rate < 1.0 {
45        trace_id.wrapping_mul(SAMPLER_HASHER) < (rate * MAX_TRACE_ID_FLOAT) as u64
46    } else {
47        true
48    }
