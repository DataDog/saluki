42/// The Datadog Agent's default zstd compression level. Because the Agent forwards its full config
43/// (including this default) over the config stream, this is indistinguishable from a user who
44/// explicitly set the level to 1, so we treat it as "unset" when resolving the level for ADP.
45pub(crate) const AGENT_DEFAULT_ZSTD_COMPRESSOR_LEVEL: i32 = 1;
46
47/// ADP's default zstd compression level. Higher than the Agent's default of 1 because ADP is more
48/// efficient and can afford better compression: level 3 yields ~6% smaller payloads without a net
49/// CPU increase.
50pub(crate) const DEFAULT_ADP_ZSTD_COMPRESSOR_LEVEL: i32 = 3;
51
52/// Resolves the effective zstd compression level for an ADP encoder.
53///
54/// Precedence:
55/// 1. `data_plane.serializer_zstd_compressor_level` (ADP-specific) when set.
56/// 2. `serializer_zstd_compressor_level` (Core Agent) when set to a non-default value, so a user who
57///    raised the Agent's level still gets it applied to ADP.
58/// 3. ADP's default of 3.
59pub(crate) fn resolve_zstd_compressor_level(data_plane_level: Option<i32>, agent_level: Option<i32>) -> i32 {
60    data_plane_level
61        .or_else(|| agent_level.filter(|&level| level != AGENT_DEFAULT_ZSTD_COMPRESSOR_LEVEL))
62        .unwrap_or(DEFAULT_ADP_ZSTD_COMPRESSOR_LEVEL)
63}

124/// Hasher used for deterministic sampling.

127/// Returns whether to keep a trace, based on its ID and a sampling rate.
128///
129/// This assumes trace IDs are nearly uniformly distributed.
130pub fn sample_by_rate(trace_id: u64, rate: f64) -> bool {
131    if rate < 1.0 {
132        trace_id.wrapping_mul(SAMPLER_HASHER) < (rate * MAX_TRACE_ID_FLOAT) as u64
133    } else {
134        true
135    }
136}

138pub fn get_trace_env(trace: &Trace, root_span_idx: usize) -> Option<&MetaString> {
139    // logic taken from here: https://github.com/DataDog/datadog-agent/blob/main/pkg/trace/traceutil/trace.go#L19-L20
140    let env = trace
141        .spans()
142        .get(root_span_idx)
143        .and_then(|span| span.attributes.get("env").and_then(AttributeValue::as_string));
144    if let Some(env) = env {
145        return Some(env);
146    }
147    for span in trace.spans().iter() {
148        if let Some(env) = span.attributes.get("env").and_then(AttributeValue::as_string) {
149            return Some(env);
150        }
151    }
152    // Fall back to the payload-level env (set from tracer payload headers or OTLP resource attributes).
153    if !trace.payload.env.is_empty() {
154        return Some(&trace.payload.env);
155    }
156    None
157}

159#[cfg(test)]
160mod tests {
161    use super::{resolve_zstd_compressor_level, DEFAULT_ADP_ZSTD_COMPRESSOR_LEVEL};
162
163    #[test]
164    fn zstd_level_defaults_to_adp_default_when_nothing_set() {
165        assert_eq!(
166            resolve_zstd_compressor_level(None, None),
167            DEFAULT_ADP_ZSTD_COMPRESSOR_LEVEL
168        );
169    }
170
171    #[test]
172    fn zstd_level_ignores_agent_default_of_one() {
173        // The Agent forwards its default of 1, which we can't distinguish from an explicit 1, so it
174        // does not override ADP's own default.
175        assert_eq!(
176            resolve_zstd_compressor_level(None, Some(1)),
177            DEFAULT_ADP_ZSTD_COMPRESSOR_LEVEL
178        );
179    }
180
181    #[test]
182    fn zstd_level_uses_agent_value_when_changed_from_default() {
183        assert_eq!(resolve_zstd_compressor_level(None, Some(5)), 5);
184    }
185
186    #[test]
187    fn zstd_level_data_plane_takes_precedence() {
188        // The ADP-specific key wins over the Agent value, even a non-default one.
189        assert_eq!(resolve_zstd_compressor_level(Some(4), Some(5)), 4);
190        // And even when it matches the Agent default of 1.
191        assert_eq!(resolve_zstd_compressor_level(Some(1), Some(5)), 1);
192    }
