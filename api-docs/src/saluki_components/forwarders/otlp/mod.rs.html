35/// Trace Agent forwarder.
36///
37/// Forwards OTLP trace payloads to the Trace Agent.
38#[derive(Clone)]
39pub struct OtlpForwarderConfiguration {
40    /// Forwarder configuration settings.
41    ///
42    /// See [`ForwarderConfiguration`] for more information about the available settings.
43    forwarder_config: ForwarderConfiguration,
44
45    configuration: Option<GenericConfiguration>,
46
47    core_agent_otlp_grpc_endpoint: String,
48
49    core_agent_traces_internal_port: u16,
50}
51
52impl OtlpForwarderConfiguration {
53    /// Creates a new `TraceAgentForwarderConfiguration` from the given configuration.
54    pub fn from_configuration(
55        config: &GenericConfiguration, core_agent_otlp_grpc_endpoint: String,
56    ) -> Result<Self, GenericError> {
57        let forwarder_config = ForwarderConfiguration::from_configuration(config)?;
58        let core_agent_traces_internal_port = config
59            .try_get_typed("otlp_config.traces.internal_port")?
60            .unwrap_or(5003);
61        Ok(Self {
62            forwarder_config,
63            configuration: Some(config.clone()),
64            core_agent_otlp_grpc_endpoint,
65            core_agent_traces_internal_port,
66        })
67    }
68    /// Overrides the default endpoint that payloads are sent to.
69    ///
70    /// This overrides any existing endpoint configuration, and manually sets the base endpoint (e.g.,
71    /// `https://api.datad0g.com`) to be used for all payloads.
72    ///
73    /// This can be used to preserve other configuration settings (forwarder settings, retry, etc) while still allowing
74    /// for overriding _where_ payloads are sent to.
75    ///
76    /// # Errors
77    ///
78    /// If the given request path is not valid, an error is returned.
79    pub fn with_endpoint_override(mut self, dd_url: String, api_key: String) -> Self {
80        // Clear any existing additional endpoints, and set the new DD URL and API key.
81        //
82        // This ensures that the only endpoint we'll send to is this one.
83        let endpoint = self.forwarder_config.endpoint_mut();
84        endpoint.clear_additional_endpoints();
85        endpoint.set_dd_url(dd_url);
86        endpoint.set_api_key(api_key);
87
88        self
89    }
90}

92#[async_trait]
93impl ForwarderBuilder for OtlpForwarderConfiguration {
94    fn input_payload_type(&self) -> PayloadType {
95        PayloadType::Http | PayloadType::Grpc
96    }
97
98    async fn build(&self, context: ComponentContext) -> Result<Box<dyn Forwarder + Send>, GenericError> {
99        let trace_agent_endpoint = format!("http://localhost:{}", self.core_agent_traces_internal_port);
100        let trace_agent_channel = Channel::from_shared(trace_agent_endpoint.clone())
101            .error_context("Failed to construct gRPC channel due to an invalid endpoint.")?
102            .connect_lazy();
103        let trace_agent_client = TraceServiceClient::new(trace_agent_channel);
104
105        let normalized_endpoint = normalize_endpoint(&self.core_agent_otlp_grpc_endpoint);
106        let core_agent_grpc_channel = Channel::from_shared(normalized_endpoint)
107            .error_context("Failed to construct gRPC channel due to an invalid endpoint.")?
108            .connect_lazy();
109        let core_agent_metrics_client = MetricsServiceClient::new(core_agent_grpc_channel.clone());
110        let core_agent_logs_client = LogsServiceClient::new(core_agent_grpc_channel);
111
112        let metrics_builder = MetricsBuilder::from_component_context(&context);
113        let telemetry = ComponentTelemetry::from_builder(&metrics_builder);
114        let forwarder = TransactionForwarder::from_config(
115            context,
116            self.forwarder_config.clone(),
117            self.configuration.clone(),
118            get_dd_endpoint_name,
119            telemetry.clone(),
120            metrics_builder,
121        )?;
122
123        Ok(Box::new(OtlpForwarder {
124            trace_agent_client,
125            core_agent_metrics_client,
126            core_agent_logs_client,
127            forwarder,
128        }))
129    }
130}

138struct OtlpForwarder {
139    trace_agent_client: TraceServiceClient<Channel>,
140    core_agent_metrics_client: MetricsServiceClient<Channel>,
141    core_agent_logs_client: LogsServiceClient<Channel>,
142    /// Forwarder for the Datadog Agent's OTLP HTTP endpoint.
143    forwarder: TransactionForwarder<FrozenChunkedBytesBuffer>,
144}
145
146#[async_trait]
147impl Forwarder for OtlpForwarder {
148    async fn run(mut self: Box<Self>, mut context: ForwarderContext) -> Result<(), GenericError> {
149        let Self {
150            mut trace_agent_client,
151            mut core_agent_metrics_client,
152            mut core_agent_logs_client,
153            forwarder,
154        } = *self;
155
156        let forwarder = forwarder.spawn().await;
157        let mut health = context.take_health_handle();
158
159        health.mark_ready();
160        debug!("Trace Agent forwarder started.");
161
162        loop {
163            select! {
164                _ = health.live() => continue,
165                maybe_payload = context.payloads().next() => match maybe_payload {
166                    Some(payload) => match payload {
167                        Payload::Grpc(grpc_payload) => {
168                            // Extract the parts of the payload, and make sure we have an OTLP payload, otherwise
169                            // we skip it and move on.
170                            let (_, endpoint, service_path, body) = grpc_payload.into_parts();
171                            match &service_path {
172                                path if *path == OTLP_TRACES_GRPC_SERVICE_PATH => {
173                                    export_traces(&mut trace_agent_client, &endpoint, &service_path, body).await;
174                                }
175                                path if *path == OTLP_METRICS_GRPC_SERVICE_PATH => {
176                                    export_metrics(&mut core_agent_metrics_client, &endpoint, &service_path, body).await;
177                                }
178                                path if *path == OTLP_LOGS_GRPC_SERVICE_PATH => {
179                                    export_logs(&mut core_agent_logs_client, &endpoint, &service_path, body).await;
180                                }
181                                _ => {
182                                    warn!(service_path = %service_path, "Received gRPC payload with unknown service path. Skipping.");
183                                    continue;
184                                }
185                            }
186
187                        },
188                        Payload::Http(http_payload) => {
189                            let (payload_meta, request) = http_payload.into_parts();
190                            let transaction_meta = Metadata::from_event_count(payload_meta.event_count());
191                            let transaction = Transaction::from_original(transaction_meta, request);
192
193                            forwarder.send_transaction(transaction).await?;
194                        }
195                        _ => continue,
196                    },
197                    None => break,
198                },
199            }
200        }
201
202        // Shutdown the forwarder gracefully.
203        forwarder.shutdown().await;
204
205        debug!("OTLP forwarder stopped.");
206
207        Ok(())
208    }
209}

211async fn export_traces(
212    trace_agent_client: &mut TraceServiceClient<Channel>, endpoint: &MetaString, service_path: &MetaString,
213    body: FrozenChunkedBytesBuffer,
214) {
215    // Decode the raw request payload into a typed body so we can export it.
216    //
217    // TODO: This is suboptimal since we know the payload should be valid as it was decoded when it
218    // was ingested, and only after that converted to raw bytes. It would be nice to just forward
219    // the bytes as-is without decoding it again here just to satisfy the client interface, but no
220    // such API currently exists.
221    let body = body.into_bytes();
222    let request = match ExportTraceServiceRequest::decode(body) {
223        Ok(req) => req,
224        Err(e) => {
225            error!(error = %e, "Failed to decode trace export request from payload.");
226            return;
227        }
228    };
229
230    match trace_agent_client.export(request).await {
231        Ok(response) => {
232            let resp = response.into_inner();
233            if let Some(partial_success) = resp.partial_success {
234                if partial_success.rejected_spans > 0 {
235                    warn!(
236                        rejected_spans = partial_success.rejected_spans,
237                        error = %partial_success.error_message,
238                        "Trace export partially failed."
239                    );
240                }
241            }
242        }
243        Err(e) => {
244            error!(error = %e, %endpoint, %service_path, "Failed to export traces to Trace Agent.");
245        }
246    }
247}

327fn get_dd_endpoint_name(uri: &Uri) -> Option<MetaString> {
328    match uri.path() {
329        "/v1/traces" => Some(MetaString::from_static("traces_v1")),
330        "/v1/metrics" => Some(MetaString::from_static("metrics_v1")),
331        "/v1/logs" => Some(MetaString::from_static("logs_v1")),
332        _ => None,
333    }
334}

336fn normalize_endpoint(endpoint: &str) -> String {
337    if endpoint.starts_with("http://") || endpoint.starts_with("https://") {
338        endpoint.to_string()
339    } else {
340        format!("https://{}", endpoint)
341    }
