1use std::collections::HashMap;
2
3use saluki_config::GenericConfiguration;
4use saluki_env::{EnvironmentProvider, HostProvider};
5use saluki_metadata;
6use tracing::warn;
7
8/// Global/shared configuration for checks.
9///
10/// This provides information to checks, either from the configuration or computed from the EnvironmentProvider if one
11/// is provided.
12#[derive(Clone)]
13pub struct ExecutionContext {
14    configuration: GenericConfiguration,
15    hostname: String,
16    http_headers: HashMap<String, String>,
17}
18
19#[allow(dead_code)]
20impl ExecutionContext {
21    pub fn new(configuration: GenericConfiguration) -> Self {
22        let http_headers = HashMap::from([
23            (
24                "User-Agent".to_string(),
25                format!("Datadog Agent/{}", saluki_metadata::get_app_details().version().raw()).to_string(),
26            ),
27            (
28                "Content-Type".to_string(),
29                "application/x-www-form-urlencoded".to_string(),
30            ),
31            ("Accept".to_string(), "text/html, */*".to_string()),
32        ]);
33
34        Self {
35            configuration,
36            hostname: "".to_string(),
37            http_headers,
38        }
39    }
40
41    /// Create an `ExecutionContext` from an `EnvironmentProvider`.
42    ///
43    /// The `EnvironmentProvider` is used to compute information from the environment, which are cached into the
44    /// `ExecutionContext`.
45    pub async fn from_environment_provider<E>(configuration: GenericConfiguration, environment_provider: &E) -> Self
46    where
47        E: EnvironmentProvider,
48        <E::Host as HostProvider>::Error: std::fmt::Debug,
49    {
50        let execution_context = Self::new(configuration);
51        let hostname = environment_provider.host().get_hostname().await.unwrap_or_else(|e| {
52            warn!("Failed to get hostname: {:?}", e);
53            "".to_string()
54        });
55
56        Self {
57            hostname,
58            ..execution_context
59        }
60    }
61
62    /// Get a reference to the configuration used to create this `ExecutionContext`.
63    pub fn configuration(&self) -> &GenericConfiguration {
64        &self.configuration
65    }
66
67    /// Get the hostname.
68    ///
69    /// Computed from the `EnvironmentProvider` if one has been provided to the `ExecutionContext` constructor.
70    pub fn hostname(&self) -> &str {
71        &self.hostname
72    }
73
74    /// Override the hostname.
75    #[allow(dead_code)]
76    pub fn with_hostname<S: AsRef<str>>(self, hostname: S) -> Self {
77        Self {
78            hostname: hostname.as_ref().to_string(),
79            ..self
80        }
81    }
82
83    /// Get a reference to the HTTP headers.
84    pub fn http_headers(&self) -> &HashMap<String, String> {
85        &self.http_headers
86    }
87}
