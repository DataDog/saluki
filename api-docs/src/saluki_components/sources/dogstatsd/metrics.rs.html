99pub(super) fn build_metrics(listen_addr: &ListenAddress, component_context: &ComponentContext) -> Metrics {
100    let builder = MetricsBuilder::from_component_context(component_context);
101
102    let listener_type = match listen_addr {
103        ListenAddress::Tcp(_) => "tcp",
104        ListenAddress::Udp(_) => "udp",
105        ListenAddress::Unix(_) => "unix",
106        ListenAddress::Unixgram(_) => "unixgram",
107    };
108
109    Metrics {
110        metrics_received: builder.register_counter_with_tags(
111            "component_events_received_total",
112            [("message_type", "metrics"), ("listener_type", listener_type)],
113        ),
114        events_received: builder.register_counter_with_tags(
115            "component_events_received_total",
116            [("message_type", "events"), ("listener_type", listener_type)],
117        ),
118        service_checks_received: builder.register_counter_with_tags(
119            "component_events_received_total",
120            [("message_type", "service_checks"), ("listener_type", listener_type)],
121        ),
122        bytes_received: builder
123            .register_counter_with_tags("component_bytes_received_total", [("listener_type", listener_type)]),
124        bytes_received_size: builder
125            .register_trace_histogram_with_tags("component_bytes_received_size", [("listener_type", listener_type)]),
126        framing_errors: builder.register_counter_with_tags(
127            "component_errors_total",
128            [("listener_type", listener_type), ("error_type", "framing")],
129        ),
130        metric_decoder_errors: builder.register_counter_with_tags(
131            "component_errors_total",
132            [
133                ("listener_type", listener_type),
134                ("error_type", "decode"),
135                ("message_type", "metrics"),
136            ],
137        ),
138        event_decoder_errors: builder.register_counter_with_tags(
139            "component_errors_total",
140            [
141                ("listener_type", listener_type),
142                ("error_type", "decode"),
143                ("message_type", "events"),
144            ],
145        ),
146        service_check_decoder_errors: builder.register_counter_with_tags(
147            "component_errors_total",
148            [
149                ("listener_type", listener_type),
150                ("error_type", "decode"),
151                ("message_type", "service_checks"),
152            ],
153        ),
154        origin_detection_errors: builder
155            .register_counter_with_tags("component_errors_total", [("error_type", ERROR_TYPE_ORIGIN_DETECTION)]),
156        connections_active: builder
157            .register_gauge_with_tags("component_connections_active", [("listener_type", listener_type)]),
158        packet_receive_success: builder.register_counter_with_tags(
159            "component_packets_received_total",
160            [("listener_type", listener_type), ("state", "ok")],
161        ),
162        packet_receive_failure: builder.register_counter_with_tags(
163            "component_packets_received_total",
164            [("listener_type", listener_type), ("state", "error")],
165        ),
166        packets_forwarded: builder.register_counter_with_tags(
167            "component_packets_forwarded_total",
168            [("listener_type", listener_type), ("state", "ok")],
169        ),
170        bytes_forwarded: builder
171            .register_counter_with_tags("component_bytes_forwarded_total", [("listener_type", listener_type)]),
172        packet_forwarding_errors: builder.register_counter_with_tags(
173            "component_packets_forwarded_total",
174            [("listener_type", listener_type), ("state", "error")],
175        ),
176        failed_context_resolve_total: builder.register_counter("component_failed_context_resolve_total"),
177    }
