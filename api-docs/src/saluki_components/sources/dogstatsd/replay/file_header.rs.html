1//! Provides parsing and writing of Datadog capture file headers.
2
3use std::io::{self, Write};
4
5use saluki_error::{generic_error, GenericError};
6
7/// Datadog capture file format version.
8pub(super) const DATADOG_FILE_VERSION: u8 = 3;
9
10/// Minimum file version that carries a tagger state trailer.
11pub(super) const MIN_STATE_VERSION: u8 = 2;
12
13/// Minimum file version that records timestamps in nanoseconds (earlier versions use seconds).
14pub(super) const MIN_NANO_VERSION: u8 = 3;
15
16pub(super) const VERSION_INDEX: usize = 4;
17pub(super) const DATADOG_HEADER: [u8; 8] = [0xD4, 0x74, 0xD0, 0x60, 0xF0, 0xFF, 0x00, 0x00];
18const ERR_HEADER_WRITE: &str = "capture file header could not be fully written to buffer";
19
20/// Returns whether the buffer begins with a valid Datadog capture header.
21pub(super) fn valid_header(buf: &[u8]) -> bool {
22    if buf.len() < DATADOG_HEADER.len() {
23        return false;
24    }
25
26    for (index, expected) in DATADOG_HEADER.iter().enumerate() {
27        let actual = buf[index];
28        if index == VERSION_INDEX {
29            if actual & expected != *expected {
30                return false;
31            }
32        } else if actual != *expected {
33            return false;
34        }
35    }
36
37    true
38}
39
40/// Parses the Datadog capture file version from the given buffer.
41pub(super) fn file_version(buf: &[u8]) -> Result<u8, GenericError> {
42    if !valid_header(buf) {
43        return Err(generic_error!(
44            "Cannot verify file version from invalid capture header."
45        ));
46    }
47
48    let version = 0xF0 ^ buf[VERSION_INDEX];
49    if version > DATADOG_FILE_VERSION {
50        return Err(generic_error!("Unsupported capture file version: {}.", version));
51    }
52
53    Ok(version)
54}
55
56/// Writes the Datadog capture file header.
57pub(super) fn write_header<W: Write>(writer: &mut W) -> io::Result<()> {
58    let mut header = DATADOG_HEADER;
59    header[VERSION_INDEX] |= DATADOG_FILE_VERSION;
60    let written = writer.write(&header)?;
61    if written < DATADOG_HEADER.len() {
62        return Err(io::Error::new(io::ErrorKind::WriteZero, ERR_HEADER_WRITE));
63    }
64
65    Ok(())
66}

69mod tests {

72    use super::{file_version, valid_header, write_header, DATADOG_FILE_VERSION, DATADOG_HEADER, VERSION_INDEX};

74    #[test]
75    fn test_header_format() {
76        let mut bytes = Vec::new();
77        write_header(&mut bytes).expect("header should write");
78
79        assert!(valid_header(&bytes));
80        assert_eq!(
81            file_version(&bytes).expect("version should parse"),
82            DATADOG_FILE_VERSION
83        );
84
85        for (index, expected) in DATADOG_HEADER.iter().enumerate() {
86            if index == VERSION_INDEX {
87                assert_eq!(bytes[index], expected | DATADOG_FILE_VERSION);
88            } else {
89                assert_eq!(bytes[index], *expected);
90            }
91        }
92    }

94    #[test]
95    fn test_header_format_error() {
96        let test_cases = [
97            (
98                "No error but less bytes written than datadogHeader",
99                1,
100                None,
101                ErrorKind::WriteZero,
102            ),
103            (
104                "Error and less bytes written than datadogHeader",
105                1,
106                Some(io::Error::new(ErrorKind::InvalidInput, "invalid")),
107                ErrorKind::InvalidInput,
108            ),
109            (
110                "Error and more bytes written than datadogHeader",
111                500,
112                Some(io::Error::new(ErrorKind::InvalidInput, "invalid")),
113                ErrorKind::InvalidInput,
114            ),
115        ];
116
117        for (name, bytes_to_write, error, expected_kind) in test_cases {
118            let error = match write_header(&mut ErrorWriter { bytes_to_write, error }) {
119                Ok(_) => panic!("{} should fail", name),
120                Err(error) => error,
121            };
122            assert_eq!(error.kind(), expected_kind, "{}", name);
123        }
124    }

134    struct ErrorWriter {
135        bytes_to_write: usize,
136        error: Option<io::Error>,
137    }
138
139    impl io::Write for ErrorWriter {
140        fn write(&mut self, _buf: &[u8]) -> io::Result<usize> {
141            match self.error.take() {
142                Some(error) => Err(error),
143                None => Ok(self.bytes_to_write),
144            }
145        }
146
147        fn flush(&mut self) -> io::Result<()> {
148            Ok(())
