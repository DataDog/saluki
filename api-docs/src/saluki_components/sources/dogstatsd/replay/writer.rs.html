61#[derive(Default)]
62struct WriterState {
63    ongoing: bool,
64    accepting: bool,
65    target_path: Option<PathBuf>,
66    compressed: bool,
67    traffic_tx: Option<SyncSender<CaptureRecord>>,
68}

70enum CaptureFileWriter {
71    Plain(BufWriter<File>),
72    Compressed(ZstdEncoder<'static, BufWriter<File>>),
73}
74
75impl CaptureFileWriter {
76    fn finish(self) -> io::Result<()> {
77        match self {
78            Self::Plain(mut writer) => writer.flush(),
79            Self::Compressed(writer) => {
80                let mut writer = writer.finish()?;
81                writer.flush()
82            }
83        }
84    }
85}

87impl Write for CaptureFileWriter {
88    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
89        match self {
90            Self::Plain(writer) => writer.write(buf),
91            Self::Compressed(writer) => writer.write(buf),
92        }
93    }
94
95    fn flush(&mut self) -> io::Result<()> {
96        match self {
97            Self::Plain(writer) => writer.flush(),
98            Self::Compressed(writer) => writer.flush(),
99        }
100    }
101}

103impl TrafficCaptureWriter {
104    /// Creates a new capture writer with the given queue depth.
105    #[cfg(test)]
106    pub(super) fn new(queue_depth: usize) -> Self {
107        Self::with_workload_provider(queue_depth, None)
108    }
109
110    /// Creates a new capture writer with the given queue depth and optional workload provider.
111    pub(super) fn with_workload_provider(
112        queue_depth: usize, workload_provider: Option<Arc<dyn WorkloadProvider + Send + Sync>>,
113    ) -> Self {
114        Self {
115            queue_depth: queue_depth.max(MIN_CAPTURE_DEPTH),
116            workload_provider,
117            state: Arc::new(Mutex::new(WriterState::default())),
118        }
119    }
120
121    /// Returns whether a capture session is currently active.
122    pub(super) fn is_ongoing(&self) -> bool {
123        let state = self.state.lock().expect("capture writer mutex poisoned");
124        state.ongoing
125    }
126
127    /// Starts a new capture session.
128    pub(super) fn start_capture(
129        &self, target_dir: CaptureTargetDir, duration: Duration, compressed: bool,
130    ) -> Result<PathBuf, GenericError> {
131        let mut state = self.state.lock().expect("capture writer mutex poisoned");
132
133        if state.ongoing {
134            return Err(generic_error!("capture already in progress"));
135        }
136
137        let target_path = resolve_target_path(&target_dir)?;
138        let mut writer = open_target_writer(&target_path, compressed)?;
139        write_header(&mut writer).map_err(|e| generic_error!("Failed to write capture file header: {}", e))?;
140        let (traffic_tx, traffic_rx) = mpsc::sync_channel(self.queue_depth);
141
142        state.ongoing = true;
143        state.accepting = true;
144        state.target_path = Some(target_path.clone());
145        state.compressed = compressed;
146        state.traffic_tx = Some(traffic_tx);
147
148        let shared_state = Arc::clone(&self.state);
149        let workload_provider = self.workload_provider.clone();
150        if let Err(e) = thread::Builder::new()
151            .name("dogstatsd-capture-writer".into())
152            .spawn(move || run_capture_loop(shared_state, traffic_rx, writer, duration, workload_provider))
153        {
154            state.ongoing = false;
155            state.accepting = false;
156            state.target_path = None;
157            state.compressed = false;
158            state.traffic_tx = None;
159            return Err(generic_error!("Failed to spawn capture writer thread: {}", e));
160        }
161
162        Ok(target_path)
163    }
164
165    /// Stops the current capture session, if one is running.
166    pub(super) fn stop_capture(&self) {
167        let traffic_tx = {
168            let mut state = self.state.lock().expect("capture writer mutex poisoned");
169
170            if !state.ongoing {
171                return;
172            }
173
174            state.accepting = false;
175            state.traffic_tx.take()
176        };
177
178        drop(traffic_tx);
179    }
180
181    /// Attempts to enqueue a captured packet for persistence.
182    pub(super) fn enqueue(&self, record: CaptureRecord) -> bool {
183        let traffic_tx = {
184            let state = self.state.lock().expect("capture writer mutex poisoned");
185            if !state.accepting {
186                return false;
187            }
188
189            state.traffic_tx.clone()
190        };
191
192        match traffic_tx {
193            Some(traffic_tx) => traffic_tx.send(record).is_ok(),
194            None => false,
195        }
196    }
197}

199fn resolve_target_path(target_dir: &CaptureTargetDir) -> Result<PathBuf, GenericError> {
200    if target_dir.as_path().as_os_str().is_empty() {
201        return Err(generic_error!(
202            "DogStatsD capture path is not configured and no explicit path was provided."
203        ));
204    }
205
206    ensure_target_dir_exists(target_dir)?;
207
208    let timestamp = get_unix_timestamp();
209
210    Ok(target_dir
211        .as_path()
212        .join(format!("{}-{}", FILE_TEMPLATE_PREFIX, timestamp)))
213}
214
215fn ensure_target_dir_exists(target_dir: &CaptureTargetDir) -> Result<(), GenericError> {
216    match fs::metadata(target_dir.as_path()) {
217        Ok(metadata) => {
218            if !metadata.is_dir() {
219                return Err(generic_error!(
220                    "DogStatsD capture path '{}' is not a directory.",
221                    target_dir.as_path().display()
222                ));
223            }
224        }
225        Err(e) if e.kind() == io::ErrorKind::NotFound => match target_dir {
226            CaptureTargetDir::Implicit(path) => {
227                fs::create_dir_all(path).map_err(|create_err| {
228                    generic_error!(
229                        "Failed to create DogStatsD capture directory '{}': {}",
230                        path.display(),
231                        create_err
232                    )
233                })?;
234            }
235            CaptureTargetDir::Explicit(path) => {
236                return Err(generic_error!(
237                    "DogStatsD capture path '{}' does not exist.",
238                    path.display()
239                ));
240            }
241        },
242        Err(e) => {
243            return Err(generic_error!(
244                "Failed to inspect DogStatsD capture path '{}': {}",
245                target_dir.as_path().display(),
246                e
247            ));
248        }
249    }
250
251    Ok(())
252}

254fn open_target_writer(target_path: &Path, compressed: bool) -> Result<CaptureFileWriter, GenericError> {
255    let file = File::options()
256        .write(true)
257        .create_new(true)
258        .open(target_path)
259        .map_err(|e| {
260            generic_error!(
261                "Failed to create DogStatsD capture file '{}': {}",
262                target_path.display(),
263                e
264            )
265        })?;
266    let writer = BufWriter::new(file);
267
268    if compressed {
269        let writer = ZstdEncoder::new(writer, 0).map_err(|e| {
270            generic_error!(
271                "Failed to create compressed DogStatsD capture writer for '{}': {}",
272                target_path.display(),
273                e
274            )
275        })?;
276        Ok(CaptureFileWriter::Compressed(writer))
277    } else {
278        Ok(CaptureFileWriter::Plain(writer))
279    }
280}

282fn run_capture_loop(
283    state: Arc<Mutex<WriterState>>, traffic_rx: Receiver<CaptureRecord>, mut writer: CaptureFileWriter,
284    duration: Duration, workload_provider: Option<Arc<dyn WorkloadProvider + Send + Sync>>,
285) {
286    let mut pid_map = FastHashMap::<i32, String>::default();
287    let start = std::time::Instant::now();
288
289    loop {
290        let remaining = duration.saturating_sub(start.elapsed());
291        if remaining.is_zero() {
292            debug!("DogStatsD capture reached requested duration.");
293            break;
294        }
295
296        match traffic_rx.recv_timeout(remaining) {
297            Ok(record) => {
298                if let Err(e) = write_record(&mut writer, record, &mut pid_map) {
299                    error!("Failed to write captured DogStatsD record: {}", e);
300                    break;
301                }
302            }
303            Err(RecvTimeoutError::Timeout) => break,
304            Err(RecvTimeoutError::Disconnected) => break,
305        }
306    }
307
308    if let Err(e) = write_state(&mut writer, pid_map, duration, workload_provider.as_deref()) {
309        warn!("Failed to write DogStatsD capture state trailer: {}", e);
310    }
311
312    if let Err(e) = writer.finish() {
313        error!("Failed to finish DogStatsD capture file: {}", e);
314    }
315
316    let mut state = state.lock().expect("capture writer mutex poisoned");
317    state.ongoing = false;
318    state.accepting = false;
319    state.target_path = None;
320    state.compressed = false;
321    state.traffic_tx = None;
322}

324fn write_record(
325    writer: &mut CaptureFileWriter, record: CaptureRecord, pid_map: &mut FastHashMap<i32, String>,
326) -> io::Result<()> {
327    let CaptureRecord {
328        timestamp_ns,
329        payload,
330        pid,
331        ancillary,
332        container_id,
333    } = record;
334
335    if let (Some(pid), Some(container_id)) = (pid, container_id) {
336        pid_map.insert(pid, container_id);
337    }
338
339    let payload_size = payload.len() as i32;
340    let ancillary_size = ancillary.len() as i32;
341    let message = UnixDogstatsdMsg {
342        timestamp: timestamp_ns,
343        payload_size,
344        payload,
345        pid: pid.unwrap_or_default(),
346        ancillary_size,
347        ancillary,
348    };
349    let serialized = message.encode_to_vec();
350
351    writer.write_all(&(serialized.len() as u32).to_le_bytes())?;
352    writer.write_all(&serialized)?;
353    Ok(())
354}

356fn write_state(
357    writer: &mut CaptureFileWriter, pid_map: FastHashMap<i32, String>, duration: Duration,
358    workload_provider: Option<&(dyn WorkloadProvider + Send + Sync)>,
359) -> io::Result<()> {
360    let state = TaggerState {
361        state: build_saved_entity_state(&pid_map, workload_provider)
362            .into_iter()
363            .collect(),
364        pid_map: pid_map.into_iter().collect(),
365        duration: duration.as_millis().min(i64::MAX as u128) as i64,
366    };
367    let serialized = state.encode_to_vec();
368
369    writer.write_all(&[0, 0, 0, 0])?;
370    writer.write_all(&serialized)?;
371    writer.write_all(&(serialized.len() as u32).to_le_bytes())?;
372    Ok(())
373}

375fn build_saved_entity_state(
376    pid_map: &FastHashMap<i32, String>, workload_provider: Option<&(dyn WorkloadProvider + Send + Sync)>,
377) -> FastHashMap<String, Entity> {
378    let Some(workload_provider) = workload_provider else {
379        return FastHashMap::default();
380    };
381
382    let mut saved_entities = FastHashMap::default();
383    let mut seen_entities = HashSet::new();
384
385    for raw_entity_id in pid_map.values() {
386        let Some(entity_id) = parse_entity_id_str(raw_entity_id) else {
387            warn!(
388                raw_entity_id,
389                "Skipping captured entity with an unsupported ID while building capture state trailer."
390            );
391            continue;
392        };
393
394        if !seen_entities.insert(entity_id.clone()) {
395            continue;
396        }
397
398        let Some(entity) = build_saved_entity(workload_provider, &entity_id) else {
399            continue;
400        };
401
402        saved_entities.insert(entity_id.to_string(), entity);
403    }
404
405    saved_entities
406}

408fn build_saved_entity(
409    workload_provider: &(dyn WorkloadProvider + Send + Sync), entity_id: &EntityId,
410) -> Option<Entity> {
411    let low_tags =
412        sorted_unique_tag_strings(workload_provider.get_tags_for_entity(entity_id, OriginTagCardinality::Low));
413    let orchestrator_cumulative =
414        sorted_unique_tag_strings(workload_provider.get_tags_for_entity(entity_id, OriginTagCardinality::Orchestrator));
415    let high_cumulative =
416        sorted_unique_tag_strings(workload_provider.get_tags_for_entity(entity_id, OriginTagCardinality::High));
417
418    let orchestrator_tags = subtract_tag_strings(orchestrator_cumulative.clone(), &low_tags);
419    let high_tags = subtract_tag_strings(high_cumulative, &orchestrator_cumulative);
420
421    if low_tags.is_empty() && orchestrator_tags.is_empty() && high_tags.is_empty() {
422        return None;
423    }
424
425    Some(Entity {
426        id: Some(entity_id_to_remote_entity_id(entity_id)),
427        low_cardinality_tags: low_tags,
428        orchestrator_cardinality_tags: orchestrator_tags,
429        high_cardinality_tags: high_tags,
430        ..Default::default()
431    })
432}

434fn sorted_unique_tag_strings(tags: Option<SharedTagSet>) -> Vec<String> {
435    let Some(tags) = tags else {
436        return Vec::new();
437    };
438
439    let unique = (&tags)
440        .into_iter()
441        .map(|tag| tag.as_str().to_string())
442        .collect::<BTreeSet<_>>();
443    unique.into_iter().collect()
444}
445
446fn subtract_tag_strings(current: Vec<String>, lower: &[String]) -> Vec<String> {
447    if lower.is_empty() {
448        return current;
449    }
450
451    let lower = lower.iter().map(String::as_str).collect::<HashSet<_>>();
452    current
453        .into_iter()
454        .filter(|tag| !lower.contains(tag.as_str()))
455        .collect()
456}

458fn entity_id_to_remote_entity_id(entity_id: &EntityId) -> RemoteEntityId {
459    match entity_id {
460        EntityId::Container(container_id) => RemoteEntityId {
461            prefix: "container_id".to_string(),
462            uid: container_id.to_string(),
463        },
464        EntityId::PodUid(pod_uid) => RemoteEntityId {
465            prefix: "kubernetes_pod_uid".to_string(),
466            uid: pod_uid.to_string(),
467        },
468        EntityId::Global => RemoteEntityId {
469            prefix: "internal".to_string(),
470            uid: "global-entity-id".to_string(),
471        },
472        EntityId::ContainerPid(pid) => RemoteEntityId {
473            prefix: "container_pid".to_string(),
474            uid: pid.to_string(),
475        },
476        EntityId::ContainerInode(inode) => RemoteEntityId {
477            prefix: "container_inode".to_string(),
478            uid: inode.to_string(),
479        },
480    }
481}

483fn parse_entity_id_str(value: &str) -> Option<EntityId> {
484    const CONTAINER_ID_PREFIX: &str = "container_id://";
485    const POD_UID_PREFIX: &str = "kubernetes_pod_uid://";
486    const CONTAINER_PID_PREFIX: &str = "container_pid://";
487    const CONTAINER_INODE_PREFIX: &str = "container_inode://";
488
489    if let Some(container_id) = value.strip_prefix(CONTAINER_ID_PREFIX) {
490        Some(EntityId::Container(container_id.into()))
491    } else if let Some(pod_uid) = value.strip_prefix(POD_UID_PREFIX) {
492        Some(EntityId::PodUid(pod_uid.into()))
493    } else if let Some(pid) = value.strip_prefix(CONTAINER_PID_PREFIX) {
494        pid.parse().ok().map(EntityId::ContainerPid)
495    } else if let Some(inode) = value.strip_prefix(CONTAINER_INODE_PREFIX) {
496        inode.parse().ok().map(EntityId::ContainerInode)
497    } else if value == "system://global" {
498        Some(EntityId::Global)
499    } else {
500        None
501    }
502}

756    fn assert_capture_contents(bytes: &[u8]) {
757        assert!(datadog_matcher(bytes));
758
759        let header_len = 8;
760        let record_size =
761            u32::from_le_bytes(bytes[header_len..header_len + 4].try_into().expect("record size")) as usize;
762        let record_start = header_len + 4;
763        let record_end = record_start + record_size;
764        let record = UnixDogstatsdMsg::decode(&bytes[record_start..record_end]).expect("record should decode");
765
766        assert_eq!(record.timestamp, 123);
767        assert_eq!(record.payload, b"test");
768        assert_eq!(record.payload_size, 4);
769        assert_eq!(record.pid, 42);
770        assert!(record.ancillary.is_empty());
771        assert_eq!(record.ancillary_size, 0);
772
773        assert_eq!(&bytes[record_end..record_end + 4], &[0, 0, 0, 0]);
774
775        let state = decode_capture_state(bytes);
776
777        assert!(
778            state.state.is_empty(),
779            "captures without a workload provider should not include entity state"
780        );
781        assert_eq!(
782            state.pid_map.get(&42).map(String::as_str),
783            Some("container_id://container-123")
784        );
785        assert!(!state.duration.is_negative());
786    }

798    fn decode_capture_state(bytes: &[u8]) -> TaggerState {
799        let state_size_offset = bytes.len() - 4;
800        let state_size = u32::from_le_bytes(bytes[state_size_offset..].try_into().expect("state size")) as usize;
801        let state_start = state_size_offset - state_size;
802        TaggerState::decode(&bytes[state_start..state_size_offset]).expect("state should decode")
803    }

809    fn wait_until_inactive(writer: &TrafficCaptureWriter) {
810        let deadline = std::time::Instant::now() + Duration::from_secs(2);
811        while writer.is_ongoing() && std::time::Instant::now() < deadline {
812            thread::sleep(Duration::from_millis(10));
813        }
814
815        assert!(!writer.is_ongoing(), "capture writer did not stop in time");
816    }
