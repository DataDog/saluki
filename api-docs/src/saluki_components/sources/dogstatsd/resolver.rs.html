10/// Context resolvers for the DogStatsD source.
11#[derive(Clone)]
12pub struct ContextResolvers {
13    primary: ContextResolver,
14    no_agg: ContextResolver,
15    tags: TagsResolver,
16}
17
18impl ContextResolvers {
19    /// Creates a new `ContextResolvers` instance from the given configuration.
20    ///
21    /// # Errors
22    ///
23    /// If the context resolver string interner size is invalid, or there is an error creating either of the context
24    /// resolvers, an error is returned.
25    pub fn new(
26        config: &DogStatsDConfiguration, context: &ComponentContext,
27        maybe_origin_tags_resolver: Option<DogStatsDOriginTagResolver>,
28    ) -> Result<Self, GenericError> {
29        // We'll use the same string interner size for both context resolvers, which does mean double the usage, but
30        // it's simpler this way for the moment.
31        let context_string_interner_size =
32            NonZeroUsize::new(config.effective_context_string_interner_bytes().as_u64() as usize)
33                .ok_or_else(|| generic_error!("context_string_interner_size must be greater than 0"))?;
34
35        let cached_contexts_limit = config.cached_contexts_limit;
36        let cached_tagsets_limit = config.cached_tagsets_limit;
37        let context_expiry_seconds = Duration::from_secs(config.context_expiry_seconds);
38        let allow_context_heap_allocations = config.allow_context_heap_allocations;
39
40        let interner = GenericMapInterner::new(context_string_interner_size);
41
42        let tags_resolver = TagsResolverBuilder::new(format!("{}/dsd/tags", context.component_id()), interner.clone())?
43            .with_cached_tagsets_limit(cached_tagsets_limit)
44            .with_idle_tagsets_expiration(context_expiry_seconds)
45            .with_heap_allocations(allow_context_heap_allocations)
46            .with_origin_tags_resolver(
47                maybe_origin_tags_resolver
48                    .map(|resolver| -> Arc<dyn saluki_context::origin::OriginTagsResolver> { Arc::new(resolver) }),
49            )
50            .build();
51
52        let primary_resolver = ContextResolverBuilder::from_name(format!("{}/dsd/primary", context.component_id()))?
53            .with_interner_capacity_bytes(context_string_interner_size)
54            .with_cached_contexts_limit(cached_contexts_limit)
55            .with_idle_context_expiration(context_expiry_seconds)
56            .with_heap_allocations(allow_context_heap_allocations)
57            .with_tags_resolver(Some(tags_resolver.clone()))
58            .with_interner(interner.clone())
59            .build();
60
61        let no_agg_resolver = ContextResolverBuilder::from_name(format!("{}/dsd/no_agg", context.component_id()))?
62            .with_interner_capacity_bytes(context_string_interner_size)
63            .without_caching()
64            .with_heap_allocations(allow_context_heap_allocations)
65            .with_tags_resolver(Some(tags_resolver.clone()))
66            .with_interner(interner)
67            .build();
68
69        Ok(ContextResolvers {
70            primary: primary_resolver,
71            no_agg: no_agg_resolver,
72            tags: tags_resolver,
73        })
74    }
75
76    #[cfg(test)]
77    pub fn manual(primary: ContextResolver, no_agg: ContextResolver, tags: TagsResolver) -> Self {
78        ContextResolvers { primary, no_agg, tags }
79    }
80
81    /// Returns a mutable reference to the primary context resolver.
82    ///
83    /// This context resolver should be used for "regular" metrics that require aggregation.
84    pub fn primary(&mut self) -> &mut ContextResolver {
85        &mut self.primary
86    }
87
88    /// Returns a mutable reference to the no-aggregation context resolver.
89    ///
90    /// This context resolver should be used for metrics that don't require aggregation, which implies the metrics had
91    /// a timestamp specified in the payload.
92    pub fn no_agg(&mut self) -> &mut ContextResolver {
93        &mut self.no_agg
94    }
95
96    /// Returns a mutable reference to the tags resolver.
97    pub fn tags(&mut self) -> &mut TagsResolver {
98        &mut self.tags
99    }
100}

113    #[test]
114    fn new_rejects_zero_interner_size() {
115        // Documented `# Errors`: an invalid (zero-byte) string interner size must be rejected.
116        let config = DogStatsDConfiguration {
117            context_string_interner_size_bytes: Some(ByteSize::b(0)),
118            ..Default::default()
119        };
120
121        let err = ContextResolvers::new(&config, &test_context(), None)
122            .err()
123            .expect("a zero interner size must be rejected");
124        assert!(
125            err.to_string()
126                .contains("context_string_interner_size must be greater than 0"),
127            "unexpected error: {}",
128            err
129        );
130    }

132    #[tokio::test]
133    async fn new_builds_working_resolvers_from_the_shared_interner() {
134        // Exercises the real production constructor (rather than the `manual` test bypass): a valid config yields
135        // usable primary and no-aggregation resolvers. The three resolvers share one interner, but that sharing isn't
136        // observable through the public API, so we assert instead that both resolvers construct and resolve contexts
137        // correctly from it.
138        //
139        // NOTE: `#[derive(Default)]` uses each field's own default (a zero interner size) rather than the serde config
140        // defaults, so we must set a non-zero interner size explicitly to reach the success path.
141        let config = DogStatsDConfiguration {
142            context_string_interner_size_bytes: Some(ByteSize::kib(64)),
143            ..Default::default()
144        };
145        let mut resolvers =
146            ContextResolvers::new(&config, &test_context(), None).expect("valid config should build resolvers");
147
148        let primary = resolvers
149            .primary()
150            .resolve("my.metric", ["env:prod"], None)
151            .expect("primary resolver should resolve a context");
152        assert_eq!(primary.name(), "my.metric");
153        assert!(primary.tags().has_tag("env:prod"));
154
155        let no_agg = resolvers
156            .no_agg()
157            .resolve("my.metric", ["env:prod"], None)
158            .expect("no-agg resolver should resolve a context");
159        assert_eq!(no_agg.name(), "my.metric");
160        assert!(no_agg.tags().has_tag("env:prod"));
161    }
