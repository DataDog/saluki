1use bytes::Bytes;
2use saluki_io::{
3    buf::ReadIoBuffer,
4    deser::framing::{Framer, FramingError, LengthDelimitedFramer, NestedFramer, NewlineFramer},
5    net::ListenAddress,
6};
7
8pub enum DsdFramer {
9    NonStream(NewlineFramer),
10    Stream(NestedFramer<NewlineFramer, LengthDelimitedFramer>),
11}
12
13impl Framer for DsdFramer {
14    fn next_frame<B: ReadIoBuffer>(&mut self, buf: &mut B, is_eof: bool) -> Result<Option<Bytes>, FramingError> {
15        match self {
16            Self::NonStream(framer) => framer.next_frame(buf, is_eof),
17            Self::Stream(framer) => framer.next_frame(buf, is_eof),
18        }
19    }
20}
21
22pub fn get_framer(listen_address: &ListenAddress) -> DsdFramer {
23    let newline_framer = NewlineFramer::default().required_on_eof(false);
24
25    match listen_address {
26        ListenAddress::Tcp(_) => DsdFramer::Stream(NestedFramer::new(newline_framer, LengthDelimitedFramer)),
27        ListenAddress::Udp(_) => DsdFramer::NonStream(newline_framer),
28        #[cfg(unix)]
29        ListenAddress::Unixgram(_) => DsdFramer::NonStream(newline_framer),
30        #[cfg(unix)]
31        ListenAddress::Unix(_) => DsdFramer::Stream(NestedFramer::new(newline_framer, LengthDelimitedFramer)),
32    }
33}
