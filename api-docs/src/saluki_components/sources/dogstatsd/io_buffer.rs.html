9/// An ergonomic wrapper around fairly utilizing I/O buffers from an object pool.
10///
11/// DogStatsD uses a bounded elastic I/O buffer pool, which can allocate additional buffers on demand until it reaches
12/// its configured cap. Once the pool reaches that cap, new connections must wait for existing connections to release
13/// buffers. If each connection simply acquired a buffer and reused it until the connection closed, waiters could be
14/// blocked indefinitely even when existing connections were idle.
15///
16/// `IoBufferManager` provides a simple, ergonomic wrapper over a basic pattern of treating the current buffer as
17/// optional, which allows the wrapper to release the current buffer back to the pool, and acquire a new one, all before
18/// returning a reference to the buffer. This provides fairness by ensuring that tasks which are waiting for a buffer
19/// can eventually acquire one once existing tasks are able to reach a consistent point that they can release their
20/// buffer.
21///
22/// ## Release behavior
23///
24/// This wrapper provides two basic behaviors:
25///
26/// - acquire a new buffer when the current buffer doesn't exist
27/// - retain the current buffer (and collapse it) if there is remaining data, _or_ release it if there is no remaining data
28///
29/// When the current buffer still has remaining data, we must preserve the buffer and its data as we could otherwise be
30/// throwing away a partial frame that will be fulfilled by the next socket read. We additionally handle collapsing the
31/// current buffer when there is remaining data, as this ensures that all available capacity is contiguous and directly
32/// follows whatever remaining data exists.
33///
34/// When the current buffer has no remaining data, we can safely release it back to the pool prior to immediately
35/// reacquiring a new buffer.
36///
37/// ## Connectionless streams
38///
39/// For connectionless streams, the current buffer is never released and is always reused. This is because
40/// connectionless streams are often higher volume in terms of the number of socket reads that occur, and are a fixed
41/// number that doesn't change with the amount of traffic that's being sent. This allows us to reduce the overhead
42/// related to acquiring and releasing buffers, ensuring that these inherently lossy streams aren't introducing
43/// additional overhead that might cause unnecessary packet loss.
44pub struct IoBufferManager<'a, O>
45where
46    O: ObjectPool<Item = BytesBuffer>,
47{
48    pool: &'a O,
49    should_retain: bool,
50    current: Option<BytesBuffer>,
51}
52
53impl<'a, O> IoBufferManager<'a, O>
54where
55    O: ObjectPool<Item = BytesBuffer>,
56{
57    /// Creates a new `IoBufferManager` for the given stream and object pool.
58    pub fn new(pool: &'a O, stream: &Stream) -> Self {
59        Self {
60            pool,
61            should_retain: stream.is_connectionless(),
62            current: None,
63        }
64    }
65
66    /// Returns a mutable reference to the current buffer.
67    ///
68    /// This method may or may not release the current buffer depending on if remaining data exists or not. When no
69    /// buffer is available (including after intentionally releasing it), a new buffer will be acquired before
70    /// returning.
71    pub async fn get_buffer_mut(&mut self) -> &mut BytesBuffer {
72        // Consume the current buffer, if it exists.
73        let current = self.current.take().and_then(|mut buffer| {
74            if buffer.has_remaining() || self.should_retain {
75                // Collapse the remaining data in the buffer and continue using it.
76                buffer.collapse();
77                Some(buffer)
78            } else {
79                // Release the buffer back to the pool.
80                None
81            }
82        });
83
84        // Determine our replacement buffer: either we're sticking with the current buffer, or reacquiring a new one.
85        let new_current = match current {
86            Some(buffer) => buffer,
87            None => {
88                let new_buffer = self.pool.acquire().await;
89                trace!(
90                    remaining = new_buffer.remaining(),
91                    capacity = new_buffer.capacity(),
92                    "Acquired new buffer from pool."
93                );
94
95                new_buffer
96            }
97        };
98
99        // Finally, store the new buffer and reference a mutable reference to it.
100        self.current = Some(new_current);
101        self.current.as_mut().unwrap()
102    }
103}

106mod tests {
107    use std::time::Duration;
108
109    use bytes::{Buf as _, BufMut as _};
110    use saluki_core::pooling::FixedSizeObjectPool;

116    const BUFFER_CAPACITY: usize = 16;
117
118    fn test_pool(name: &'static str, count: usize) -> FixedSizeObjectPool<BytesBuffer> {
119        FixedSizeObjectPool::with_builder(name, count, || FixedSizeVec::with_capacity(BUFFER_CAPACITY))
120    }
121
122    #[tokio::test]
123    async fn get_buffer_mut_acquires_buffer_when_none_is_held() {
124        let pool = test_pool("io-buffer-acquire", 1);
125        let mut manager = IoBufferManager {
126            pool: &pool,
127            should_retain: false,
128            current: None,
129        };
130
131        let buffer = manager.get_buffer_mut().await;
132        assert_eq!(buffer.remaining(), 0, "a freshly acquired buffer has nothing to read");
133        assert_eq!(buffer.capacity(), BUFFER_CAPACITY);
134    }
135
136    #[tokio::test]
137    async fn get_buffer_mut_collapses_and_retains_remaining_data() {
138        // When the current buffer still has unread data, get_buffer_mut must keep that buffer, collapse it so the
139        // unread tail moves to the front, and expose the freed capacity for the next socket read -- never dropping a
140        // partial frame.
141        let pool = test_pool("io-buffer-collapse", 1);
142        let mut manager = IoBufferManager {
143            pool: &pool,
144            should_retain: false,
145            current: None,
146        };
147
148        {
149            let buffer = manager.get_buffer_mut().await;
150            buffer.put_slice(b"framed-data"); // 11 bytes written
151            buffer.advance(6); // consume "framed", leaving "-data" as a partial frame
152            assert_eq!(buffer.chunk(), b"-data");
153            // Pre-collapse, only the trailing free space (capacity - written length) is writable.
154            assert_eq!(buffer.remaining_mut(), BUFFER_CAPACITY - 11);
155        }
156
157        let buffer = manager.get_buffer_mut().await;
158        assert_eq!(
159            buffer.chunk(),
160            b"-data",
161            "unread data must be preserved across the call"
162        );
163        assert_eq!(buffer.remaining(), 5);
164        // Post-collapse, the consumed prefix has been reclaimed as writable capacity.
165        assert_eq!(buffer.remaining_mut(), BUFFER_CAPACITY - 5);
166    }
167
168    #[tokio::test]
169    async fn get_buffer_mut_releases_drained_buffer_before_reacquiring() {
170        // Fairness contract (see module docs): a connection-oriented reader that has fully drained its buffer must
171        // release it back to the pool *before* acquiring a replacement. With a single-slot pool this only makes
172        // progress if the release happens first -- an acquire-before-release ordering would deadlock.
173        let pool = test_pool("io-buffer-release", 1);
174        let mut manager = IoBufferManager {
175            pool: &pool,
176            should_retain: false,
177            current: None,
178        };
179
180        {
181            let buffer = manager.get_buffer_mut().await;
182            buffer.put_slice(b"done");
183            buffer.advance(4); // fully consumed: nothing remaining
184            assert_eq!(buffer.remaining(), 0);
185        }
186
187        let reacquired = timeout(Duration::from_secs(5), manager.get_buffer_mut())
188            .await
189            .expect("draining then reacquiring on a saturated pool must not deadlock");
190        assert_eq!(reacquired.remaining(), 0);
191        assert_eq!(reacquired.capacity(), BUFFER_CAPACITY);
192    }
193
194    #[tokio::test]
195    async fn connectionless_stream_reuses_buffer_across_reads() {
196        // Connectionless streams keep their buffer (`should_retain = true`) rather than releasing it, even once
197        // drained. The retain-vs-release decision for an empty buffer isn't observable from the buffer alone (both
198        // yield an empty buffer), but this confirms the connectionless path preserves unread data across calls and
199        // keeps returning a usable buffer once drained.
200        let pool = test_pool("io-buffer-connectionless", 1);
201        let mut manager = IoBufferManager {
202            pool: &pool,
203            should_retain: true,
204            current: None,
205        };
206
207        {
208            let buffer = manager.get_buffer_mut().await;
209            buffer.put_slice(b"partial");
210            buffer.advance(3); // leave "tial" unread
211        }
212
213        {
214            let buffer = manager.get_buffer_mut().await;
215            assert_eq!(
216                buffer.chunk(),
217                b"tial",
218                "connectionless reuse must preserve unread data"
219            );
220            buffer.advance(4); // drain it fully
221        }
222
223        let buffer = manager.get_buffer_mut().await;
224        assert_eq!(buffer.remaining(), 0);
225        assert_eq!(buffer.capacity(), BUFFER_CAPACITY);
226    }
