1//! OTLP `instrumentation_scope` metadata.
2
3use otlp_protos::opentelemetry::proto::common::v1 as otlp_common;
4use otlp_protos::opentelemetry::proto::common::v1::any_value::Value;
5
6use crate::sources::otlp::metrics::internal::utils;
7
8const SCOPE_NAME_TAG: &str = "instrumentation_scope";
9const SCOPE_VERSION_TAG: &str = "instrumentation_scope_version";
10
11/// Creates a new slice of tags from an OTLP `InstrumentationScope`.
12pub fn tags_from_instrumentation_scope_metadata(scope: &otlp_common::InstrumentationScope) -> Vec<String> {
13    let mut tags = Vec::new();
14    tags.push(utils::format_key_value_tag(SCOPE_NAME_TAG, &scope.name));
15    tags.push(utils::format_key_value_tag(SCOPE_VERSION_TAG, &scope.version));
16    for kv in &scope.attributes {
17        if let Some(Value::StringValue(v)) = kv.value.as_ref().and_then(|av| av.value.as_ref()) {
18            tags.push(utils::format_key_value_tag(&kv.key, v));
19        }
20    }
21    tags
22}
