8/// The state we store for each unique time series.
9#[derive(Clone, Debug)]
10pub struct NumberCounter {
11    pub value: f64,
12    pub timestamp: u64,
13    pub start_timestamp: u64,
14}

16/// The state we store for min/max values.
17#[derive(Clone, Debug)]
18pub struct Extrema {
19    pub timestamp: u64,
20    pub start_timestamp: u64,
21    pub stored_extrema: f64,
22}

24/// A cache for storing previous data points to calculate deltas for cumulative metrics.
25pub struct PointsCache {
26    number_points: Cache<String, NumberCounter>,
27    extrema_points: Cache<String, Extrema>,
28}

30impl PointsCache {
31    pub fn from_config(config: OtlpTranslatorConfig) -> Self {
32        let ttl = config.delta_ttl;
33        let interval = config.sweep_interval;
34
35        let number_points = CacheBuilder::from_identifier("otlp/metrics/number_points")
36            .expect("identifier cannot be invalid")
37            .with_time_to_idle(Some(ttl))
38            .with_expiration_interval(interval)
39            .build();
40        let extrema_points = CacheBuilder::from_identifier("otlp/metrics/extrema_points")
41            .expect("identifier cannot be invalid")
42            .with_time_to_idle(Some(ttl))
43            .with_expiration_interval(interval)
44            .build();
45
46        Self {
47            number_points,
48            extrema_points,
49        }
50    }

52    // Submits a new value for a given monotonic metric
53    //
54    // Returns the difference with the last submitted value (ordered by timestamp),
55    // whether this is a first point and whether it should be dropped.
56    pub fn monotonic_diff(
57        &mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64,
58    ) -> (f64, bool, bool) {
59        self.put_and_get_monotonic(dims, start_timestamp, timestamp, value, false)
60    }

63    //
64    // Returns the rate per second since the last submitted value (ordered by timestamp),

66    pub fn monotonic_rate(
67        &mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64,

69        self.put_and_get_monotonic(dims, start_timestamp, timestamp, value, true)

72    // Diff submits a new value for a given non-monotonic metric and returns the difference with the

74    pub fn diff(&mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64) -> (f64, bool) {
75        self.put_and_get_diff(dims, start_timestamp, timestamp, value)
76    }

82    pub fn put_and_check_max(&mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64) -> bool {
83        self.put_and_check_extrema(dims, start_timestamp, timestamp, value, false)
84    }

86    fn put_and_get_diff(&mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64) -> (f64, bool) {
87        let key = dims.get_cache_key();
88
89        let mut dx = 0.0;
90        let mut ok = false;
91
92        if let Some(prev_counter) = self.number_points.get(&key) {
93            if prev_counter.timestamp > timestamp {
94                // Point is older than the one in memory, drop it.
95                return (0.0, false);
96            }
97            dx = value - prev_counter.value;
98            ok = is_not_first_point(start_timestamp, timestamp, prev_counter.start_timestamp);
99        }
100
101        self.number_points.insert(
102            key,
103            NumberCounter {
104                value,
105                timestamp,
106                start_timestamp,
107            },
108        );
109
110        (dx, ok)
111    }
112
113    fn put_and_get_monotonic(
114        &mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, value: f64, rate: bool,
115    ) -> (f64, bool, bool) {
116        // dx, firstPoint, dropPoint
117        let mut dx = 0.0;
118        let mut first_point = true;
119        let drop_point = false;
120
121        let key = dims.get_cache_key();
122
123        if let Some(prev_counter) = self.number_points.get(&key) {
124            if prev_counter.timestamp >= timestamp {
125                // We were given a point with a timestamp older or equal to the one in the cache. This point
126                // should be dropped. We keep the current point in cache.
127                return (0.0, false, true);
128            }
129            dx = value - prev_counter.value;
130            if rate {
131                let time_delta = Duration::from_nanos(timestamp - prev_counter.timestamp);
132                dx /= time_delta.as_secs_f64();
133            }
134            // If !isNotFirstPoint or dx < 0, there has been a reset. We cache the new value, and firstPoint is true.
135            first_point = !is_not_first_point(start_timestamp, timestamp, prev_counter.start_timestamp) || dx < 0.0;
136        }
137
138        self.number_points.insert(
139            key,
140            NumberCounter {
141                value,
142                timestamp,
143                start_timestamp,
144            },
145        );
146
147        (dx, first_point, drop_point)
148    }
149
150    fn put_and_check_extrema(
151        &mut self, dims: &Dimensions, start_timestamp: u64, timestamp: u64, current_extrema: f64, is_min: bool,
152    ) -> bool {
153        let key = dims.get_cache_key();
154        let mut from_last_window = false;
155
156        if let Some(prev_extrema) = self.extrema_points.get(&key) {
157            if prev_extrema.timestamp > timestamp {
158                return false;
159            }
160
161            let is_not_first = is_not_first_point(start_timestamp, timestamp, prev_extrema.start_timestamp);
162            if is_min {
163                from_last_window = (is_not_first && current_extrema < prev_extrema.stored_extrema)
164                    || (current_extrema > prev_extrema.stored_extrema);
165            } else {
166                from_last_window = (is_not_first && current_extrema > prev_extrema.stored_extrema)
167                    || (current_extrema < prev_extrema.stored_extrema);
168            }
169        }
170
171        self.extrema_points.insert(
172            key,
173            Extrema {
174                stored_extrema: current_extrema,
175                timestamp,
176                start_timestamp,
177            },
178        );
179
180        from_last_window
181    }

192/// isNotFirstPoint determines if this is NOT the first point on a cumulative series:
193/// https://github.com/open-telemetry/opentelemetry-specification/blob/v1.19.0/specification/metrics/data-model.md#resets-and-gaps
194fn is_not_first_point(start_ts: u64, ts: u64, old_start_ts: u64) -> bool {
195    if start_ts == 0 {
196        // We don't know the start time, assume the sequence has not been restarted.
197        return true;
198    } else if start_ts != ts && start_ts == old_start_ts {
199        // Since startTs != 0 we know the start time, thus we apply the following rules from the spec:
200        //  - "When StartTimeUnixNano equals TimeUnixNano, a new unbroken sequence of observations begins with a reset at an unknown start time."
201        //  - "[for cumulative series] the StartTimeUnixNano of each point matches the StartTimeUnixNano of the initial observation."
202        return true;
203    }
204    false
205}

445    fn test_diff_known_start() {
446        let mut start_ts = 1;
447        let mut prev_pts = PointsCache::for_tests();
448        let dims = Dimensions {
449            name: "test".to_string(),
450            tags: Default::default(),
451            host: None,
452            origin_id: None,
453        };
454
455        let (_, ok) = prev_pts.diff(&dims, start_ts, 1, 5.0);
456        assert!(!ok, "expected no diff: first point");
457
458        let (_, ok) = prev_pts.diff(&dims, start_ts, 0, 0.0);
459        assert!(!ok, "expected no diff: old point");
460
461        let (dx, ok) = prev_pts.diff(&dims, start_ts, 2, 2.0);
462        assert!(ok, "expected diff: same startTs, not monotonic");
463        assert_eq!(-3.0, dx, "expected diff -3.0 with (1,1,5) point");
464
465        let (dx, ok) = prev_pts.diff(&dims, start_ts, 3, 4.0);
466        assert!(ok, "expected diff: same startTs, not monotonic");
467        assert_eq!(2.0, dx, "expected diff 2.0 with (0,2,2) value");
468
469        // simulate reset with startTs = ts
470        start_ts = 4;
471        let (_, ok) = prev_pts.diff(&dims, start_ts, start_ts, 8.0);
472        assert!(!ok, "expected no diff: reset with unknown start");
473        let (dx, ok) = prev_pts.diff(&dims, start_ts, 5, 9.0);
474        assert!(ok, "expected diff: same startTs, not monotonic");
475        assert_eq!(1.0, dx, "expected diff 1.0 with (4,4,8) value");
476
477        // simulate reset with known start
478        start_ts = 6;
479        let (_, ok) = prev_pts.diff(&dims, start_ts, 7, 1.0);
480        assert!(!ok, "expected no diff: reset with known start");
481        let (dx, ok) = prev_pts.diff(&dims, start_ts, 8, 10.0);
482        assert!(ok, "expected diff: same startTs, not monotonic");
483        assert_eq!(9.0, dx, "expected diff 9.0 with (6,7,1) value");
484    }
