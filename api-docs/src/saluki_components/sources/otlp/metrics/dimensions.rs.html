1//! OTLP metric dimensions.
2
3use otlp_protos::opentelemetry::proto::common::v1 as otlp_common;
4use saluki_context::tags::{SharedTagSet, TagSet};
5use stringtheory::MetaString;
6
7use super::internal::utils::format_key_value_tag;
8
9/// A helper struct for building the identity of a metric.
10#[derive(Clone, Debug, Default)]
11pub struct Dimensions {
12    pub name: String,
13    pub tags: SharedTagSet,
14    pub host: Option<MetaString>,
15    #[allow(dead_code)]
16    pub origin_id: Option<String>,
17}
18
19impl Dimensions {
20    /// Creates a new `Dimensions` with a suffix added to the name.
21    pub fn with_suffix(&self, suffix: &str) -> Self {
22        Self {
23            name: format!("{}.{}", self.name, suffix),
24            tags: self.tags.clone(),
25            host: self.host.clone(),
26            origin_id: self.origin_id.clone(),
27        }
28    }
29
30    /// Creates a new `Dimensions` with additional tags.
31    #[allow(dead_code)]
32    pub fn add_tags(&self, tags_to_add: &[String]) -> Self {
33        let mut new_tags = TagSet::default();
34        for tag in &self.tags {
35            new_tags.insert_tag(tag.clone());
36        }
37        for tag in tags_to_add {
38            new_tags.insert_tag(tag.clone());
39        }
40
41        Self {
42            name: self.name.clone(),
43            tags: new_tags.into_shared(),
44            host: self.host.clone(),
45            origin_id: self.origin_id.clone(),
46        }
47    }
48
49    /// Creates a new `Dimensions` with tags from an OTLP attribute map.
50    ///
51    /// When `shadowing_resource_attributes` is `Some`, any data-point attribute whose key also
52    /// appears in the resource attributes is skipped. This matches the Agent, where resource
53    /// attributes overwrite colliding data-point attributes when they are added as tags.
54    pub fn with_attribute_map(
55        &self, attributes: &[otlp_common::KeyValue], shadowing_resource_attributes: Option<&[otlp_common::KeyValue]>,
56    ) -> Self {
57        if attributes.is_empty() {
58            return self.clone();
59        }
60
61        let mut tag_set = TagSet::default();
62        for kv in attributes {
63            if let Some(resource_attributes) = shadowing_resource_attributes {
64                if resource_attributes.iter().any(|resource_kv| resource_kv.key == kv.key) {
65                    continue;
66                }
67            }
68
69            if let Some(value) = kv.value.as_ref().and_then(|v| v.value.as_ref()) {
70                let v_str = match value {
71                    otlp_common::any_value::Value::StringValue(s) => s.clone(),
72                    otlp_common::any_value::Value::BoolValue(b) => b.to_string(),
73                    otlp_common::any_value::Value::IntValue(i) => i.to_string(),
74                    otlp_common::any_value::Value::DoubleValue(d) => d.to_string(),
75                    // Bytes, arrays, and key-value lists are not converted to tags.
76                    _ => continue,
77                };
78                // Empty values render as `n/a`, matching the Agent's `FormatKeyValueTag`.
79                tag_set.insert_tag(format_key_value_tag(&kv.key, &v_str));
80            }
81        }
82
83        let mut tags = tag_set.into_shared();
84        tags.extend_from_shared(&self.tags);
85
86        Self {
87            name: self.name.clone(),
88            tags,
89            host: self.host.clone(),
90            origin_id: self.origin_id.clone(),
91        }
92    }
93
94    /// Creates a canonical cache key for the dimension set.
95    pub fn get_cache_key(&self) -> String {
96        let mut dimensions: Vec<String> = self.tags.into_iter().map(|t| t.to_string()).collect();
97
98        dimensions.push(format!("name:{}", self.name));
99
100        if let Some(host) = &self.host {
101            dimensions.push(format!("host:{}", host));
102        }
103        if let Some(origin_id) = &self.origin_id {
104            dimensions.push(format!("originID:{}", origin_id));
105        }
106
107        // Sort the dimensions alphabetically to ensure a canonical key.
108        dimensions.sort();
109
110        // Join with a null character separator.
111        dimensions.join("\0")
112    }
113}
