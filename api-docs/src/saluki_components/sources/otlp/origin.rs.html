1use std::sync::Arc;
2
3use saluki_context::{
4    origin::{OriginTagCardinality, OriginTagsResolver, RawOrigin},
5    tags::SharedTagSet,
6};
7use saluki_env::WorkloadProvider;
8use tracing::trace;
9
10#[derive(Clone)]
11pub(super) struct OtlpOriginTagResolver {
12    workload_provider: Arc<dyn WorkloadProvider + Send + Sync>,
13}
14
15impl OtlpOriginTagResolver {
16    pub fn new(workload_provider: Arc<dyn WorkloadProvider + Send + Sync>) -> Self {
17        Self { workload_provider }
18    }
19
20    fn collect_origin_tags(&self, resolved_origin: &saluki_env::workload::origin::ResolvedOrigin) -> SharedTagSet {
21        let mut collected_tags = SharedTagSet::default();
22        let tag_cardinality = resolved_origin.cardinality().unwrap_or(OriginTagCardinality::Low);
23
24        // TODO: Add additional entity IDs that are specific to OTLP.
25        // https://github.com/DataDog/datadog-agent/blob/main/comp/otelcol/otlp/components/processor/infraattributesprocessor/common.go#L158
26        let entity_ids = [
27            resolved_origin.container_id(),
28            resolved_origin.pod_uid(),
29            resolved_origin.process_id(),
30        ];
31
32        for entity_id in entity_ids.iter().flatten() {
33            if let Some(tags) = self.workload_provider.get_tags_for_entity(entity_id, tag_cardinality) {
34                if !tags.is_empty() {
35                    collected_tags.extend_from_shared(&tags);
36                    return collected_tags;
37                }
38            } else {
39                trace!(
40                    ?entity_id,
41                    cardinality = tag_cardinality.as_str(),
42                    "No tags found for entity."
43                );
44            }
45        }
46        collected_tags
47    }
48}
49
50impl OriginTagsResolver for OtlpOriginTagResolver {
51    fn resolve_origin_tags(&self, origin: RawOrigin<'_>) -> SharedTagSet {
52        match self.workload_provider.get_resolved_origin(origin.clone()) {
53            Some(resolved_origin) => self.collect_origin_tags(&resolved_origin),
54            None => {
55                trace!(?origin, "No resolved origin found for raw origin.");
56                SharedTagSet::default()
57            }
58        }
59    }
60}
