1use std::{num::NonZeroUsize, sync::Arc, time::Duration};

3use saluki_context::{ContextResolver, ContextResolverBuilder, TagsResolverBuilder};
4use saluki_core::components::ComponentContext;
5use saluki_error::{generic_error, GenericError};
6use stringtheory::interning::GenericMapInterner;
7
8use crate::sources::{otlp::OtlpOriginTagResolver, OtlpConfiguration};
9
10const RESOLVER_CACHE_EXPIRATION: Duration = Duration::from_secs(30);
11
12/// Creates a new `ContextResolver` instance from the given configuration.
13///
14/// # Errors
15///
16/// If the context resolver string interner size is invalid, or there is an error creating either of the context
17/// resolvers, an error is returned.
18pub fn build_context_resolver(
19    config: &OtlpConfiguration, context: &ComponentContext, maybe_origin_tags_resolver: Option<OtlpOriginTagResolver>,
20) -> Result<ContextResolver, GenericError> {
21    let context_string_interner_size = NonZeroUsize::new(config.context_string_interner_bytes.as_u64() as usize)
22        .ok_or_else(|| generic_error!("context_string_interner_size must be greater than 0"))?;
23
24    let cached_contexts_limit = config.cached_contexts_limit;
25    let cached_tagsets_limit = config.cached_tagsets_limit;
26
27    let interner = GenericMapInterner::new(context_string_interner_size);
28
29    let tags_resolver = TagsResolverBuilder::new(format!("{}/otlp/tags", context.component_id()), interner.clone())?
30        .with_cached_tagsets_limit(cached_tagsets_limit)
31        .with_idle_tagsets_expiration(RESOLVER_CACHE_EXPIRATION)
32        .with_heap_allocations(config.allow_context_heap_allocations)
33        .with_origin_tags_resolver(
34            maybe_origin_tags_resolver
35                .map(|resolver| -> Arc<dyn saluki_context::origin::OriginTagsResolver> { Arc::new(resolver) }),
36        )
37        .build();
38
39    let resolver = ContextResolverBuilder::from_name(format!("{}/otlp/primary", context.component_id()))?
40        .with_interner_capacity_bytes(context_string_interner_size)
41        .with_cached_contexts_limit(cached_contexts_limit)
42        .with_idle_context_expiration(RESOLVER_CACHE_EXPIRATION)
43        .with_heap_allocations(config.allow_context_heap_allocations)
44        .with_tags_resolver(Some(tags_resolver.clone()))
45        .with_interner(interner.clone())
46        .build();
47
48    Ok(resolver)
49}
