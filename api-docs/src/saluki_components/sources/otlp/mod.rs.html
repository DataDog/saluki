55/// Parses `otlp_config.metrics.tags` into a set of tags added to every emitted metric.
56///
57/// The value is a comma-separated list. An empty configuration yields no tags.
58fn parse_configured_metric_tags(raw: &str) -> SharedTagSet {
59    let mut tags = TagSet::default();
60    for tag in raw.split(',') {
61        let tag = tag.trim();
62        if !tag.is_empty() {
63            tags.insert_tag(tag);
64        }
65    }
66    tags.into_shared()
67}

85/// Configuration for the OTLP source.
86#[derive(Deserialize, Default)]
87#[cfg_attr(test, derive(derive_where::DeriveWhere, serde::Serialize))]
88#[cfg_attr(test, derive_where(PartialEq))]
89pub struct OtlpConfiguration {
90    #[serde(skip)]
91    default_hostname: MetaString,
92
93    otlp_config: OtlpConfig,
94
95    /// Total size of the string interner used for contexts.
96    ///
97    /// This controls the amount of memory that can be used to intern metric names and tags. If the interner is full,
98    /// metrics with contexts that haven't already been resolved may or may not be dropped, depending on the value of
99    /// `allow_context_heap_allocations`.
100    #[serde(
101        rename = "otlp_string_interner_size",
102        default = "default_context_string_interner_size"
103    )]
104    context_string_interner_bytes: ByteSize,
105
106    /// The maximum number of cached contexts to allow.
107    ///
108    /// This is the maximum number of resolved contexts that can be cached at any given time. This limit doesn't affect
109    /// the total number of contexts that can be _alive_ at any given time, which is dependent on the interner capacity
110    /// and whether or not heap allocations are allowed.
111    ///
112    /// Defaults to 500,000.
113    #[serde(rename = "otlp_cached_contexts_limit", default = "default_cached_contexts_limit")]
114    cached_contexts_limit: usize,
115
116    /// The maximum number of cached tagsets to allow.
117    ///
118    /// This is the maximum number of resolved tagsets that can be cached at any given time. This limit doesn't affect
119    /// the total number of tagsets that can be _alive_ at any given time, which is dependent on the interner capacity
120    /// and whether or not heap allocations are allowed.
121    ///
122    /// Defaults to 500,000.
123    #[serde(rename = "otlp_cached_tagsets_limit", default = "default_cached_tagsets_limit")]
124    cached_tagsets_limit: usize,
125
126    /// Whether or not to allow heap allocations when resolving contexts.
127    ///
128    /// When resolving contexts during parsing, the metric name and tags are interned to reduce memory usage. The
129    /// interner has a fixed size, however, which means some strings can fail to be interned if the interner is full.
130    /// When set to `true`, we allow these strings to be allocated on the heap like normal, but this can lead to
131    /// increased (unbounded) memory usage. When set to `false`, if the metric name and all of its tags can't be
132    /// interned, the metric is skipped.
133    ///
134    /// Defaults to `true`.
135    #[serde(
136        rename = "otlp_allow_context_heap_allocs",
137        default = "default_allow_context_heap_allocations"
138    )]
139    allow_context_heap_allocations: bool,
140
141    /// Workload provider to utilize for origin detection/enrichment.
142    #[serde(skip)]
143    #[cfg_attr(test, derive_where(skip))]
144    workload_provider: Option<Arc<dyn WorkloadProvider + Send + Sync>>,
145}

147impl OtlpConfiguration {
148    /// Creates a new `OTLPConfiguration` from the given configuration.
149    pub fn from_configuration(config: &GenericConfiguration) -> Result<Self, GenericError> {
150        let mut cfg: Self = config.as_typed()?;
151        cfg.otlp_config.traces.apply_env_overrides(config)?;
152        Ok(cfg)
153    }

155    /// Sets the default hostname used when OTLP metrics do not carry a resource hostname.
156    pub fn with_default_hostname(mut self, hostname: impl Into<MetaString>) -> Self {
157        self.default_hostname = hostname.into();
158        self
159    }
160
161    /// Sets the workload provider to use for configuring origin detection/enrichment.
162    ///
163    /// A workload provider must be set otherwise origin detection/enrichment won't be enabled.
164    ///
165    /// Defaults to unset.
166    pub fn with_workload_provider<W>(mut self, workload_provider: W) -> Self
167    where
168        W: WorkloadProvider + Send + Sync + 'static,
169    {
170        self.workload_provider = Some(Arc::new(workload_provider));
171        self
172    }

176impl SourceBuilder for OtlpConfiguration {
177    fn outputs(&self) -> &[OutputDefinition<EventType>] {
178        static OUTPUTS: LazyLock<Vec<OutputDefinition<EventType>>> = LazyLock::new(|| {
179            vec![
180                OutputDefinition::named_output("metrics", EventType::Metric),
181                OutputDefinition::named_output("logs", EventType::Log),
182                OutputDefinition::named_output("traces", EventType::Trace),
183            ]
184        });
185
186        &OUTPUTS
187    }
188
189    async fn build(&self, context: ComponentContext) -> Result<Box<dyn Source + Send>, GenericError> {
190        if !self.otlp_config.metrics.enabled && !self.otlp_config.logs.enabled && !self.otlp_config.traces.enabled {
191            return Err(generic_error!(
192                "OTLP metrics, logs and traces support is disabled. Please enable at least one of them."
193            ));
194        }
195
196        let grpc_listen_str = format!(
197            "{}://{}",
198            self.otlp_config.receiver.protocols.grpc.transport, self.otlp_config.receiver.protocols.grpc.endpoint
199        );
200        let grpc_endpoint = ListenAddress::try_from(grpc_listen_str.as_str())
201            .map_err(|e| generic_error!("Invalid gRPC endpoint address '{}': {}", grpc_listen_str, e))?;
202
203        // Enforce the current limitation that we only support TCP for gRPC.
204        if !matches!(grpc_endpoint, ListenAddress::Tcp(_)) {
205            return Err(generic_error!("Only 'tcp' transport is supported for OTLP gRPC"));
206        }
207
208        let http_endpoint_str = &self.otlp_config.receiver.protocols.http.endpoint;
209        let http_socket_addr = http_endpoint_str
210            .to_socket_addrs()
211            .map_err(|e| generic_error!("Invalid HTTP endpoint address '{}': {}", http_endpoint_str, e))?
212            .next()
213            .ok_or_else(|| generic_error!("No addresses resolved for HTTP endpoint '{}'", http_endpoint_str))?;
214
215        let maybe_origin_tags_resolver = self.workload_provider.clone().map(OtlpOriginTagResolver::new);
216
217        let context_resolver = build_context_resolver(self, &context, maybe_origin_tags_resolver.clone())?;
218        let metrics_translator_config = metrics::config::OtlpMetricsTranslatorConfig::default()
219            .with_remapping(true)
220            .with_quantiles(true)
221            .with_resource_attributes_as_tags(self.otlp_config.metrics.resource_attributes_as_tags);
222
223        let metric_tags = parse_configured_metric_tags(&self.otlp_config.metrics.tags);
224        let traces_interner_size =
225            std::num::NonZeroUsize::new(self.otlp_config.traces.string_interner_bytes.as_u64() as usize)
226                .ok_or_else(|| generic_error!("otlp_config.traces.string_interner_size must be greater than 0"))?;
227        let traces_translator = OtlpTracesTranslator::new(self.otlp_config.traces.clone(), traces_interner_size);
228        let grpc_max_recv_msg_size_bytes =
229            self.otlp_config.receiver.protocols.grpc.max_recv_msg_size_mib as usize * 1024 * 1024;
230        let metrics = build_metrics(&context);
231
232        Ok(Box::new(Otlp {
233            context_resolver,
234            origin_tag_resolver: maybe_origin_tags_resolver,
235            grpc_endpoint,
236            http_endpoint: ListenAddress::Tcp(http_socket_addr),
237            grpc_max_recv_msg_size_bytes,
238            metrics_translator_config,
239            metric_tags,
240            default_hostname: self.default_hostname.clone(),
241            traces_translator,
242            metrics,
243        }))
244    }
245}

269#[async_trait]
270impl Source for Otlp {
271    async fn run(self: Box<Self>, mut context: SourceContext) -> Result<(), GenericError> {
272        let Self {
273            context_resolver,
274            origin_tag_resolver,
275            grpc_endpoint,
276            http_endpoint,
277            grpc_max_recv_msg_size_bytes,
278            metrics_translator_config,
279            metric_tags,
280            default_hostname,
281            traces_translator,
282            metrics,
283        } = *self;
284
285        let global_shutdown = context.take_shutdown_handle();
286        pin!(global_shutdown);
287
288        let mut health = context.take_health_handle();
289        let memory_limiter = context.topology_context().memory_limiter();
290
291        // Create the internal channel for decoupling the servers from the converter.
292        let (tx, rx) = mpsc::channel::<OtlpResource>(1024);
293
294        let mut converter_shutdown_coordinator = ShutdownCoordinator::default();
295
296        let metrics_translator = OtlpMetricsTranslator::new(
297            metrics_translator_config,
298            default_hostname,
299            context_resolver,
300            metric_tags,
301        )?;
302
303        let thread_pool_handle = context.topology_context().global_thread_pool().clone();
304
305        // Spawn the converter task. This task is shared by both servers.
306        thread_pool_handle.spawn_traced_named(
307            "otlp-resource-converter",
308            run_converter(
309                rx,
310                context.clone(),
311                origin_tag_resolver,
312                converter_shutdown_coordinator.register(),
313                metrics_translator,
314                metrics.clone(),
315                traces_translator,
316            ),
317        );
318
319        let handler = SourceHandler::new(tx);
320        let server_builder = OtlpServerBuilder::new(http_endpoint, grpc_endpoint, grpc_max_recv_msg_size_bytes);
321
322        let (http_shutdown, mut http_error) = server_builder
323            .build(handler, memory_limiter.clone(), thread_pool_handle, metrics)
324            .await?;
325
326        health.mark_ready();
327        debug!("OTLP source started.");
328
329        // Wait for the global shutdown signal, then notify converter to shutdown.
330        loop {
331            select! {
332                _ = &mut global_shutdown => {
333                    debug!("Received shutdown signal.");
334                    break
335                },
336                error = &mut http_error => {
337                    if let Some(error) = error {
338                        debug!(%error, "HTTP server error.");
339                    }
340                    break;
341                },
342                _ = health.live() => continue,
343            }
344        }
345
346        debug!("Stopping OTLP source...");
347
348        http_shutdown.shutdown();
349        converter_shutdown_coordinator.shutdown_and_wait().await;
350
351        debug!("OTLP source stopped.");
352
353        Ok(())
354    }
355}

374#[async_trait]
375impl OtlpHandler for SourceHandler {
376    async fn handle_metrics(&self, body: Bytes) -> Result<(), GenericError> {
377        let request =
378            ExportMetricsServiceRequest::decode(body).error_context("Failed to decode metrics export request.")?;
379
380        for resource_metrics in request.resource_metrics {
381            self.tx
382                .send(OtlpResource::Metrics(resource_metrics))
383                .await
384                .error_context("Failed to send resource metrics to converter: channel is closed.")?;
385        }
386        Ok(())
387    }
388
389    async fn handle_logs(&self, body: Bytes) -> Result<(), GenericError> {
390        let request = ExportLogsServiceRequest::decode(body).error_context("Failed to decode logs export request.")?;
391
392        for resource_logs in request.resource_logs {
393            self.tx
394                .send(OtlpResource::Logs(resource_logs))
395                .await
396                .error_context("Failed to send resource logs to converter: channel is closed.")?;
397        }
398        Ok(())
399    }
400
401    async fn handle_traces(&self, body: Bytes) -> Result<(), GenericError> {
402        let request =
403            ExportTraceServiceRequest::decode(body).error_context("Failed to decode trace export request.")?;
404
405        for resource_spans in request.resource_spans {
406            self.tx
407                .send(OtlpResource::Traces(resource_spans))
408                .await
409                .error_context("Failed to send resource spans to converter: channel is closed.")?;
410        }
411        Ok(())
412    }
413}

415async fn run_converter(
416    mut receiver: mpsc::Receiver<OtlpResource>, source_context: SourceContext,
417    origin_tag_resolver: Option<OtlpOriginTagResolver>, shutdown_handle: ShutdownHandle,
418    mut metrics_translator: OtlpMetricsTranslator, metrics: Metrics, mut traces_translator: OtlpTracesTranslator,
419) {
420    pin!(shutdown_handle);
421
422    debug!("OTLP resource converter task started.");
423
424    // Set a buffer flush interval of 100ms, which will ensure we always flush buffered events at least every 100ms if
425    // we're otherwise idle and not receiving packets from the client.
426    let mut buffer_flush = interval(Duration::from_millis(100));
427    buffer_flush.set_missed_tick_behavior(MissedTickBehavior::Delay);
428
429    let mut metrics_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
430    let mut logs_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
431    let mut traces_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
432
433    loop {
434        select! {
435            Some(otlp_resource) = receiver.recv() => {
436                match otlp_resource {
437                    OtlpResource::Metrics(resource_metrics) => {
438                        match metrics_translator.translate_metrics(resource_metrics, &metrics) {
439                            Ok(events) => {
440                                for event in events {
441                                    let dispatcher = metrics_dispatcher.get_or_insert_with(|| {
442                                        source_context
443                                            .dispatcher()
444                                            .buffered_named("metrics")
445                                            .expect("metrics output should exist")
446                                    });
447                                    if let Err(e) = dispatcher.push(event).await {
448                                        error!(error = %e, "Failed to dispatch metric event.");
449                                    }
450                                }
451                            }
452                            Err(e) => {
453                                error!(error = %e, "Failed to handle resource metrics.");
454                            }
455                        }
456                    }
457                    OtlpResource::Logs(resource_logs) => {
458                        let translator = OtlpLogsTranslator::from_resource_logs(resource_logs, origin_tag_resolver.as_ref());
459                        for log_event in translator {
460                            metrics.logs_received().increment(1);
461
462                            let dispatcher = logs_dispatcher.get_or_insert_with(|| {
463                                source_context
464                                    .dispatcher()
465                                    .buffered_named("logs")
466                                    .expect("logs output should exist")
467                            });
468                            if let Err(e) = dispatcher.push(log_event).await {
469                                error!(error = %e, "Failed to dispatch log event.");
470                            }
471                        }
472                    }
473                    OtlpResource::Traces(resource_spans) => {
474                        for trace_event in traces_translator.translate_spans(resource_spans, &metrics) {
475                            let dispatcher = traces_dispatcher.get_or_insert_with(|| {
476                                source_context
477                                    .dispatcher()
478                                    .buffered_named("traces")
479                                    .expect("traces output should exist")
480                            });
481                            if let Err(e) = dispatcher.push(trace_event).await {
482                                error!(error = %e, "Failed to dispatch trace event.");
483                            }
484                        }
485                    }
486                }
487            },
488            _ = buffer_flush.tick() => {
489                if let Some(dispatcher) = metrics_dispatcher.take() {
490                    if let Err(e) = dispatcher.flush().await {
491                        error!(error = %e, "Failed to flush metric events.");
492                    }
493                }
494                if let Some(dispatcher) = logs_dispatcher.take() {
495                    if let Err(e) = dispatcher.flush().await {
496                        error!(error = %e, "Failed to flush log events.");
497                    }
498                }
499                if let Some(dispatcher) = traces_dispatcher.take() {
500                    if let Err(e) = dispatcher.flush().await {
501                        error!(error = %e, "Failed to flush trace events.");
502                    }
503                }
504            },
505            _ = &mut shutdown_handle => {
506                debug!("Converter task received shutdown signal.");
507                break;
508            }
509        }
510    }
511
512    if let Some(dispatcher) = metrics_dispatcher.take() {
513        if let Err(e) = dispatcher.flush().await {
514            error!(error = %e, "Failed to flush metric events.");
515        }
516    }
517    if let Some(dispatcher) = logs_dispatcher.take() {
518        if let Err(e) = dispatcher.flush().await {
519            error!(error = %e, "Failed to flush log events.");
520        }
521    }
522    if let Some(dispatcher) = traces_dispatcher.take() {
523        if let Err(e) = dispatcher.flush().await {
524            error!(error = %e, "Failed to flush trace events.");
525        }
526    }
527
528    debug!("OTLP resource converter task stopped.");
529}
