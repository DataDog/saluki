55/// Parses `otlp_config.metrics.tags` into a set of tags added to every emitted metric.
56///
57/// The value is a comma-separated list. An empty configuration yields no tags.
58fn parse_configured_metric_tags(raw: &str) -> SharedTagSet {
59    let mut tags = TagSet::default();
60    for tag in raw.split(',') {
61        let tag = tag.trim();
62        if !tag.is_empty() {
63            tags.insert_tag(tag);
64        }
65    }
66    tags.into_shared()
67}

85/// Configuration for the OTLP source.
86#[derive(Deserialize, Default)]
87#[cfg_attr(test, derive(derive_where::DeriveWhere, serde::Serialize))]
88#[cfg_attr(test, derive_where(PartialEq))]
89pub struct OtlpConfiguration {
90    #[serde(skip)]
91    default_hostname: MetaString,
92
93    otlp_config: OtlpConfig,
94
95    /// Total size of the string interner used for contexts.
96    ///
97    /// This controls the amount of memory that can be used to intern metric names and tags. If the interner is full,
98    /// metrics with contexts that haven't already been resolved may or may not be dropped, depending on the value of
99    /// `allow_context_heap_allocations`.
100    #[serde(
101        rename = "otlp_string_interner_size",
102        default = "default_context_string_interner_size"
103    )]
104    context_string_interner_bytes: ByteSize,
105
106    /// The maximum number of cached contexts to allow.
107    ///
108    /// This is the maximum number of resolved contexts that can be cached at any given time. This limit doesn't affect
109    /// the total number of contexts that can be _alive_ at any given time, which is dependent on the interner capacity
110    /// and whether or not heap allocations are allowed.
111    ///
112    /// Defaults to 500,000.
113    #[serde(rename = "otlp_cached_contexts_limit", default = "default_cached_contexts_limit")]
114    cached_contexts_limit: usize,
115
116    /// The maximum number of cached tagsets to allow.
117    ///
118    /// This is the maximum number of resolved tagsets that can be cached at any given time. This limit doesn't affect
119    /// the total number of tagsets that can be _alive_ at any given time, which is dependent on the interner capacity
120    /// and whether or not heap allocations are allowed.
121    ///
122    /// Defaults to 500,000.
123    #[serde(rename = "otlp_cached_tagsets_limit", default = "default_cached_tagsets_limit")]
124    cached_tagsets_limit: usize,
125
126    /// Whether or not to allow heap allocations when resolving contexts.
127    ///
128    /// When resolving contexts during parsing, the metric name and tags are interned to reduce memory usage. The
129    /// interner has a fixed size, however, which means some strings can fail to be interned if the interner is full.
130    /// When set to `true`, we allow these strings to be allocated on the heap like normal, but this can lead to
131    /// increased (unbounded) memory usage. When set to `false`, if the metric name and all of its tags can't be
132    /// interned, the metric is skipped.
133    ///
134    /// Defaults to `true`.
135    #[serde(
136        rename = "otlp_allow_context_heap_allocs",
137        default = "default_allow_context_heap_allocations"
138    )]
139    allow_context_heap_allocations: bool,
140
141    /// Workload provider to utilize for origin detection/enrichment.
142    #[serde(skip)]
143    #[cfg_attr(test, derive_where(skip))]
144    workload_provider: Option<Arc<dyn WorkloadProvider + Send + Sync>>,
145}

200impl SourceBuilder for OtlpConfiguration {
201    fn outputs(&self) -> &[OutputDefinition<EventType>] {
202        static OUTPUTS: LazyLock<Vec<OutputDefinition<EventType>>> = LazyLock::new(|| {
203            vec![
204                OutputDefinition::named_output("metrics", EventType::Metric),
205                OutputDefinition::named_output("logs", EventType::Log),
206                OutputDefinition::named_output("traces", EventType::Trace),
207            ]
208        });
209
210        &OUTPUTS
211    }
212
213    async fn build(&self, context: ComponentContext) -> Result<Box<dyn Source + Send>, GenericError> {
214        if !self.otlp_config.metrics.enabled && !self.otlp_config.logs.enabled && !self.otlp_config.traces.enabled {
215            return Err(generic_error!(
216                "OTLP metrics, logs and traces support is disabled. Please enable at least one of them."
217            ));
218        }
219
220        let grpc_listen_str = format!(
221            "{}://{}",
222            self.otlp_config.receiver.protocols.grpc.transport, self.otlp_config.receiver.protocols.grpc.endpoint
223        );
224        let grpc_endpoint = ListenAddress::try_from(grpc_listen_str.as_str())
225            .map_err(|e| generic_error!("Invalid gRPC endpoint address '{}': {}", grpc_listen_str, e))?;
226
227        // Enforce the current limitation that we only support TCP for gRPC.
228        if !matches!(grpc_endpoint, ListenAddress::Tcp(_)) {
229            return Err(generic_error!("Only 'tcp' transport is supported for OTLP gRPC"));
230        }
231
232        let http_endpoint_str = &self.otlp_config.receiver.protocols.http.endpoint;
233        let http_socket_addr = http_endpoint_str
234            .to_socket_addrs()
235            .map_err(|e| generic_error!("Invalid HTTP endpoint address '{}': {}", http_endpoint_str, e))?
236            .next()
237            .ok_or_else(|| generic_error!("No addresses resolved for HTTP endpoint '{}'", http_endpoint_str))?;
238
239        let maybe_origin_tags_resolver = self.workload_provider.clone().map(OtlpOriginTagResolver::new);
240
241        let context_resolver = build_context_resolver(self, &context, maybe_origin_tags_resolver.clone())?;
242        let metrics_translator_config = self.metrics_translator_config();
243
244        let metric_tags = parse_configured_metric_tags(&self.otlp_config.metrics.tags);
245        let traces_interner_size =
246            std::num::NonZeroUsize::new(self.otlp_config.traces.string_interner_bytes.as_u64() as usize)
247                .ok_or_else(|| generic_error!("otlp_config.traces.string_interner_size must be greater than 0"))?;
248        let traces_translator = OtlpTracesTranslator::new(self.otlp_config.traces.clone(), traces_interner_size);
249        let grpc_max_recv_msg_size_bytes =
250            self.otlp_config.receiver.protocols.grpc.max_recv_msg_size_mib as usize * 1024 * 1024;
251        let metrics = build_metrics(&context);
252
253        Ok(Box::new(Otlp {
254            context_resolver,
255            origin_tag_resolver: maybe_origin_tags_resolver,
256            grpc_endpoint,
257            http_endpoint: ListenAddress::Tcp(http_socket_addr),
258            grpc_max_recv_msg_size_bytes,
259            metrics_translator_config,
260            metric_tags,
261            default_hostname: self.default_hostname.clone(),
262            traces_translator,
263            metrics,
264        }))
265    }
266}

290#[async_trait]
291impl Source for Otlp {
292    async fn run(self: Box<Self>, mut context: SourceContext) -> Result<(), GenericError> {
293        let Self {
294            context_resolver,
295            origin_tag_resolver,
296            grpc_endpoint,
297            http_endpoint,
298            grpc_max_recv_msg_size_bytes,
299            metrics_translator_config,
300            metric_tags,
301            default_hostname,
302            traces_translator,
303            metrics,
304        } = *self;
305
306        let global_shutdown = context.take_shutdown_handle();
307        pin!(global_shutdown);
308
309        let mut health = context.take_health_handle();
310        let memory_limiter = context.topology_context().memory_limiter();
311
312        // Create the internal channel for decoupling the servers from the converter.
313        let (tx, rx) = mpsc::channel::<OtlpResource>(1024);
314
315        let mut converter_shutdown_coordinator = ShutdownCoordinator::default();
316
317        let metrics_translator = OtlpMetricsTranslator::new(
318            metrics_translator_config,
319            default_hostname,
320            context_resolver,
321            metric_tags,
322        )?;
323
324        let thread_pool_handle = context.topology_context().global_thread_pool().clone();
325
326        // Spawn the converter task. This task is shared by both servers.
327        thread_pool_handle.spawn_traced_named(
328            "otlp-resource-converter",
329            run_converter(
330                rx,
331                context.clone(),
332                origin_tag_resolver,
333                converter_shutdown_coordinator.register(),
334                metrics_translator,
335                metrics.clone(),
336                traces_translator,
337            ),
338        );
339
340        let handler = SourceHandler::new(tx);
341        let server_builder = OtlpServerBuilder::new(http_endpoint, grpc_endpoint, grpc_max_recv_msg_size_bytes);
342
343        let (http_shutdown, mut http_error) = server_builder
344            .build(handler, memory_limiter.clone(), thread_pool_handle, metrics)
345            .await?;
346
347        health.mark_ready();
348        debug!("OTLP source started.");
349
350        // Wait for the global shutdown signal, then notify converter to shutdown.
351        loop {
352            select! {
353                _ = &mut global_shutdown => {
354                    debug!("Received shutdown signal.");
355                    break
356                },
357                error = &mut http_error => {
358                    if let Some(error) = error {
359                        debug!(%error, "HTTP server error.");
360                    }
361                    break;
362                },
363                _ = health.live() => continue,
364            }
365        }
366
367        debug!("Stopping OTLP source...");
368
369        http_shutdown.shutdown();
370        converter_shutdown_coordinator.shutdown_and_wait().await;
371
372        debug!("OTLP source stopped.");
373
374        Ok(())
375    }
376}

395#[async_trait]
396impl OtlpHandler for SourceHandler {
397    async fn handle_metrics(&self, body: Bytes) -> Result<(), GenericError> {
398        let request =
399            ExportMetricsServiceRequest::decode(body).error_context("Failed to decode metrics export request.")?;
400
401        for resource_metrics in request.resource_metrics {
402            self.tx
403                .send(OtlpResource::Metrics(resource_metrics))
404                .await
405                .error_context("Failed to send resource metrics to converter: channel is closed.")?;
406        }
407        Ok(())
408    }
409
410    async fn handle_logs(&self, body: Bytes) -> Result<(), GenericError> {
411        let request = ExportLogsServiceRequest::decode(body).error_context("Failed to decode logs export request.")?;
412
413        for resource_logs in request.resource_logs {
414            self.tx
415                .send(OtlpResource::Logs(resource_logs))
416                .await
417                .error_context("Failed to send resource logs to converter: channel is closed.")?;
418        }
419        Ok(())
420    }
421
422    async fn handle_traces(&self, body: Bytes) -> Result<(), GenericError> {
423        let request =
424            ExportTraceServiceRequest::decode(body).error_context("Failed to decode trace export request.")?;
425
426        for resource_spans in request.resource_spans {
427            self.tx
428                .send(OtlpResource::Traces(resource_spans))
429                .await
430                .error_context("Failed to send resource spans to converter: channel is closed.")?;
431        }
432        Ok(())
433    }
434}

436async fn run_converter(
437    mut receiver: mpsc::Receiver<OtlpResource>, source_context: SourceContext,
438    origin_tag_resolver: Option<OtlpOriginTagResolver>, shutdown_handle: ShutdownHandle,
439    mut metrics_translator: OtlpMetricsTranslator, metrics: Metrics, mut traces_translator: OtlpTracesTranslator,
440) {
441    pin!(shutdown_handle);
442
443    debug!("OTLP resource converter task started.");
444
445    // Set a buffer flush interval of 100ms, which will ensure we always flush buffered events at least every 100ms if
446    // we're otherwise idle and not receiving packets from the client.
447    let mut buffer_flush = interval(Duration::from_millis(100));
448    buffer_flush.set_missed_tick_behavior(MissedTickBehavior::Delay);
449
450    let mut metrics_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
451    let mut logs_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
452    let mut traces_dispatcher: Option<BufferedDispatcher<'_, EventsBuffer>> = None;
453
454    loop {
455        select! {
456            Some(otlp_resource) = receiver.recv() => {
457                match otlp_resource {
458                    OtlpResource::Metrics(resource_metrics) => {
459                        match metrics_translator.translate_metrics(resource_metrics, &metrics) {
460                            Ok(events) => {
461                                for event in events {
462                                    let dispatcher = metrics_dispatcher.get_or_insert_with(|| {
463                                        source_context
464                                            .dispatcher()
465                                            .buffered_named("metrics")
466                                            .expect("metrics output should exist")
467                                    });
468                                    if let Err(e) = dispatcher.push(event).await {
469                                        error!(error = %e, "Failed to dispatch metric event.");
470                                    }
471                                }
472                            }
473                            Err(e) => {
474                                error!(error = %e, "Failed to handle resource metrics.");
475                            }
476                        }
477                    }
478                    OtlpResource::Logs(resource_logs) => {
479                        let translator = OtlpLogsTranslator::from_resource_logs(resource_logs, origin_tag_resolver.as_ref());
480                        for log_event in translator {
481                            metrics.logs_received().increment(1);
482
483                            let dispatcher = logs_dispatcher.get_or_insert_with(|| {
484                                source_context
485                                    .dispatcher()
486                                    .buffered_named("logs")
487                                    .expect("logs output should exist")
488                            });
489                            if let Err(e) = dispatcher.push(log_event).await {
490                                error!(error = %e, "Failed to dispatch log event.");
491                            }
492                        }
493                    }
494                    OtlpResource::Traces(resource_spans) => {
495                        for trace_event in traces_translator.translate_spans(resource_spans, &metrics) {
496                            let dispatcher = traces_dispatcher.get_or_insert_with(|| {
497                                source_context
498                                    .dispatcher()
499                                    .buffered_named("traces")
500                                    .expect("traces output should exist")
501                            });
502                            if let Err(e) = dispatcher.push(trace_event).await {
503                                error!(error = %e, "Failed to dispatch trace event.");
504                            }
505                        }
506                    }
507                }
508            },
509            _ = buffer_flush.tick() => {
510                if let Some(dispatcher) = metrics_dispatcher.take() {
511                    if let Err(e) = dispatcher.flush().await {
512                        error!(error = %e, "Failed to flush metric events.");
513                    }
514                }
515                if let Some(dispatcher) = logs_dispatcher.take() {
516                    if let Err(e) = dispatcher.flush().await {
517                        error!(error = %e, "Failed to flush log events.");
518                    }
519                }
520                if let Some(dispatcher) = traces_dispatcher.take() {
521                    if let Err(e) = dispatcher.flush().await {
522                        error!(error = %e, "Failed to flush trace events.");
523                    }
524                }
525            },
526            _ = &mut shutdown_handle => {
527                debug!("Converter task received shutdown signal.");
528                break;
529            }
530        }
531    }
532
533    if let Some(dispatcher) = metrics_dispatcher.take() {
534        if let Err(e) = dispatcher.flush().await {
535            error!(error = %e, "Failed to flush metric events.");
536        }
537    }
538    if let Some(dispatcher) = logs_dispatcher.take() {
539        if let Err(e) = dispatcher.flush().await {
540            error!(error = %e, "Failed to flush log events.");
541        }
542    }
543    if let Some(dispatcher) = traces_dispatcher.take() {
544        if let Err(e) = dispatcher.flush().await {
545            error!(error = %e, "Failed to flush trace events.");
546        }
547    }
548
549    debug!("OTLP resource converter task stopped.");
550}
