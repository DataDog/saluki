10pub fn convert_trace_id(trace_id: &[u8]) -> u64 {
11    if trace_id.len() < 8 {
12        return 0;
13    }
14    u64::from_be_bytes((&trace_id[(trace_id.len() - 8)..]).try_into().unwrap_or_default())
15}
16
17pub fn convert_span_id(span_id: &[u8]) -> u64 {
18    if span_id.len() != 8 {
19        return 0;
20    }
21    u64::from_be_bytes(span_id.try_into().unwrap_or_default())
22}

25    config: OtlpTracesTranslatorConfig,
26}

28impl OtlpTracesTranslator {
29    pub fn new(config: OtlpTracesTranslatorConfig) -> Self {
30        Self { config }
31    }
32
33    pub fn translate_resource_spans(&self, resource_spans: ResourceSpans, metrics: &Metrics) -> Vec<Event> {
34        let resource = resource_spans.resource.unwrap_or_default();
35        let mut traces_by_id: FastHashMap<u64, Vec<DdSpan>> = FastHashMap::default();
36        let ignore_missing_fields = self.config.ignore_missing_datadog_fields;
37
38        for scope_spans in resource_spans.scope_spans {
39            let scope = scope_spans.scope;
40            let scope_ref = scope.as_ref();
41            metrics.spans_received().increment(scope_spans.spans.len() as u64);
42            for span in scope_spans.spans {
43                let trace_id = convert_trace_id(&span.trace_id);
44                let dd_span = otel_span_to_dd_span(
45                    &span,
46                    &resource,
47                    scope_ref,
48                    ignore_missing_fields,
49                    self.config.compute_top_level_by_span_kind,
50                );
51                traces_by_id.entry(trace_id).or_default().push(dd_span);
52            }
53        }
54
55        traces_by_id
56            .into_iter()
57            .filter_map(|(_, spans)| {
58                if spans.is_empty() {
59                    None
60                } else {
61                    Some(Event::Trace(Trace::new(spans)))
62                }
63            })
64            .collect()
65    }
66}
