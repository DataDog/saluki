14pub fn convert_trace_id(trace_id: &[u8]) -> u64 {
15    if trace_id.len() < 8 {
16        return 0;
17    }
18    u64::from_be_bytes((&trace_id[(trace_id.len() - 8)..]).try_into().unwrap_or_default())
19}
20
21pub fn convert_span_id(span_id: &[u8]) -> u64 {
22    if span_id.len() != 8 {
23        return 0;
24    }
25    u64::from_be_bytes(span_id.try_into().unwrap_or_default())
26}

28pub fn resource_attributes_to_tagset(attributes: &[otlp_common::KeyValue]) -> TagSet {
29    let mut tags = TagSet::with_capacity(attributes.len());
30    for kv in attributes {
31        if let Some(key_value) = &kv.value {
32            if let Some(value) = &key_value.value {
33                if let Some(string_value) = otlp_value_to_string(value) {
34                    tags.insert_tag(format!("{}:{}", kv.key, string_value));
35                }
36            }
37        }
38    }
39    tags
40}

51    pub fn translate_resource_spans(&self, resource_spans: ResourceSpans, metrics: &Metrics) -> Vec<Event> {
52        let resource: OtlpResource = resource_spans.resource.unwrap_or_default();
53        let resource_tags: TagSet = resource_attributes_to_tagset(&resource.attributes);
54        let mut traces_by_id: FastHashMap<u64, Vec<DdSpan>> = FastHashMap::default();
55        let ignore_missing_fields = self.config.ignore_missing_datadog_fields;
56
57        for scope_spans in resource_spans.scope_spans {
58            let scope = scope_spans.scope;
59            let scope_ref = scope.as_ref();
60            metrics.spans_received().increment(scope_spans.spans.len() as u64);
61            for span in scope_spans.spans {
62                let trace_id = convert_trace_id(&span.trace_id);
63                let dd_span = otel_span_to_dd_span(
64                    &span,
65                    &resource,
66                    scope_ref,
67                    ignore_missing_fields,
68                    self.config.compute_top_level_by_span_kind,
69                );
70                traces_by_id.entry(trace_id).or_default().push(dd_span);
71            }
72        }
73
74        traces_by_id
75            .into_iter()
76            .filter_map(|(_, spans)| {
77                if spans.is_empty() {
78                    None
79                } else {
80                    Some(Event::Trace(Trace::new(spans, resource_tags.clone())))
81                }
82            })
83            .collect()
84    }
