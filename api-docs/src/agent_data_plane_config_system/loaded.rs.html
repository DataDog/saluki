1//! [`LoadedConfiguration`]: the local configuration sources loaded once, ready to be bound to a
2//! runtime authority.
3//!
4//! The configuration system owns loading. [`LoadedConfiguration::load`] reads the local file and
5//! environment using the requested precedence, then one of the two terminals binds those sources to
6//! an authority: [`LoadedConfiguration::run`] uses the Datadog Agent's config stream, while
7//! [`LoadedConfiguration::standalone`] treats the local sources as authoritative.
8

22// The environment-variable prefix ADP reads (`DD_`). Mirrors
23// `PlatformSettings::get_env_var_prefix()`; hardcoded so the configuration system need not depend on
24// `datadog-agent-commons` for a single constant.
25const ENV_VAR_PREFIX: &str = "DD";
26
27// Bound on the internal channel that forwards Agent updates into the compatibility map. Matches the
28// Agent stream's own channel depth (`RemoteAgentBootstrap::create_config_stream`).
29const COMPAT_FORWARD_CHANNEL_SIZE: usize = 100;
30
31/// Where environment variables sit relative to the configuration file.
32///
33/// One setting, applied identically to the Figment provider order (which flat key wins) and to the
34/// typed env-key overlay. Replaces the raw `EnvOverlayMode` at the configuration system's boundary.
35#[derive(Clone, Copy, Debug, Eq, PartialEq)]
36pub enum EnvPrecedence {
37    /// Environment variables are read below the file: the file wins.
38    ///
39    /// The file takes precedence over environment variables.
40    BeforeFile,
41    /// Environment variables are read after the file: the environment wins. Matches the Datadog
42    /// Agent's precedence and is the value ADP uses.
43    AfterFile,
44    /// Environment variables are not read.
45    Disabled,
46}
47
48impl EnvPrecedence {
49    /// The overlay flag the typed deserializer consumes for the same precedence.
50    //
51    // The remote Agent layer takes precedence over local environment values, so the environment
52    // fills only a nested slot that no higher-authority source supplied. Both file precedence modes
53    // use this rule; `Disabled` suppresses environment relocation entirely.
54    fn overlay_mode(self) -> EnvOverlayMode {
55        match self {
56            EnvPrecedence::Disabled => EnvOverlayMode::Disabled,
57            EnvPrecedence::AfterFile | EnvPrecedence::BeforeFile => EnvOverlayMode::Fallback,
58        }
59    }
60}

62/// Local configuration sources loaded once, not yet bound to a runtime authority.
63///
64/// Holds both the nested typed value and the by-key view so they resolve the local sources
65/// identically.
66pub struct LoadedConfiguration {
67    loader: ConfigurationLoader,
68    // The complete configuration read from the file and environment at startup. It is retained so
69    // the typed path and the by-key compatibility view use the same local sources.
70    base: Value,
71    env: EnvPrecedence,
72}
73
74impl LoadedConfiguration {
75    /// Loads the local sources once from `path` and the environment, at the given precedence.
76    ///
77    /// # Errors
78    ///
79    /// Returns an error if the file cannot be read or is not valid YAML, or if the environment
80    /// prefix is invalid.
81    pub async fn load(path: impl AsRef<Path>, env: EnvPrecedence) -> Result<Self, Error> {
82        let loader = build_loader(path.as_ref(), env)?;
83        let base = build_base(path.as_ref(), env)?;
84        Ok(Self { loader, base, env })
85    }
86
87    /// A static snapshot of the local sources for startup phase configuration.
88    // TODO: remove this backdoor and use typed configuration for the bootstrap phase
89    pub fn raw_config(&self) -> GenericConfiguration {
90        self.loader.bootstrap_generic()
91    }
92
93    /// Connected authority: attach the Datadog Agent's config stream, block for the first
94    /// authoritative snapshot, deserialize and translate it, and start the update task. This is the
95    /// strict startup gate: a snapshot that cannot be received, deserialized, or translated fails
96    /// the boot.
97    ///
98    /// # Errors
99    ///
100    /// Returns an error if the initial configuration cannot be built.
101    pub async fn run(self, config_stream: mpsc::Receiver<ConfigUpdate>) -> Result<ConfigurationSystem, Error> {
102        // The configuration system owns the Agent stream and forwards each update into this
103        // compatibility map, so `raw_map()` keeps serving un-migrated components.
104        let (compat_tx, compat_rx) = mpsc::channel(COMPAT_FORWARD_CHANNEL_SIZE);
105        let compat_map = self.loader.with_dynamic_configuration(compat_rx).into_generic().await?;
106
107        ConfigurationSystem::connected(config_stream, compat_tx, compat_map, self.base, self.env.overlay_mode()).await
108    }
109
110    /// Standalone authority: the local sources are authoritative, with no stream and no update task.
111    /// Also a strict startup gate.
112    ///
113    /// # Errors
114    ///
115    /// Returns an error if the configuration cannot be built from the local sources.
116    pub async fn standalone(self) -> Result<ConfigurationSystem, Error> {
117        let compat_map = self.loader.into_generic().await?;
118        ConfigurationSystem::standalone(compat_map, self.base, self.env.overlay_mode())
119    }
120}

122/// Builds the typed base: the configuration file parsed to its nested shape, with environment
123/// variables read directly and decoded into the schema's shapes on top.
124///
125/// It reads the same file as the by-key compatibility view, normalizes it the same way (drop
126/// null-valued keys; an empty file is an empty object), and then overlays the environment via the
127/// generated Datadog reader, the Saluki-only reader, and the
128/// canonical proxy variables. `env` sets whether the environment overwrites the file (`AfterFile`)
129/// or only fills absent keys (`BeforeFile`); `Disabled` skips the environment entirely.
130fn build_base(path: &Path, env: EnvPrecedence) -> Result<Value, Error> {
131    let text = std::fs::read_to_string(path).map_err(|e| Error::Base {
132        message: format!("read `{}`: {e}", path.display()),
133    })?;
134    let mut base: Value = serde_yaml::from_str(&text).map_err(|e| Error::Base {
135        message: format!("parse `{}`: {e}", path.display()),
136    })?;
137    drop_nulls(&mut base);
138    if base.is_null() {
139        base = Value::Object(serde_json::Map::new());
140    }
141
142    let overwrite = match env {
143        EnvPrecedence::Disabled => return Ok(base),
144        EnvPrecedence::AfterFile => true,
145        EnvPrecedence::BeforeFile => false,
146    };
147    apply_datadog_env(&mut base, overwrite).map_err(|message| Error::Base { message })?;
148    saluki_env_overlay::apply_env(&mut base, overwrite).map_err(|message| Error::Base { message })?;
149    Ok(base)
150}
151
152/// Recursively removes object entries whose value is JSON null, mirroring the compatibility loader's
153/// file normalization: an explicitly null YAML key must not override a model default with null.
154fn drop_nulls(value: &mut Value) {
155    if let Value::Object(map) = value {
156        map.retain(|_, v| !v.is_null());
157        for v in map.values_mut() {
158            drop_nulls(v);
159        }
160    }
161}

163/// Builds the by-key configuration view at the given precedence. File keys are normalized to the
164/// names used by environment variables, and later sources override earlier ones.
165fn build_loader(path: &Path, env: EnvPrecedence) -> Result<ConfigurationLoader, Error> {
166    let loader = ConfigurationLoader::default().with_key_aliases(KEY_ALIASES);
167    let loader = match env {
168        EnvPrecedence::AfterFile => loader
169            .from_yaml(path)?
170            .add_providers([DatadogRemapper::new()])
171            .from_environment(ENV_VAR_PREFIX)?,
172        EnvPrecedence::BeforeFile => loader
173            .add_providers([DatadogRemapper::new()])
174            .from_environment(ENV_VAR_PREFIX)?
175            .from_yaml(path)?,
176        EnvPrecedence::Disabled => loader.from_yaml(path)?,
177    };
178    Ok(loader)
179}

181#[cfg(test)]
182mod tests {
183    use serde_json::json;
184
185    use super::*;
186
187    // The environment is process-global; serialize the tests that mutate it.
188    static ENV_MUTEX: std::sync::Mutex<()> = std::sync::Mutex::new(());
189
190    #[test]
191    fn build_base_composes_file_and_environment_by_precedence() {
192        let _guard = ENV_MUTEX.lock().unwrap();
193        let path = std::env::temp_dir().join(format!("adp_build_base_{}.yaml", std::process::id()));
194        std::fs::write(
195            &path,
196            "dogstatsd_port: 8125\ndogstatsd_non_local_traffic: false\nempty_key:\n",
197        )
198        .unwrap();
199        std::env::set_var("DD_DOGSTATSD_PORT", "9125");
200
201        // AfterFile: the environment wins over the file, the decoded value is a real number, an
202        // explicitly null key is dropped, and an unrelated file key is preserved.
203        let base = build_base(&path, EnvPrecedence::AfterFile).expect("base builds");
204        assert_eq!(base.get("dogstatsd_port"), Some(&json!(9125)));
205        assert_eq!(base.get("dogstatsd_non_local_traffic"), Some(&json!(false)));
206        assert!(base.get("empty_key").is_none());
207
208        // BeforeFile: the file wins over the environment.
209        let base = build_base(&path, EnvPrecedence::BeforeFile).expect("base builds");
210        assert_eq!(base.get("dogstatsd_port"), Some(&json!(8125)));
211
212        // Disabled: the environment is ignored entirely.
213        let base = build_base(&path, EnvPrecedence::Disabled).expect("base builds");
214        assert_eq!(base.get("dogstatsd_port"), Some(&json!(8125)));
215
216        std::env::remove_var("DD_DOGSTATSD_PORT");
217        std::fs::remove_file(&path).ok();
218    }
219
220    #[test]
221    fn build_base_rejects_a_malformed_environment_value() {
222        let _guard = ENV_MUTEX.lock().unwrap();
223        let path = std::env::temp_dir().join(format!("adp_build_base_bad_{}.yaml", std::process::id()));
224        std::fs::write(&path, "dogstatsd_port: 8125\n").unwrap();
225        std::env::set_var("DD_DOGSTATSD_PORT", "not-a-number");
226
227        let result = build_base(&path, EnvPrecedence::AfterFile);
228
229        std::env::remove_var("DD_DOGSTATSD_PORT");
230        std::fs::remove_file(&path).ok();
231        assert!(matches!(result, Err(Error::Base { .. })));
232    }
