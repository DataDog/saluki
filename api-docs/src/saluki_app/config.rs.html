20/// State used for the config API handler.
21#[derive(Clone)]
22pub struct ConfigState {
23    config: GenericConfiguration,
24}
25
26/// An API handler for returning the current configuration.
27///
28/// This handler exposes a single route -- `/config` -- that returns the current configuration in its serialized JSON
29/// form. This allows determining exactly how the process' configuration looks based on the various providers being
30/// used, including any dynamic changes being applied.
31pub struct ConfigAPIHandler {
32    state: ConfigState,
33}
34
35impl ConfigAPIHandler {
36    fn new(config: GenericConfiguration) -> Self {
37        Self {
38            state: ConfigState { config },
39        }
40    }
41
42    async fn config_handler(State(state): State<ConfigState>) -> impl IntoResponse {
43        match state.config.as_typed::<Value>() {
44            Ok(config) => (StatusCode::OK, serde_json::to_string(&config).unwrap()).into_response(),
45            Err(e) => (
46                StatusCode::INTERNAL_SERVER_ERROR,
47                format!("Failed to get configuration: {}", e),
48            )
49                .into_response(),
50        }
51    }
52}
53
54impl APIHandler for ConfigAPIHandler {
55    type State = ConfigState;
56
57    fn generate_initial_state(&self) -> Self::State {
58        self.state.clone()
59    }
60
61    fn generate_routes(&self) -> Router<Self::State> {
62        Router::new().route("/config", get(Self::config_handler))
63    }
64}
65
66/// A worker for exposing an endpoint that returns the current configuration.
67///
68/// When running, the worker asserts a set of routes (based on [`ConfigAPIHandler`]) that allow querying the current
69/// configuration. As the configuration may contain sensitive data, these routes are only present on the privileged API
70/// endpoint.
71pub struct ConfigWorker {
72    handler: ConfigAPIHandler,
73}
74
75impl ConfigWorker {
76    /// Creates a new [`ConfigWorker`] with the given configuration.
77    pub fn new(config: GenericConfiguration) -> Self {
78        Self {
79            handler: ConfigAPIHandler::new(config),
80        }
81    }
82}
83
84#[async_trait]
85impl Supervisable for ConfigWorker {
86    fn name(&self) -> &str {
87        "config-api"
88    }
89
90    async fn initialize(&self, process_shutdown: ShutdownHandle) -> Result<SupervisorFuture, InitializationError> {
91        let config_route = DynamicRoute::http(EndpointType::Privileged, &self.handler);
92
93        let config = self.handler.state.config.clone();
94        let flare_handle = DiagnosticHandle::new("runtime_config_dump.yaml", move || {
95            config
96                .as_typed::<serde_json::Value>()
97                .map(|v| serde_json::to_vec_pretty(&v).unwrap_or_default())
98                .unwrap_or_default()
99        });
100
101        Ok(Box::pin(async move {
102            let dataspace =
103                DataspaceRegistry::try_current().ok_or_else(|| generic_error!("Dataspace not available."))?;
104
105            dataspace.assert(config_route, "config-api");
106            dataspace.assert(flare_handle, "diag-config");
107
108            process_shutdown.await;
109            Ok(())
110        }))
111    }
112}
