15/// Initializes the metrics subsystem for `metrics`.
16///
17/// The given prefix is used to namespace all metrics that are emitted by the application, and is prepended to all
18/// metrics, followed by a period (e.g. `<prefix>.<metric name>`).
19///
20/// ## Errors
21///
22/// If the metrics subsystem was already initialized, an error will be returned.
23pub async fn initialize_metrics(
24    metrics_prefix: impl Into<String>,
25) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
26    // We forward to the implementation in `saluki_core` so that we can have this crate be the collection point of all
27    // helpers/types that are specific to generic application setup/initialization.
28    //
29    // The implementation itself has to live in `saluki_core`, however, to have access to all of the underlying types
30    // that are created and used to install the global recorder, such that they need not be exposed publicly.
31    let filter_handle = saluki_core::observability::metrics::initialize_metrics(metrics_prefix.into()).await?;
32    API_HANDLER
33        .lock()
34        .unwrap()
35        .replace(MetricsAPIHandler::new(filter_handle));
36
37    // We also spawn a background task that collects and emits the Tokio runtime metrics.
38    spawn_traced_named(
39        "tokio-runtime-metrics-collector-primary",
40        collect_runtime_metrics("primary"),
41    );
42
43    Ok(())
44}

46/// Acquires the metrics API handler.
47///
48/// This function is mutable, and consumes the handler if it's present. This means it should only be called once, and
49/// only after metrics have been initialized via `initialize_metrics`.
50///
51/// The metrics API handler can be used to install API routes which allow dynamically controlling the metrics level
52/// filtering. See [`MetricsAPIHandler`] for more information.
53pub fn acquire_metrics_api_handler() -> Option<MetricsAPIHandler> {
54    API_HANDLER.lock().unwrap().take()
55}

57/// Emits the startup metrics for the application.
58///
59/// This is generally meant to be called after the application has been initialized, in order to indicate the
60/// application has completed start-up and is now running.
61///
62/// Must be called after the metrics subsystem has been initialized.
63pub fn emit_startup_metrics() {
64    let app_details = saluki_metadata::get_app_details();
65    let app_version = if app_details.is_dev_build() {
66        format!("{}-dev-{}", app_details.version().raw(), app_details.git_hash(),)
67    } else {
68        app_details.version().raw().to_string()
69    };
70
71    // Emit a "running" metric to indicate that the application is running.
72    gauge!("running", "version" => app_version).set(1.0);
73}

75/// Collects Tokio runtime metrics on the current Tokio runtime.
76///
77/// All metrics generate will include a `runtime_id` label which maps to the given runtime ID. This allows for
78/// differentiating between multiple runtimes that may be running in the same process.
79pub async fn collect_runtime_metrics(runtime_id: &str) {
80    // Get the handle to the runtime we're executing in, and then grab the total number of runtime workers.
81    //
82    // We need the number of workers to properly initialize/register our metrics.
83    let handle = Handle::current();
84    let runtime_metrics = RuntimeMetrics::with_workers(runtime_id, handle.metrics().num_workers());
85
86    // With our metrics registered, enter the main loop where we periodically scrape the metrics.
87    loop {
88        let latest_runtime_metrics = handle.metrics();
89        runtime_metrics.update(&latest_runtime_metrics);
90
91        sleep(Duration::from_secs(5)).await;
92    }
93}

114impl WorkerMetrics {
115    fn with_worker_idx(runtime_id: &str, worker_idx: usize) -> Self {
116        Self::new(runtime_id.to_string(), worker_idx.to_string())
117    }
118
119    fn update(&self, worker_idx: usize, metrics: &tokio::runtime::RuntimeMetrics) {
120        self.local_queue_depth()
121            .set(metrics.worker_local_queue_depth(worker_idx) as f64);
122        self.local_schedule_count()
123            .set(metrics.worker_local_schedule_count(worker_idx) as f64);
124        self.mean_poll_time()
125            .set(metrics.worker_mean_poll_time(worker_idx).as_nanos() as f64);
126        self.noop_count().set(metrics.worker_noop_count(worker_idx) as f64);
127        self.overflow_count()
128            .set(metrics.worker_overflow_count(worker_idx) as f64);
129        self.park_count().set(metrics.worker_park_count(worker_idx) as f64);
130        self.park_unpark_count()
131            .set(metrics.worker_park_unpark_count(worker_idx) as f64);
132        self.poll_count().set(metrics.worker_poll_count(worker_idx) as f64);
133        self.steal_count().set(metrics.worker_steal_count(worker_idx) as f64);
134        self.steal_operations()
135            .set(metrics.worker_steal_operations(worker_idx) as f64);
136        self.total_busy_duration()
137            .set(metrics.worker_total_busy_duration(worker_idx).as_nanos() as f64);
138    }
139}

161struct RuntimeMetrics {
162    global: GlobalRuntimeMetrics,
163    workers: Vec<WorkerMetrics>,
164}
165
166impl RuntimeMetrics {
167    fn with_workers(runtime_id: &str, workers_len: usize) -> Self {
168        let mut workers = Vec::with_capacity(workers_len);
169        for i in 0..workers_len {
170            workers.push(WorkerMetrics::with_worker_idx(runtime_id, i));
171        }
172
173        Self {
174            global: GlobalRuntimeMetrics::new(runtime_id.to_string()),
175            workers,
176        }
177    }
178
179    fn update(&self, metrics: &tokio::runtime::RuntimeMetrics) {
180        self.global.num_alive_tasks().set(metrics.num_alive_tasks() as f64);
181        self.global
182            .blocking_queue_depth()
183            .set(metrics.blocking_queue_depth() as f64);
184        self.global
185            .budget_forced_yield_count()
186            .set(metrics.budget_forced_yield_count() as f64);
187        self.global
188            .global_queue_depth()
189            .set(metrics.global_queue_depth() as f64);
190        self.global
191            .io_driver_fd_deregistered_count()
192            .set(metrics.io_driver_fd_deregistered_count() as f64);
193        self.global
194            .io_driver_fd_registered_count()
195            .set(metrics.io_driver_fd_registered_count() as f64);
196        self.global
197            .io_driver_ready_count()
198            .set(metrics.io_driver_ready_count() as f64);
199        self.global
200            .num_blocking_threads()
201            .set(metrics.num_blocking_threads() as f64);
202        self.global
203            .num_idle_blocking_threads()
204            .set(metrics.num_idle_blocking_threads() as f64);
205        self.global.num_workers().set(metrics.num_workers() as f64);
206        self.global
207            .remote_schedule_count()
208            .set(metrics.remote_schedule_count() as f64);
209        self.global
210            .spawned_tasks_count()
211            .set(metrics.spawned_tasks_count() as f64);
212
213        for (worker_idx, worker) in self.workers.iter().enumerate() {
214            worker.update(worker_idx, metrics);
215        }
216    }
