18enum SyslogDestination {
19    Udp(String, u16),
20    Tcp(String, u16),
21    #[cfg(unix)]
22    Unixgram(PathBuf),
23    #[cfg(unix)]
24    Unix(PathBuf),
25}
26
27impl SyslogDestination {
28    fn parse(uri: &str) -> Result<Self, SyslogError> {
29        let parsed = Url::parse(uri).map_err(|source| SyslogError::InvalidUri {
30            uri: uri.to_string(),
31            reason: source.to_string(),
32        })?;
33
34        match parsed.scheme() {
35            "udp" => parse_network_endpoint(&parsed, uri).map(|(host, port)| Self::Udp(host, port)),
36            "tcp" => parse_network_endpoint(&parsed, uri).map(|(host, port)| Self::Tcp(host, port)),
37            #[cfg(unix)]
38            "unixgram" => parse_unix_path(&parsed, uri).map(Self::Unixgram),
39            #[cfg(unix)]
40            "unix" => parse_unix_path(&parsed, uri).map(Self::Unix),
41            scheme => Err(SyslogError::UnsupportedScheme {
42                scheme: scheme.to_string(),
43            }),
44        }
45    }
46
47    fn connect(&self) -> io::Result<SyslogConnection> {
48        match self {
49            Self::Udp(host, port) => connect_first((host.as_str(), *port), |addr| {
50                let bind_addr = if addr.is_ipv4() { "0.0.0.0:0" } else { "[::]:0" };
51                let socket = UdpSocket::bind(bind_addr)?;
52                socket.connect(addr)?;
53                Ok(socket)
54            })
55            .map(SyslogConnection::Udp),
56            Self::Tcp(host, port) => {
57                connect_first((host.as_str(), *port), TcpStream::connect).map(SyslogConnection::Tcp)
58            }
59            #[cfg(unix)]
60            Self::Unixgram(path) => connect_unixgram(path).map(SyslogConnection::Unixgram),
61            #[cfg(unix)]
62            Self::Unix(path) => UnixStream::connect(path).map(SyslogConnection::Unix),
63        }
64    }
65}

67fn parse_network_endpoint(parsed: &Url, uri: &str) -> Result<(String, u16), SyslogError> {
68    let host = parsed
69        .host_str()
70        .filter(|host| !host.is_empty())
71        .ok_or_else(|| SyslogError::InvalidUri {
72            uri: uri.to_string(),
73            reason: "network syslog URI must include a host".to_string(),
74        })?;
75    let port = parsed.port().ok_or_else(|| SyslogError::InvalidUri {
76        uri: uri.to_string(),
77        reason: "network syslog URI must include a port".to_string(),
78    })?;
79
80    Ok((host.to_string(), port))
81}

83#[cfg(unix)]
84fn parse_unix_path(parsed: &Url, uri: &str) -> Result<PathBuf, SyslogError> {
85    let path = parsed.path();
86    if path.is_empty() {
87        return Err(SyslogError::InvalidUri {
88            uri: uri.to_string(),
89            reason: "Unix socket path cannot be empty".to_string(),
90        });
91    }
92
93    let path = PathBuf::from(path);
94    if !path.is_absolute() {
95        return Err(SyslogError::InvalidUri {
96            uri: uri.to_string(),
97            reason: "Unix socket path must be absolute".to_string(),
98        });
99    }
100
101    Ok(path)
102}

104fn connect_first<T>(addrs: impl ToSocketAddrs, mut connect: impl FnMut(SocketAddr) -> io::Result<T>) -> io::Result<T> {
105    let mut last_error = None;
106    for addr in addrs.to_socket_addrs()? {
107        match connect(addr) {
108            Ok(conn) => return Ok(conn),
109            Err(err) => last_error = Some(err),
110        }
111    }
112
113    Err(last_error.unwrap_or_else(|| {
114        io::Error::new(
115            io::ErrorKind::InvalidInput,
116            "network syslog URI did not resolve to any socket addresses",
117        )
118    }))
119}

121#[cfg(unix)]
122fn connect_unixgram(path: &PathBuf) -> io::Result<UnixDatagram> {
123    let socket = UnixDatagram::unbound()?;
124    socket.connect(path)?;
125    Ok(socket)
126}

128fn write_all_as_len(writer: &mut impl Write, buf: &[u8]) -> io::Result<usize> {
129    writer.write_all(buf)?;
130    Ok(buf.len())
131}

133enum SyslogConnection {
134    Udp(UdpSocket),
135    Tcp(TcpStream),
136    #[cfg(unix)]
137    Unixgram(UnixDatagram),
138    #[cfg(unix)]
139    Unix(UnixStream),
140}
141
142impl Write for SyslogConnection {
143    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
144        match self {
145            Self::Udp(socket) => socket.send(buf),
146            Self::Tcp(stream) => write_all_as_len(stream, buf),
147            #[cfg(unix)]
148            Self::Unixgram(socket) => socket.send(buf),
149            #[cfg(unix)]
150            Self::Unix(stream) => write_all_as_len(stream, buf),
151        }
152    }
153
154    fn flush(&mut self) -> io::Result<()> {
155        match self {
156            Self::Udp(_) => Ok(()),
157            Self::Tcp(stream) => stream.flush(),
158            #[cfg(unix)]
159            Self::Unixgram(_) => Ok(()),
160            #[cfg(unix)]
161            Self::Unix(stream) => stream.flush(),
162        }
163    }
164}
165
166impl Drop for SyslogConnection {
167    fn drop(&mut self) {
168        match self {
169            Self::Tcp(stream) => {
170                let _ = stream.shutdown(Shutdown::Both);
171            }
172            #[cfg(unix)]
173            Self::Unix(stream) => {
174                let _ = stream.shutdown(Shutdown::Both);
175            }
176            Self::Udp(_) => {}
177            #[cfg(unix)]
178            Self::Unixgram(_) => {}
179        }
180    }
181}

183pub(super) struct SyslogWriter {
184    destination: SyslogDestination,
185    connection: Option<SyslogConnection>,
186}
187
188impl SyslogWriter {
189    pub(super) fn from_uri(uri: &str) -> Result<Self, SyslogError> {
190        let destination = SyslogDestination::parse(uri)?;
191        let connection = destination.connect().ok();
192
193        Ok(Self {
194            destination,
195            connection,
196        })
197    }
198}
199
200impl Write for SyslogWriter {
201    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
202        if self.connection.is_none() {
203            self.connection = Some(self.destination.connect()?);
204        }
205
206        let result = self.connection.as_mut().expect("connection should exist").write(buf);
207        if result.is_err() {
208            self.connection = None;
209        }
210
211        result
212    }
213
214    fn flush(&mut self) -> io::Result<()> {
215        match self.connection.as_mut() {
216            Some(connection) => connection.flush(),
217            None => Ok(()),
218        }
219    }
220}

222#[derive(Debug)]
223pub(super) enum SyslogError {
224    InvalidUri { uri: String, reason: String },
225    UnsupportedScheme { scheme: String },
226}
227
228impl fmt::Display for SyslogError {
229    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
230        match self {
231            Self::InvalidUri { uri, reason } => write!(f, "Invalid syslog URI '{}': {}", uri, reason),
232            Self::UnsupportedScheme { scheme } => write!(
233                f,
234                "Unsupported syslog URI scheme '{}'. Supported schemes: {}.",
235                scheme, SUPPORTED_SCHEMES
236            ),
237        }
238    }
239}

443    struct PartialWriter {
444        max_chunk: usize,
445        bytes: Vec<u8>,
446        writes: usize,
447    }

449    impl Write for PartialWriter {
450        fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
451            let len = self.max_chunk.min(buf.len());
452            self.bytes.extend_from_slice(&buf[..len]);
453            self.writes += 1;
454            Ok(len)
455        }

457        fn flush(&mut self) -> io::Result<()> {
458            Ok(())
459        }
