50impl LoggingConfiguration {
51    pub fn from_config(config: &GenericConfiguration) -> Result<Self, GenericError> {
52        let logging_config = config.as_typed()?;
53        Ok(logging_config)
54    }
55
56    pub async fn from_environment() -> Result<Self, GenericError> {
57        // TODO: For the sake of transitioning to this new bootstrapping pattern, we're just creating a configuration
58        // manually here so that we can drive everything through use of `LoggingConfiguration` instead of two different
59        // code paths. That means we want to use `GenericConfiguration` to source our environment variables instead of
60        // querying them manually... mostly to ensure that doing it that way (the way we want to do it overall) is
61        // consistent with how we're doing it by hand at the moment.
62        let config = ConfigurationLoader::default()
63            .from_environment("DD")
64            .expect("Environment variable prefix is not empty.")
65            .into_generic()
66            .await?;
67        Self::from_config(&config)
68    }
69}

71#[derive(Deserialize)]

75impl LogLevel {

77        self.0.clone()

81impl From<LevelFilter> for LogLevel {
82    fn from(level: LevelFilter) -> Self {
83        Self(EnvFilter::default().add_directive(level.into()))
84    }
85}
86
87impl TryFrom<String> for LogLevel {
88    type Error = GenericError;
89
90    fn try_from(value: String) -> Result<Self, Self::Error> {
91        if value.is_empty() {
92            return Err(generic_error!("Log level cannot be empty."));
93        }
94
95        EnvFilter::builder()
96            .parse(value)
97            .map(Self)
98            .error_context("Failed to parse valid log level.")
99    }
