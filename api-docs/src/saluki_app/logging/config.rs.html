62#[derive(Deserialize)]

66impl LogLevel {

72impl From<LevelFilter> for LogLevel {
73    fn from(level: LevelFilter) -> Self {
74        Self(EnvFilter::default().add_directive(level.into()))
75    }
76}
77
78impl TryFrom<String> for LogLevel {
79    type Error = GenericError;
80
81    fn try_from(value: String) -> Result<Self, Self::Error> {
82        if value.is_empty() {
83            return Err(generic_error!("Log level cannot be empty."));
84        }
85
86        EnvFilter::builder()
87            .parse(value)
88            .map(Self)
89            .error_context("Failed to parse valid log level.")
90    }
