6const DEFAULT_LOG_FILE_MAX_SIZE: ByteSize = ByteSize::mib(10);
7const DEFAULT_LOG_FILE_MAX_ROLLS: usize = 1;

9/// Logging configuration.
10///
11/// This is a plain value type. Callers construct instances directly, either via [`simple`] for a basic
12/// console-only configuration or via a translator that maps from the application's wider configuration into this
13/// type, applying any application-specific rules.
14///
15/// [`simple`]: LoggingConfiguration::simple
16pub struct LoggingConfiguration {
17    /// Verbosity directives (e.g., `info`, `debug`, `saluki=trace`).
18    pub log_level: LogLevel,
19
20    /// Whether to emit log records as JSON instead of the default human-readable format.
21    pub log_format_json: bool,
22
23    /// Whether to write log records to standard output.
24    pub log_to_console: bool,
25
26    /// Path to the log file to write to, or empty to disable file logging.
27    pub log_file: String,
28
29    /// Maximum size of a log file before it is rolled over.
30    pub log_file_max_size: ByteSize,
31
32    /// Maximum number of rolled-over log files to retain.
33    pub log_file_max_rolls: usize,
34}
35
36impl LoggingConfiguration {
37    /// Returns a configuration that writes only to the console in human-readable format at INFO level.
38    ///
39    /// Used as a safe default when an application has not yet supplied an explicit configuration.
40    pub fn simple() -> Self {
41        Self {
42            log_level: LevelFilter::INFO.into(),
43            log_format_json: false,
44            log_to_console: true,
45            log_file: String::new(),
46            log_file_max_size: DEFAULT_LOG_FILE_MAX_SIZE,
47            log_file_max_rolls: DEFAULT_LOG_FILE_MAX_ROLLS,
48        }
49    }
50}

52/// A parsed `tracing` log level filter.
53///
54/// Wraps [`EnvFilter`] so it can be deserialized from a string (e.g., `"info"`, `"saluki=trace,info"`).
55#[derive(Deserialize)]
56#[serde(try_from = "String")]
57pub struct LogLevel(EnvFilter);
58
59impl LogLevel {
60    /// Returns the underlying `EnvFilter`.
61    pub fn as_env_filter(&self) -> EnvFilter {
62        self.0.clone()
63    }
64}
65
66impl From<LevelFilter> for LogLevel {
67    fn from(level: LevelFilter) -> Self {
68        Self(EnvFilter::default().add_directive(level.into()))
69    }
70}
71
72impl TryFrom<String> for LogLevel {
73    type Error = GenericError;
74
75    fn try_from(value: String) -> Result<Self, Self::Error> {
76        if value.is_empty() {
77            return Err(generic_error!("Log level cannot be empty."));
78        }
79
80        EnvFilter::builder()
81            .parse(value)
82            .map(Self)
83            .error_context("Failed to parse valid log level.")
84    }
