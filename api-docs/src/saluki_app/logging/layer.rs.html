19pub fn build_formatting_layer<S, W>(config: &LoggingConfiguration, writer: W) -> Box<dyn layer::Layer<S> + Send + Sync>
20where
21    S: Subscriber + for<'a> LookupSpan<'a>,
22    W: for<'writer> MakeWriter<'writer> + Send + Sync + 'static,
23{
24    if config.log_format_json {
25        Layer::new()
26            .json()
27            .flatten_event(true)
28            .with_target(true)
29            .with_file(true)
30            .with_line_number(true)
31            .with_writer(writer)
32            .boxed()
33    } else {
34        Layer::new()
35            .event_format(AgentLikeFormatter::new())
36            .with_writer(writer)
37            .boxed()
38    }
39}

41struct AgentLikeFormatter {
42    app_name: String,
43}
44
45impl AgentLikeFormatter {
46    fn new() -> Self {
47        // Get the configured short name for the current data plane and transform it to a consistent format.
48        //
49        // This will take something like "data-plane" or "Data Plane" and turn it into "DATAPLANE".
50        let app_details = saluki_metadata::get_app_details();
51        let app_name = app_details
52            .short_name()
53            .to_uppercase()
54            .replace("-", "")
55            .replace(" ", "");
56
57        Self { app_name }
58    }
59}

61impl<S, N> FormatEvent<S, N> for AgentLikeFormatter
62where
63    S: Subscriber + for<'a> LookupSpan<'a>,
64    N: for<'a> FormatFields<'a> + 'static,
65{
66    fn format_event(&self, _ctx: &FmtContext<'_, S, N>, mut writer: Writer<'_>, event: &Event<'_>) -> fmt::Result {
67        let metadata = event.metadata();
68
69        // Write the basic log header bits: time, data plane identifier, level, and file/line information:
70        write!(
71            writer,
72            "{} | {} | {} | ",
73            get_delayed_format_now(),
74            self.app_name,
75            metadata.level()
76        )?;
77
78        if let (Some(file), Some(line)) = (metadata.file(), metadata.line()) {
79            write!(writer, "({}:{})", file, line)?;
80        } else {
81            write!(writer, "(unknown:0)")?;
82        }
83
84        // Write the span fields, non-message event fields, and the message field itself:
85        let mut v = AgentLikeFieldVisitor::new(writer.by_ref());
86        event.record(&mut v);
87        v.finish()?;
88
89        writeln!(writer)
90    }
91}

93/// Field visitor that writes event fields similar to the Datadog Agent.
94///
95/// Structured fields in Agent log messages are written as colon-separated key/value pairs, which are then separated by
96/// commas, like so: `key:value,key2:value2,...`. Structured fields are also written before the message, and both are
97/// separated by a pipe character, like other sections of the overall log format.
98///
99/// This means that both structured fields and the log message are written as `| <text>`, leading to something equivalent
100/// to the following when both are present:
101///
102/// ```text
103/// | key:value,key2:value2 | message
104/// ```
105///
106/// # Errors
107///
108/// Errors with writing fields to the given writer are tracked internally. If any operation hits an error during
109/// writing, the error is captured and returned when the visitor is finished. All subsequent operations after an error
110/// are no-ops
111struct AgentLikeFieldVisitor<'writer> {
112    writer: Writer<'writer>,
113    fields_written: usize,
114    message: String,
115    last_result: fmt::Result,
116}
117
118impl<'writer> AgentLikeFieldVisitor<'writer> {
119    fn new(writer: Writer<'writer>) -> Self {
120        Self {
121            writer,
122            fields_written: 0,
123            message: String::new(),
124            last_result: Ok(()),
125        }
126    }
127
128    fn needs_prefix(&self) -> bool {
129        self.fields_written == 0
130    }
131
132    fn needs_separator(&self) -> bool {
133        self.fields_written > 0
134    }
135
136    /// Writes the given field to the writer.
137    ///
138    /// `f` is expected to write the field value in whatever the appropriate format is, and should only write the field
139    /// value: all other aspects -- field name, separators, etc -- are handled outside of the closure.
140    fn try_write(&mut self, field: &field::Field, f: impl FnOnce(&mut dyn fmt::Write) -> fmt::Result) {
141        if self.last_result.is_err() {
142            return;
143        }
144
145        if field.name() == "message" {
146            // We store the `message` field until the end when flushing, as it must come last.
147            //
148            // We still format it here, though.
149            self.last_result = f(&mut self.message);
150        } else {
151            let prefix = if self.needs_prefix() { " | " } else { "" };
152            let separator = if self.needs_separator() { "," } else { "" };
153
154            self.last_result = write!(self.writer, "{}{}{}:", prefix, separator, field.name());
155            if self.last_result.is_err() {
156                return;
157            }
158
159            self.last_result = f(&mut self.writer);
160            if self.last_result.is_ok() {
161                self.fields_written += 1;
162            }
163        }
164    }
165}

167impl field::Visit for AgentLikeFieldVisitor<'_> {
168    fn record_debug(&mut self, field: &field::Field, value: &dyn fmt::Debug) {
169        // Small behavior tweak: we don't care about quoting the message field.
170        self.try_write(field, |w| {
171            if field.name() == "message" {
172                write!(w, "{:?}", value)
173            } else {
174                write!(w, "\"{:?}\"", value)
175            }
176        });
177    }
178
179    fn record_str(&mut self, field: &field::Field, value: &str) {
180        // Small behavior tweak: we don't care about quoting the message field.
181        self.try_write(field, |w| {
182            if field.name() == "message" {
183                w.write_str(value)
184            } else {
185                write!(w, "\"{}\"", value)
186            }
187        });
188    }
189
190    fn record_f64(&mut self, field: &field::Field, value: f64) {
191        let mut float_writer = ryu::Buffer::new();
192        let float_str = float_writer.format(value);
193        self.try_write(field, |w| w.write_str(float_str));
194    }
195
196    fn record_i64(&mut self, field: &field::Field, value: i64) {
197        let mut int_writer = itoa::Buffer::new();
198        let int_str = int_writer.format(value);
199        self.try_write(field, |w| w.write_str(int_str));
200    }
201
202    fn record_u64(&mut self, field: &field::Field, value: u64) {
203        let mut int_writer = itoa::Buffer::new();
204        let int_str = int_writer.format(value);
205        self.try_write(field, |w| w.write_str(int_str));
206    }
207
208    fn record_i128(&mut self, field: &field::Field, value: i128) {
209        let mut int_writer = itoa::Buffer::new();
210        let int_str = int_writer.format(value);
211        self.try_write(field, |w| w.write_str(int_str));
212    }
213
214    fn record_bool(&mut self, field: &field::Field, value: bool) {
215        self.try_write(field, |w| if value { write!(w, "true") } else { write!(w, "false") });
216    }
217
218    fn record_u128(&mut self, field: &field::Field, value: u128) {
219        let mut int_writer = itoa::Buffer::new();
220        let int_str = int_writer.format(value);
221        self.try_write(field, |w| w.write_str(int_str));
222    }
223}

225impl VisitOutput<fmt::Result> for AgentLikeFieldVisitor<'_> {
226    fn finish(mut self) -> fmt::Result {
227        // Check to see if our last write operation was successful or not before trying to write the `message` field.
228        self.last_result?;
229
230        if !self.message.is_empty() {
231            write!(self.writer, " | {}", self.message)
232        } else {
233            Ok(())
234        }
235    }
236}

238/// Gets a delayed formatter for the current time.
239fn get_delayed_format_now() -> DelayedFormat<impl Iterator<Item = &'static Item<'static>> + Clone> {
240    // Determine the system's timezone.
241    //
242    // We fallback to using UTC if something goes wrong during timezone detection.
243    static SYSTEM_TZ: OnceLock<Tz> = OnceLock::new();
244    let system_tz = SYSTEM_TZ.get_or_init(|| {
245        iana_time_zone::get_timezone()
246            .map_err(|_| ())
247            .and_then(|raw_tz| Tz::from_str(&raw_tz).map_err(|_| ()))
248            .unwrap_or(Tz::UTC)
249    });
250
251    // Timestamp format to end up with the equivalent of `2024-12-31 23:59:59 UTC`.
252    static FORMAT_ITEMS: OnceLock<Vec<Item<'static>>> = OnceLock::new();
253    let format_items = FORMAT_ITEMS.get_or_init(|| {
254        let parser = StrftimeItems::new("%Y-%m-%d %H:%M:%S %Z");
255        parser.parse().expect("should not fail to parse datetime format")
256    });
257
258    let now = Utc::now().with_timezone(system_tz);
259    now.format_with_items(format_items.iter())
260}
