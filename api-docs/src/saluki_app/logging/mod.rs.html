31// Number of buffered lines in each non-blocking log writer.
32//
33// This directly influences the idle memory usage since each logging backend (console, file, etc) will have a bounded
34// channel that can hold this many elements, and each element is roughly 32 bytes, so 1,000 elements/lines consumes a
35// minimum of ~32KB, etc.
36const NB_LOG_WRITER_BUFFER_SIZE: usize = 4096;
37
38type OutputStack = Vec<Box<dyn Layer<Registry> + Send + Sync>>;
39
40pub(crate) struct LoggingGuard {
41    worker_guards: Vec<WorkerGuard>,
42    stack_handle: reload::Handle<OutputStack, Registry>,
43    filter_handle: reload::Handle<EnvFilter, Registry>,
44    base_filter: Arc<Mutex<EnvFilter>>,
45}
46
47impl LoggingGuard {
48    /// Reloads the logging subsystem from the given configuration.
49    ///
50    /// Rebuilds the output layer stack and updates the level filter from `config`, swapping both atomically into the
51    /// already-installed `tracing` subscriber. Worker guards for the previous outputs are dropped after the swap, which
52    /// flushes any buffered log lines to their original destinations.
53    pub(crate) fn reload(&mut self, config: LoggingConfiguration) -> Result<(), GenericError> {
54        let (new_stack, new_guards) = build_output_stack(&config)?;
55        let new_filter = config.log_level.as_env_filter();
56
57        self.stack_handle
58            .reload(new_stack)
59            .map_err(|e| generic_error!("Failed to swap logging output stack: {}", e))?;
60        self.filter_handle
61            .reload(new_filter.clone())
62            .map_err(|e| generic_error!("Failed to swap logging filter: {}", e))?;
63        *self.base_filter.lock().unwrap() = new_filter;
64
65        // Drop the old worker guards _after_ the swap so any buffered lines are flushed to their original destinations
66        // before the worker threads exit.
67        let _old_guards = std::mem::replace(&mut self.worker_guards, new_guards);
68
69        Ok(())
70    }
71}

73/// Initializes the logging subsystem for `tracing` with the ability to dynamically update the log filtering directives
74/// at runtime.
75///
76/// An API handler can be acquired (via [`acquire_logging_api_handler`]) to install the API routes which allow for
77/// dynamically controlling the logging level filtering. See [`LoggingAPIHandler`] for more information.
78///
79/// Returns a [`LoggingGuard`] which must be held until the application is about to shutdown, ensuring that any
80/// configured logging backends are able to completely flush any pending logs before the application exits.
81///
82/// # Errors
83///
84/// If the logging subsystem was already initialized, an error will be returned.
85pub(crate) async fn initialize_logging(config: LoggingConfiguration) -> Result<LoggingGuard, GenericError> {
86    // TODO: Support for logging to syslog.
87
88    // Build the initial output stack from the supplied configuration. This is later swappable via
89    // `BootstrapGuard::reload_logging` once the Datadog Agent provides authoritative configuration.
90    let (output_stack, worker_guards) = build_output_stack(&config)?;
91    let (output_layer, stack_handle) = reload::Layer::new(output_stack);
92
93    // Set up our log level filtering and dynamic filter layer.
94    let level_filter = config.log_level.as_env_filter();
95    let (filter_layer, filter_handle) = reload::Layer::new(level_filter.clone());
96
97    // The base filter is the level the override-restore should land on after a `/logging/override` timeout. It starts
98    // as the bootstrap level, and is updated by `LoggingGuard::reload` once the Agent's configuration is applied.
99    let base_filter = Arc::new(Mutex::new(level_filter));
100    set_logging_api_handler(LoggingAPIHandler::new(base_filter.clone(), filter_handle.clone()));
101
102    tracing_subscriber::registry()
103        .with(output_layer.with_filter(filter_layer))
104        .try_init()?;
105
106    Ok(LoggingGuard {
107        worker_guards,
108        stack_handle,
109        filter_handle,
110        base_filter,
111    })
112}

114fn build_output_stack(config: &LoggingConfiguration) -> Result<(OutputStack, Vec<WorkerGuard>), GenericError> {
115    let mut layers: OutputStack = Vec::new();
116    let mut guards = Vec::new();
117
118    if config.log_to_console {
119        let (nb_stdout, guard) = writer_to_nonblocking("console", std::io::stdout());
120        guards.push(guard);
121        layers.push(build_formatting_layer(config, nb_stdout));
122    }
123
124    if !config.log_file.is_empty() {
125        let appender_builder = RollingFileAppenderBase::builder();
126        let appender = appender_builder
127            .filename(config.log_file.clone())
128            .max_filecount(config.log_file_max_rolls)
129            .condition_max_file_size(config.log_file_max_size.as_u64())
130            .build()
131            .map_err(|e| generic_error!("Failed to build log file appender: {}", e))?;
132
133        let (nb_appender, guard) = writer_to_nonblocking("file", appender);
134        guards.push(guard);
135        layers.push(build_formatting_layer(config, nb_appender));
136    }
137
138    Ok((layers, guards))
139}

141fn writer_to_nonblocking<W>(writer_name: &'static str, writer: W) -> (NonBlocking, WorkerGuard)
142where
143    W: Write + Send + 'static,
144{
145    let thread_name = format!("log-writer-{}", writer_name);
146    NonBlockingBuilder::default()
147        .thread_name(&thread_name)
148        .buffered_lines_limit(NB_LOG_WRITER_BUFFER_SIZE)
149        .lossy(true)
150        .finish(writer)
151}
