29/// Logs a message to standard error and exits the process with a non-zero exit code.

32    std::process::exit(1);

35/// Initializes the logging subsystem for `tracing` with the ability to dynamically update the log filtering directives
36/// at runtime.
37///
38/// This function reads the `DD_LOG_LEVEL` environment variable to determine the log level to use. If the environment
39/// variable is not set, the default log level is `INFO`. Additionally, it reads the `DD_LOG_FORMAT_JSON` environment
40/// variable to determine which output format to use. If it is set to `json` (case insensitive), the logs will be
41/// formatted as JSON. If it is set to any other value, or not set at all, the logs will default to a rich, colored,
42/// human-readable format.
43///
44/// An API handler can be acquired (via [`acquire_logging_api_handler`]) to install the API routes which allow for
45/// dynamically controlling the logging level filtering. See [`LoggingAPIHandler`] for more information.
46///
47/// # Errors
48///
49/// If the logging subsystem was already initialized, an error will be returned.
50pub async fn initialize_logging() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
51    // TODO: Support for logging to syslog.
52
53    let logging_config = LoggingConfiguration::from_environment()
54        .await
55        .map_err(|e| format!("Failed to initialize logging configuration: {}", e))?;
56
57    // Set up our log level filtering and dynamic filter layer.
58    let level_filter = logging_config.log_level.as_env_filter();
59    let (filter_layer, reload_handle) = ReloadLayer::new(level_filter.clone());
60    set_logging_api_handler(LoggingAPIHandler::new(level_filter, reload_handle));
61
62    // Build all configured layers: one per output mechanism (console, file, etc).
63    let mut configured_layers = Vec::new();
64
65    if logging_config.log_to_console {
66        let nb_stdout = convert_to_non_blocking_writer(std::io::stdout()).await;
67        configured_layers.push(build_formatting_layer(&logging_config, nb_stdout));
68    }
69
70    if !logging_config.log_file.is_empty() {
71        let appender_builder = RollingFileAppenderBase::builder();
72        let appender = appender_builder
73            .filename(logging_config.log_file.clone())
74            .max_filecount(logging_config.log_file_max_rolls)
75            .condition_max_file_size(logging_config.log_file_max_size.as_u64())
76            .build()?;
77
78        let nb_appender = convert_to_non_blocking_writer(appender).await;
79        configured_layers.push(build_formatting_layer(&logging_config, nb_appender));
80    }
81
82    // `tracing` accepts a `Vec<L>` where `L` implements `Layer<S>`, which acts as a fanout.. and then we're applying
83    // our filter layer on top of that, so that we filter out events once rather than per output layer.
84    tracing_subscriber::registry()
85        .with(configured_layers.with_filter(filter_layer))
86        .try_init()?;
87
88    Ok(())
89}

91async fn convert_to_non_blocking_writer<W>(writer: W) -> NonBlocking
92where
93    W: Write + Send + 'static,
94{
95    let (nb_writer, worker_guard) = tracing_appender::non_blocking(writer);
96
97    // TODO: Right now, we're just shoving this worker guard into an asynchronous task to keep it alive,
98    // but we really should return it so that it can be held on to and only dropped at process exit... we
99    // lack a mechanism to return it cleanly, though, so we'll tackle this once we have a proper bootstrap
100    // procedure.
101    tokio::spawn(async move {
102        let _worker_guard = worker_guard;
103        pending::<()>().await;
104    });
105
106    nb_writer
107}
