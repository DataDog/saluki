6/// Initializes the TLS subsystem.
7///
8/// This ensures the correct cryptography provider is selected for use by the TLS subsystem, which is important for
9/// ensuring, in certain cases, that a validated cryptographic provider is used (e.g., when operating in FIPS mode).
10/// Additionally, it ensures that the platform's root certificate store can be loaded for validating connections.
11///
12/// ## Errors
13///
14/// If the TLS subsystem was already initialized, or if there was an error loading the platform's native certificate
15/// store, an error will be returned.
16pub fn initialize_tls() -> Result<(), GenericError> {
17    initialize_default_crypto_provider()?;
18    load_platform_root_certificates()
19}
