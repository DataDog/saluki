16/// A listener error.
17#[derive(Debug, Snafu)]
18#[snafu(context(suffix(false)))]
19pub enum ListenerError {
20    /// An invalid configuration was given when creating the listener.
21    #[snafu(display("invalid configuration: {}", reason))]
22    InvalidConfiguration {
23        /// Cause of the invalid configuration.
24        reason: &'static str,
25    },
26
27    /// Failed to bind to the listen address.
28    #[snafu(display("failed to bind to listen address {}: {}", address, source))]
29    FailedToBind {
30        /// Listen address.
31        address: ListenAddress,
32
33        /// Source of the error.
34        source: io::Error,
35    },
36
37    /// Failed to configure a setting on the listening socket.
38    #[snafu(display("failed to configure {} for listener on address {}: {}", setting, address, source))]
39    FailedToConfigureListener {
40        /// Listen address.
41        address: ListenAddress,
42
43        /// Name of the setting.
44        setting: &'static str,
45
46        /// Source of the error.
47        source: io::Error,
48    },
49
50    /// Failed to configure a setting on an accepted stream.
51    #[snafu(display("failed to configure {} for {} stream: {}", setting, stream_type, source))]
52    FailedToConfigureStream {
53        /// Name of the setting.
54        setting: &'static str,
55
56        /// Type of stream.
57        stream_type: &'static str,
58
59        /// Source of the error.
60        source: io::Error,
61    },
62
63    /// Failed to accept a new stream from the listener.
64    #[snafu(display("failed to accept new stream for listener on address {}: {}", address, source))]
65    FailedToAccept {
66        /// Listen address.
67        address: ListenAddress,
68
69        /// Source of the error.
70        source: io::Error,
71    },
72}

83/// A network listener.
84///
85/// `Listener` is a abstract listener that works in conjunction with `Stream`, providing the ability to listen on
86/// arbitrary addresses and accept new streams of that address family.
87///
88/// ## Connection-oriented vs connectionless listeners
89///
90/// For listeners on connection-oriented address families (for example, TCP, Unix domain sockets in stream mode), the listener
91/// will listen for an accept new connections in the typical fashion. However, for connectionless address families
92/// (for example, UDP, Unix domain sockets in datagram mode), there is no concept of a "connection" and so nothing to be
93/// continually "accepted". Instead, `Listener` will emit a single `Stream` that can be used to send and receive data
94/// from multiple remote peers.
95///
96/// ## Missing
97///
98/// - Ability to configure `Listener` to emit multiple streams for connectionless address families, allowing for load
99///   balancing. (Only possible for UDP via SO_REUSEPORT, as UDS does not support SO_REUSEPORT.)
100pub struct Listener {
101    listen_address: ListenAddress,
102    inner: ListenerInner,
103    socket_receive_buffer_size: Option<usize>,
104}
105
106impl Listener {
107    /// Creates a new `Listener` from the given listen address.
108    ///
109    /// ## Errors
110    ///
111    /// If the listen address cannot be bound, or if the listener cannot be configured correctly, an error is returned.
112    pub async fn from_listen_address(listen_address: ListenAddress) -> Result<Self, ListenerError> {
113        let inner = match &listen_address {
114            ListenAddress::Tcp(addr) => {
115                TcpListener::bind(addr)
116                    .await
117                    .map(ListenerInner::Tcp)
118                    .context(FailedToBind {
119                        address: listen_address.clone(),
120                    })?
121            }
122            ListenAddress::Udp(addr) => TokioUdpSocket::bind(addr)
123                .await
124                .map(Some)
125                .map(ListenerInner::Udp)
126                .context(FailedToBind {
127                    address: listen_address.clone(),
128                })?,
129            #[cfg(unix)]
130            ListenAddress::Unixgram(addr) => {
131                ensure_unix_socket_free(addr).await.context(FailedToBind {
132                    address: listen_address.clone(),
133                })?;
134
135                let listener = tokio::net::UnixDatagram::bind(addr)
136                    .map(Some)
137                    .map(ListenerInner::Unixgram)
138                    .context(FailedToBind {
139                        address: listen_address.clone(),
140                    })?;
141
142                set_unix_socket_write_only(addr)
143                    .await
144                    .context(FailedToConfigureListener {
145                        address: listen_address.clone(),
146                        setting: "read/write permissions",
147                    })?;
148
149                listener
150            }
151            #[cfg(unix)]
152            ListenAddress::Unix(addr) => {
153                ensure_unix_socket_free(addr).await.context(FailedToBind {
154                    address: listen_address.clone(),
155                })?;
156
157                let listener = tokio::net::UnixListener::bind(addr)
158                    .map(ListenerInner::Unix)
159                    .context(FailedToBind {
160                        address: listen_address.clone(),
161                    })?;
162                set_unix_socket_write_only(addr)
163                    .await
164                    .context(FailedToConfigureListener {
165                        address: listen_address.clone(),
166                        setting: "read/write permissions",
167                    })?;
168
169                listener
170            }
171        };
172
173        Ok(Self {
174            listen_address,
175            inner,
176            socket_receive_buffer_size: None,
177        })
178    }
179
180    /// Sets the socket receive buffer size for this listener.
181    ///
182    /// The receive buffer size applies to accepted streams. `None` keeps the OS default.
183    pub fn with_receive_buffer_size(mut self, socket_receive_buffer_size: Option<usize>) -> Self {
184        self.socket_receive_buffer_size = socket_receive_buffer_size;
185        self
186    }
187
188    /// Gets a reference to the listen address.
189    pub fn listen_address(&self) -> &ListenAddress {
190        &self.listen_address
191    }
192
193    /// Accepts a new stream from the listener.
194    ///
195    /// For connection-oriented address families, this will accept a new connection and return a `Stream` that is bound
196    /// to that remote peer. For connectionless address families, this will return a single `Stream` that will receive
197    /// from multiple remote peers, and no further streams will be emitted.
198    ///
199    /// ## Errors
200    ///
201    /// If the listener fails to accept a new stream, or if the accepted stream cannot be configured correctly, an error
202    /// is returned.
203    pub async fn accept(&mut self) -> Result<Stream, ListenerError> {
204        let stream_type = self.listen_address.listener_type();
205        match &mut self.inner {
206            ListenerInner::Tcp(tcp) => {
207                let (socket, addr) = tcp.accept().await.context(FailedToAccept {
208                    address: self.listen_address.clone(),
209                })?;
210                configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
211                Ok((socket, addr).into())
212            }
213            ListenerInner::Udp(udp) => {
214                // TODO: We only emit a single stream here, but we _could_ do something like an internal configuration
215                // to allow for multiple streams to be emitted, where the socket is bound via SO_REUSEPORT and then we
216                // get load balancing between the sockets.... basically make it possible to parallelize UDP handling if
217                // that's a thing we want to do.
218                if let Some(socket) = udp.take() {
219                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
220                    Ok(socket.into())
221                } else {
222                    pending().await
223                }
224            }
225            #[cfg(unix)]
226            ListenerInner::Unixgram(unix) => {
227                if let Some(socket) = unix.take() {
228                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
229                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
230                        setting: "SO_PASSCRED",
231                        stream_type,
232                    })?;
233                    Ok(socket.into())
234                } else {
235                    pending().await
236                }
237            }
238            #[cfg(unix)]
239            ListenerInner::Unix(unix) => unix
240                .accept()
241                .await
242                .context(FailedToAccept {
243                    address: self.listen_address.clone(),
244                })
245                .and_then(|(socket, _)| {
246                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
247                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
248                        setting: "SO_PASSCRED",
249                        stream_type,
250                    })?;
251                    Ok(socket.into())
252                }),
253        }
254    }
255}
256
257fn configure_stream_socket_receive_buffer_size<'sock, S>(
258    socket: &'sock S, recv_buffer_size: Option<usize>, stream_type: &'static str,
259) -> Result<(), ListenerError>
260where
261    SockRef<'sock>: From<&'sock S>,
262{
263    if let Some(size) = recv_buffer_size {
264        SockRef::from(socket)
265            .set_recv_buffer_size(size)
266            .context(FailedToConfigureStream {
267                setting: SOCKET_RECV_BUFFER_SIZE_SETTING,
268                stream_type,
269            })?;
270    }
271
272    Ok(())
273}

281/// A connection-oriented network listener.
282///
283/// `ConnectionOrientedListener` is conceptually the same as `Listener`, but specifically works with connection-oriented
284/// protocols. This variant is provided to facilitate usages where only a connection-oriented stream makes sense, such
285/// as an HTTP server.
286pub struct ConnectionOrientedListener {
287    listen_address: ListenAddress,
288    inner: ConnectionOrientedListenerInner,
289}
290
291impl ConnectionOrientedListener {
292    /// Creates a new `ConnectionOrientedListener` from the given listen address.
293    ///
294    /// ## Errors
295    ///
296    /// If the listen address is not a connection-oriented address family, or if the listen address cannot be bound, or
297    /// if the listener cannot be configured correctly, an error is returned.
298    pub async fn from_listen_address(listen_address: ListenAddress) -> Result<Self, ListenerError> {
299        let inner = match &listen_address {
300            ListenAddress::Tcp(addr) => TcpListener::bind(addr)
301                .await
302                .map(ConnectionOrientedListenerInner::Tcp)
303                .context(FailedToBind {
304                    address: listen_address.clone(),
305                })?,
306            #[cfg(unix)]
307            ListenAddress::Unix(addr) => {
308                ensure_unix_socket_free(addr).await.context(FailedToBind {
309                    address: listen_address.clone(),
310                })?;
311
312                let listener = tokio::net::UnixListener::bind(addr)
313                    .map(ConnectionOrientedListenerInner::Unix)
314                    .context(FailedToBind {
315                        address: listen_address.clone(),
316                    })?;
317                set_unix_socket_write_only(addr)
318                    .await
319                    .context(FailedToConfigureListener {
320                        address: listen_address.clone(),
321                        setting: "read/write permissions",
322                    })?;
323
324                listener
325            }
326            _ => {
327                return Err(ListenerError::InvalidConfiguration {
328                    reason: "only TCP and Unix listen addresses are supported",
329                })
330            }
331        };
332
333        Ok(Self { listen_address, inner })
334    }
335
336    /// Gets a reference to the listen address.
337    pub fn listen_address(&self) -> &ListenAddress {
338        &self.listen_address
339    }
340
341    /// Returns the actual local socket address this listener is bound to.
342    ///
343    /// This is useful when binding to port `0` to discover the ephemeral port assigned by the OS.
344    pub fn local_addr(&self) -> Result<SocketAddr, ListenerError> {
345        match &self.inner {
346            ConnectionOrientedListenerInner::Tcp(tcp) => tcp.local_addr().context(FailedToConfigureListener {
347                address: self.listen_address.clone(),
348                setting: "local_addr",
349            }),
350            #[cfg(unix)]
351            ConnectionOrientedListenerInner::Unix(_) => Err(ListenerError::InvalidConfiguration {
352                reason: "local_addr is not supported for Unix listeners",
353            }),
354        }
355    }
356
357    /// Accepts a new connection from the listener.
358    ///
359    /// ## Errors
360    ///
361    /// If the listener fails to accept a new connection, or if the accepted connection cannot be configured correctly,
362    /// an error is returned.
363    pub async fn accept(&mut self) -> Result<Connection, ListenerError> {
364        match &mut self.inner {
365            ConnectionOrientedListenerInner::Tcp(tcp) => tcp
366                .accept()
367                .await
368                .map(|(stream, addr)| Connection::Tcp(stream, addr))
369                .context(FailedToAccept {
370                    address: self.listen_address.clone(),
371                }),
372            #[cfg(unix)]
373            ConnectionOrientedListenerInner::Unix(unix) => unix
374                .accept()
375                .await
376                .context(FailedToAccept {
377                    address: self.listen_address.clone(),
378                })
379                .and_then(|(socket, _)| {
380                    let stream_type = self.listen_address.listener_type();
381                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
382                        setting: "SO_PASSCRED",
383                        stream_type,
384                    })?;
385                    Ok(Connection::Unix(socket))
386                }),
387        }
388    }
389}

404    async fn zero_receive_buffer_size_preserves_udp_default() {
405        let default_address = ListenAddress::Udp(([127, 0, 0, 1], 0).into());
406        let mut default_listener = Listener::from_listen_address(default_address)

419            .await
420            .expect("zero-sized listener should bind")
421            .with_receive_buffer_size(None);
422        let zero_stream = zero_listener
423            .accept()
424            .await
425            .expect("zero-sized stream should be accepted");
426        let zero_recv_buffer_size = zero_stream

566    fn tcp_local_addr(listener: &Listener) -> io::Result<SocketAddr> {
567        match &listener.inner {

569            _ => panic!("expected TCP listener"),
