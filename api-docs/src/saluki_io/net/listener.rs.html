19/// A listener error.
20#[derive(Debug, Snafu)]
21#[snafu(context(suffix(false)))]
22pub enum ListenerError {
23    /// An invalid configuration was given when creating the listener.
24    #[snafu(display("invalid configuration: {}", reason))]
25    InvalidConfiguration {
26        /// Cause of the invalid configuration.
27        reason: &'static str,
28    },
29
30    /// Failed to bind to the listen address.
31    #[snafu(display("failed to bind to listen address {}: {}", address, source))]
32    FailedToBind {
33        /// Listen address.
34        address: ListenAddress,
35
36        /// Source of the error.
37        source: io::Error,
38    },
39
40    /// Failed to configure a setting on the listening socket.
41    #[snafu(display("failed to configure {} for listener on address {}: {}", setting, address, source))]
42    FailedToConfigureListener {
43        /// Listen address.
44        address: ListenAddress,
45
46        /// Name of the setting.
47        setting: &'static str,
48
49        /// Source of the error.
50        source: io::Error,
51    },
52
53    /// Failed to configure a setting on an accepted stream.
54    #[snafu(display("failed to configure {} for {} stream: {}", setting, stream_type, source))]
55    FailedToConfigureStream {
56        /// Name of the setting.
57        setting: &'static str,
58
59        /// Type of stream.
60        stream_type: &'static str,
61
62        /// Source of the error.
63        source: io::Error,
64    },
65
66    /// Failed to accept a new stream from the listener.
67    #[snafu(display("failed to accept new stream for listener on address {}: {}", address, source))]
68    FailedToAccept {
69        /// Listen address.
70        address: ListenAddress,
71
72        /// Source of the error.
73        source: io::Error,
74    },

86/// A network listener.
87///
88/// `Listener` is a abstract listener that works in conjunction with `Stream`, providing the ability to listen on
89/// arbitrary addresses and accept new streams of that address family.
90///
91/// ## Connection-oriented vs connectionless listeners
92///
93/// For listeners on connection-oriented address families (for example, TCP, Unix domain sockets in stream mode), the listener
94/// will listen for an accept new connections in the typical fashion. However, for connectionless address families
95/// (for example, UDP, Unix domain sockets in datagram mode), there is no concept of a "connection" and so nothing to be
96/// continually "accepted". Instead, `Listener` will emit a single `Stream` that can be used to send and receive data
97/// from multiple remote peers.
98///
99/// ## UDP autoscaling
100///
101/// On Linux, UDP listeners can be configured to bind multiple sockets to the same address using `SO_REUSEPORT`,
102/// allowing the kernel to load-balance incoming datagrams across them. The configured number of sockets are yielded
103/// one at a time from successive calls to [`Listener::accept`] before the listener returns pending forever. See
104/// the `udp_streams` parameter of [`Listener::from_listen_address`].
105pub struct Listener {
106    listen_address: ListenAddress,
107    inner: ListenerInner,
108    socket_receive_buffer_size: Option<usize>,
109}
110
111impl Listener {
112    /// Creates a new `Listener` from the given listen address.
113    ///
114    /// ## UDP streams
115    ///
116    /// For UDP listen addresses, `udp_streams` controls how many sockets are bound to the address and how many
117    /// `Stream`s the listener will yield from [`accept`](Self::accept) before going pending forever. `None` behaves
118    /// like `Some(1)`: a single socket is bound normally and one stream is yielded.
119    ///
120    /// When `Some(N)` with N > 1 is requested on Linux, the listener binds N sockets with `SO_REUSEPORT` set before
121    /// `bind`, so the kernel will hash-load-balance incoming datagrams across them. On non-Linux platforms,
122    /// `SO_REUSEPORT` doesn't provide load balancing, so the request is downgraded to a single socket.
123    ///
124    /// For non-UDP listen addresses, `udp_streams` is ignored.
125    ///
126    /// ## Errors
127    ///
128    /// If the listen address can't be bound, or if the listener can't be configured correctly, an error is returned.
129    pub async fn from_listen_address(
130        listen_address: ListenAddress, mut udp_streams: Option<NonZeroUsize>,
131    ) -> Result<Self, ListenerError> {
132        let inner = match &listen_address {
133            ListenAddress::Tcp(addr) => {
134                TcpListener::bind(addr)
135                    .await
136                    .map(ListenerInner::Tcp)
137                    .context(FailedToBind {
138                        address: listen_address.clone(),
139                    })?
140            }
141            ListenAddress::Udp(addr) => {
142                // See if we have platform support for SO_REUSEPORT, and if not, fall back to the default behavior.
143                if !socket_reuseport_supported() {
144                    udp_streams = None;
145                    warn!("SO_REUSEPORT not supported on the current platform. Falling back to the default behavior.");
146                }
147
148                let sockets = bind_udp_sockets(*addr, udp_streams).await.context(FailedToBind {
149                    address: listen_address.clone(),
150                })?;
151                ListenerInner::Udp(sockets)
152            }
153            #[cfg(unix)]
154            ListenAddress::Unixgram(addr) => {
155                ensure_unix_socket_free(addr).await.context(FailedToBind {
156                    address: listen_address.clone(),
157                })?;
158
159                let listener = tokio::net::UnixDatagram::bind(addr)
160                    .map(Some)
161                    .map(ListenerInner::Unixgram)
162                    .context(FailedToBind {
163                        address: listen_address.clone(),
164                    })?;
165
166                set_unix_socket_write_only(addr)
167                    .await
168                    .context(FailedToConfigureListener {
169                        address: listen_address.clone(),
170                        setting: "read/write permissions",
171                    })?;
172
173                listener
174            }
175            #[cfg(unix)]
176            ListenAddress::Unix(addr) => {
177                ensure_unix_socket_free(addr).await.context(FailedToBind {
178                    address: listen_address.clone(),
179                })?;
180
181                let listener = tokio::net::UnixListener::bind(addr)
182                    .map(ListenerInner::Unix)
183                    .context(FailedToBind {
184                        address: listen_address.clone(),
185                    })?;
186                set_unix_socket_write_only(addr)
187                    .await
188                    .context(FailedToConfigureListener {
189                        address: listen_address.clone(),
190                        setting: "read/write permissions",
191                    })?;
192
193                listener
194            }
195        };
196
197        Ok(Self {
198            listen_address,
199            inner,
200            socket_receive_buffer_size: None,
201        })
202    }

204    /// Sets the socket receive buffer size for this listener.
205    ///
206    /// The receive buffer size applies to accepted streams. `None` keeps the OS default.
207    pub fn with_receive_buffer_size(mut self, socket_receive_buffer_size: Option<usize>) -> Self {
208        self.socket_receive_buffer_size = socket_receive_buffer_size;
209        self
210    }

212    /// Gets a reference to the listen address.
213    pub fn listen_address(&self) -> &ListenAddress {
214        &self.listen_address
215    }
216
217    /// Minimum number of I/O buffers needed to safely service every stream this listener will yield.
218    ///
219    /// Connectionless streams permanently retain their buffer for the lifetime of the stream, so the reservation
220    /// matches the total number of yielded streams. Connection-oriented streams return their buffer to the pool
221    /// between connections, so the reservation is a lower bound of `1` per listener.
222    pub fn min_buffer_reservation(&self) -> usize {
223        match &self.inner {
224            ListenerInner::Tcp(_) => 1,
225            ListenerInner::Udp(sockets) => sockets.len(),
226            #[cfg(unix)]
227            ListenerInner::Unixgram(_) => 1,
228            #[cfg(unix)]
229            ListenerInner::Unix(_) => 1,
230        }
231    }
232
233    /// Accepts a new stream from the listener.
234    ///
235    /// For connection-oriented address families, this will accept a new connection and return a `Stream` that's bound
236    /// to that remote peer. For connectionless address families, this will yield up to the configured number of
237    /// pre-bound `Stream`s—one per call—before returning pending forever.
238    ///
239    /// ## Errors
240    ///
241    /// If the listener fails to accept a new stream, or if the accepted stream can't be configured correctly, an error
242    /// is returned.
243    pub async fn accept(&mut self) -> Result<Stream, ListenerError> {
244        let stream_type = self.listen_address.listener_type();
245        match &mut self.inner {
246            ListenerInner::Tcp(tcp) => {
247                let (socket, addr) = tcp.accept().await.context(FailedToAccept {
248                    address: self.listen_address.clone(),
249                })?;
250                configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
251                Ok((socket, addr).into())
252            }
253            ListenerInner::Udp(udp) => {
254                if let Some(socket) = udp.pop_front() {
255                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
256                    Ok(socket.into())
257                } else {
258                    pending().await
259                }
260            }
261            #[cfg(unix)]
262            ListenerInner::Unixgram(unix) => {
263                if let Some(socket) = unix.take() {
264                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
265                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
266                        setting: "SO_PASSCRED",
267                        stream_type,
268                    })?;
269                    Ok(socket.into())
270                } else {
271                    pending().await
272                }
273            }
274            #[cfg(unix)]
275            ListenerInner::Unix(unix) => unix
276                .accept()
277                .await
278                .context(FailedToAccept {
279                    address: self.listen_address.clone(),
280                })
281                .and_then(|(socket, _)| {
282                    configure_stream_socket_receive_buffer_size(&socket, self.socket_receive_buffer_size, stream_type)?;
283                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
284                        setting: "SO_PASSCRED",
285                        stream_type,
286                    })?;
287                    Ok(socket.into())
288                }),
289        }
290    }
291}

293fn configure_stream_socket_receive_buffer_size<'sock, S>(
294    socket: &'sock S, recv_buffer_size: Option<usize>, stream_type: &'static str,
295) -> Result<(), ListenerError>
296where
297    SockRef<'sock>: From<&'sock S>,
298{
299    if let Some(size) = recv_buffer_size {
300        SockRef::from(socket)
301            .set_recv_buffer_size(size)
302            .context(FailedToConfigureStream {
303                setting: SOCKET_RECV_BUFFER_SIZE_SETTING,
304                stream_type,
305            })?;
306    }
307
308    Ok(())
309}

311async fn bind_udp_socket_standard(addr: SocketAddr) -> io::Result<VecDeque<TokioUdpSocket>> {
312    let socket = TokioUdpSocket::bind(addr).await?;
313    let mut sockets = VecDeque::with_capacity(1);
314    sockets.push_back(socket);
315    Ok(sockets)
316}

318#[cfg(target_os = "linux")]
319async fn bind_udp_sockets(
320    addr: SocketAddr, maybe_socket_count: Option<NonZeroUsize>,
321) -> io::Result<VecDeque<TokioUdpSocket>> {
322    use socket2::{Domain, Protocol, SockAddr, Socket, Type};
323
324    fn bind_one(addr: SocketAddr) -> io::Result<TokioUdpSocket> {
325        let socket = Socket::new(Domain::for_address(addr), Type::DGRAM, Some(Protocol::UDP))?;
326        socket.set_reuse_address(true)?;
327        socket.set_reuse_port(true)?;
328        socket.set_nonblocking(true)?;
329        socket.bind(&SockAddr::from(addr))?;
330        let std_socket: std::net::UdpSocket = socket.into();
331        TokioUdpSocket::from_std(std_socket)
332    }
333
334    let socket_count = maybe_socket_count.map(NonZeroUsize::get).unwrap_or(1);
335    if socket_count == 1 {
336        return bind_udp_socket_standard(addr).await;
337    }
338
339    let mut sockets = VecDeque::with_capacity(socket_count);
340
341    // Bind the first socket to learn the effective address. When the caller passed port `0`, the OS assigns an
342    // ephemeral port; every remaining socket must bind to that same port for SO_REUSEPORT load balancing to work
343    // (otherwise each subsequent socket would receive its own distinct ephemeral port).
344    let first = bind_one(addr)?;
345    let effective_addr = first.local_addr()?;
346    sockets.push_back(first);
347
348    for _ in 1..socket_count {
349        sockets.push_back(bind_one(effective_addr)?);
350    }
351
352    Ok(sockets)
353}

355#[cfg(not(target_os = "linux"))]
356async fn bind_udp_sockets(addr: SocketAddr, _: Option<NonZeroUsize>) -> io::Result<VecDeque<TokioUdpSocket>> {
357    bind_udp_socket_standard(addr).await
358}

366/// A connection-oriented network listener.
367///
368/// `ConnectionOrientedListener` is conceptually the same as `Listener`, but specifically works with connection-oriented
369/// protocols. This variant is provided to facilitate usages where only a connection-oriented stream makes sense, such
370/// as an HTTP server.
371pub struct ConnectionOrientedListener {
372    listen_address: ListenAddress,
373    inner: ConnectionOrientedListenerInner,
374}
375
376impl ConnectionOrientedListener {
377    /// Creates a new `ConnectionOrientedListener` from the given listen address.
378    ///
379    /// ## Errors
380    ///
381    /// If the listen address isn't a connection-oriented address family, or if the listen address can't be bound, or
382    /// if the listener can't be configured correctly, an error is returned.
383    pub async fn from_listen_address(listen_address: ListenAddress) -> Result<Self, ListenerError> {
384        let inner = match &listen_address {
385            ListenAddress::Tcp(addr) => TcpListener::bind(addr)
386                .await
387                .map(ConnectionOrientedListenerInner::Tcp)
388                .context(FailedToBind {
389                    address: listen_address.clone(),
390                })?,
391            #[cfg(unix)]
392            ListenAddress::Unix(addr) => {
393                ensure_unix_socket_free(addr).await.context(FailedToBind {
394                    address: listen_address.clone(),
395                })?;
396
397                let listener = tokio::net::UnixListener::bind(addr)
398                    .map(ConnectionOrientedListenerInner::Unix)
399                    .context(FailedToBind {
400                        address: listen_address.clone(),
401                    })?;
402                set_unix_socket_write_only(addr)
403                    .await
404                    .context(FailedToConfigureListener {
405                        address: listen_address.clone(),
406                        setting: "read/write permissions",
407                    })?;
408
409                listener
410            }
411            _ => {
412                return Err(ListenerError::InvalidConfiguration {
413                    reason: "only TCP and Unix listen addresses are supported",
414                })
415            }
416        };
417
418        Ok(Self { listen_address, inner })
419    }
420
421    /// Gets a reference to the listen address.
422    pub fn listen_address(&self) -> &ListenAddress {
423        &self.listen_address
424    }
425
426    /// Returns the actual local socket address this listener is bound to.
427    ///
428    /// This is useful when binding to port `0` to discover the ephemeral port assigned by the OS.
429    pub fn local_addr(&self) -> Result<SocketAddr, ListenerError> {
430        match &self.inner {
431            ConnectionOrientedListenerInner::Tcp(tcp) => tcp.local_addr().context(FailedToConfigureListener {
432                address: self.listen_address.clone(),
433                setting: "local_addr",
434            }),
435            #[cfg(unix)]
436            ConnectionOrientedListenerInner::Unix(_) => Err(ListenerError::InvalidConfiguration {
437                reason: "local_addr is not supported for Unix listeners",
438            }),
439        }
440    }
441
442    /// Accepts a new connection from the listener.
443    ///
444    /// ## Errors
445    ///
446    /// If the listener fails to accept a new connection, or if the accepted connection can't be configured correctly,
447    /// an error is returned.
448    pub async fn accept(&mut self) -> Result<Connection, ListenerError> {
449        match &mut self.inner {
450            ConnectionOrientedListenerInner::Tcp(tcp) => tcp
451                .accept()
452                .await
453                .map(|(stream, addr)| Connection::Tcp(stream, addr))
454                .context(FailedToAccept {
455                    address: self.listen_address.clone(),
456                }),
457            #[cfg(unix)]
458            ConnectionOrientedListenerInner::Unix(unix) => unix
459                .accept()
460                .await
461                .context(FailedToAccept {
462                    address: self.listen_address.clone(),
463                })
464                .and_then(|(socket, _)| {
465                    let stream_type = self.listen_address.listener_type();
466                    enable_uds_socket_credentials(&socket).context(FailedToConfigureStream {
467                        setting: "SO_PASSCRED",
468                        stream_type,
469                    })?;
470                    Ok(Connection::Unix(socket))
471                }),
472        }
473    }
474}

724    #[cfg(target_os = "linux")]
725    fn udp_socket_ports(listener: &Listener) -> Vec<u16> {
726        match &listener.inner {
727            ListenerInner::Udp(sockets) => sockets
728                .iter()
729                .map(|s| s.local_addr().expect("socket should have local addr").port())
730                .collect(),
731            _ => panic!("expected UDP listener"),
732        }
733    }

738            _ => panic!("expected TCP listener"),
739        }
