7/// Stack allocated structure for ancillary (out-of-band) data.
8pub struct AncillaryData<const N: usize> {
9    buf: [MaybeUninit<u8>; N],
10    len: usize,
11}
12
13impl<const N: usize> AncillaryData<N> {
14    /// Creates a new `AncillaryData` structure of the given size.
15    pub fn new() -> Self {
16        Self {
17            buf: [MaybeUninit::uninit(); N],
18            len: 0,
19        }
20    }
21
22    /// Gets a mutable reference to the underlying buffer as a slice of uninitialized bytes.
23    pub fn as_mut_uninit(&mut self) -> &mut [MaybeUninit<u8>] {
24        &mut self.buf[..]
25    }
26
27    /// Sets the number of bytes that have been filled in the buffer.
28    ///
29    /// ## Safety
30    ///
31    /// The caller must ensure that the number of bytes filled is greater than or equal to the value of `new_len`.
32    ///
33    /// ## Panics
34    ///
35    /// If `new_len` is greater than the length of the buffer itself, this function will panic.
36    pub unsafe fn set_len(&mut self, new_len: usize) {
37        if new_len > self.buf.len() {
38            panic!("new length exceeds buffer length");
39        }
40
41        self.len = new_len;
42    }
43
44    /// Gets an iterator over any control messages in the buffer.
45    pub unsafe fn messages(&self) -> ControlMessages<'_> {
46        let buf = std::slice::from_raw_parts(self.buf.as_ptr() as *const _, self.len);
47        ControlMessages::new(buf)
48    }
49}

51/// An iterator over control messages in an ancillary data buffer.
52pub struct ControlMessages<'a> {
53    buf: &'a [u8],
54    current: Option<&'a libc::cmsghdr>,
55}
56
57impl<'a> ControlMessages<'a> {
58    fn new(buf: &'a [u8]) -> Self {
59        Self { buf, current: None }
60    }
61}
62
63impl<'a> Iterator for ControlMessages<'a> {
64    type Item = ControlMessage<'a>;
65
66    fn next(&mut self) -> Option<Self::Item> {
67        unsafe {
68            // Create a temporary message header that we can use to pull out control message headers from.
69            let mut msg: libc::msghdr = mem::zeroed();
70            msg.msg_control = self.buf.as_ptr() as *mut _;
71            msg.msg_controllen = self.buf.len() as _;
72
73            let cmsg = if let Some(current_cmsg) = self.current {
74                // Get the next control message header after the current one.
75                libc::CMSG_NXTHDR(&msg, current_cmsg)
76            } else {
77                // We haven't read a control message header yet, so take the first one.
78                libc::CMSG_FIRSTHDR(&msg)
79            };
80
81            let cmsg = cmsg.as_ref()?;
82            self.current = Some(cmsg);
83
84            ControlMessage::try_from_cmsghdr(cmsg)
85        }
86    }
87}

89/// Control message.
90pub enum ControlMessage<'a> {
91    /// UNIX socket credentials.
92    ///
93    /// This captures the process ID, user ID, and group ID of the peer process on the other end of a Unix domain
94    /// socket.
95    Credentials(&'a libc::ucred),
96}
97
98impl<'a> ControlMessage<'a> {
99    fn try_from_cmsghdr(cmsg: &'a libc::cmsghdr) -> Option<Self> {
100        unsafe {
101            // Calculate the size of the control message header, so we can figure out the byte offset to actually get at
102            // the raw message data, and then create a slice to that data.
103            let cmsg_len_offset = libc::CMSG_LEN(0) as usize;
104            let data_len = cmsg.cmsg_len - cmsg_len_offset;
105            let data_ptr = libc::CMSG_DATA(cmsg).cast();
106            let data = std::slice::from_raw_parts(data_ptr, data_len);
107
108            // Currently, all we handle is socket credentials.
109            match cmsg.cmsg_level {
110                libc::SOL_SOCKET => match cmsg.cmsg_type {
111                    libc::SCM_CREDENTIALS => ControlMessage::as_credentials(data),
112                    _ => None,
113                },
114                _ => None,
115            }
116        }
117    }
118
119    fn as_credentials(buf: &'a [u8]) -> Option<Self> {
120        if buf.len() == mem::size_of::<libc::ucred>() {
121            // SAFETY: We've already checked that the buffer is long enough to be mapped to `ucred`, and we're only here
122            // if `cmsg_type` was SCM_CREDENTIALS, and our reference is safe to take because it's tied to the lifetime
123            // of the buffer we're taking a pointer to.
124            unsafe {
125                let ucred_ptr: *const libc::ucred = buf.as_ptr().cast();
126                ucred_ptr.as_ref().map(Self::Credentials)
127            }
128        } else {
129            None
130        }
131    }
132}

134const fn get_ucred_struct_size() -> usize {
135    let ucred_raw_size = mem::size_of::<libc::ucred>();
136    let ucred_raw_size = if ucred_raw_size.wrapping_shr(u32::BITS) != 0 {
137        // We do a const shift of the raw size to see if it has any additional bits past what we can fit in u32, and
138        // this way we know that it's safe to directly cast the value to u32 without having truncated any bits.
139        panic!("size of `ucred` struct greater than u32::MAX");
140    } else {
141        ucred_raw_size as u32
142    };
143
144    // SAFETY: This is part of a blanket "unsafe" wrapper around libc functions, but it's safe to call since it boils
145    // down to a bunch of `size_of` calls and arithmetic for ensuring the values take alignment into consideration, etc.
146    unsafe { libc::CMSG_SPACE(ucred_raw_size) as usize }
147}
