13/// A listen address.
14///
15/// Listen addresses are used to bind listeners to specific local addresses and ports, and multiple address families and
16/// protocols are supported. In textual form, listen addresses are represented as URLs, with the scheme indicating the
17/// protocol and the authority/path representing the address to listen on.
18///
19/// ## Examples
20///
21/// - `tcp://127.0.0.1:6789` (listen on IPv4 loopback, TCP port 6789)
22/// - `udp://[::1]:53` (listen on IPv6 loopback, UDP port 53)
23/// - `unixgram:///tmp/app.socket` (listen on a Unix datagram socket at `/tmp/app.socket`)
24/// - `unix:///tmp/app.socket` (listen on a Unix stream socket at `/tmp/app.socket`)
25#[derive(Clone, Debug, Deserialize)]
26#[serde(try_from = "String")]
27pub enum ListenAddress {
28    /// A TCP listen address.
29    Tcp(SocketAddr),
30
31    /// A UDP listen address.
32    Udp(SocketAddr),
33
34    /// A Unix datagram listen address.
35    #[cfg(unix)]
36    Unixgram(PathBuf),
37
38    /// A Unix stream listen address.
39    #[cfg(unix)]
40    Unix(PathBuf),
41}
42
43impl ListenAddress {
44    /// Creates a TCP address for the given port that listens on all interfaces.
45    pub const fn any_tcp(port: u16) -> Self {
46        Self::Tcp(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::UNSPECIFIED, port)))
47    }
48
49    /// Returns the socket type of the listen address.
50    pub const fn listener_type(&self) -> &'static str {
51        match self {
52            Self::Tcp(_) => "tcp",
53            Self::Udp(_) => "udp",
54            #[cfg(unix)]
55            Self::Unixgram(_) => "unixgram",
56            #[cfg(unix)]
57            Self::Unix(_) => "unix",
58        }
59    }
60
61    /// Returns a socket address that can be used to connect to the configured listen address with a bias for local clients.
62    ///
63    /// When the listen address is a TCP or UDP address, this method returns a socket address that can be used to
64    /// connect to the listener bound to this listen addresss, such that if the listen address is unspecified
65    /// (`0.0.0.0`), the client will connect locally using "localhost". When the listen address is not "unspecified" or
66    /// already uses "localhost", this method returns the listen address as-is.
67    ///
68    /// If the address is a Unix domain socket, this method returns `None`.
69    pub fn as_local_connect_addr(&self) -> Option<SocketAddr> {
70        match self {
71            Self::Tcp(addr) | Self::Udp(addr) => {
72                let mut connect_addr = *addr;
73                if connect_addr.ip().is_unspecified() {
74                    let localhost_ip = match connect_addr.is_ipv4() {
75                        true => IpAddr::V4(Ipv4Addr::LOCALHOST),
76                        false => IpAddr::V6(Ipv6Addr::LOCALHOST),
77                    };
78
79                    connect_addr.set_ip(localhost_ip);
80                }
81
82                Some(connect_addr)
83            }
84            #[cfg(unix)]
85            Self::Unixgram(_) => None,
86            #[cfg(unix)]
87            Self::Unix(_) => None,
88        }
89    }
90}

92impl fmt::Display for ListenAddress {
93    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
94        match self {
95            Self::Tcp(addr) => write!(f, "tcp://{}", addr),
96            Self::Udp(addr) => write!(f, "udp://{}", addr),
97            #[cfg(unix)]

99            #[cfg(unix)]
100            Self::Unix(path) => write!(f, "unix://{}", path.display()),
101        }
102    }

113impl<'a> TryFrom<&'a str> for ListenAddress {
114    type Error = String;
115
116    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
117        let url = match Url::parse(value) {
118            Ok(url) => url,
119            Err(e) => match e {
120                url::ParseError::RelativeUrlWithoutBase => {
121                    Url::parse(&format!("unixgram://{}", value)).map_err(|e| e.to_string())?
122                }
123                _ => return Err(e.to_string()),
124            },
125        };
126
127        match url.scheme() {
128            "tcp" => {
129                let mut socket_addresses = url.socket_addrs(|| None).map_err(|e| e.to_string())?;
130                if socket_addresses.is_empty() {
131                    Err("listen address must resolve to at least one valid IP address/port pair".to_string())
132                } else {
133                    Ok(Self::Tcp(socket_addresses.swap_remove(0)))
134                }
135            }
136            "udp" => {
137                let mut socket_addresses = url.socket_addrs(|| None).map_err(|e| e.to_string())?;
138                if socket_addresses.is_empty() {
139                    Err("listen address must resolve to at least one valid IP address/port pair".to_string())
140                } else {
141                    Ok(Self::Udp(socket_addresses.swap_remove(0)))
142                }
143            }
144            #[cfg(unix)]
145            "unixgram" => {
146                let path = url.path();
147                if path.is_empty() {
148                    return Err("socket path cannot be empty".to_string());
149                }
150
151                let path_buf = PathBuf::from(path);
152                if !path_buf.is_absolute() {
153                    return Err("socket path must be absolute".to_string());
154                }
155
156                Ok(Self::Unixgram(path_buf))
157            }
158            #[cfg(unix)]
159            "unix" => {
160                let path = url.path();
161                if path.is_empty() {
162                    return Err("socket path cannot be empty".to_string());
163                }
164
165                let path_buf = PathBuf::from(path);
166                if !path_buf.is_absolute() {
167                    return Err("socket path must be absolute".to_string());
168                }
169
170                Ok(Self::Unix(path_buf))
171            }
172            scheme => Err(format!("unknown/unsupported address scheme '{}'", scheme)),
173        }
174    }
175}

177/// Process credentials for a Unix domain socket connection.
178///
179/// When dealing with Unix domain sockets, they can be configured such that the "process credentials" of the remote peer
180/// are sent as part of each received message. These "credentials" are the process ID of the remote peer, and the user
181/// ID and group ID that the process is running as.
182///
183/// In some cases, this information can be useful for identifying the remote peer and enriching the received data in an
184/// automatic way.
185#[cfg(unix)]
186#[derive(Clone)]
187pub struct ProcessCredentials {
188    /// Process ID of the remote peer.
189    pub pid: i32,
190
191    /// User ID of the remote peer process.
192    pub uid: u32,
193
194    /// Group ID of the remote peer process.
195    pub gid: u32,
196}
197
198/// Connection address.
199///
200/// A generic representation of the address of a remote peer. This can either be a typical socket address (used for
201/// IPv4/IPv6), or potentially the process credentials of a Unix domain socket connection.
202#[derive(Clone)]
203pub enum ConnectionAddress {
204    /// A socket-like address.
205    SocketLike(SocketAddr),
206
207    /// A process-like address.
208    #[cfg(unix)]
209    ProcessLike(Option<ProcessCredentials>),
210}
211
212impl fmt::Display for ConnectionAddress {
213    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
214        match self {
215            Self::SocketLike(addr) => write!(f, "{}", addr),
216            #[cfg(unix)]
217            Self::ProcessLike(maybe_creds) => match maybe_creds {
218                None => write!(f, "<unbound>"),
219                Some(creds) => write!(f, "<pid={} uid={} gid={}>", creds.pid, creds.uid, creds.gid),
220            },
221        }
222    }
223}
224
225impl From<SocketAddr> for ConnectionAddress {
226    fn from(value: SocketAddr) -> Self {
227        Self::SocketLike(value)
228    }
229}
230
231#[cfg(unix)]
232impl From<ProcessCredentials> for ConnectionAddress {
233    fn from(creds: ProcessCredentials) -> Self {
234        Self::ProcessLike(Some(creds))
235    }
236}

238impl<'a> Connected<&'a Connection> for ConnectionAddress {
239    fn connect_info(target: &'a Connection) -> Self {
240        target.remote_addr()
241    }

249    fn test_as_local_connect_addr() {
250        let tcp_any_addr = ListenAddress::try_from("tcp://0.0.0.0:1234").unwrap();
251        assert_eq!(
252            tcp_any_addr.as_local_connect_addr(),
253            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::LOCALHOST, 1234)))
254        );
255
256        let tcp_localhost_addr = ListenAddress::try_from("tcp://127.0.0.1:2345").unwrap();
257        assert_eq!(
258            tcp_localhost_addr.as_local_connect_addr(),
259            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::LOCALHOST, 2345)))
260        );
261
262        let tcp_private_addr = ListenAddress::try_from("tcp://192.168.10.2:3456").unwrap();
263        assert_eq!(
264            tcp_private_addr.as_local_connect_addr(),
265            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::new(192, 168, 10, 2), 3456)))
266        );
267
268        let udp_any_addr = ListenAddress::try_from("udp://0.0.0.0:4567").unwrap();
269        assert_eq!(
270            udp_any_addr.as_local_connect_addr(),
271            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::LOCALHOST, 4567)))
272        );
273
274        let udp_localhost_addr = ListenAddress::try_from("udp://127.0.0.1:5678").unwrap();
275        assert_eq!(
276            udp_localhost_addr.as_local_connect_addr(),
277            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::LOCALHOST, 5678)))
278        );
279
280        let udp_private_addr = ListenAddress::try_from("udp://192.168.10.2:6789").unwrap();
281        assert_eq!(
282            udp_private_addr.as_local_connect_addr(),
283            Some(SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::new(192, 168, 10, 2), 6789)))
