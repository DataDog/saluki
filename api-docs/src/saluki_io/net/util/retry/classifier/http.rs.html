7/// A predicate that decides whether a response should be treated as retriable.
8///
9/// The predicate receives the response and returns `true` if the response should be retried.
10pub type HttpRetryPredicate<B = ()> = Arc<dyn Fn(&Response<B>) -> bool + Send + Sync>;
11
12fn default_should_retry<B>(response: &Response<B>) -> bool {
13    let status = response.status();
14
15    match status {
16        // There are some status codes that likely indicate a fundamental misconfiguration or bug on the client side
17        // which won't be resolved by retrying the request.
18        http::StatusCode::BAD_REQUEST
19        | http::StatusCode::UNAUTHORIZED
20        | http::StatusCode::FORBIDDEN
21        | http::StatusCode::PAYLOAD_TOO_LARGE => {
22            // These statuses are permanent failures — the transaction is dropped, not retried (a data-loss path).
23            // Anchor that the run reaches it.
24            #[cfg(feature = "antithesis")]
25            antithesis_sdk::assert_sometimes!(
26                true,
27                "transaction permanently dropped — non-retryable status",
28                &serde_json::json!({ "status": status.as_u16() })
29            );
30            false
31        }
32
33        // For all other status codes, we'll only retry if they're in the client/server error range.
34        _ => status.is_client_error() || status.is_server_error(),
35    }
36}

38/// A standard HTTP response classifier.
39///
40/// Generally treats all client (4xx) and server (5xx) errors as retriable, with the exception of a few specific client
41/// errors that shouldn't be retried:
42///
43/// - 400 Bad Request (likely a client-side bug)
44/// - 401 Unauthorized (likely a client-side misconfiguration)
45/// - 403 Forbidden (likely a client-side misconfiguration)
46/// - 413 Payload Too Large (likely a client-side bug)
47///
48/// Additional [`HttpRetryPredicate`]s can be registered via [`StandardHttpClassifier::with_predicate`]. A response is
49/// retried if any predicate—including the default—returns `true` (OR semantics). This allows callers to
50/// selectively unlock retries for status codes that the default predicate would not retry, without affecting other
51/// status codes.
52pub struct StandardHttpClassifier<B = ()> {
53    predicates: Vec<HttpRetryPredicate<B>>,
54}
55
56impl<B> Clone for StandardHttpClassifier<B> {
57    fn clone(&self) -> Self {
58        Self {
59            predicates: self.predicates.clone(),
60        }
61    }
62}
63
64impl<B: 'static> Default for StandardHttpClassifier<B> {
65    fn default() -> Self {
66        Self::new()
67    }
68}
69
70impl<B: 'static> StandardHttpClassifier<B> {
71    /// Creates a new [`StandardHttpClassifier`] with the default status-code predicate installed.
72    pub fn new() -> Self {
73        Self {
74            predicates: vec![Arc::new(default_should_retry::<B>)],
75        }
76    }
77
78    /// Adds a predicate.
79    ///
80    /// A response is retried if any predicate—including the default—returns `true` (OR semantics).
81    pub fn with_predicate(mut self, predicate: HttpRetryPredicate<B>) -> Self {
82        self.predicates.push(predicate);
83        self
84    }
85}
86
87impl<B, Error> RetryClassifier<http::Response<B>, Error> for StandardHttpClassifier<B> {
88    fn should_retry(&self, response: &Result<http::Response<B>, Error>) -> bool {
89        match response {
90            Ok(resp) => self.predicates.iter().any(|p| p(resp)),
91            Err(_) => true,
92        }
93    }
94}

104    type TestResponse = Result<http::Response<()>, ()>;
105
106    fn ok(status: StatusCode) -> TestResponse {
107        Ok(http::Response::builder().status(status).body(()).unwrap())
108    }
109
110    fn err() -> TestResponse {
111        Err(())
112    }
113
114    fn classify(classifier: &StandardHttpClassifier<()>, response: &TestResponse) -> bool {
115        <StandardHttpClassifier<()> as RetryClassifier<http::Response<()>, ()>>::should_retry(classifier, response)
116    }
117
118    #[test]
119    fn default_classifier_retries_5xx_and_most_4xx() {
120        let classifier = StandardHttpClassifier::new();
121
122        assert!(!classify(&classifier, &ok(StatusCode::OK)));
123        assert!(!classify(&classifier, &ok(StatusCode::NO_CONTENT)));
124
125        for status in [
126            StatusCode::INTERNAL_SERVER_ERROR,
127            StatusCode::BAD_GATEWAY,
128            StatusCode::SERVICE_UNAVAILABLE,
129            StatusCode::GATEWAY_TIMEOUT,
130        ] {
131            assert!(classify(&classifier, &ok(status)), "{} should be retried", status);
132        }
133
134        for status in [
135            StatusCode::REQUEST_TIMEOUT,
136            StatusCode::TOO_MANY_REQUESTS,
137            StatusCode::NOT_FOUND,
138        ] {
139            assert!(classify(&classifier, &ok(status)), "{} should be retried", status);
140        }
141    }
142
143    #[test]
144    fn default_classifier_does_not_retry_known_client_misconfig() {
145        let classifier = StandardHttpClassifier::new();
146
147        for status in [
148            StatusCode::BAD_REQUEST,
149            StatusCode::UNAUTHORIZED,
150            StatusCode::FORBIDDEN,
151            StatusCode::PAYLOAD_TOO_LARGE,
152        ] {
153            assert!(!classify(&classifier, &ok(status)), "{} should not be retried", status);
154        }
155    }
156
157    #[test]
158    fn default_classifier_retries_transport_error() {
159        let classifier = StandardHttpClassifier::new();
160        assert!(classify(&classifier, &err()));
161    }
162
163    #[test]
164    fn predicate_adds_retry_for_403() {
165        let classifier = StandardHttpClassifier::new()
166            .with_predicate(Arc::new(|response| response.status() == StatusCode::FORBIDDEN));
167
168        assert!(classify(&classifier, &ok(StatusCode::FORBIDDEN)));
169        // Sibling client-misconfig statuses without a matching predicate keep their default (non-retriable) behavior.
170        assert!(!classify(&classifier, &ok(StatusCode::UNAUTHORIZED)));
171        assert!(!classify(&classifier, &ok(StatusCode::BAD_REQUEST)));
172        // Status codes that are retried by default are unaffected.
173        assert!(classify(&classifier, &ok(StatusCode::INTERNAL_SERVER_ERROR)));
174    }
175
176    #[test]
177    fn predicate_is_re_evaluated_each_call() {
178        let flag = Arc::new(AtomicBool::new(false));
179        let flag_clone = Arc::clone(&flag);
180        let predicate: HttpRetryPredicate =
181            Arc::new(move |response| response.status() == StatusCode::FORBIDDEN && flag_clone.load(Ordering::SeqCst));
182
183        let classifier = StandardHttpClassifier::new().with_predicate(predicate);
184
185        assert!(!classify(&classifier, &ok(StatusCode::FORBIDDEN)));
186
187        flag.store(true, Ordering::SeqCst);
188        assert!(classify(&classifier, &ok(StatusCode::FORBIDDEN)));
189
190        flag.store(false, Ordering::SeqCst);
191        assert!(!classify(&classifier, &ok(StatusCode::FORBIDDEN)));
192    }
