8pub trait RetryLifecycle<Req, Res, Error> {
9    fn before_retry(&self, req: &Req, res: &Result<Res, Error>, retry_backoff: Duration, error_count: u32);
10    fn after_success(&self, req: &Req, res: &Result<Res, Error>);
11}
12
13/// A default retry lifecycle that emits basic debug logs when retrying requests and when requests succeed.
14///
15/// This lifecycle emits an extremely minimal amount of information, and is generally only useful for debugging purposes
16/// to understand if/when retries are happening, but gives no information about the request/response, why a retry
17/// decision was made, the code that is ultimately using this retry logic, and so on.
18#[derive(Clone)]
19pub struct DefaultDebugRetryLifecycle;
20
21impl<Req, Res, Error> RetryLifecycle<Req, Res, Error> for DefaultDebugRetryLifecycle {
22    fn before_retry(&self, _: &Req, _: &Result<Res, Error>, retry_backoff: Duration, error_count: u32) {
23        debug!(error_count, "Retrying request after {:?}.", retry_backoff);
24    }
25
26    fn after_success(&self, _: &Req, _: &Result<Res, Error>) {
27        debug!("Request succeeded.");
28    }
29}
