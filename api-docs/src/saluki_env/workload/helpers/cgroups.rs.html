30/// Linux Control Groups-specific configuration.
31///
32/// Provides environment-specific paths to both "procfs" and "cgroupfs" filesystems, necessary for querying the Linux
33/// Control Groups v2 unified hierarchy.
34pub struct CgroupsConfiguration {
35    procfs_root: PathBuf,
36    cgroupfs_root: PathBuf,
37}
38
39impl CgroupsConfiguration {
40    /// Creates a new `CgroupsConfiguration` from the given configuration.
41    ///
42    /// # Errors
43    ///
44    /// If any of the paths in the configuration are not valid, an error will be returned. This does not include,
45    /// however, if any of the configured paths do not _exist_.
46    pub fn from_configuration(
47        config: &GenericConfiguration, feature_detector: FeatureDetector,
48    ) -> Result<Self, GenericError> {
49        let procfs_root = match config.try_get_typed::<PathBuf>("container_proc_root")? {
50            Some(path) => path,
51            None => {
52                if feature_detector.is_feature_available(Feature::HostMappedProcfs) {
53                    PathBuf::from(DEFAULT_HOST_MAPPED_PROCFS_ROOT)
54                } else {
55                    PathBuf::from(DEFAULT_PROCFS_ROOT)
56                }
57            }
58        };
59
60        let cgroupfs_root = match config.try_get_typed::<PathBuf>("container_cgroup_root")? {
61            Some(path) => path,
62            None => {
63                if feature_detector.is_feature_available(Feature::HostMappedProcfs) {
64                    PathBuf::from(DEFAULT_HOST_MAPPED_CGROUPFS_ROOT)
65                } else {
66                    // TODO: Consider if we need to do anything specific for Amazon Linux [1] or does the referenced code only
67                    // matter for cgroups v1?
68                    //
69                    // [1]: https://github.com/DataDog/datadog-agent/blob/fe75b815c2f135f0d2ea85d7a57a8fc8cbf56bd9/pkg/config/setup/config.go#L1172-L1173
70                    PathBuf::from(DEFAULT_CGROUPFS_ROOT)
71                }
72            }
73        };
74
75        Ok(Self {
76            procfs_root,
77            cgroupfs_root,
78        })
79    }
80
81    /// Returns the path to the "procfs" filesystem.
82    pub fn procfs_path(&self) -> &Path {
83        self.procfs_root.as_path()
84    }
85
86    /// Returns the path to the "cgroupfs" filesystem.
87    pub fn cgroupfs_path(&self) -> &Path {
88        self.cgroupfs_root.as_path()
89    }
90}

92/// Reader for querying control groups being used for containerization.
93///
94/// This reader is capable of querying both cgroups v1 and v2 hierarchies, and can be used to find cgroups -- either
95/// within the entire hierarchy, or for a specific process ID -- that are mapped specifically to containers. A simple
96/// naming heuristic is used to both identify and extract container IDs from cgroup names.
97#[derive(Clone)]
98pub struct CgroupsReader {
99    procfs_path: PathBuf,
100    hierarchy_reader: HierarchyReader,
101    interner: GenericMapInterner,
102}
103
104impl CgroupsReader {
105    /// Creates a new `CgroupsReader` from the given configuration and interner.
106    ///
107    /// If either a valid cgroups v1 or v2 hierarchy is found, `Ok(Some)` is returned with the reader. Otherwise,
108    /// `Ok(None)` is returned.
109    ///
110    /// The provided interner will be used exclusively for handling container IDs.
111    ///
112    /// # Errors
113    ///
114    /// If there is an I/O error while attempting to query the current cgroups hierarchy, an error will be returned.
115    pub fn try_from_config(
116        config: &CgroupsConfiguration, interner: GenericMapInterner,
117    ) -> Result<Option<Self>, GenericError> {
118        let hierarchy_reader = HierarchyReader::try_from_config(config)?;
119        Ok(hierarchy_reader.map(|hierarchy_reader| Self {
120            procfs_path: config.procfs_path().to_path_buf(),
121            hierarchy_reader,
122            interner,
123        }))
124    }
125
126    fn try_cgroup_from_path(&self, cgroup_path: &Path) -> Option<Cgroup> {
127        if let Some(name) = cgroup_path.file_name().and_then(|s| s.to_str()) {
128            if let Some(container_id) = extract_container_id(name, &self.interner) {
129                let metadata = match cgroup_path.metadata() {
130                    Ok(metadata) => metadata,
131                    Err(e) => {
132                        trace!(error = %e, cgroup_controller_path = %cgroup_path.display(), "Failed to get metadata for possible cgroup controller path.");
133                        return None;
134                    }
135                };
136
137                trace!(
138                    controller_inode = metadata.ino(),
139                    %container_id,
140                    cgroup_controller_path = %cgroup_path.display(),
141                    "Found valid cgroups controller for container.",
142                );
143
144                return Some(Cgroup {
145                    ino: Some(metadata.ino()),
146                    container_id,
147                });
148            }
149        }
150
151        None
152    }
153
154    /// Gets a cgroup for the given process ID.
155    ///
156    /// This method will attempt to find the cgroup for the given process ID by looking at the `/proc/<pid>/cgroup`
157    /// file. If the process ID does not exist or is not attached to a cgroup, `None` will be returned.
158    pub fn get_cgroup_by_pid(&self, pid: u32) -> Option<Cgroup> {
159        // See if the given process ID exists in the proc filesystem _and_ if there's a cgroup path for it.
160        let proc_pid_cgroup_path = self.procfs_path.join(pid.to_string()).join("cgroup");
161        let lines = match read_lines(&proc_pid_cgroup_path) {
162            Ok(lines) => lines,
163            Err(e) => match e.kind() {
164                io::ErrorKind::NotFound => {
165                    debug!(pid, cgroup_lookup_path = %proc_pid_cgroup_path.display(), "Process does not exist or is not attached to a cgroup.");
166                    return None;
167                }
168                _ => {
169                    debug!(error = %e, pid, cgroup_lookup_path = %proc_pid_cgroup_path.display(), "Failed to read cgroup file for process.");
170                    return None;
171                }
172            },
173        };
174
175        let base_controller_name = self.hierarchy_reader.base_controller();
176
177        // We're looking for the first line that matches our base controller name, and then we'll see if it's attached
178        // to the container based on the name, and if so, return it.
179        for entry in lines.iter().filter_map(|s| CgroupControllerEntry::try_from_str(s)) {
180            if entry.name == base_controller_name {
181                // We explicitly try to extract the container ID from the reported cgroup controller path, rather than
182                // trying to stick it on the end of our configured root cgroups path. This is because unless we're in the
183                // host's cgroup namespace, the path we get here will be the leaf directory -- the part with the
184                // container ID in it -- but it will be relative in a way that doesn't allow it to be appended to the
185                // root cgroups path, and so trying to query it to get the controller inode, and all of that, will fail.
186                //
187                // All we need is the leaf directory anyways.
188                if let Some(proc_cgroup_path) = entry.path.file_name().and_then(|s| s.to_str()) {
189                    if let Some(container_id) = extract_container_id(proc_cgroup_path, &self.interner) {
190                        return Some(Cgroup {
191                            ino: None,
192                            container_id,
193                        });
194                    }
195                }
196            } else {
197                debug!(pid, cgroup_lookup_path = %proc_pid_cgroup_path.display(), base_controller_name, "Found cgroup controller for process, but it doesn't match the base controller.");
198            }
199        }
200
201        debug!(pid, cgroup_lookup_path = %proc_pid_cgroup_path.display(), base_controller_name, "Could not find matching base cgroup controller for process.");
202
203        None
204    }

206    /// Gets all child cgroups in the current cgroups hierarchy
207    pub fn get_child_cgroups(&self) -> Vec<Cgroup> {
208        let mut cgroups = Vec::new();
209        let mut visit = |path: &Path| {
210            if let Some(cgroup) = self.try_cgroup_from_path(path) {
211                cgroups.push(cgroup);
212            }
213        };
214
215        // Walk the cgroups hierarchy and collect all cgroups that we can find that are related to containers..
216        let root_path = match &self.hierarchy_reader {
217            HierarchyReader::V1 {
218                base_controller_path, ..
219            } => base_controller_path.as_path(),
220            HierarchyReader::V2 { root, .. } => root.as_path(),
221        };
222
223        if let Err(e) = visit_subdirectories(root_path, &mut visit) {
224            error!(error = %e, "Failed to visit cgroups hierarchy.");
225        }
226
227        cgroups
228    }
229}

231#[derive(Clone)]
232enum HierarchyReader {
233    V1 {
234        base_controller_path: PathBuf,
235        controllers: HashMap<String, PathBuf>,
236    },
237
238    V2 {
239        root: PathBuf,
240        controllers: Vec<String>,
241    },
242}
243
244impl HierarchyReader {
245    fn try_from_config(config: &CgroupsConfiguration) -> Result<Option<Self>, GenericError> {
246        // Open the mount file from procfs to scan through and find any cgroups subsystems.
247        let mounts_path = config.procfs_path().join("mounts");
248        let mount_entries = read_lines(&mounts_path)
249            .with_error_context(|| format!("Failed to read mount entries from procfs ({})", mounts_path.display()))?;
250
251        let mut controllers = HashMap::new();
252        let mut maybe_cgroups_v2 = None;
253
254        // For each mount line, check if its of the `cgroup` or `cgroup2` type. Skip everything else.
255        for mount_entry in mount_entries {
256            // Split the line into fields, and take the second and third values. We always expect at least three fields
257            // in a line if it's a line that might possibly be a cgroup mount.
258            let mut fields = mount_entry.split_whitespace();
259            let maybe_cgroup_path = fields.nth(1);
260            let maybe_fs_type = fields.nth(0);
261
262            if let (Some(raw_cgroup_path), Some(fs_type)) = (maybe_cgroup_path, maybe_fs_type) {
263                let cgroup_path = Path::new(raw_cgroup_path);
264
265                // Make sure this path is rooted within our configured cgroupfs path.
266                //
267                // When we're inside a container that has a host-mapped cgroupfs path, the `mounts` file might end up
268                // having duplicate entries (like one set as `/sys/fs/cgroup` and another set as `/host/sys/fs/cgroup`,
269                // etc)... and we want to use the one that matches our configured cgroupfs path as that's the one that
270                // will actually have the cgroups we care about.
271                if !cgroup_path.starts_with(config.cgroupfs_path()) {
272                    continue;
273                }
274
275                match fs_type {
276                    // For cgroups v1, we have to go through all mounts we see to build a full list of enabled controlled.
277                    "cgroup" => process_cgroupv1_mount_entry(cgroup_path, &mut controllers),
278                    // For cgroups v2, we only need to find the unified root mountpoint, and then we can create our reader.
279                    "cgroup2" => maybe_cgroups_v2 = process_cgroupv2_mount_entry(cgroup_path)?,
280                    _ => {}
281                }
282            }
283        }
284
285        // If we didn't find any cgroups v1 controllers, then we potentially return the cgroups v2 hierarchy if found...
286        // otherwise, this will just return `None`.
287        if controllers.is_empty() {
288            if maybe_cgroups_v2.is_some() {
289                debug!("Using cgroups v2 hierarchy.");
290            }
291
292            return Ok(maybe_cgroups_v2);
293        }
294
295        // If we're here, we potentially have a cgroups v1 hierarchy.  Find our base controller -- the memory controller
296        // -- and once we do that, we can create our reader.
297        let base_controller_path = controllers
298            .get(CGROUPS_V1_BASE_CONTROLLER_NAME)
299            .cloned()
300            .ok_or_else(|| {
301                generic_error!(
302                    "Failed to find base controller ({}) in cgroups v1 hierarchy.",
303                    CGROUPS_V1_BASE_CONTROLLER_NAME
304                )
305            })?;
306
307        debug!(root = %base_controller_path.display(), controllers_len = controllers.len(), "Using cgroups v1 hierarchy.");
308
309        Ok(Some(HierarchyReader::V1 {
310            base_controller_path,
311            controllers,
312        }))
313    }
314
315    fn base_controller(&self) -> Option<&'static str> {
316        match self {
317            Self::V1 { .. } => Some(CGROUPS_V1_BASE_CONTROLLER_NAME),
318
319            // Since cgroups v2 is "unified", there's no base controller path.
320            Self::V2 { .. } => None,
321        }
322    }
323
324    fn root_path(&self) -> &Path {
325        match self {
326            Self::V1 {
327                base_controller_path, ..
328            } => base_controller_path.as_path(),
329            Self::V2 { root, .. } => root.as_path(),
330        }
331    }
332}

352struct CgroupControllerEntry<'a> {
353    id: usize,
354    name: Option<&'a str>,
355    path: &'a Path,
356}
357
358impl<'a> CgroupControllerEntry<'a> {
359    fn try_from_str(line: &'a str) -> Option<Self> {
360        let mut fields = line.splitn(3, ':');
361
362        let id = fields.next()?.parse::<usize>().ok()?;
363        let name = fields.next().map(|s| if s.is_empty() { None } else { Some(s) })?;
364        let path = fields.next()?;
365
366        if path.is_empty() {
367            return None;
368        }
369
370        Some(Self {
371            id,
372            name,
373            path: Path::new(path),
374        })
375    }
376}

378fn process_cgroupv1_mount_entry(cgroup_path: &Path, controllers: &mut HashMap<String, PathBuf>) {
379    // Split the cgroup path, since there can be multiple controllers mounted at the same path.
380    let path_controllers = cgroup_path
381        .file_name()
382        .and_then(|s| s.to_str().map(|s| s.split(',')))
383        .into_iter()
384        .flatten();
385    for path_controller in path_controllers {
386        // If we have an existing path mapping for this controller, keep whichever one is the
387        // shortest, as we want the more generic path.
388        if let Some(existing_path) = controllers.get(path_controller) {
389            if existing_path.as_os_str().len() < cgroup_path.as_os_str().len() {
390                continue;
391            }
392        }
393
394        controllers.insert(path_controller.to_string(), PathBuf::from(cgroup_path));
395    }
396}

398fn process_cgroupv2_mount_entry(cgroup_path: &Path) -> Result<Option<HierarchyReader>, GenericError> {
399    // Read and get the list of active/enabled controllers.
400    let controllers_path = cgroup_path.join(CGROUPS_V2_CONTROLLERS_FILE);
401    let controllers = read_lines(&controllers_path)
402        .with_error_context(|| {
403            format!(
404                "Failed to read controllers from cgroups v2 hierarchy ({}).",
405                controllers_path.display()
406            )
407        })?
408        .into_iter()
409        .flat_map(|s| s.split_whitespace().map(|s| s.to_string()).collect::<Vec<_>>())
410        .collect::<Vec<_>>();
411
412    Ok(Some(HierarchyReader::V2 {
413        root: cgroup_path.to_path_buf(),
414        controllers,
415    }))
416}

418fn read_lines(path: &Path) -> io::Result<Vec<String>> {
419    let file = OpenOptions::new().read(true).open(path)?;
420
421    let reader = BufReader::new(file).lines();
422
423    let mut lines = Vec::new();
424    for line in reader {
425        lines.push(line?);
426    }
427
428    Ok(lines)
429}

431fn visit_subdirectories<P, F>(path: P, mut visit: F) -> io::Result<()>
432where
433    P: AsRef<Path>,
434    F: FnMut(&Path),
435{
436    // We can only visit directories, so if the initial path we're given isn't a directory, then we can't do anything.
437    let metadata = fs::metadata(path.as_ref())?;
438    if !metadata.is_dir() {
439        return Ok(());
440    }
441
442    // Do an initial pass on our path to get all of its subdirectories, which we'll visit, and then also use as the seed
443    // for further visiting.
444    let mut stack = vec![path.as_ref().to_path_buf()];
445    while let Some(path) = stack.pop() {
446        let dir_reader = fs::read_dir(path)?;
447        for entry in dir_reader {
448            let entry = entry?;
449            if entry.file_type()?.is_dir() {
450                let path = entry.path();
451                visit(&path);
452                stack.push(path);
453            }
454        }
455    }
456
457    Ok(())
458}

460fn extract_container_id(cgroup_name: &str, interner: &GenericMapInterner) -> Option<MetaString> {
461    // This regular expression is meant to capture:
462    // - 64 character hexadecimal strings (standard format for container IDs almost everywhere)
463    // - 32 character hexadecimal strings followed by a dash and a number (used by AWS ECS)
464    // - 8 character hexadecimal strings followed by up to four groups of 4 character hexadecimal strings separated by
465    //   dashes (essentially a UUID, used by Pivotal Cloud Foundry's Garden technology)
466    static CONTAINER_REGEX: LazyLock<Regex> =
467        LazyLock::new(|| Regex::new("([0-9a-f]{64})|([0-9a-f]{32}-\\d+)|([0-9a-f]{8}(-[0-9a-f]{4}){4}$)").unwrap());
468
469    CONTAINER_REGEX
470        .find(cgroup_name)
471        .filter(|name| {
472            // Filter out any systemd-managed cgroups, as well as CRI-O conmon cgroups, as they don't represent containers.
473            !name.as_str().ends_with(".mount") && !name.as_str().starts_with("crio-conmon-")
474        })
475        .and_then(|name| match interner.try_intern(name.as_str()) {
476            Some(interned) => Some(MetaString::from(interned)),
477            None => {
478                error!(container_id = %name.as_str(), "Failed to intern container ID.");
479                None
480            }
481        })
482}
