9/// Containerd event topics.
10pub enum ContainerdTopic {
11    /// Container task started.
12    TaskStarted,
13
14    /// Container task deleted.
15    TaskDeleted,
16}
17
18impl ContainerdTopic {
19    /// Returns the topic as a string.
20    ///
21    /// This can be used to match against the topic string present in the envelope.
22    pub fn as_topic_str(&self) -> &'static str {
23        match self {
24            Self::TaskStarted => "/tasks/start",
25            Self::TaskDeleted => "/tasks/delete",
26        }
27    }
28
29    /// Attempts to parse a topic string into a `ContainerdTopic`.
30    ///
31    /// Returns `None` if the topic string is not recognized/supported.
32    pub fn from_topic_str(topic: &str) -> Option<Self> {
33        match topic {
34            "/tasks/start" => Some(Self::TaskStarted),
35            "/tasks/delete" => Some(Self::TaskDeleted),
36            _ => None,
37        }
38    }
39}

41/// A containerd event.
42///
43/// This is a minimal unified representation of all envelope events that we have decoding support for. Generally, the
44/// event includes only the necessary information to allow a subscriber to react to the event, such as the container ID
45/// for any container-specific events, which then allows the subscriber to fetch additional information about the
46/// container, and so on.
47#[derive(Debug)]
48pub enum ContainerdEvent {
49    /// A task was started.
50    TaskStarted { id: MetaString, pid: u32 },
51
52    /// A task was deleted.
53    TaskDeleted { pid: u32 },
54}
55
56/// Decodes an event envelope into a minimal internal representation.
57///
58/// If the topic of the event is not recognized/supported, `Ok(None)` is returned.
59///
60/// ## Errors
61///
62/// If the envelope payload cannot be decoded, `Err` is returned with the decoder error.
63pub fn decode_envelope_to_event(mut envelope: Envelope) -> Result<Option<ContainerdEvent>, DecodeError> {
64    let topic = match ContainerdTopic::from_topic_str(envelope.topic.as_str()) {
65        Some(topic) => topic,
66        None => return Ok(None),
67    };
68
69    // Alter the envelope event's type URL, if present, to make `prost` happy.
70    //
71    // This boils down to the Protocol Buffer's specification around `Any`, and how it wants the "type URL" field -- the
72    // field that declares the FQDN of the message type's Protocol Buffers definition, essentially -- to include at
73    // least one forward slash.. but containerd doesn't send payloads with the type URL containing a forward slash at
74    // all.. so `prost`, following the specification, generates the expected type URL as "/foo.bar" while te event has
75    // "foo.bar" and decoding fails.
76    if let Some(event) = &mut envelope.event {
77        if !event.type_url.starts_with('/') {
78            event.type_url.insert(0, '/');
79        }
80    }
81
82    match topic {
83        ContainerdTopic::TaskStarted => decode_envelope::<containerd_events::TaskStart>(envelope),
84        ContainerdTopic::TaskDeleted => decode_envelope::<containerd_events::TaskDelete>(envelope),
85    }
86}
87
88fn decode_envelope<D>(envelope: Envelope) -> Result<Option<ContainerdEvent>, DecodeError>
89where
90    D: Default + Name,
91    ContainerdEvent: From<D>,
92{
93    match envelope.event {
94        Some(event) => event.to_msg::<D>().map(|v| Some(v.into())),
95        None => Ok(None),
96    }
97}
