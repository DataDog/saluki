12const OPERATIONS_CHANNEL_SIZE: usize = 128;

14/// Aggregates the metadata from multiple collectors.
15///
16/// Metadata collectors are used to either scrape or listen for changes in workload metadata, and convert those into
17/// [`MetadataOperation`]s that are applied to a [`TagStore`]. [`MetadataAggregator`] is a simple manager type that
18/// controls the lifecycle of each collector added, and processes the metadata operations produced by them in order to
19/// update a single, unified tag store.
20///
21/// Tags can then be accessed through [`TagSnapshot`] (via [`MetadataAggregator::tags`]), which is a shared reference
22/// to the most up-to-date, consistent view of the tag store. This tag snapshot can be used to query for entity tags
23/// directly, with an equivalent API to [`WorkloadProvider`].
24pub struct MetadataAggregator {
25    stores: Vec<Box<dyn MetadataStore + Send>>,
26    operations_tx: mpsc::Sender<MetadataOperation>,
27    operations_rx: mpsc::Receiver<MetadataOperation>,
28    health: Health,
29}
30
31impl MetadataAggregator {
32    /// Create a new `MetadataAggregator`.
33    pub fn new(health: Health) -> Self {
34        let (operations_tx, operations_rx) = mpsc::channel(OPERATIONS_CHANNEL_SIZE);
35        Self {
36            stores: Vec::new(),
37            operations_tx,
38            operations_rx,
39            health,
40        }
41    }
42
43    /// Adds a metadata collector to the aggregator.
44    pub fn add_collector<C>(&mut self, collector: C)
45    where
46        C: MetadataCollector + Send + 'static,
47    {
48        let worker = MetadataCollectorWorker::new(collector);
49        tokio::spawn(worker.run(self.operations_tx.clone()));
50    }
51
52    /// Adds a metadata store to the aggregator.
53    ///
54    /// This store will receive a copy of every metadata operation that is emitted from the configured metadata
55    /// collectors.
56    pub fn add_store<S>(&mut self, store: S)
57    where
58        S: MetadataStore + Send + 'static,
59    {
60        self.stores.push(Box::new(store));
61    }
62
63    /// Runs the aggregator.
64    ///
65    /// This method will run indefinitely, watching for metadata changes seen by the collectors and updating the tag
66    /// store based on those changes. Changes to the tag store will be captured as a tag "snapshot", to which a shared
67    /// reference can be acquired ([`MetadataAggregator::tags`]), and the snapshot allows querying for the tags of a
68    /// specific entity.
69    pub async fn run(mut self) {
70        debug!("Metadata aggregator started.");
71        self.health.mark_ready();
72
73        loop {
74            select! {
75                _ = self.health.live() => {},
76                maybe_operation = self.operations_rx.recv() => match maybe_operation {
77                    Some(operation) => {
78                        // Send the operation to all stores, taking care to only clone the operation if we have two or
79                        // more stores configured.
80                        let stores_to_clone_for = self.stores.len().saturating_sub(1);
81                        for store in self.stores.iter_mut().take(stores_to_clone_for) {
82                            store.process_operation(operation.clone());
83                        }
84
85                        if let Some(last_store) = self.stores.last_mut() {
86                            last_store.process_operation(operation);
87                        }
88                    },
89                    None => {
90                        debug!("Metadata aggregator operations channel closed. Stopping...");
91                        break
92                    },
93                },
94            }
95        }
96
97        self.health.mark_not_ready();
98
99        debug!("Metadata aggregator stopped.");
100    }
101}

103impl MemoryBounds for MetadataAggregator {
104    fn specify_bounds(&self, builder: &mut MemoryBoundsBuilder) {
105        builder
106            .firm()
107            // Operations channel.
108            .with_array::<MetadataOperation>("metadata ops channel", OPERATIONS_CHANNEL_SIZE);
109
110        for store in &self.stores {
111            builder.with_subcomponent(store.name(), store);
112        }
113    }
114}
