13// TODO: Make this configurable.
14const OPERATIONS_CHANNEL_SIZE: usize = 128;
15
16/// Aggregates the metadata from multiple collectors.
17///
18/// Metadata collectors are used to either scrape or listen for changes in workload metadata, and convert those into
19/// [`MetadataOperation`]s that are applied to a [`MetadataStore`]. [`MetadataAggregator`] is a simple manager type that
20/// controls the lifecycle of each collector added, and processes the metadata operations produced by them in order to
21/// update a set of [`MetadataStore`]s.
22pub struct MetadataAggregator {
23    stores: Vec<Box<dyn MetadataStore + Send>>,
24    operations_tx: mpsc::Sender<MetadataOperation>,
25    operations_rx: mpsc::Receiver<MetadataOperation>,
26    health: Health,
27}
28
29impl MetadataAggregator {
30    /// Create a new `MetadataAggregator`.
31    pub fn new(health: Health) -> Self {
32        let (operations_tx, operations_rx) = mpsc::channel(OPERATIONS_CHANNEL_SIZE);
33        Self {
34            stores: Vec::new(),
35            operations_tx,
36            operations_rx,
37            health,
38        }
39    }
40
41    /// Adds a metadata collector to the aggregator.
42    pub fn add_collector<C>(&mut self, collector: C)
43    where
44        C: MetadataCollector + Send + 'static,
45    {
46        let worker = MetadataCollectorWorker::new(collector);
47        tokio::spawn(worker.run(self.operations_tx.clone()));
48    }
49
50    /// Adds a metadata store to the aggregator.
51    ///
52    /// This store will receive a copy of every metadata operation that is emitted from the configured metadata
53    /// collectors.
54    pub fn add_store<S>(&mut self, store: S)
55    where
56        S: MetadataStore + Send + 'static,
57    {
58        self.stores.push(Box::new(store));
59    }
60
61    /// Runs the aggregator.
62    ///
63    /// This method will run indefinitely, watching for metadata changes seen by the collectors and updating the
64    /// configured metadata stores based on those changes.
65    pub async fn run(mut self) {
66        debug!("Metadata aggregator started.");
67        self.health.mark_ready();
68
69        loop {
70            select! {
71                _ = self.health.live() => {},
72                maybe_operation = self.operations_rx.recv() => match maybe_operation {
73                    Some(operation) => {
74                        // Send the operation to all stores, taking care to only clone the operation if we have two or
75                        // more stores configured.
76                        let stores_to_clone_for = self.stores.len().saturating_sub(1);
77                        for store in self.stores.iter_mut().take(stores_to_clone_for) {
78                            store.process_operation(operation.clone());
79                        }
80
81                        if let Some(last_store) = self.stores.last_mut() {
82                            last_store.process_operation(operation);
83                        }
84                    },
85                    None => {
86                        debug!("Metadata aggregator operations channel closed. Stopping...");
87                        break
88                    },
89                },
90            }
91        }
92
93        self.health.mark_not_ready();
94
95        debug!("Metadata aggregator stopped.");
96    }
97}

99impl MemoryBounds for MetadataAggregator {
100    fn specify_bounds(&self, builder: &mut MemoryBoundsBuilder) {
101        builder
102            .firm()
103            // Operations channel.
104            .with_array::<MetadataOperation>("metadata ops channel", OPERATIONS_CHANNEL_SIZE);
105
106        for store in &self.stores {
107            builder.with_subcomponent(store.name(), store);
108        }
109    }
110}

112/// A store which receives a stream of metadata operations.

114    /// Returns the name of the store.
115    fn name(&self) -> &'static str;
116
117    /// Process a metadata operation.
118    fn process_operation(&mut self, operation: MetadataOperation);
