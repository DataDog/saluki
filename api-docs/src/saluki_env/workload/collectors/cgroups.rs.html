23/// A metadata collector that observes Linux "Control Groups" (cgroups).
24///
25/// This collector specifically tracks cgroup controllers attached to container workloads using simple regex-based
26/// matching against the controller path, and maintains a mapping of controller inodes to the container ID extracted
27/// from the controller path.
28///
29/// This is specifically used to support client-based Origin Detection in DogStatsD, where clients will either send
30/// their detected container ID _or_ the inode of their cgroup controller. A canonical container ID must always be used
31/// for origin enrichment, so this mapping allows resolving controller inodes to their canonical container ID.
32pub struct CgroupsMetadataCollector {
33    reader: CgroupsReader,
34    health: Health,
35}
36
37impl CgroupsMetadataCollector {
38    /// Creates a new `CgroupsMetadataCollector` from the given configuration.
39    ///
40    /// # Errors
41    ///
42    /// If a valid cgroups hierarchy can not be located at the configured path, an error will be returned.
43    pub async fn from_configuration(
44        config: &GenericConfiguration, feature_detector: FeatureDetector, health: Health, interner: GenericMapInterner,
45    ) -> Result<Self, GenericError> {
46        let cgroups_config = CgroupsConfiguration::from_configuration(config, feature_detector)?;
47        let reader = match CgroupsReader::try_from_config(&cgroups_config, interner)? {
48            Some(reader) => reader,
49            None => {
50                return Err(generic_error!("Failed to detect any cgroups v1/v2 hierarchy. "));
51            }
52        };
53
54        Ok(Self { reader, health })
55    }
56}

58#[async_trait]
59impl MetadataCollector for CgroupsMetadataCollector {
60    fn name(&self) -> &'static str {
61        "cgroups"
62    }
63
64    async fn watch(&mut self, operations_tx: &mut mpsc::Sender<MetadataOperation>) -> Result<(), GenericError> {
65        self.health.mark_ready();
66
67        let mut poller_handle = None;
68
69        // Drive a blocking background task that polls the cgroups hierarchy on a regular interval, and sends metadata
70        // updates when cgroups are created or deleted. We do this in a blocking task since all of the I/O operations
71        // are synchronous.
72        //
73        // Our loop here ensures that we always keep a single instance of the poller running.
74        loop {
75            // Ensure that we have a poller task running.
76            if poller_handle.is_none() {
77                let mut cgroups_manager = SynchronousCgroupsManager::from_reader(self.reader.clone());
78                let operations_tx = operations_tx.clone();
79
80                poller_handle = Some(tokio::task::spawn_blocking(move || cgroups_manager.poll(operations_tx)));
81
82                debug!("Spawned cgroups background poller task.");
83            }
84
85            select! {
86                _ = self.health.live() => {},
87                result = poller_handle.as_mut().unwrap() => {
88                    // The poller task completed -- for some reason -- so drop our handle to signal that the next loop
89                    // iteration needs to recreate it, and provide some information on _why_ the previous task stopped.
90                    poller_handle = None;
91
92                    match result {
93                        Ok(Ok(())) => unreachable!("Cgroups background poller task should run indefinitely"),
94                        Ok(Err(e)) => error!(error = ?e, "Cgroups background poller task encountered an error."),
95                        Err(e) => error!(error = ?e, "Cgroups background poller task panicked."),
96                    }
97                }
98            }
99        }
100    }
101}

103impl MemoryBounds for CgroupsMetadataCollector {
104    fn specify_bounds(&self, builder: &mut MemoryBoundsBuilder) {
105        builder
106            .minimum()
107            // Pre-allocated operation batch buffer. This is only the minimum, as it could grow larger.
108            .with_array::<MetadataOperation>("metadata operations", 64);
109        // TODO: Kind of a throwaway calculation because nothing about the reader can really be bounded at the moment.
110        //
111        // Specifically, we don't know the number of cgroups that will be present... and we have both a map that holds
112        // the active cgroups _and_ a map to track the cgroups seen during a single traversal, which we need to
113        // determine which cgroups have been removed. This means we might end up with like 3 copies of the same cgroup
114        // times however many cgroups there are at peak.
115        builder.firm().with_single_value::<Self>("component struct");
116    }
117}

119struct SynchronousCgroupsManager {
120    reader: CgroupsReader,
121    active_cgroups: FastHashMap<u64, MetaString>,
122    operations: Vec<MetadataOperation>,
123}
124
125impl SynchronousCgroupsManager {
126    fn from_reader(reader: CgroupsReader) -> Self {
127        Self {
128            reader,
129            active_cgroups: FastHashMap::default(),
130            operations: Vec::with_capacity(64),
131        }
132    }
133
134    fn poll(&mut self, operations_tx: mpsc::Sender<MetadataOperation>) -> Result<(), GenericError> {
135        let mut traversed_cgroups = FastHashSet::default();
136        let mut cgroups_to_delete = Vec::new();
137
138        loop {
139            // Make sure we should still be running.
140            if operations_tx.is_closed() {
141                return Ok(());
142            }
143
144            traversed_cgroups.clear();
145
146            let start = std::time::Instant::now();
147
148            // Traverse the cgroups hierarchy and collect all child cgroups that we can find that are attached to a
149            // container and have a controller inode for us to attach an alias to.
150            let child_cgroups = self.reader.get_child_cgroups();
151            let child_cgroups_len = child_cgroups.len();
152            for child_cgroup in child_cgroups {
153                if let Some(cgroup_inode) = child_cgroup.inode() {
154                    traversed_cgroups.insert(cgroup_inode);
155
156                    // If we haven't seen this cgroup before, start tracking it.
157                    if !self.active_cgroups.contains_key(&cgroup_inode) {
158                        let container_id = child_cgroup.into_container_id();
159                        debug!(%cgroup_inode, %container_id, "Found new container-based cgroup.");
160
161                        self.active_cgroups.insert(cgroup_inode, container_id.clone());
162
163                        // Emit an operation to add an alias between the cgroup inode and the container ID.
164                        let entity_id = EntityId::ContainerInode(cgroup_inode);
165                        let ancestor_entity_id = EntityId::Container(container_id);
166
167                        let operation = MetadataOperation::add_alias(entity_id, ancestor_entity_id);
168                        self.operations.push(operation);
169                    }
170                } else {
171                    // If the cgroup has no inode, we can't track it.
172                    let container_id = child_cgroup.into_container_id();
173                    warn!(%container_id, "Encountered cgroup without controller inode during metadata traversal. This is unexpected.");
174                    continue;
175                }
176            }
177
178            // Figure out which cgroups are no longer active and mark them for deletion.
179            for cgroup_inode in self.active_cgroups.keys() {
180                if !traversed_cgroups.contains(cgroup_inode) {
181                    // This cgroup is no longer present, so we need to delete it.
182                    cgroups_to_delete.push(*cgroup_inode);
183                }
184            }
185
186            // Process the deletions.
187            for cgroup_inode in cgroups_to_delete.drain(..) {
188                if let Some(container_id) = self.active_cgroups.remove(&cgroup_inode) {
189                    debug!(%cgroup_inode, %container_id, "Removing old container-based cgroup.");
190
191                    // Emit a metadata operation to remove the alias between the cgroup inode and the container ID.
192                    let entity_id = EntityId::ContainerInode(cgroup_inode);
193                    let ancestor_entity_id = EntityId::Container(container_id);
194
195                    let operation = MetadataOperation::remove_alias(entity_id, ancestor_entity_id);
196                    self.operations.push(operation);
197                } else {
198                    warn!(%cgroup_inode, "Tried to remove a cgroup that was not in the active set.");
199                }
200            }
201
202            let elapsed = start.elapsed();
203            debug!(elapsed = ?elapsed, child_cgroups_len, "Traversed cgroups.");
204
205            // Send all collected operations to the channel.
206            for operation in self.operations.drain(..) {
207                if operations_tx.blocking_send(operation).is_err() {
208                    return Err(GenericError::msg("Operations channel unexpectedly closed."));
209                }
210            }
211
212            std::thread::sleep(Duration::from_secs(2));
213        }
214    }
