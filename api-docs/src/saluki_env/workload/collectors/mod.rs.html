27/// A metadata collector.
28///
29/// Metadata collectors are responsible for collecting metadata from the environment, both at startup and over time as
30/// changes to the workload occur. This metadata can represent many things, from basic key/value pairs about specific
31/// entities, to fixed relationships between entities, to more dynamic information like the current state of a workload.
32#[async_trait]
33pub trait MetadataCollector {
34    /// Get the name of this collector.
35    fn name(&self) -> &'static str;
36
37    /// Watch for metadata changes.
38    async fn watch(&mut self, operations_tx: &mut mpsc::Sender<MetadataOperation>) -> Result<(), GenericError>;
39}
40
41/// A worker that drives a [`MetadataCollector`] and forwards metadata operations to a central aggregator.
42pub struct MetadataCollectorWorker {
43    name: &'static str,
44    state: Arc<Mutex<MetadataCollectorState>>,
45}
46
47struct MetadataCollectorState {
48    collector: Box<dyn MetadataCollector + Send>,
49    operations_tx: mpsc::Sender<MetadataOperation>,
50}
51
52impl MetadataCollectorWorker {
53    /// Create a new `MetadataCollectorWorker` from the given `collector` and operations sender.
54    pub fn new<MC>(collector: MC, operations_tx: mpsc::Sender<MetadataOperation>) -> Self
55    where
56        MC: MetadataCollector + Send + 'static,
57    {
58        let name = collector.name();
59        Self {
60            name,
61            state: Arc::new(Mutex::new(MetadataCollectorState {
62                collector: Box::new(collector),
63                operations_tx,
64            })),
65        }
66    }
67}
68
69#[async_trait]
70impl Supervisable for MetadataCollectorWorker {
71    fn name(&self) -> &str {
72        self.name
73    }
74
75    async fn initialize(&self, process_shutdown: ProcessShutdown) -> Result<SupervisorFuture, InitializationError> {
76        let state = Arc::clone(&self.state);
77
78        Ok(Box::pin(async move {
79            let mut state_guard = state.lock_owned().await;
80
81            select! {
82                _ = process_shutdown => Ok(()),
83                result = run_collector(&mut state_guard) => result,
84            }
85        }))
86    }
87}
88
89async fn run_collector(state: &mut MetadataCollectorState) -> Result<(), GenericError> {
90    debug!(
91        collector_name = state.collector.name(),
92        "Starting metadata collector worker."
93    );
94
95    let MetadataCollectorState {
96        collector,
97        operations_tx,
98    } = state;
99
100    let result = collector.watch(operations_tx).await;
101    if let Err(e) = &result {
102        warn!(
103            error = %e,
104            collector_name = collector.name(),
105            "Failed to collect metadata. Sleeping 2s before retrying...",
106        );
107        sleep(Duration::from_secs(2)).await;
108    }
109
110    debug!(collector_name = collector.name(), "Metadata collector worker stopped.");
111
112    result
113}
