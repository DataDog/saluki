42/// A workload provider that uses the remote tagger API from a Datadog Agent to provide workload information.
43pub struct RemoteAgentTaggerMetadataCollector {
44    client: RemoteAgentClient,
45    interner: GenericMapInterner,
46    health: Health,
47    telemetry: Telemetry,
48}
49
50impl RemoteAgentTaggerMetadataCollector {
51    /// Creates a new `RemoteAgentTaggerMetadataCollector` from the given configuration.
52    ///
53    /// ## Errors
54    ///
55    /// If the Agent gRPC client cannot be created (invalid API endpoint, missing authentication token, etc), or if the
56    /// authentication token is invalid, an error will be returned.
57    pub async fn from_configuration(
58        config: &GenericConfiguration, health: Health, interner: GenericMapInterner,
59    ) -> Result<Self, GenericError> {
60        let client = RemoteAgentClient::from_configuration(config).await?;
61
62        Ok(Self {
63            client,
64            interner,
65            health,
66            telemetry: Telemetry::new(),
67        })
68    }
69
70    fn try_intern(&self, value: &str) -> Option<MetaString> {
71        match self.interner.try_intern(value) {
72            Some(interned) => Some(MetaString::from(interned)),
73            None => {
74                self.telemetry.intern_failed_total().increment(1);
75                None
76            }
77        }
78    }
79
80    fn owned_tags_into_tagset(&self, tags: Vec<String>) -> Option<TagSet> {
81        // We'll either inline the tags if they're short enough, otherwise we intern them.
82        let mut new_tags = Vec::with_capacity(tags.len());
83        for tag in tags {
84            let new_tag = match MetaString::try_inline(&tag) {
85                Some(s) => Tag::from(s),
86                None => {
87                    let interned = self.try_intern(&tag)?;
88                    Tag::from(interned)
89                }
90            };
91
92            new_tags.push(new_tag);
93        }
94
95        Some(TagSet::from_iter(new_tags))
96    }
97
98    fn track_event(&self, event_type: EventType) {
99        match event_type {
100            EventType::Added => {
101                self.telemetry.events_added_total().increment(1);
102            }
103            EventType::Modified => {
104                self.telemetry.events_modified_total().increment(1);
105            }
106            EventType::Deleted => {
107                self.telemetry.events_deleted_total().increment(1);
108            }
109        }
110    }
111}

113#[async_trait]
114impl MetadataCollector for RemoteAgentTaggerMetadataCollector {
115    fn name(&self) -> &'static str {
116        "remote-agent-tags"
117    }
118
119    async fn watch(&mut self, operations_tx: &mut mpsc::Sender<MetadataOperation>) -> Result<(), GenericError> {
120        self.health.mark_ready();
121
122        let mut entity_stream = self
123            .client
124            .get_tagger_stream(RemoteTagCardinality::High)
125            .map_err(StatusError::from);
126        debug!("Established tagger entity stream.");
127
128        loop {
129            select! {
130                _ = self.health.live() => {},
131                maybe_response = entity_stream.next() => match maybe_response {
132                    Some(Ok(response)) => {
133                        trace!("Received tagger stream event.");
134
135                        for event in response.events {
136                            let event_type = match EventType::try_from(event.r#type) {
137                                Ok(event_type) => event_type,
138                                Err(_) => {
139                                    debug!("Received tagger stream event with unknown type: {}", event.r#type);
140                                    continue;
141                                }
142                            };
143
144                            let entity = match event.entity {
145                                Some(entity) => entity,
146                                None => {
147                                    debug!("Received tagger stream event with no entity.");
148                                    continue;
149                                }
150                            };
151
152                            let entity_id = match entity.id.and_then(remote_entity_id_to_entity_id) {
153                                Some(entity_id) => entity_id,
154                                None => {
155                                    debug!("Received tagger stream event with missing or invalid entity ID.");
156                                    continue;
157                                }
158                            };
159
160                            self.track_event(event_type);
161
162                            let maybe_operation = match event_type {
163                                EventType::Added | EventType::Modified => {
164                                    let entity_tags = [
165                                        (OriginTagCardinality::Low, entity.low_cardinality_tags),
166                                        (OriginTagCardinality::Orchestrator, entity.orchestrator_cardinality_tags),
167                                        (OriginTagCardinality::High, entity.high_cardinality_tags),
168                                    ];
169
170                                    let mut actions = Vec::new();
171                                    for (cardinality, tags) in entity_tags {
172                                        if !tags.is_empty() {
173                                            match self.owned_tags_into_tagset(tags) {
174                                                Some(tags) => actions.push(MetadataAction::SetTags { cardinality, tags }),
175                                                None => {
176                                                    warn!(%entity_id, %cardinality, "Failed to intern tags for entity. Tags will not be present.");
177                                                }
178                                            }
179                                        }
180                                    }
181
182                                    if actions.is_empty() {
183                                        None
184                                    } else {
185                                        Some(MetadataOperation {
186                                            entity_id,
187                                            actions: actions.into(),
188                                        })
189                                    }
190                                }
191                                EventType::Deleted => Some(MetadataOperation::delete(entity_id)),
192                            };
193
194                            if let Some(operation) = maybe_operation {
195                                if let Err(e) = operations_tx.send(operation).await {
196                                    debug!(error = %e, "Failed to send metadata operation.");
197                                }
198                            }
199                        }
200
201                        trace!("Processed tagger stream event.");
202                    },
203                    Some(Err(e)) => {
204                        self.telemetry.rpc_errors_total().increment(1);
205                        return Err(e.into())
206                    },
207                    None => break,
208                }
209            }
210        }
211
212        self.health.mark_not_ready();
213
214        Ok(())
215    }
216}

228fn remote_entity_id_to_entity_id(remote_entity_id: RemoteEntityId) -> Option<EntityId> {
229    // TODO: In the future, it would be nice to do zero-copy deserialization so that we could just intern them (or
230    // inline them) directly instead of having to deal with the owned strings... but for now, we can transparently
231    // convert the owned `String`s to `MetaString`s so it's not a huge deal.
232    match remote_entity_id.prefix.as_str() {
233        "container_id" => Some(EntityId::Container(remote_entity_id.uid.into())),
234        "kubernetes_pod_uid" => Some(EntityId::PodUid(remote_entity_id.uid.into())),
235        "internal" => match remote_entity_id.uid.as_str() {
236            "global-entity-id" => Some(EntityId::Global),
237            uid => {
238                warn!("Unhandled internal entity ID: internal://{}", uid);
239                None
240            }
241        },
242        // We don't care about these, so we just ignore them.
243        "container_image_metadata" => None,
244        prefix => {
245            warn!("Unhandled entity ID prefix: {}://{}", prefix, remote_entity_id.uid);
246            None
247        }
248    }
