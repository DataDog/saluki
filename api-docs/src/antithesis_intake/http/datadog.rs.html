1//! Datadog-compatible HTTP intake routes.
2//!
3//! This module owns the public routes that Datadog Agent and ADP send to:
4//!
5//! - `POST /api/v2/series`: accepts metric series payloads and records payload
6//!   shape assertions.
7//! - `POST /api/beta/sketches`: accepts distribution sketch payloads.
8//! - `POST /api/v1/events_batch`: accepts protobuf event batches.
9//! - `POST /api/v1/events`: accepts JSON event intake payloads and rejects
10//!   malformed bodies.
11//! - `POST /intake/`: accepts the shared JSON intake endpoint and ignores
12//!   non-event bodies.
13//! - `POST /api/v1/check_run`: accepts service check payloads.
14//! - `GET /api/v1/validate`: accepts Datadog Agent connectivity validation.
15
16use axum::{
17    extract::DefaultBodyLimit,
18    http::StatusCode,
19    middleware::from_fn,
20    routing::{get, post},
21    Router,
22};
23use tower::ServiceBuilder;
24use tower_http::{decompression::RequestDecompressionLayer, limit::RequestBodyLimitLayer};
25
26use super::middleware::measure_compressed_size;
27use super::state::AppState;
28use super::MAX_COMPRESSED_BODY_BYTES;
29
30mod events;
31mod metrics;
32mod service_checks;
33
34/// Build Datadog-compatible intake routes.
35pub(crate) fn routes() -> Router<AppState> {
36    Router::new()
37        .merge(series_route())
38        .merge(decoded_payload_routes())
39        .route("/api/v1/validate", get(|| async { StatusCode::OK }))
40}
41
42/// The `/api/v2/series` route. Pyld01-Pyld06 and Pyld22 need the compressed body and raw headers.
43/// `measure_compressed_size` runs outermost, then decompression. The route lifts the default body
44/// limit. The middleware cap is the backstop.
45fn series_route() -> Router<AppState> {
46    let layers = ServiceBuilder::new()
47        .layer(from_fn(measure_compressed_size))
48        .layer(RequestDecompressionLayer::new().pass_through_unaccepted(true))
49        .layer(DefaultBodyLimit::disable());
50    Router::new().route("/api/v2/series", post(metrics::handle_series).layer(layers))
51}
52
53/// Routes that decompress and parse a body without recording `Measurements`. One shared stack
54/// caps the wire body with `RequestBodyLimitLayer` as it streams. Decompression follows. Each
55/// handler caps the decompressed body with `to_bytes`.
56fn decoded_payload_routes() -> Router<AppState> {
57    Router::new()
58        .route("/api/beta/sketches", post(metrics::handle_sketches))
59        .route("/api/v1/events_batch", post(events::handle_events_batch))
60        .route("/api/v1/events", post(events::handle_events_v1))
61        .route("/intake/", post(events::handle_intake))
62        .route("/api/v1/check_run", post(service_checks::handle_check_run_v1))
63        .layer(
64            ServiceBuilder::new()
65                .layer(RequestBodyLimitLayer::new(MAX_COMPRESSED_BODY_BYTES))
66                .layer(RequestDecompressionLayer::new().pass_through_unaccepted(true)),
67        )
68}
