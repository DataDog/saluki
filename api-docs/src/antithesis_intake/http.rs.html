1//! Axum HTTP surface for the intake.
2//!
3//! The `/api/v2/series` route stacks measurement middleware ahead of the
4//! decompression layer so Pyld05 (compressed size), Pyld06 (uncompressed size),
5//! and Pyld22 (content-length) can read both the on-the-wire and decompressed
6//! body lengths, recorded as request extensions before `RequestDecompressionLayer`
7//! consumes the encoding headers.
8
9use std::sync::Arc;
10
11use axum::{
12    body::Body,
13    extract::{DefaultBodyLimit, Request},
14    http::StatusCode,
15    middleware::{from_fn, Next},
16    response::{IntoResponse, Response},
17    routing::post,
18    Router,
19};
20use headers::{ContentEncoding, ContentLength, HeaderMapExt};
21use tower::ServiceBuilder;
22use tower_http::decompression::RequestDecompressionLayer;
23
24use crate::intake;
25
26/// Memory backstop on the compressed body buffered before decompression, above any Pyld05 spec limit
27const MAX_COMPRESSED_BODY_BYTES: usize = 64 * 1024 * 1024;
28
29/// Wire measurements recorded before decompression, attached as a request extension for Pyld05/Pyld06/Pyld22
30#[derive(Clone, Copy, Debug)]
31pub(crate) struct Measurements {
32    /// Compressed, on-the-wire body length, read before decompression.
33    pub(crate) compressed_len: u64,
34    /// Whether the request entered the decompression path.
35    pub(crate) decompression_applied: bool,
36    /// The declared `Content-Length`, or `None` when the header was absent.
37    pub(crate) declared_content_length: Option<u64>,
38}
39
40/// Build the intake router, `/api/v2/series` for payload assertions, others return 200 OK
41pub fn build_router(hostname: Arc<str>) -> Router {
42    // Pyld01-Pyld06 and Pyld22 need the compressed body and raw headers, so the series
43    // route runs `measure_compressed_size` outermost, then decompresses, then
44    // lifts the body limit (the middleware's own cap is the backstop).
45    let series = post(intake::handle_series).layer(
46        ServiceBuilder::new()
47            .layer(from_fn(measure_compressed_size))
48            .layer(RequestDecompressionLayer::new().pass_through_unaccepted(true))
49            .layer(DefaultBodyLimit::disable()),
50    );
51
52    Router::new()
53        .route("/api/v2/series", series)
54        .fallback(|| async { StatusCode::OK })
55        .with_state(hostname)
56}
57
58/// Buffer the body and record compressed size, encoding, and content-length before decompression
59async fn measure_compressed_size(req: Request, next: Next) -> Response {
60    let (parts, body) = req.into_parts();
61    let Ok(bytes) = axum::body::to_bytes(body, MAX_COMPRESSED_BODY_BYTES).await else {
62        return StatusCode::PAYLOAD_TOO_LARGE.into_response();
63    };
64    let len = bytes.len() as u64;
65    let applied = parts
66        .headers
67        .typed_get::<ContentEncoding>()
68        .is_some_and(|enc| enc.contains("deflate") || enc.contains("gzip") || enc.contains("zstd"));
69    let declared = parts.headers.typed_get::<ContentLength>().map(|cl| cl.0);
70    let mut req = Request::from_parts(parts, Body::from(bytes));
71    req.extensions_mut().insert(Measurements {
72        compressed_len: len,
73        decompression_applied: applied,
74        declared_content_length: declared,
75    });
76    next.run(req).await
77}
