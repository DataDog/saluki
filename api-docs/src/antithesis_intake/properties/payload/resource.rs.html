1//! Resource-level checks
2
3use std::sync::OnceLock;
4
5use antithesis_sdk::prelude::*;
6use datadog_protos::metrics::metric_payload::{MetricSeries, Resource};
7use datadog_protos::metrics::MetricPayload;
8use serde_json::json;
9
10/// Pyld18 -- maximum resources per series.
11const MAX_RESOURCES_PER_SERIES: usize = 500;
12/// Pyld19 -- maximum host name length in bytes.
13const MAX_HOST_NAME_BYTES: usize = 255;
14
15/// The first resource with type "host", or None.
16fn host_resource(series: &MetricSeries) -> Option<&Resource> {
17    series.resources.iter().find(|r| r.type_() == "host")
18}
19
20/// Pyld17 -- every metric context across all inbound traffic on a lane resolves a non-empty host,
21/// and they all share one host. `established` is set once from the first host seen so the check spans
22/// every request. Cross-lane host equality is checked by the differential oracle.
23pub(crate) fn host_consistent(payload: &MetricPayload, established: &OnceLock<String>) {
24    let mut observed = None;
25    for ms in &payload.series {
26        let name = host_resource(ms).map_or("", Resource::name);
27        if name.is_empty() {
28            observed = Some(json!({ "metric": ms.metric(), "issue": "empty_or_missing" }));
29            break;
30        }
31        let prev = established.get_or_init(|| name.to_owned());
32        if prev != name {
33            observed =
34                Some(json!({ "metric": ms.metric(), "issue": "mismatch", "established": prev, "resolved": name }));
35            break;
36        }
37    }
38    assert_always!(
39        observed.is_none(),
40        "Pyld17.host_resource_resolved",
41        &json!({ "series": payload.series.len(), "host": established.get(), "observed": observed })
42    );
43}
44
45/// Pyld18 -- resource count at most MaxResources(orgID).
46pub(crate) fn resource_count(ms: &MetricSeries) {
47    let over = (ms.resources.len() > MAX_RESOURCES_PER_SERIES).then_some(ms.resources.len());
48    assert_always!(
49        over.is_none(),
50        "Pyld18.resource_count",
51        &json!({ "metric": ms.metric(), "max_resources": MAX_RESOURCES_PER_SERIES, "observed": over })
52    );
53}
54
55/// Pyld19 -- host name at most 255 bytes.
56pub(crate) fn host_name_length(ms: &MetricSeries) {
57    let over = host_resource(ms)
58        .map(Resource::name)
59        .filter(|name| name.len() > MAX_HOST_NAME_BYTES);
60    assert_always!(
61        over.is_none(),
62        "Pyld19.host_name_length",
63        &json!({ "metric": ms.metric(), "observed": over })
64    );
65}
