8/// Hashes a metric context.
9///
10/// Takes a metric name, an iterator of tags, and an iterator of origin tags, and returns a tuple containing a unique
11/// hash key for the overall context, and a unique hash key for the non-origin tags by themselves.
12///
13/// All tags are hashed in an order-oblivious (XOR) manner, which allows tags to be hashed in any order while still
14/// resulting in the same overall hash. This function is _not_ oblivious to the actual tag values themselves, though, so
15/// differences such as case (lower vs upper) or leading/trailing whitespace will influence the resulting hash.
16///
17/// If a tag is seen more than once, it will be ignored and not included in the overall hash. This function allocates a
18/// hash set in order to track which tags have already been hashed, so it is preferable to allocate a single
19/// [`PrehashedHashSet`] and use it with [`hash_context_with_seen`] in order to amortize the cost of allocating the hash
20/// set.
21///
22/// Returns a hash that uniquely identifies the combination of name, tags, and origin of the value.
23pub fn hash_context<I, I2, T, T2>(name: &str, tags: I, origin_tags: I2) -> (ContextKey, TagSetKey)
24where
25    I: IntoIterator<Item = T>,
26    T: AsRef<str>,
27    I2: IntoIterator<Item = T2>,
28    T2: AsRef<str>,
29{
30    let mut seen = PrehashedHashSet::default();
31    hash_context_with_seen(name, tags, origin_tags, &mut seen)
32}

34/// Hashes a metric context, using a provided set to track which tags have already been hashed.
35///
36/// Takes a metric name, an iterator of tags, and an iterator of origin tags, and returns a tuple containing a unique
37/// hash key for the overall context, and a unique hash key for the non-origin tags by themselves.
38///
39/// All tags are hashed in an order-oblivious (XOR) manner, which allows tags to be hashed in any order while still
40/// resulting in the same overall hash. This function is _not_ oblivious to the actual tag values themselves, though, so
41/// differences such as case (lower vs upper) or leading/trailing whitespace will influence the resulting hash.
42///
43/// If a tag is seen more than once, it will be ignored and not included in the overall hash. This function requires
44/// the caller to provide the hash set used for tracking duplicates, and is more efficient than [`hash_context`] which
45/// allocates a new hash set each time.
46///
47/// Returns a hash that uniquely identifies the combination of name, tags, and origin of the value.
48pub(super) fn hash_context_with_seen<I, I2, T, T2>(
49    name: &str, tags: I, origin_tags: I2, seen: &mut PrehashedHashSet<u64>,
50) -> (ContextKey, TagSetKey)
51where
52    I: IntoIterator<Item = T>,
53    T: AsRef<str>,
54    I2: IntoIterator<Item = T2>,
55    T2: AsRef<str>,
56{
57    seen.clear();
58
59    let mut hasher = get_fast_hasher();
60
61    // Hash the metric name.
62    name.hash(&mut hasher);
63
64    // Hash the metric tags individually and XOR their hashes together, which allows us to be order-oblivious:
65    let mut combined_tags_hash = 0;
66
67    for tag in tags {
68        let tag_hash = hash_single_fast(tag.as_ref());
69
70        // If we've already seen this tag before, skip combining it again.
71        if !seen.insert(tag_hash) {
72            continue;
73        }
74
75        combined_tags_hash ^= tag_hash;
76    }
77
78    hasher.write_u64(combined_tags_hash);
79
80    // Finally, hash the origin tags.
81    //
82    // We also hash these in an order-oblivious manner.
83    seen.clear();
84    let mut combined_origin_tags_hash = 0;
85
86    for tag in origin_tags {
87        let tag_hash = hash_single_fast(tag.as_ref());
88
89        // If we've already seen this tag before, skip combining it again.
90        if !seen.insert(tag_hash) {
91            continue;
92        }
93
94        combined_origin_tags_hash ^= tag_hash;
95    }
96
97    hasher.write_u64(combined_origin_tags_hash);
98
99    let context_key = ContextKey { hash: hasher.finish() };
100    let tagset_key = TagSetKey {
101        hash: combined_tags_hash,
102    };
103
104    (context_key, tagset_key)
105}
