11const BASE_CONTEXT_SIZE: usize = std::mem::size_of::<Context>() + std::mem::size_of::<ContextInner>();

13/// A metric context.
14#[derive(Clone, Debug, Eq, Hash, PartialEq)]
15pub struct Context {
16    inner: Arc<ContextInner>,
17}
18
19impl Context {
20    /// Creates a new `Context` from the given static name.
21    pub fn from_static_name(name: &'static str) -> Self {
22        let tags = TagSet::default();
23        let origin_tags = TagSet::default();
24
25        let (key, _) = hash_context(name, &tags, &origin_tags);
26        Self {
27            inner: Arc::new(ContextInner {
28                name: MetaString::from_static(name),
29                tags,
30                origin_tags,
31                key,
32                active_count: Gauge::noop(),
33            }),
34        }
35    }
36
37    /// Creates a new `Context` from the given static name and given static tags.
38    pub fn from_static_parts(name: &'static str, tags: &[&'static str]) -> Self {
39        let mut tag_set = TagSet::with_capacity(tags.len());
40        for tag in tags {
41            tag_set.insert_tag(MetaString::from_static(tag));
42        }
43
44        let origin_tags = TagSet::default();
45
46        let (key, _) = hash_context(name, tags, &origin_tags);
47        Self {
48            inner: Arc::new(ContextInner {
49                name: MetaString::from_static(name),
50                tags: tag_set,
51                origin_tags,
52                key,
53                active_count: Gauge::noop(),
54            }),
55        }
56    }
57
58    /// Creates a new `Context` from the given name and given tags.
59    pub fn from_parts<S: Into<MetaString>>(name: S, tags: impl Into<TagSet>) -> Self {
60        let name = name.into();
61        let tags = tags.into();
62        let origin_tags = TagSet::default();
63        let (key, _) = hash_context(&name, &tags, &origin_tags);
64        Self {
65            inner: Arc::new(ContextInner {
66                name,
67                tags,
68                origin_tags,
69                key,
70                active_count: Gauge::noop(),
71            }),
72        }
73    }
74
75    /// Clones this context, and uses the given name for the cloned context.
76    pub fn with_name<S: Into<MetaString>>(&self, name: S) -> Self {
77        // Regenerate the context key to account for the new name.
78        let name = name.into();
79        let tags = self.inner.tags.clone();
80        let origin_tags = self.inner.origin_tags.clone();
81        let (key, _) = hash_context(&name, &tags, &origin_tags);
82
83        Self {
84            inner: Arc::new(ContextInner {
85                name,
86                tags,
87                origin_tags,
88                key,
89                active_count: Gauge::noop(),
90            }),
91        }
92    }
93
94    pub(crate) fn from_inner(inner: ContextInner) -> Self {
95        Self { inner: Arc::new(inner) }
96    }
97
98    #[cfg(test)]
99    pub(crate) fn ptr_eq(&self, other: &Self) -> bool {
100        Arc::ptr_eq(&self.inner, &other.inner)
101    }
102
103    /// Returns the name of this context.
104    pub fn name(&self) -> &MetaString {
105        &self.inner.name
106    }
107
108    /// Returns the instrumented tags of this context.
109    pub fn tags(&self) -> &TagSet {
110        &self.inner.tags
111    }
112
113    /// Returns the origin tags of this context.
114    pub fn origin_tags(&self) -> &TagSet {
115        &self.inner.origin_tags
116    }
117
118    /// Mutates the instrumented tags of this context via a closure.
119    ///
120    /// Uses copy-on-write semantics: if this context shares its inner data with other clones, the
121    /// inner data is cloned first so that mutations do not affect other holders. If this context is
122    /// the sole owner, the mutation happens in place.
123    ///
124    /// The context key is automatically recomputed after the closure returns.
125    pub fn mutate_tags(&mut self, f: impl FnOnce(&mut TagSet)) {
126        self.mutate_inner(|inner| f(&mut inner.tags));
127    }
128
129    /// Mutates the origin tags of this context via a closure.
130    ///
131    /// Uses copy-on-write semantics: if this context shares its inner data with other clones, the
132    /// inner data is cloned first so that mutations do not affect other holders. If this context is
133    /// the sole owner, the mutation happens in place.
134    ///
135    /// The context key is automatically recomputed after the closure returns.
136    pub fn mutate_origin_tags(&mut self, f: impl FnOnce(&mut TagSet)) {
137        self.mutate_inner(|inner| f(&mut inner.origin_tags));
138    }
139
140    /// Runs the given closure on the inner context data, recomputing the context key afterwards.
141    ///
142    /// When the inner context state is shared (we aren't the only ones with a strong reference), we clone the inner
143    /// data first to have our own copy. Otherwise, we modify the inner data in place.
144    fn mutate_inner(&mut self, f: impl FnOnce(&mut ContextInner)) {
145        let inner = Arc::make_mut(&mut self.inner);
146        f(inner);
147        let (key, _) = hash_context(&inner.name, &inner.tags, &inner.origin_tags);
148        inner.key = key;
149    }
150
151    /// Returns the size of this context in bytes.
152    ///
153    /// A context's size is the sum of the sizes of its fields and the size of the `Context` struct itself, and
154    /// includes:
155    /// - the context name
156    /// - the context tags (both instrumented and origin)
157    ///
158    /// Since origin tags can potentially be expensive to calculate, this method will cache the size of the origin tags
159    /// when this method is first called.
160    ///
161    /// Additionally, the value returned by this method does not compensate for externalities such as origin tags
162    /// potentially being shared by multiple contexts, or whether or not tags are inlined, interned, or heap
163    /// allocated. This means that the value returned is essentially the worst-case usage, and should be used as a rough
164    /// estimate.
165    pub fn size_of(&self) -> usize {
166        let name_size = self.inner.name.len();
167        let tags_size = self.inner.tags.size_of();
168        let origin_tags_size = self.inner.origin_tags.size_of();
169
170        BASE_CONTEXT_SIZE + name_size + tags_size + origin_tags_size
171    }
172}

174impl From<&'static str> for Context {

180impl<'a> From<(&'static str, &'a [&'static str])> for Context {
181    fn from((name, tags): (&'static str, &'a [&'static str])) -> Self {

186impl fmt::Display for Context {
187    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
188        write!(f, "{}", self.inner.name)?;
189        if !self.inner.tags.is_empty() {
190            write!(f, "{{")?;
191
192            let mut needs_separator = false;
193            for tag in &self.inner.tags {
194                if needs_separator {
195                    write!(f, ", ")?;
196                } else {
197                    needs_separator = true;
198                }
199
200                write!(f, "{}", tag)?;
201            }
202
203            write!(f, "}}")?;
204        }
205
206        Ok(())
207    }
208}

218impl ContextInner {
219    pub fn from_parts(
220        key: ContextKey, name: MetaString, tags: TagSet, origin_tags: TagSet, active_count: Gauge,
221    ) -> Self {
222        Self {
223            key,
224            name,
225            tags,
226            origin_tags,
227            active_count,
228        }

232impl Clone for ContextInner {
233    fn clone(&self) -> Self {
234        Self {
235            key: self.key,
236            name: self.name.clone(),
237            tags: self.tags.clone(),
238            origin_tags: self.origin_tags.clone(),
239
240            // We're specifically detaching this context from the statistics of the resolver from which `self`
241            // originated, as we only want to track the statistics of the contexts created _directly_ through the
242            // resolver.
243            active_count: Gauge::noop(),
244        }
245    }
246}
247
248impl Drop for ContextInner {
249    fn drop(&mut self) {
250        self.active_count.decrement(1);
251    }
252}
253
254impl PartialEq<ContextInner> for ContextInner {
255    fn eq(&self, other: &ContextInner) -> bool {
256        // TODO: Note about why we consider the hash good enough for equality.
257        self.key == other.key
258    }
259}

400        // Build a context and mutate it to add a tag.
401        let mut mutated = Context::from_static_parts("metric", &["env:prod"]);
402        mutated.mutate_tags(|tags| {

406        // Build an equivalent context from scratch with both tags.

409        // The recomputed key should match a freshly-constructed context with the same state.
