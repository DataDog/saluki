58/// Component data type.
59///
60/// This is used to determine the type of data that a component can produce and/or consume.
61#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
62pub enum DataType {
63    /// Events.
64    Event(EventType),
65
66    /// Payloads.
67    Payload(PayloadType),
68}
69
70impl DataType {
71    /// Returns `true` if `self` is the same integral data type as `other`, and `other` has at least one overlapping
72    /// subtype with `self`.
73    ///
74    /// This means that event and payload types are always disjoint.
75    fn intersects(&self, other: DataType) -> bool {
76        match (self, other) {
77            (DataType::Event(a), DataType::Event(b)) => !a.is_none() && !b.is_none() && a.intersects(b),
78            (DataType::Payload(a), DataType::Payload(b)) => !a.is_none() && !b.is_none() && a.intersects(b),
79            _ => false,
80        }
81    }
82}
83
84impl From<EventType> for DataType {
85    fn from(ty: EventType) -> Self {
86        DataType::Event(ty)
87    }
88}
89
90impl From<PayloadType> for DataType {
91    fn from(ty: PayloadType) -> Self {
92        DataType::Payload(ty)
93    }
94}

98        match self {
99            DataType::Event(ty) => write!(f, "event({})", ty),
100            DataType::Payload(ty) => write!(f, "payload({})", ty),
101        }
102    }

325    /// Adds an edge to the graph.
326    ///
327    /// # Errors
328    ///
329    /// If either the `from` or `to` component IDs do not exist in the graph, or are invalid, an error is returned.
330    pub fn add_edge<F, T>(&mut self, from: F, to: T) -> Result<(), GraphError>
331    where
332        F: AsRef<str>,
333        T: AsRef<str>,
334    {
335        let component_id = try_into_component_id(to)?;
336        let component_output_id = try_into_component_output_id(from)?;
337
338        if !self.nodes.contains_key(&component_id) {
339            return Err(GraphError::NonexistentComponentId { component_id });
340        }
341
342        // Find the node that the "from" side of the edge is connected to.
343        match self.nodes.get(&component_output_id.component_id()) {
344            Some(node) => match node {
345                // Sources, relays, and transforms support named outputs, so we have to dig into their outputs to make
346                // sure the given output ID exists.
347                Node::Source { outputs } | Node::Relay { outputs } | Node::Transform { outputs, .. } => {
348                    if !outputs
349                        .iter()
350                        .any(|output| output.component_output() == &component_output_id)
351                    {
352                        return Err(GraphError::NonexistentComponentOutputId { component_output_id });
353                    }
354                }
355                // Decoders and encoders only have default outputs, so make sure the "from" side is referencing a
356                // default output.
357                Node::Decoder { .. } | Node::Encoder { .. } => {
358                    if !component_output_id.is_default() {
359                        return Err(GraphError::NonexistentComponentOutputId { component_output_id });
360                    }
361                }
362                // Destinations and forwarders are terminal nodes, so they can't have outbound edges.
363                Node::Destination { .. } | Node::Forwarder { .. } => {
364                    return Err(GraphError::NonexistentComponentOutputId { component_output_id })
365                }
366            },
367            None => {
368                return Err(GraphError::NonexistentComponentId {
369                    component_id: component_output_id.component_id(),
370                })
371            }
372        }
373
374        self.edges.push(Edge {
375            from: component_output_id,
376            to: component_id,
377        });
378        Ok(())
379    }

381    fn get_input_type(&self, id: &ComponentId) -> DataType {
382        match self.nodes[id] {
383            Node::Source { .. } | Node::Relay { .. } => panic!("no inputs on sources/relays"),
384            Node::Decoder { input_ty, .. } => DataType::Payload(input_ty),
385            Node::Transform { input_ty, .. } => DataType::Event(input_ty),
386            Node::Encoder { input_ty, .. } => DataType::Event(input_ty),
387            Node::Forwarder { input_ty } => DataType::Payload(input_ty),
388            Node::Destination { input_ty } => DataType::Event(input_ty),
389        }
390    }
391
392    fn get_output_type(&self, id: &ComponentOutputId) -> DataType {
393        match &self.nodes[&id.component_id()] {
394            Node::Source { outputs } | Node::Relay { outputs } | Node::Transform { outputs, .. } => outputs
395                .iter()
396                .find(|output| output.component_output().output() == id.output())
397                .map(|output| output.output_ty())
398                .expect("output didn't exist"),
399            Node::Decoder { output_ty, .. } => {
400                if id.is_default() {
401                    (*output_ty).into()
402                } else {
403                    panic!("decoder should only have default output")
404                }
405            }
406            Node::Encoder { output_ty, .. } => {
407                if id.is_default() {
408                    (*output_ty).into()
409                } else {
410                    panic!("encoder should only have default output")
411                }
412            }
413            Node::Destination { .. } | Node::Forwarder { .. } => panic!("no outputs on destinations/forwarders"),
414        }
415    }

417    /// Validates the graph.
418    ///
419    /// Several invariants are checked:
420    ///
421    /// - All edges must have compatible data types.
422    /// - No cycles are allowed in the graph.
423    /// - All components must be connected to at least one other component.
424    ///
425    /// ## Data types
426    ///
427    /// Components can have two possible data types: events and payloads. Each of these has a number of subtypes, such
428    /// as a "metric" event, or a "raw" payload. When two components are connected by an edge, they must both have the
429    /// same data type, and have an overlap in data subtypes.
430    ///
431    /// For example, sources, transforms, and destinations all deal exclusively with events, which means that to be
432    /// connected, they must simply have an overlap in the event subtypes they support, such as a downstream component
433    /// at least supporting metric events if the upstream only emits metrics.
434    ///
435    /// Additionally, there are encoder and forwarder components, which deal with events and payloads, respectively.
436    /// Forwarders can only accept payloads, so they cannot be connected directly to "event" components like sources or
437    /// transforms, but they can be connected to encoders. Encoders can accept events and emit payloads, so they can
438    /// be connected to sources and transforms as a downstream component (same "event" data type) but cannot be
439    /// connected to another encoder (different data types). Like "event" components, encoders and forwarders must
440    /// have an overlap in "payload" subtypes to be connected.
441    ///
442    /// # Errors
443    ///
444    /// If any of the invariants are violated, an error is returned.
445    pub fn validate(&self) -> Result<(), GraphError> {
446        self.check_event_types()?;
447        self.check_for_cycles()?;
448        self.check_for_disconnected()?;
449
450        Ok(())
451    }
452
453    fn check_event_types(&self) -> Result<(), GraphError> {
454        // Check that all connected edges have compatible data types.
455        for edge in &self.edges {
456            let from_ty = self.get_output_type(&edge.from);
457            let to_ty = self.get_input_type(&edge.to);
458
459            if !from_ty.intersects(to_ty) {
460                return Err(GraphError::DataTypeMismatch {
461                    from_component_output_id: edge.from.clone(),
462                    from_ty,
463                    to_component_id: edge.to.clone(),
464                    to_ty,
465                });
466            }
467        }
468
469        Ok(())
470    }
471
472    fn check_for_cycles(&self) -> Result<(), GraphError> {
473        let destinations = self.nodes.iter().filter_map(|(id, node)| match node {
474            Node::Destination { .. } => Some(id.clone()),
475            _ => None,
476        });
477
478        // Run a depth-first search from each destination while keep tracking the current stack to detect cycles.
479        for d in destinations {
480            let mut traversal = VecDeque::new();
481            let mut visited = HashSet::new();
482            let mut stack = IndexSet::new();
483
484            traversal.push_back(d);
485            while !traversal.is_empty() {
486                let n = traversal.back().cloned().expect("can't be empty");
487                if !visited.contains(&n) {
488                    visited.insert(n.clone());
489                    stack.insert(n.clone());
490                } else {
491                    // We came back to the node after exploring all its children: remove it from the stack and
492                    // traversal.
493                    stack.shift_remove(&n);
494                    traversal.pop_back();
495                }
496                let inputs = self.edges.iter().filter(|e| e.to == n).map(|e| &e.from);
497                for input in inputs {
498                    let input_component_id = input.component_id();
499                    if !visited.contains(&input_component_id) {
500                        traversal.push_back(input_component_id);
501                    } else if stack.contains(&input_component_id) {
502                        // We reached the node while it is on the current stack: it's a cycle.
503                        let mut path = stack
504                            .iter()
505                            .skip(1) // skip the sink
506                            .rev()
507                            .cloned()
508                            .collect::<Vec<_>>();
509                        path.insert(0, input_component_id.clone());
510
511                        return Err(GraphError::Cycle { path });
512                    }
513                }
514            }
515        }
516
517        Ok(())
518    }

520    fn check_for_disconnected(&self) -> Result<(), GraphError> {
521        let mut unchecked_nodes = self.nodes.keys().cloned().collect::<HashSet<_>>();
522
523        for edge in &self.edges {
524            unchecked_nodes.remove(&edge.from.component_id());
525            unchecked_nodes.remove(&edge.to);
526        }
527
528        if unchecked_nodes.is_empty() {
529            Ok(())
530        } else {
531            let mut component_ids = unchecked_nodes.into_iter().collect::<Vec<_>>();
532            component_ids.dedup();
533            component_ids.sort();
534            Err(GraphError::DisconnectedComponents { component_ids })
535        }
536    }

553    /// Returns a mapping of component IDs to their outputs and the component IDs that are connected to them.
554    pub fn get_outbound_directed_edges(&self) -> HashMap<TypedComponentId, HashMap<OutputName, Vec<TypedComponentId>>> {
555        let mut mappings: HashMap<TypedComponentId, HashMap<OutputName, Vec<TypedComponentId>>> = HashMap::new();
556
557        // Populate the map so that every component has its outputs represented.
558        //
559        // We want to ensure that the caller can reconstruct all of the _configured_ outputs of a component,
560        // even if an output doesn't have any edges (connections).
561        for (component_id, node) in &self.nodes {
562            let component_type = self.get_component_type(component_id).unwrap();
563            let typed_component_id = TypedComponentId::new(component_id.clone(), component_type);
564
565            match node {
566                Node::Source { outputs } | Node::Relay { outputs } | Node::Transform { outputs, .. } => {
567                    let per_component = mappings.entry(typed_component_id).or_default();
568                    for output in outputs {
569                        per_component.insert(output.component_output().output(), Vec::new());
570                    }
571                }
572                Node::Decoder { .. } | Node::Encoder { .. } => {
573                    // Decoders and encoders have a default output
574                    let per_component = mappings.entry(typed_component_id).or_default();
575                    per_component.insert(OutputName::Default, Vec::new());
576                }
577                _ => {}
578            }
579        }
580
581        for edge in &self.edges {
582            let raw_from_id = edge.from.component_id();
583            let from_type = self.get_component_type(&raw_from_id).expect("component ID not found");
584            let from_id = TypedComponentId::new(raw_from_id, from_type);
585
586            let per_component = mappings.entry(from_id).or_default();
587            let per_output = per_component.entry(edge.from.output()).or_default();
588
589            let raw_to_id = edge.to.clone();
590            let to_type = self.get_component_type(&raw_to_id).expect("component ID not found");
591            let to_id = TypedComponentId::new(raw_to_id, to_type);
592
593            per_output.push(to_id);
594        }
595
596        mappings
597    }
598}

600fn try_into_component_id<I>(component_id: I) -> Result<ComponentId, GraphError>
601where
602    I: AsRef<str>,
603{
604    ComponentId::try_from(component_id.as_ref()).map_err(|e| GraphError::InvalidComponentId {
605        input: component_id.as_ref().to_string(),
606        reason: e,
607    })
608}

610fn try_into_component_output_id<I>(component_output_id: I) -> Result<ComponentOutputId, GraphError>
611where
612    I: AsRef<str>,
613{
614    ComponentOutputId::try_from(component_output_id.as_ref()).map_err(|e| GraphError::InvalidComponentOutputId {
615        input: component_output_id.as_ref().into(),
616        reason: e,
617    })
618}

620fn construct_typed_output_ids<T>(
621    component_id: &ComponentId, outputs: &[OutputDefinition<T>],
622) -> Result<Vec<TypedComponentOutputId>, GraphError>
623where
624    T: Into<DataType> + Copy,
625{
626    let mut typed_output_ids = Vec::new();
627    for output in outputs {
628        let output_id = ComponentOutputId::from_definition(component_id.clone(), output)
629            .map_err(|(input, reason)| GraphError::InvalidComponentOutputId { input, reason })?;
630        typed_output_ids.push(TypedComponentOutputId::new(output_id, output.data_ty().into()));
631    }
632
633    Ok(typed_output_ids)
634}
