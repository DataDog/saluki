61const FLUSH_INTERVAL: Duration = Duration::from_secs(1);
62const INTERNAL_METRICS_INTERNER_SIZE: NonZeroUsize = NonZeroUsize::new(16_384).unwrap();

64/// Number of consecutive flush intervals that a metric must go without a live handle and without being re-registered
65/// before it is evicted from the registry.
66///
67/// This grace period prevents metrics that are emitted through `metrics` macros without caching the returned handle --
68/// and that therefore re-register on every call -- from being churned in and out of the registry while they are still
69/// being emitted. Naturally, this only applies to metrics which re-register within `IDLE_EVICT_INTERVALS *
70/// FLUSH_INTERVAL`, which is 3 seconds by default... but that's a tradeoff we're making here to avoid unbounded growth.
71const IDLE_EVICT_INTERVALS: u32 = 3;

88struct Handle<T> {
89    inner: T,
90    level: Level,
91    idle: AtomicU32,
92}
93
94impl<T> Handle<T> {
95    const fn new(inner: T, level: Level) -> Self {
96        Self {
97            inner,
98            level,
99            idle: AtomicU32::new(0),
100        }
101    }
102
103    fn level(&self) -> Level {
104        self.level
105    }
106
107    fn reset_idle(&self) {
108        self.idle.store(0, Ordering::Relaxed);
109    }
110
111    fn bump_idle(&self) -> u32 {
112        self.idle.fetch_add(1, Ordering::Relaxed).saturating_add(1)
113    }
114}

116struct CounterInner {
117    value: AtomicU64,
118}
119
120impl Handle<CounterInner> {
121    const fn new_counter(level: Level) -> Self {
122        Self::new(
123            CounterInner {
124                value: AtomicU64::new(0),
125            },
126            level,
127        )
128    }
129
130    /// Consumes the accumulated delta since the last flush, resetting the counter to zero.
131    fn consume(&self) -> u64 {
132        self.inner.value.swap(0, Ordering::Relaxed)
133    }
134}
135
136impl CounterFn for Handle<CounterInner> {
137    fn increment(&self, value: u64) {
138        CounterFn::increment(&self.inner.value, value);
139    }
140
141    fn absolute(&self, value: u64) {
142        CounterFn::absolute(&self.inner.value, value);
143    }
144}

146struct GaugeInner {
147    value: AtomicU64,
148    // Tracks whether or not the gauge has been written since the last flush, since the gauge _could_
149    // be modified in a way where the value seen by two consecutive flushes is the same, even though
150    // it _was_ modified between the two and should be considered non-idle.
151    dirty: AtomicBool,
152}
153
154impl Handle<GaugeInner> {
155    const fn new_gauge(level: Level) -> Self {
156        Self::new(
157            GaugeInner {
158                value: AtomicU64::new(0),
159                dirty: AtomicBool::new(false),
160            },
161            level,
162        )
163    }
164
165    /// Reads the current gauge value.
166    fn load(&self) -> f64 {
167        f64::from_bits(self.inner.value.load(Ordering::Relaxed))
168    }
169
170    /// Returns whether the gauge was written since the last flush, clearing the flag.
171    fn take_dirty(&self) -> bool {
172        self.inner.dirty.swap(false, Ordering::Relaxed)
173    }
174}
175
176impl GaugeFn for Handle<GaugeInner> {
177    fn increment(&self, value: f64) {
178        GaugeFn::increment(&self.inner.value, value);
179        self.inner.dirty.store(true, Ordering::Relaxed);
180    }
181
182    fn decrement(&self, value: f64) {
183        GaugeFn::decrement(&self.inner.value, value);
184        self.inner.dirty.store(true, Ordering::Relaxed);
185    }
186
187    fn set(&self, value: f64) {
188        GaugeFn::set(&self.inner.value, value);
189        self.inner.dirty.store(true, Ordering::Relaxed);
190    }
191}

193/// Storage for a single histogram, shared between the registry and every caller-held [`Histogram`].
194struct HistogramInner {
195    value: AtomicBucket<f64>,
196}
197
198impl Handle<HistogramInner> {
199    fn new_histogram(level: Level) -> Self {
200        Self::new(
201            HistogramInner {
202                value: AtomicBucket::new(),
203            },
204            level,
205        )
206    }
207
208    /// Drains all recorded samples since the last flush into `out`, clearing the histogram.
209    fn drain_into(&self, out: &mut Vec<f64>) {
210        self.inner.value.clear_with(|samples| out.extend(samples));
211    }
212}

214impl HistogramFn for Handle<HistogramInner> {
215    fn record(&self, value: f64) {
216        self.inner.value.push(value);
217    }

220/// A registry for all internal metrics.
221///
222/// Optimized for simplicity and the ability to efficiently track metric state and evict idle metrics. Not optimized for
223/// high concurrency with regards to registration, so metric handles should always be held for as long as possible, when
224/// possible.
225#[derive(Default)]
226struct MetricsRegistry {
227    maps: Mutex<RegistryMaps>,
228}
229
230#[derive(Default)]
231struct RegistryMaps {
232    counters: FastHashMap<Key, Arc<Handle<CounterInner>>>,
233    gauges: FastHashMap<Key, Arc<Handle<GaugeInner>>>,
234    histograms: FastHashMap<Key, Arc<Handle<HistogramInner>>>,
235}
236
237impl MetricsRegistry {
238    /// Returns a handle to the counter for `key`, creating it at `level` if it doesn't yet exist.
239    ///
240    /// The level is fixed when the metric is first created; re-registration only refreshes the idle
241    /// counter (the activity signal that keeps actively emitted metrics from being evicted).
242    fn get_or_create_counter(&self, key: &Key, level: Level) -> Counter {
243        let mut maps = self.maps.lock().unwrap();
244        let handle = if let Some(handle) = maps.counters.get(key) {
245            handle.reset_idle();
246            Arc::clone(handle)
247        } else {
248            let handle = Arc::new(Handle::new_counter(level));
249            maps.counters.insert(key.clone(), Arc::clone(&handle));
250            handle
251        };
252
253        Counter::from_arc(handle)
254    }
255
256    /// Returns a handle to the gauge for `key`, creating it at `level` if it doesn't yet exist.
257    fn get_or_create_gauge(&self, key: &Key, level: Level) -> Gauge {
258        let mut maps = self.maps.lock().unwrap();
259        let handle = if let Some(handle) = maps.gauges.get(key) {
260            handle.reset_idle();
261            Arc::clone(handle)
262        } else {
263            let handle = Arc::new(Handle::new_gauge(level));
264            maps.gauges.insert(key.clone(), Arc::clone(&handle));
265            handle
266        };
267
268        Gauge::from_arc(handle)
269    }
270
271    /// Returns a handle to the histogram for `key`, creating it at `level` if it doesn't yet exist.
272    fn get_or_create_histogram(&self, key: &Key, level: Level) -> Histogram {
273        let mut maps = self.maps.lock().unwrap();
274        let handle = if let Some(handle) = maps.histograms.get(key) {
275            handle.reset_idle();
276            Arc::clone(handle)
277        } else {
278            let handle = Arc::new(Handle::new_histogram(level));
279            maps.histograms.insert(key.clone(), Arc::clone(&handle));
280            handle
281        };
282
283        Histogram::from_arc(handle)
284    }
285}

321impl MetricsRecorder {
322    fn new(metrics_prefix: String, default_level: Level) -> Self {
323        let (flush_tx, _) = broadcast::channel(2);
324        Self {
325            state: Arc::new(State {
326                registry: MetricsRegistry::default(),
327                flush_tx,
328                metrics_prefix,
329                default_level,
330                current_level: Mutex::new(default_level),
331                idle_evict_intervals: IDLE_EVICT_INTERVALS,
332                flush_interval: FLUSH_INTERVAL,
333            }),
334        }
335    }
336
337    fn filter_handle(&self) -> FilterHandle {
338        FilterHandle {
339            state: Arc::clone(&self.state),
340        }
341    }
342
343    fn install(self) -> Result<(), SetRecorderError<Self>> {
344        let state = Arc::clone(&self.state);
345        metrics::set_global_recorder(self)?;
346
347        if RECEIVER_STATE.set(state).is_err() {
348            panic!("metrics receiver should never be set prior to global recorder being installed");
349        }
350
351        Ok(())
352    }
353
354    fn prefix_key(&self, key: &Key) -> Key {
355        Key::from_parts(format!("{}.{}", self.state.metrics_prefix, key.name()), key.labels())
356    }

386/// Internal metrics stream
387///
388/// Used to receive periodic snapshots of the internal metrics registry, which contains all metrics that are currently
389/// active within the process.
390pub struct MetricsStream {
391    inner: ReusableBoxFuture<
392        'static,
393        (
394            Result<SharedMetricsSnapshot, RecvError>,
395            Receiver<SharedMetricsSnapshot>,
396        ),
397    >,
398}
399
400impl MetricsStream {
401    /// Creates a new `MetricsStream` that receives updates from the internal metrics registry.
402    pub fn register() -> Self {
403        let state = RECEIVER_STATE.get().expect("metrics receiver should be set");
404        Self {
405            inner: ReusableBoxFuture::new(make_rx_future(state.flush_tx.subscribe())),
406        }
407    }
408}
409
410impl Stream for MetricsStream {
411    type Item = SharedMetricsSnapshot;
412
413    fn poll_next(mut self: Pin<&mut Self>, cx: &mut task::Context<'_>) -> Poll<Option<Self::Item>> {
414        loop {
415            // Poll the receiver, and rearm the future once we actually resolve it.
416            let (result, rx) = ready!(self.inner.poll(cx));
417            self.inner.set(make_rx_future(rx));
418
419            match result {
420                Ok(item) => return Poll::Ready(Some(item)),
421                Err(RecvError::Closed) => return Poll::Ready(None),
422                Err(RecvError::Lagged(n)) => {
423                    debug!(
424                        missed_payloads = n,
425                        "Stream lagging behind internal metrics producer. Internal metrics may have been lost."
426                    );
427                    continue;
428                }
429            }
430        }
431    }
432}

434async fn make_rx_future(
435    mut rx: Receiver<SharedMetricsSnapshot>,
436) -> (
437    Result<SharedMetricsSnapshot, RecvError>,
438    Receiver<SharedMetricsSnapshot>,
439) {
440    let result = rx.recv().await;
441    (result, rx)
442}

461async fn flush_metrics() {
462    let mut context_resolver = MetricsContextResolver::new(INTERNAL_METRICS_INTERNER_SIZE);
463
464    let state = RECEIVER_STATE.get().expect("metrics receiver should be set");
465
466    let mut flush_interval = tokio::time::interval(state.flush_interval);
467    flush_interval.tick().await;
468
469    let mut flush_state = FlushState::default();
470
471    loop {
472        flush_interval.tick().await;
473        flush_once(state, &mut context_resolver, &mut flush_state);
474    }
475}

477/// Performs a single flush pass over the registry: consumes each metric's value, evicts metrics that
478/// are no longer referenced and have gone idle, and broadcasts the resulting metric values and
479/// evictions to downstream consumers.
480fn flush_once(state: &State, context_resolver: &mut MetricsContextResolver, flush_state: &mut FlushState) {
481    let has_listeners = state.flush_tx.receiver_count() > 0;
482    let current_level = *state.current_level.lock().unwrap();
483    let idle_threshold = state.idle_evict_intervals;
484    let mut snapshot = MetricsSnapshot::default();
485
486    flush_state.clear();
487    let FlushState {
488        counter_emits,
489        gauge_emits,
490        histogram_emits,
491        evicted_keys,
492    } = flush_state;
493
494    {
495        let mut maps = state.registry.maps.lock().unwrap();
496
497        // Counters: a counter is never idle so long as it has a non-zero delta during a flush _or_ has
498        // an active reference (strong count > 1).
499        //
500        // The strong-count check precedes `consume()`, with an Acquire fence on the orphaned path:
501        // observing `strong_count == 1` synchronizes-with the dropping thread's release, which
502        // guarantees the following `consume()` observes that thread's final increment.
503        maps.counters.retain(|key, handle| {
504            let idle = handle.bump_idle();
505            let orphaned = Arc::strong_count(handle) == 1;
506            if orphaned {
507                std::sync::atomic::fence(Ordering::Acquire);
508            }
509            let delta = handle.consume();
510
511            if orphaned && delta == 0 && idle >= idle_threshold {
512                evicted_keys.push(key.clone());
513                return false;
514            }
515
516            if has_listeners && handle.level() >= current_level {
517                counter_emits.push((key.clone(), delta as f64));
518            }
519            true
520        });
521
522        // Gauges: a gauge is never idle so long as it was touched ("dirty") prior to a flush _or_ has
523        // an active reference (strong count > 1).
524        maps.gauges.retain(|key, handle| {
525            let idle = handle.bump_idle();
526            let orphaned = Arc::strong_count(handle) == 1;
527            if orphaned {
528                std::sync::atomic::fence(Ordering::Acquire);
529            }
530            let value = handle.load();
531            let written = handle.take_dirty();
532
533            if orphaned && !written && idle >= idle_threshold {
534                evicted_keys.push(key.clone());
535                return false;
536            }
537
538            if has_listeners && handle.level() >= current_level {
539                gauge_emits.push((key.clone(), value));
540            }
541            true
542        });
543
544        // Histograms: a histogram is never idle so long as it has recorded samples prior to a flush _or_
545        // has an active reference (strong count > 1).
546        //
547        // The strong-count check precedes the drain, with an Acquire fence on the orphaned path, so the
548        // drain observes a dropping thread's final `record()`.
549        maps.histograms.retain(|key, handle| {
550            let idle = handle.bump_idle();
551            let orphaned = Arc::strong_count(handle) == 1;
552            if orphaned {
553                std::sync::atomic::fence(Ordering::Acquire);
554            }
555            let mut histogram_samples = Vec::new();
556            handle.drain_into(&mut histogram_samples);
557            let had_samples = !histogram_samples.is_empty();
558
559            if orphaned && !had_samples && idle >= idle_threshold {
560                evicted_keys.push(key.clone());
561                return false;
562            }
563
564            if has_listeners && had_samples && handle.level() >= current_level {
565                histogram_emits.push((key.clone(), histogram_samples));
566            }
567            true
568        });
569    }
570
571    // For every evicted key we collected, take its cached context (if it exists) and add it to the
572    // snapshot's evictions list.
573    for key in evicted_keys {
574        if let Some(context) = context_resolver.take(key) {
575            if has_listeners {
576                snapshot.evictions.push(context);
577            }
578        }
579    }
580
581    if !has_listeners {
582        return;
583    }
584
585    for (key, value) in counter_emits.drain(..) {
586        let context = context_resolver.resolve_from_key(key);
587        snapshot.upserts.push(Event::Metric(Metric::counter(context, value)));
588    }
589    for (key, value) in gauge_emits.drain(..) {
590        let context = context_resolver.resolve_from_key(key);
591        snapshot.upserts.push(Event::Metric(Metric::gauge(context, value)));
592    }
593    for (key, samples) in histogram_emits.drain(..) {
594        let context = context_resolver.resolve_from_key(key);
595        snapshot
596            .upserts
597            .push(Event::Metric(Metric::histogram(context, &samples[..])));
598    }
599
600    if !snapshot.upserts.is_empty() || !snapshot.evictions.is_empty() {
601        let _ = state.flush_tx.send(Arc::new(snapshot));
602    }
603}

605struct MetricsContextResolver {
606    context_resolver: ContextResolver,
607    key_context_cache: FastHashMap<Key, Context>,
608}
609
610impl MetricsContextResolver {
611    fn new(resolver_interner_size_bytes: NonZeroUsize) -> Self {
612        Self {
613            // Set up our context resolver without caching, since we will be caching the contexts ourselves.
614            context_resolver: ContextResolverBuilder::from_name("core/internal_metrics")
615                .expect("resolver name is not empty")
616                .with_interner_capacity_bytes(resolver_interner_size_bytes)
617                .without_caching()
618                .build(),
619            key_context_cache: FastHashMap::default(),
620        }
621    }
622
623    fn resolve_from_key(&mut self, key: Key) -> Context {
624        static SELF_ORIGIN_INFO: LazyLock<RawOrigin<'static>> = LazyLock::new(|| {
625            let mut origin_info = RawOrigin::default();
626            origin_info.set_process_id(std::process::id());
627            origin_info
628        });
629
630        // Check the cache first.
631        if let Some(context) = self.key_context_cache.get(&key) {
632            return context.clone();
633        }
634
635        // We don't have the context cached, so we need to resolve it.
636        let tags = key
637            .labels()
638            .map(|l| Tag::from(format!("{}:{}", l.key(), l.value())))
639            .collect::<TagSet>();
640
641        let context = self
642            .context_resolver
643            .resolve(key.name(), &tags, Some(SELF_ORIGIN_INFO.clone()))
644            .expect("resolver should always allow falling back");
645
646        self.key_context_cache.insert(key, context.clone());
647        context
648    }
649
650    /// Removes a key's cached context, returning it if present.
651    ///
652    /// Returns `Some` only if the key had been resolved before (that is, the metric was emitted at
653    /// least once), which is what tells the flush loop whether a downstream eviction needs to be sent.
654    fn take(&mut self, key: &Key) -> Option<Context> {
655        self.key_context_cache.remove(key)
656    }
657}

659/// Initializes the metrics subsystem with the given metrics prefix and default filter level.
660///
661/// `default_level` sets the initial filter level for emitted metrics, and is also what the filter is restored to when
662/// [`FilterHandle::reset_filter`] is invoked. Metrics whose level is more verbose than this default are filtered out
663/// until a runtime override is applied via [`FilterHandle::override_filter`].
664///
665/// Returns a [`FilterHandle`] for adjusting the runtime metrics filter, plus a [`MetricsFlusherWorker`]
666/// that must be added to a [`Supervisor`][crate::runtime::Supervisor] in order to drive the periodic
667/// flush loop. Internal metrics aren't propagated to subscribers until the worker is running.
668///
669/// # Errors
670///
671/// If a global recorder was already installed, an error will be returned.
672pub async fn initialize_metrics(
673    metrics_prefix: String, default_level: Level,
674) -> Result<(FilterHandle, MetricsFlusherWorker), GenericError> {
675    let recorder = MetricsRecorder::new(metrics_prefix, default_level);
676    let filter_handle = recorder.filter_handle();
677    recorder.install()?;
678
679    Ok((filter_handle, MetricsFlusherWorker))
680}

682/// A worker that periodically flushes the internal metrics registry to broadcast subscribers.
683///
684/// Wraps the internal flush loop and runs it under a [`Supervisor`][crate::runtime::Supervisor]. Must
685/// only be added to a supervisor after [`initialize_metrics`] has been called -- the flush loop
686/// reads from the global recorder state set by that call.
687pub struct MetricsFlusherWorker;
688
689#[async_trait]
690impl Supervisable for MetricsFlusherWorker {
691    fn name(&self) -> &str {
692        "internal-telemetry-metrics-flusher"
693    }
694
695    async fn initialize(&self, process_shutdown: ShutdownHandle) -> Result<SupervisorFuture, InitializationError> {
696        Ok(Box::pin(async move {
697            select! {
698                _ = process_shutdown => {},

933// Loom model of the counter eviction path in `flush_once`.
934//
935// In `flush_once`, the registry `maps` lock is held during the drain, but the increment/drop path
936// does not take that lock: `Counter::increment` writes straight through the `Arc<Handle>`, and
937// dropping a caller's `Counter` only decrements the Arc strong count. A component holding a cached
938// handle can therefore land a final increment and drop the handle while a flush is mid-pass over it.
939//
940// The model transcribes the counter branch with loom primitives instead of exercising the real types,
941// which loom cannot instrument here:
942//
943// - The strong count is an explicit `AtomicUsize`. loom's `Arc::strong_count` does not reflect a
944//   concurrent decrement from another thread (it models Arc's drop synchronization, not the observable
945//   count value), so a flush reading it never sees the orphaned (`== 1`) state mid-race. The explicit
946//   atomic matches `std`'s `Arc`: clone increments (Relaxed), drop decrements with `Release`,
947//   observation is a `Relaxed` load -- the same shape as how `flush_once` reads `Arc::strong_count`.
948// - The real `Handle` is wrapped by the `metrics` crate's `Counter`/`CounterFn`, which construct
949//   `std::sync::Arc` internally. loom only instruments atomics and `Arc`s swapped to `loom::sync::*`
950//   behind the `loom` cfg, not those inside a third-party crate.
951//
952// `flush_decision_consume_first` and `flush_decision_strong_count_first` mirror the two possible
953// orderings of `flush_once`'s counter branch and must stay in lockstep with it.
954#[cfg(all(test, feature = "loom"))]
955mod loom_tests {
956    use loom::sync::atomic::{fence, AtomicU64, AtomicUsize, Ordering};
957    use loom::sync::Arc;
958
959    /// The shared state behind a counter handle: the accumulator the flush loop drains, and the Arc
960    /// strong count it consults to decide whether the handle is orphaned.
961    struct Shared {
962        value: AtomicU64,
963        strong: AtomicUsize,
964    }
965
966    /// A component holding a cached handle lands one final increment, then drops the handle.
967    fn caller_increment_then_drop(shared: &Arc<Shared>, amount: u64) {
968        shared.value.fetch_add(amount, Ordering::Relaxed);
969        shared.strong.fetch_sub(1, Ordering::Release); // Arc clone drop
970    }
971
972    /// Consumes the value, then checks the strong count (value read before the liveness check).
973    fn flush_decision_consume_first(shared: &Arc<Shared>) -> (u64, bool) {
974        let delta = shared.value.swap(0, Ordering::Relaxed);
975        let orphaned = shared.strong.load(Ordering::Relaxed) == 1;
976        (delta, orphaned && delta == 0)
977    }
978
979    /// Checks the strong count first; if orphaned, Acquire-fences to synchronize with the dropping
980    /// thread's `Release`, then consumes the value.
981    fn flush_decision_strong_count_first(shared: &Arc<Shared>) -> (u64, bool) {
982        let orphaned = shared.strong.load(Ordering::Relaxed) == 1;
983        if orphaned {
984            fence(Ordering::Acquire);
985        }
986        let delta = shared.value.swap(0, Ordering::Relaxed);
987        (delta, orphaned && delta == 0)
988    }
989
990    fn model(decision: fn(&Arc<Shared>) -> (u64, bool)) {
991        loom::model(move || {
992            const FINAL: u64 = 7;
993
994            // strong count starts at 2: one ref in the registry map, one held by the caller.
995            let shared = Arc::new(Shared {
996                value: AtomicU64::new(0),
997                strong: AtomicUsize::new(2),
998            });
999            let caller_shared = Arc::clone(&shared);
1000
1001            let caller = loom::thread::spawn(move || {
1002                caller_increment_then_drop(&caller_shared, FINAL);
1003            });
1004
1005            // The flush task evaluates this counter while the component races.
1006            let (delta, evicted) = decision(&shared);
1007            caller.join().unwrap();
1008
1009            // Invariant: a counter may only be evicted once everything it accumulated has been
1010            // emitted. If the flush evicts it, the delta it emitted downstream must already include
1011            // the final increment -- otherwise those counts are silently discarded with the handle.
1012            if evicted {
1013                assert_eq!(
1014                    delta,
1015                    FINAL,
1016                    "counter evicted while {} counts were never emitted -> permanently lost",
1017                    FINAL - delta
1018                );
1019            }
1020        });
1021    }
1022
1023    // Consuming the value before checking the strong count is lossy: loom finds an interleaving where
1024    // the handle is evicted while a final increment goes unemitted. `#[should_panic]` asserts loom
1025    // reaches that interleaving.
1026    #[test]
1027    #[should_panic(expected = "permanently lost")]
1028    fn consume_first_ordering_loses_a_final_increment() {
1029        model(flush_decision_consume_first);
1030    }
1031
1032    // Checking the strong count first (Acquire-fencing when orphaned) before consuming preserves every
1033    // increment across all interleavings loom explores.
1034    #[test]
1035    fn strong_count_first_ordering_preserves_a_final_increment() {
1036        model(flush_decision_strong_count_first);
1037    }
