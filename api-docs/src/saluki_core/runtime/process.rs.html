16static GLOBAL_PROCESS_ID_COUNTER: AtomicUsize = AtomicUsize::new(0);

18/// Process identifier.
19///
20/// A simple, numeric identifier that uniquely identifies a process.
21#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
22pub struct Id(usize);
23
24impl Id {
25    /// Creates a new process identifier.
26    pub fn new() -> Self {
27        let id = GLOBAL_PROCESS_ID_COUNTER.fetch_add(1, Relaxed);
28        Self(id)
29    }
30
31    fn as_usize(&self) -> usize {
32        self.0
33    }
34}

36/// Process name.
37///
38/// A human-readable name for a process that only contains alphanumeric characters, underscores, and periods.
39///
40/// Process names are scoped, such that the resulting process name is nested. For example, if a supervisor has a name of
41/// `topology_sup`, and a child process is added to that supervisor with a name of `worker`, the resulting process name
42/// for the child process will be `topology_sup.worker`. Process names can be arbitrarily nested in this way.
43///
44/// Process names will be sanitized if they contain invalid characters, such as hyphens or spaces. Invalid characters
45/// will be replaced with underscores.
46///
47/// Not guaranteed to be unique.
48#[derive(Clone, Debug, PartialEq, Eq, Hash)]
49pub struct Name(Arc<str>);
50
51impl Name {
52    pub(crate) fn root<N: AsRef<str>>(name: N) -> Option<Self> {
53        let name = name.as_ref();
54        if name.is_empty() {
55            return None;
56        }
57
58        Some(Self(get_sanitized_name(name)))
59    }
60
61    pub(crate) fn scoped<N: AsRef<str>>(parent: &Name, name: N) -> Option<Self> {
62        let name = name.as_ref();
63        if name.is_empty() {
64            return None;
65        }
66
67        let sanitized_name = get_sanitized_name(name);
68        let scoped_name = format!("{}.{}", parent.0, sanitized_name);
69        Some(Self(scoped_name.into()))
70    }
71}
72
73impl Deref for Name {
74    type Target = str;
75
76    fn deref(&self) -> &Self::Target {
77        &self.0
78    }
79}

81/// A runtime process.
82#[derive(Clone)]
83pub struct Process {
84    id: Id,
85    name: Name,
86    alloc_group_token: AllocationGroupToken,
87}
88
89impl Process {
90    pub(crate) fn supervisor<N: AsRef<str>>(name: N, parent: Option<&Process>) -> Option<Self> {
91        let name = parent
92            .and_then(|p| Name::scoped(&p.name, &name))
93            .or_else(|| Name::root(name))?;
94        let alloc_group_token = AllocationGroupRegistry::global().register_allocation_group(&*name);
95        Some(Self::from_parts(Id::new(), name, alloc_group_token))
96    }
97
98    pub(crate) fn worker<N: AsRef<str>>(name: N, parent: &Process) -> Option<Self> {
99        let name = Name::scoped(&parent.name, name)?;
100        Some(Self::from_parts(Id::new(), name, parent.alloc_group_token))
101    }
102
103    fn from_parts(id: Id, name: Name, alloc_group_token: AllocationGroupToken) -> Self {
104        Self {
105            id,
106            name,
107            alloc_group_token,
108        }
109    }
110
111    /// Returns the process identifier.
112    pub fn id(&self) -> &Id {
113        &self.id
114    }
115
116    pub fn into_instrumented<F>(self, inner: F) -> InstrumentedProcess<F>
117    where
118        F: Future,
119    {
120        InstrumentedProcess::new(self, inner)
121    }
122}

124/// An instrumented process.
125#[pin_project]
126pub struct InstrumentedProcess<F> {
127    #[pin]
128    inner: Instrumented<Tracked<F>>,
129}
130
131impl<F> InstrumentedProcess<F>
132where
133    F: Future,
134{
135    pub(crate) fn new(process: Process, inner: F) -> Self {
136        let span = debug_span!(
137            "process",
138            process_id = process.id().as_usize(),
139            process_name = &*process.name,
140        );
141
142        let inner = inner.track_allocations(process.alloc_group_token).instrument(span);
143
144        Self { inner }
145    }
146}
147
148impl<F> Future for InstrumentedProcess<F>
149where
150    F: Future,
151{
152    type Output = F::Output;
153
154    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
155        self.project().inner.poll(cx)
156    }
157}

179fn is_process_name_segment_valid(name: &str) -> bool {
180    // Process name cannot be empty.
181    if name.is_empty() {
182        return false;
183    }
184
185    // Process names cannot start or end with anything other than alphanumeric characters.
186    if !name.starts_with(|c: char| c.is_alphanumeric()) || !name.ends_with(|c: char| c.is_alphanumeric()) {
187        return false;
188    }
189
190    // Process name segments can only include alphanumeric characters and underscores.
191    //
192    // Periods are allowed in process names overall, but they're only used as separators between segments.
193    for c in name.chars() {
194        if !c.is_alphanumeric() && c != '_' {
195            return false;
196        }
197    }
198
199    true
200}
201
202fn get_sanitized_name(name: &str) -> Arc<str> {
203    if is_process_name_segment_valid(name) {
204        name.into()
205    } else {
206        // Replace invalid characters with underscores, and collapses multiple underscores into a single one.
207        let raw_sanitized = name
208            .chars()
209            .map(|c| if c.is_alphanumeric() || c == '_' { c } else { '_' });
210        let mut sanitized = String::with_capacity(name.len());
211
212        let mut last_was_underscore = true;
213        for c in raw_sanitized {
214            if c == '_' {
215                if !last_was_underscore {
216                    sanitized.push(c);
217                    last_was_underscore = true;
218                }
219            } else {
220                sanitized.push(c);
221                last_was_underscore = false;
222            }
223        }
224
225        // Remove all non-alphanumeric characters from beginning and end.
226        let trimmed = sanitized.trim_matches(|c: char| !c.is_alphanumeric());
227        Arc::from(trimmed)
228    }
229}

235    #[test]
236    fn test_process_name_root() {
237        let cases = [
238            ("topology_sup", Some("topology_sup")),
239            ("worker", Some("worker")),
240            ("worker.", Some("worker")),
241            ("_worker_", Some("worker")),
242            ("worker-123", Some("worker_123")),
243            ("--worker_123", Some("worker_123")),
244            ("worker 123", Some("worker_123")),
245            ("worker===123", Some("worker_123")),
246            ("topology.worker", Some("topology_worker")),
247            ("", None),
248        ];
249
250        for (input, expected) in cases {
251            let name = Name::root(input);
252            assert_eq!(name.as_deref(), expected);
253        }
254    }
255
256    #[test]
257    fn test_process_name_scoped() {
258        let parent = Name::root("topology_sup").unwrap();
259        let cases = [
260            ("worker", Some("topology_sup.worker")),
261            ("worker.", Some("topology_sup.worker")),
262            ("_worker_", Some("topology_sup.worker")),
263            ("worker-123", Some("topology_sup.worker_123")),
264            ("--worker_123", Some("topology_sup.worker_123")),
265            ("worker 123", Some("topology_sup.worker_123")),
266            ("worker===123", Some("topology_sup.worker_123")),
267            ("nested.worker", Some("topology_sup.nested_worker")),
268            ("", None),
269        ];
270
271        for (input, expected) in cases {
