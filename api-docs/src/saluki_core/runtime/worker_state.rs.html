1//! Worker-tracking state for the supervisor.
2//!
3//! `WorkerState` owns the set of running child tasks for a [`Supervisor`](super::Supervisor) and provides the common
4//! operations it needs: spawning a child, awaiting the next child to finish, and shutting all children down (either in
5//! order or concurrently). It is deliberately agnostic about restart policy -- the supervisor decides what to do when a
6//! worker exits.
7
8use std::future::pending;
9use std::time::Duration;
10
11use saluki_common::collections::FastIndexMap;
12use saluki_common::sync::shutdown::{ShutdownCoordinator, ShutdownHandle};
13use tokio::{
14    pin, select,
15    task::{AbortHandle, Id, JoinSet},
16};
17use tracing::debug;
18
19use super::process::{Process, ProcessExt as _};
20use super::supervisor::{ProcessError, ShutdownMode, ShutdownStrategy, SupervisedChild, SupervisorError, WorkerError};

22/// Per-worker bookkeeping held by a [`WorkerState`].
23struct ProcessState {
24    /// Caller-assigned identifier for the worker.
25    ///
26    /// Opaque to `WorkerState`: the supervisor assigns each child a stable id from a monotonic counter. It is returned
27    /// from [`WorkerState::wait_for_next_worker`] so the caller can correlate the exit with its own bookkeeping.
28    worker_id: u64,
29    shutdown_strategy: ShutdownStrategy,
30    shutdown_coordinator: ShutdownCoordinator,
31    abort_handle: AbortHandle,
32}
33
34/// Tracks the set of running child tasks for a supervisor.
35pub(super) struct WorkerState {
36    process: Process,
37    shutdown_mode: ShutdownMode,
38    worker_tasks: JoinSet<Result<(), WorkerError>>,
39    worker_map: FastIndexMap<Id, ProcessState>,
40}
41
42impl WorkerState {
43    pub(super) fn new(process: Process, shutdown_mode: ShutdownMode) -> Self {
44        Self {
45            process,
46            shutdown_mode,
47            worker_tasks: JoinSet::new(),
48            worker_map: FastIndexMap::default(),
49        }
50    }
51
52    /// Spawns the child described by `child_spec`, tracking it under the given `worker_id`.
53    pub(super) fn add_worker(&mut self, worker_id: u64, child_spec: &SupervisedChild) -> Result<(), SupervisorError> {
54        let (shutdown_coordinator, shutdown_handle) = ShutdownHandle::paired();
55        let process = child_spec.create_process(&self.process)?;
56        let worker_future = child_spec.create_worker_future(process.clone(), shutdown_handle)?;
57        let shutdown_strategy = child_spec.shutdown_strategy();
58        let abort_handle = self.worker_tasks.spawn(worker_future.into_process_future(process));
59        self.worker_map.insert(
60            abort_handle.id(),
61            ProcessState {
62                worker_id,
63                shutdown_strategy,
64                shutdown_coordinator,
65                abort_handle,
66            },
67        );
68        Ok(())
69    }
70
71    /// Awaits the next worker to finish, returning its `worker_id` and result.
72    pub(super) async fn wait_for_next_worker(&mut self) -> (u64, Result<(), WorkerError>) {
73        debug!("Waiting for next process to complete.");
74
75        // If there are no workers to wait on, park indefinitely so the supervisor's select loop only proceeds via its
76        // other arms (shutdown, or a newly-added dynamic child). Without this guard, `join_next_with_id` would return
77        // `None` immediately on an empty set and the supervisor would busy-loop. The set legitimately empties when all
78        // children are non-restartable (e.g. `RestartType::Temporary`) and have exited.
79        if self.worker_tasks.is_empty() {
80            pending::<()>().await;
81        }
82
83        match self.worker_tasks.join_next_with_id().await {
84            Some(Ok((worker_task_id, worker_result))) => {
85                let process_state = self
86                    .worker_map
87                    .shift_remove(&worker_task_id)
88                    .expect("worker task ID not found");
89                (process_state.worker_id, worker_result)
90            }
91            Some(Err(e)) => {
92                let worker_task_id = e.id();
93                let process_state = self
94                    .worker_map
95                    .shift_remove(&worker_task_id)
96                    .expect("worker task ID not found");
97                let e = if e.is_cancelled() {
98                    ProcessError::Aborted
99                } else {
100                    ProcessError::Panicked
101                };
102                (process_state.worker_id, Err(WorkerError::Runtime(e.into())))
103            }
104            None => unreachable!(
105                "join set is non-empty here: we park above while empty, and only this method removes workers"
106            ),
107        }
108    }

110    /// Shuts down all workers, honoring each worker's shutdown strategy and the configured [`ShutdownMode`].
111    pub(super) async fn shutdown_workers(&mut self) {
112        debug!(shutdown_mode = ?self.shutdown_mode, "Shutting down all processes.");
113
114        match self.shutdown_mode {
115            ShutdownMode::Ordered => self.shutdown_workers_ordered().await,
116            ShutdownMode::Concurrent => self.shutdown_workers_concurrent().await,
117        }
118
119        debug_assert!(self.worker_map.is_empty(), "worker map should be empty after shutdown");
120        debug_assert!(
121            self.worker_tasks.is_empty(),
122            "worker tasks should be empty after shutdown"
123        );
124    }
125
126    /// Shuts down all workers one at a time, in reverse order of insertion, honoring each worker's shutdown strategy.
127    async fn shutdown_workers_ordered(&mut self) {
128        // Pop entries from the worker map, which grabs us workers in the reverse order they were added. This lets us
129        // ensure we're shutting down any _dependent_ processes (processes which depend on previously-started processes)
130        // first.
131        //
132        // For each entry, we trigger shutdown in whatever way necessary, and then wait for the process to exit by
133        // driving the `JoinSet`. If other workers complete while we're waiting, we'll simply remove them from the
134        // worker map and continue waiting for the current worker we're shutting down.
135        //
136        // We do this until the worker map is empty, at which point we can be sure that all processes have exited.
137        while let Some((current_worker_task_id, process_state)) = self.worker_map.pop() {
138            let ProcessState {
139                worker_id,
140                shutdown_strategy,
141                shutdown_coordinator,
142                abort_handle,
143            } = process_state;
144
145            // Trigger the process to shutdown based on the configured shutdown strategy.
146            let shutdown_deadline = match shutdown_strategy {
147                ShutdownStrategy::Graceful(timeout) => {
148                    debug!(worker_id, shutdown_timeout = ?timeout, "Gracefully shutting down process.");
149                    shutdown_coordinator.shutdown();
150
151                    tokio::time::sleep(timeout)
152                }
153                ShutdownStrategy::Brutal => {
154                    debug!(worker_id, "Forcefully aborting process.");
155                    abort_handle.abort();
156
157                    // We have to return a future that never resolves, since we're already aborting it. This is a little
158                    // hacky but it's also difficult to do an optional future, so this is what we're going with for now.
159                    tokio::time::sleep(Duration::MAX)
160                }
161            };
162            pin!(shutdown_deadline);
163
164            // Wait for the process to exit by driving the `JoinSet`. If other workers complete while we're waiting,
165            // we'll simply remove them from the worker map and continue waiting.
166            let mut aborted = false;
167            loop {
168                select! {
169                    worker_result = self.worker_tasks.join_next_with_id() => {
170                        match worker_result {
171                            Some(Ok((worker_task_id, _))) => {
172                                if worker_task_id == current_worker_task_id {
173                                    debug!(?worker_task_id, "Target process exited successfully.");
174                                    break;
175                                } else {
176                                    debug!(?worker_task_id, "Non-target process exited successfully. Continuing to wait.");
177                                    let removed = self.worker_map.shift_remove(&worker_task_id);
178                                    debug_assert!(removed.is_some(), "non-target worker must be in the worker map");
179                                }
180                            },
181                            Some(Err(e)) => {
182                                let worker_task_id = e.id();
183                                if worker_task_id == current_worker_task_id {
184                                    debug!(?worker_task_id, "Target process exited with error.");
185                                    break;
186                                } else {
187                                    debug!(?worker_task_id, "Non-target process exited with error. Continuing to wait.");
188                                    let removed = self.worker_map.shift_remove(&worker_task_id);
189                                    debug_assert!(removed.is_some(), "non-target worker must be in the worker map");
190                                }
191                            }
192                            None => unreachable!("worker task must exist in join set if we are waiting for it"),
193                        }
194                    },
195                    // We've exceeded the shutdown timeout, so we abort the process. The `if !aborted` guard stops this
196                    // arm from re-firing on every poll once the deadline has elapsed (an elapsed `Sleep` stays ready),
197                    // which would otherwise spin re-aborting until the task is reaped.
198                    _ = &mut shutdown_deadline, if !aborted => {
199                        debug!(worker_id, "Shutdown timeout expired, forcefully aborting process.");
200                        abort_handle.abort();
201                        aborted = true;
202                    }
203                }
204            }
205        }
206    }

208    /// Shuts down all workers at once, waiting for them concurrently.
209    ///
210    /// Each worker is signalled up front, then awaited concurrently under its **own** graceful deadline, so a worker
211    /// that ignores shutdown is aborted at its configured timeout regardless of its siblings. Total shutdown time is
212    /// therefore bounded by the slowest individual worker rather than the sum of all timeouts, which suits large,
213    /// independent worker sets (for example, one task per network connection).
214    ///
215    /// A worker whose graceful timeout is effectively unbounded (such as a nested supervisor, which uses
216    /// `Duration::MAX` because it bounds itself via its own children's deadlines) is waited on indefinitely and is
217    /// never aborted by this method.
218    async fn shutdown_workers_concurrent(&mut self) {
219        // Take ownership of all worker bookkeeping so we can consume each worker's shutdown coordinator. Signal every
220        // graceful worker and immediately abort brutal ones, recording a per-worker abort deadline so each is held to
221        // its own timeout rather than a single shared one.
222        let now = tokio::time::Instant::now();
223        let mut pending: FastIndexMap<Id, (AbortHandle, Option<tokio::time::Instant>)> = FastIndexMap::default();
224        for (task_id, process_state) in std::mem::take(&mut self.worker_map) {
225            let ProcessState {
226                worker_id,
227                shutdown_strategy,
228                shutdown_coordinator,
229                abort_handle,
230            } = process_state;
231
232            match shutdown_strategy {
233                ShutdownStrategy::Graceful(timeout) => {
234                    debug!(worker_id, shutdown_timeout = ?timeout, "Gracefully shutting down process.");
235                    shutdown_coordinator.shutdown();
236                    // An effectively-infinite timeout (`Duration::MAX`) maps to `None` -- "never abort" -- which is
237                    // correct for nested supervisors, since they bound themselves via their own children's deadlines.
238                    let deadline = (timeout != Duration::MAX).then(|| now + timeout);
239                    pending.insert(task_id, (abort_handle, deadline));
240                }
241                ShutdownStrategy::Brutal => {
242                    debug!(worker_id, "Forcefully aborting process.");
243                    abort_handle.abort();
244                }
245            }
246        }
247
248        // Wait for every task to exit. Each iteration sleeps until the earliest still-pending abort deadline; when it
249        // fires we abort exactly those workers whose own deadline has passed (their tasks are then reaped by a later
250        // `join_next`). Brutal workers were aborted above and aren't tracked here.
251        while !self.worker_tasks.is_empty() {
252            match pending.values().filter_map(|(_, deadline)| *deadline).min() {
253                Some(deadline) => {
254                    select! {
255                        joined = self.worker_tasks.join_next_with_id() => {
256                            let task_id = match joined {
257                                Some(Ok((task_id, _))) => Some(task_id),
258                                Some(Err(e)) => Some(e.id()),
259                                None => None,
260                            };
261                            if let Some(task_id) = task_id {
262                                pending.swap_remove(&task_id);
263                            }
264                        }
265                        _ = tokio::time::sleep_until(deadline) => {
266                            let now = tokio::time::Instant::now();
267                            pending.retain(|_, (abort_handle, deadline)| {
268                                if deadline.is_some_and(|deadline| deadline <= now) {
269                                    debug!("Shutdown timeout expired, forcefully aborting process.");
270                                    abort_handle.abort();
271                                    false
272                                } else {
273                                    true
274                                }
275                            });
276                        }
277                    }
278                }
279                // Only workers with no finite deadline remain (e.g. nested supervisors); wait for them to exit on their
280                // own.
281                None => {
282                    let _ = self.worker_tasks.join_next_with_id().await;
283                }
284            }
285        }
286    }
