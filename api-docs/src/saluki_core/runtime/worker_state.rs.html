1//! Worker-tracking state for the supervisor.
2//!
3//! `WorkerState` owns the set of running child tasks for a [`Supervisor`](super::Supervisor) and provides the common
4//! operations it needs: spawning a child, awaiting the next child to finish, and shutting all children down (either in
5//! order or concurrently). It is deliberately agnostic about restart policy -- the supervisor decides what to do when a
6//! worker exits.
7
8use std::future::pending;
9use std::time::Duration;
10
11use saluki_common::collections::FastIndexMap;
12use saluki_common::sync::shutdown::{ShutdownCoordinator, ShutdownHandle};
13use tokio::{
14    pin, select,
15    task::{AbortHandle, Id, JoinSet},
16};
17use tracing::{debug, warn};

19use super::process::{Process, ProcessExt as _};
20use super::supervisor::{ProcessError, ShutdownMode, ShutdownStrategy, SupervisedChild, SupervisorError, WorkerError};

22/// Per-worker bookkeeping held by a [`WorkerState`].
23struct ProcessState {
24    /// Caller-assigned identifier for the worker.
25    ///
26    /// Opaque to `WorkerState`: the supervisor assigns each child a stable id from a monotonic counter. It is returned
27    /// from [`WorkerState::wait_for_next_worker`] so the caller can correlate the exit with its own bookkeeping.
28    worker_id: u64,
29    /// Fully qualified process name, retained so shutdown can name precisely which worker had to be forcefully
30    /// aborted.
31    worker_name: String,
32    shutdown_strategy: ShutdownStrategy,
33    shutdown_coordinator: ShutdownCoordinator,
34    abort_handle: AbortHandle,
35}
36
37/// Tracks the set of running child tasks for a supervisor.
38pub(super) struct WorkerState {
39    process: Process,
40    shutdown_mode: ShutdownMode,
41    worker_tasks: JoinSet<Result<(), WorkerError>>,
42    worker_map: FastIndexMap<Id, ProcessState>,
43}
44
45impl WorkerState {
46    pub(super) fn new(process: Process, shutdown_mode: ShutdownMode) -> Self {
47        Self {
48            process,
49            shutdown_mode,
50            worker_tasks: JoinSet::new(),
51            worker_map: FastIndexMap::default(),
52        }
53    }
54
55    /// Spawns the child described by `child_spec`, tracking it under the given `worker_id`.
56    pub(super) fn add_worker(&mut self, worker_id: u64, child_spec: &SupervisedChild) -> Result<(), SupervisorError> {
57        let (shutdown_coordinator, shutdown_handle) = ShutdownHandle::paired();
58        let process = child_spec.create_process(&self.process)?;
59        let worker_name = process.name().to_string();
60        let worker_future = child_spec.create_worker_future(process.clone(), shutdown_handle)?;
61        let shutdown_strategy = child_spec.shutdown_strategy();
62        let abort_handle = self.worker_tasks.spawn(worker_future.into_process_future(process));
63        self.worker_map.insert(
64            abort_handle.id(),
65            ProcessState {
66                worker_id,
67                worker_name,
68                shutdown_strategy,
69                shutdown_coordinator,
70                abort_handle,
71            },
72        );
73        Ok(())
74    }
75
76    /// Awaits the next worker to finish, returning its `worker_id` and result.
77    pub(super) async fn wait_for_next_worker(&mut self) -> (u64, Result<(), WorkerError>) {
78        debug!("Waiting for next process to complete.");
79
80        // If there are no workers to wait on, park indefinitely so the supervisor's select loop only proceeds via its
81        // other arms (shutdown, or a newly-added dynamic child). Without this guard, `join_next_with_id` would return
82        // `None` immediately on an empty set and the supervisor would busy-loop. The set legitimately empties when all
83        // children are non-restartable (e.g. `RestartType::Temporary`) and have exited.
84        if self.worker_tasks.is_empty() {
85            pending::<()>().await;
86        }
87
88        match self.worker_tasks.join_next_with_id().await {
89            Some(Ok((worker_task_id, worker_result))) => {
90                let process_state = self
91                    .worker_map
92                    .shift_remove(&worker_task_id)
93                    .expect("worker task ID not found");
94                (process_state.worker_id, worker_result)
95            }
96            Some(Err(e)) => {
97                let worker_task_id = e.id();
98                let process_state = self
99                    .worker_map
100                    .shift_remove(&worker_task_id)
101                    .expect("worker task ID not found");
102                let e = if e.is_cancelled() {
103                    ProcessError::Aborted
104                } else {
105                    ProcessError::Panicked
106                };
107                (process_state.worker_id, Err(WorkerError::Runtime(e.into())))
108            }
109            None => unreachable!(
110                "join set is non-empty here: we park above while empty, and only this method removes workers"
111            ),
112        }
113    }

115    /// Shuts down all workers, honoring each worker's shutdown strategy and the configured [`ShutdownMode`].
116    ///
117    /// Returns the number of workers that had to be forcefully aborted because they exceeded their graceful shutdown
118    /// timeout. The count includes aborts reported by nested child supervisors (which surface their own tally via
119    /// [`WorkerError::ShutdownTimedOut`]), so the value reflects the entire supervision tree rooted at this supervisor.
120    pub(super) async fn shutdown_workers(&mut self) -> usize {
121        debug!(shutdown_mode = ?self.shutdown_mode, "Shutting down all processes.");
122
123        let aborted = match self.shutdown_mode {
124            ShutdownMode::Ordered => self.shutdown_workers_ordered().await,
125            ShutdownMode::Concurrent => self.shutdown_workers_concurrent().await,
126        };
127
128        debug_assert!(self.worker_map.is_empty(), "worker map should be empty after shutdown");
129        debug_assert!(
130            self.worker_tasks.is_empty(),
131            "worker tasks should be empty after shutdown"
132        );
133
134        aborted
135    }
136
137    /// Shuts down all workers one at a time, in reverse order of insertion, honoring each worker's shutdown strategy.
138    ///
139    /// Returns the number of workers forcefully aborted after exceeding their graceful timeout, including counts
140    /// merged from nested child supervisors that timed out.
141    async fn shutdown_workers_ordered(&mut self) -> usize {
142        // Pop entries from the worker map, which grabs us workers in the reverse order they were added. This lets us
143        // ensure we're shutting down any _dependent_ processes (processes which depend on previously-started processes)
144        // first.
145        //
146        // For each entry, we trigger shutdown in whatever way necessary, and then wait for the process to exit by
147        // driving the `JoinSet`. If other workers complete while we're waiting, we'll simply remove them from the
148        // worker map and continue waiting for the current worker we're shutting down.
149        //
150        // We do this until the worker map is empty, at which point we can be sure that all processes have exited.
151        let mut aborted_total = 0;
152        while let Some((current_worker_task_id, process_state)) = self.worker_map.pop() {
153            let ProcessState {
154                worker_id,
155                worker_name,
156                shutdown_strategy,
157                shutdown_coordinator,
158                abort_handle,
159            } = process_state;
160
161            // Trigger the process to shutdown based on the configured shutdown strategy.
162            let shutdown_deadline = match shutdown_strategy {
163                ShutdownStrategy::Graceful(timeout) => {
164                    debug!(worker_id, shutdown_timeout = ?timeout, "Gracefully shutting down process.");
165                    shutdown_coordinator.shutdown();
166
167                    tokio::time::sleep(timeout)
168                }
169                ShutdownStrategy::Brutal => {
170                    debug!(worker_id, "Forcefully aborting process.");
171                    abort_handle.abort();
172
173                    // We have to return a future that never resolves, since we're already aborting it. This is a little
174                    // hacky but it's also difficult to do an optional future, so this is what we're going with for now.
175                    tokio::time::sleep(Duration::MAX)
176                }
177            };
178            pin!(shutdown_deadline);
179
180            // Wait for the process to exit by driving the `JoinSet`. If other workers complete while we're waiting,
181            // we'll simply remove them from the worker map and continue waiting.
182            let mut aborted = false;
183            loop {
184                select! {
185                    worker_result = self.worker_tasks.join_next_with_id() => {
186                        match worker_result {
187                            Some(Ok((worker_task_id, output))) => {
188                                // A nested child supervisor that timed out reports its own abort tally here; merge it.
189                                aborted_total += reported_abort_count(&output);
190                                if worker_task_id == current_worker_task_id {
191                                    debug!(?worker_task_id, "Target process exited successfully.");
192                                    break;
193                                } else {
194                                    debug!(?worker_task_id, "Non-target process exited successfully. Continuing to wait.");
195                                    let removed = self.worker_map.shift_remove(&worker_task_id);
196                                    debug_assert!(removed.is_some(), "non-target worker must be in the worker map");
197                                }
198                            },
199                            Some(Err(e)) => {
200                                let worker_task_id = e.id();
201                                if worker_task_id == current_worker_task_id {
202                                    debug!(?worker_task_id, "Target process exited with error.");
203                                    break;
204                                } else {
205                                    debug!(?worker_task_id, "Non-target process exited with error. Continuing to wait.");
206                                    let removed = self.worker_map.shift_remove(&worker_task_id);
207                                    debug_assert!(removed.is_some(), "non-target worker must be in the worker map");
208                                }
209                            }
210                            None => unreachable!("worker task must exist in join set if we are waiting for it"),
211                        }
212                    },
213                    // We've exceeded the shutdown timeout, so we abort the process. The `if !aborted` guard stops this
214                    // arm from re-firing on every poll once the deadline has elapsed (an elapsed `Sleep` stays ready),
215                    // which would otherwise spin re-aborting until the task is reaped.
216                    _ = &mut shutdown_deadline, if !aborted => {
217                        warn!(worker_id, worker_name = %worker_name, "Worker ignored graceful shutdown; forcefully aborting after timeout.");
218                        abort_handle.abort();
219                        aborted = true;
220                        aborted_total += 1;
221                    }
222                }
223            }
224        }
225
226        aborted_total
227    }

229    /// Shuts down all workers at once, waiting for them concurrently.
230    ///
231    /// Each worker is signalled up front, then awaited concurrently under its **own** graceful deadline, so a worker
232    /// that ignores shutdown is aborted at its configured timeout regardless of its siblings. Total shutdown time is
233    /// therefore bounded by the slowest individual worker rather than the sum of all timeouts, which suits large,
234    /// independent worker sets (for example, one task per network connection).
235    ///
236    /// A worker whose graceful timeout is effectively unbounded (such as a nested supervisor, which uses
237    /// `Duration::MAX` because it bounds itself via its own children's deadlines) is waited on indefinitely and is
238    /// never aborted by this method.
239    ///
240    /// Returns the number of workers forcefully aborted after exceeding their graceful timeout, including counts
241    /// merged from nested child supervisors that timed out.
242    async fn shutdown_workers_concurrent(&mut self) -> usize {
243        // Take ownership of all worker bookkeeping so we can consume each worker's shutdown coordinator. Signal every
244        // graceful worker and immediately abort brutal ones, recording a per-worker abort deadline so each is held to
245        // its own timeout rather than a single shared one.
246        let now = tokio::time::Instant::now();
247        let mut pending: FastIndexMap<Id, (u64, String, AbortHandle, Option<tokio::time::Instant>)> =
248            FastIndexMap::default();
249        for (task_id, process_state) in std::mem::take(&mut self.worker_map) {
250            let ProcessState {
251                worker_id,
252                worker_name,
253                shutdown_strategy,
254                shutdown_coordinator,
255                abort_handle,
256            } = process_state;
257
258            match shutdown_strategy {
259                ShutdownStrategy::Graceful(timeout) => {
260                    debug!(worker_id, shutdown_timeout = ?timeout, "Gracefully shutting down process.");
261                    shutdown_coordinator.shutdown();
262                    // An effectively-infinite timeout (`Duration::MAX`) maps to `None` -- "never abort" -- which is
263                    // correct for nested supervisors, since they bound themselves via their own children's deadlines.
264                    let deadline = (timeout != Duration::MAX).then(|| now + timeout);
265                    pending.insert(task_id, (worker_id, worker_name, abort_handle, deadline));
266                }
267                ShutdownStrategy::Brutal => {
268                    debug!(worker_id, "Forcefully aborting process.");
269                    abort_handle.abort();
270                }
271            }
272        }
273
274        // Wait for every task to exit. Each iteration sleeps until the earliest still-pending abort deadline; when it
275        // fires we abort exactly those workers whose own deadline has passed (their tasks are then reaped by a later
276        // `join_next`). Brutal workers were aborted above and aren't tracked here.
277        let mut aborted_total = 0;
278        while !self.worker_tasks.is_empty() {
279            match pending.values().filter_map(|(_, _, _, deadline)| *deadline).min() {
280                Some(deadline) => {
281                    select! {
282                        joined = self.worker_tasks.join_next_with_id() => {
283                            let task_id = match joined {
284                                Some(Ok((task_id, output))) => {
285                                    // A nested child supervisor that timed out reports its abort tally here; merge it.
286                                    aborted_total += reported_abort_count(&output);
287                                    Some(task_id)
288                                }
289                                Some(Err(e)) => Some(e.id()),
290                                None => None,
291                            };
292                            if let Some(task_id) = task_id {
293                                pending.swap_remove(&task_id);
294                            }
295                        }
296                        _ = tokio::time::sleep_until(deadline) => {
297                            let now = tokio::time::Instant::now();
298                            pending.retain(|_, (worker_id, worker_name, abort_handle, deadline)| {
299                                if deadline.is_some_and(|deadline| deadline <= now) {
300                                    warn!(worker_id = *worker_id, worker_name = %worker_name, "Worker ignored graceful shutdown; forcefully aborting after timeout.");
301                                    abort_handle.abort();
302                                    aborted_total += 1;
303                                    false
304                                } else {
305                                    true
306                                }
307                            });
308                        }
309                    }
310                }
311                // Only workers with no finite deadline remain (e.g. nested supervisors); wait for them to exit on their
312                // own. This is the path the topology takes -- each per-component supervisor is graceful-with-`MAX`, so
313                // its own forced-abort tally is reported here and merged into ours.
314                None => {
315                    if let Some(Ok((_, output))) = self.worker_tasks.join_next_with_id().await {
316                        aborted_total += reported_abort_count(&output);
317                    }
318                }
319            }
320        }
321
322        aborted_total
323    }
324}

326/// Extracts the number of force-aborts a reaped child reported.
327///
328/// A nested child supervisor that completes a requested shutdown after forcefully aborting one or more of its own
329/// workers returns [`WorkerError::ShutdownTimedOut`]; its tally is merged into the parent's so the count aggregates
330/// across the whole supervision tree. Any other completion (clean exit, our own abort surfacing as a cancellation,
331/// panic) contributes nothing here.
332fn reported_abort_count(output: &Result<(), WorkerError>) -> usize {
333    match output {
334        Err(WorkerError::ShutdownTimedOut { aborted }) => *aborted,
335        // A `Supervisable` worker that internally drives a supervisor (such as a topology blueprint) flattens that
336        // supervisor's `SupervisorError` into a `GenericError` at its boundary, so it surfaces here as `Runtime`
337        // rather than `ShutdownTimedOut`. Recover the structured count via downcast so it still aggregates upward; the
338        // concrete error type is preserved because the boundary converts with a plain `Into` (no added context).
339        Err(WorkerError::Runtime(e)) => match e.downcast_ref::<SupervisorError>() {
340            Some(SupervisorError::ShutdownTimedOut { aborted }) => *aborted,
341            _ => 0,
342        },
343        _ => 0,
344    }
